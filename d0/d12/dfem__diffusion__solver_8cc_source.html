47void
48Solver::Initialize()
49{
50 const std::string fname = "Solver::Initialize";
51 log.Log() << "\n"
52 << program_timer.GetTimeString() << " " << TextName()
53 << ": Initializing DFEM Diffusion solver ";
54
55 // Get grid
56 grid_ptr_ = GetCurrentHandler().GetGrid();
57 const auto& grid = *grid_ptr_;
58 if (grid_ptr_ == nullptr)
59 throw std::logic_error(std::string(__PRETTY_FUNCTION__) + " No grid defined.");
60
61 log.Log() << "Global num cells: " << grid.GetGlobalNumberOfCells();
62
63 // BIDs
64 auto globl_unique_bndry_ids = grid.GetDomainUniqueBoundaryIDs();
65
66 const auto& grid_boundary_id_map = grid_ptr_->GetBoundaryIDMap();
67 for (uint64_t bndry_id : globl_unique_bndry_ids)
68 {
69 if (grid_boundary_id_map.count(bndry_id) == 0)
70 throw std::logic_error(fname + ": Boundary id " + std::to_string(bndry_id) +
71 " does not have a name-assignment.");
72
73 const auto& bndry_name = grid_boundary_id_map.at(bndry_id);
74 if (boundary_preferences_.find(bndry_name) != boundary_preferences_.end())
75 {
76 BoundaryInfo bndry_info = boundary_preferences_.at(bndry_name);
77 auto& bndry_vals = bndry_info.second;
78 switch (bndry_info.first)
79 {
80 case BoundaryType::Reflecting:
81 {
82 boundaries_.insert(
83 std::make_pair(bndry_id, Boundary{BoundaryType::Reflecting, {0., 0., 0.}}));
84 log.Log() << "Boundary " << bndry_name << " set to reflecting.";
85 break;
86 }
87 case BoundaryType::Dirichlet:
88 {
89 if (bndry_vals.empty()) bndry_vals.resize(1, 0.0);
90 boundaries_.insert(
91 std::make_pair(bndry_id, Boundary{BoundaryType::Dirichlet, {bndry_vals[0], 0., 0.}}));
92 log.Log() << "Boundary " << bndry_name << " set to dirichlet.";
93 break;
94 }
95 case BoundaryType::Robin:
96 {
97 if (bndry_vals.size() != 3)
98 throw std::logic_error(std::string(__PRETTY_FUNCTION__) +
99 " Robin needs 3 values in bndry vals.");
100 boundaries_.insert(std::make_pair(
101 bndry_id,
102 Boundary{BoundaryType::Robin, {bndry_vals[0], bndry_vals[1], bndry_vals[2]}}));
103 log.Log() << "Boundary " << bndry_name << " set to robin." << bndry_vals[0] << ","
104 << bndry_vals[1] << "," << bndry_vals[2];
105 break;
106 }
107 case BoundaryType::Vacuum:
108 {
109 boundaries_.insert(
110 std::make_pair(bndry_id, Boundary{BoundaryType::Robin, {0.25, 0.5, 0.}}));
111 log.Log() << "Boundary " << bndry_name << " set to vacuum.";
112 break;
113 }
114 case BoundaryType::Neumann:
115 {
116 if (bndry_vals.size() != 3)
117 throw std::logic_error(std::string(__PRETTY_FUNCTION__) +
118 " Neumann needs 3 values in bndry vals.");
119 boundaries_.insert(std::make_pair(
120 bndry_id, Boundary{BoundaryType::Robin, {0., bndry_vals[0], bndry_vals[1]}}));
121 log.Log() << "Boundary " << bndry_name << " set to neumann." << bndry_vals[0];
122 break;
123 }
124 } // switch boundary type
125 }
126 else
127 {
128 boundaries_.insert(std::make_pair(bndry_id, Boundary{BoundaryType::Dirichlet, {0., 0., 0.}}));
129 log.Log0Verbose1() << "No boundary preference found for boundary index " << bndry_name
130 << "Dirichlet boundary added with zero boundary value.";
131 }
132 } // for bndry
133
134 // Make SDM
135 sdm_ptr_ = PieceWiseLinearDiscontinuous::New(*grid_ptr_);
136 const auto& sdm = *sdm_ptr_;
137
138 const auto& OneDofPerNode = sdm.UNITARY_UNKNOWN_MANAGER;
139
140 num_local_dofs_ = sdm.GetNumLocalDOFs(OneDofPerNode);
141 num_globl_dofs_ = sdm.GetNumGlobalDOFs(OneDofPerNode);
142
143 log.Log() << "Num local DOFs: " << num_local_dofs_;
144 log.Log() << "Num globl DOFs: " << num_globl_dofs_;
145
146 // Initializes Mats and Vecs
147 const auto n = static_cast<int64_t>(num_local_dofs_);
148 const auto N = static_cast<int64_t>(num_globl_dofs_);
149
150 A_ = CreateSquareMatrix(n, N);
151 x_ = CreateVector(n, N);
152 b_ = CreateVector(n, N);
153
154 std::vector<int64_t> nodal_nnz_in_diag;
155 std::vector<int64_t> nodal_nnz_off_diag;
156 sdm.BuildSparsityPattern(nodal_nnz_in_diag, nodal_nnz_off_diag, OneDofPerNode);
157
158 InitMatrixSparsity(A_, nodal_nnz_in_diag, nodal_nnz_off_diag);
159
160 if (field_functions_.empty())
161 {
162 std::string solver_name;
163 if (not TextName().empty()) solver_name = TextName() + "-";
164
165 std::string text_name = solver_name + "phi";
166
167 auto initial_field_function =
168 std::make_shared<FieldFunctionGridBased>(text_name, sdm_ptr_, Unknown(UnknownType::SCALAR));
169
170 field_functions_.push_back(initial_field_function);
171 field_function_stack.push_back(initial_field_function);
172 } // if not ff set
173}

175void
176Solver::Execute()
177{
178 log.Log() << "\nExecuting DFEM IP Diffusion solver";
179
180 const auto& grid = *grid_ptr_;
181 const auto& sdm = *sdm_ptr_;
182
183 // Assemble the system
184 // is this needed?
185 VecSet(b_, 0.0);
186
187 log.Log() << "Assembling system: ";
188
189 for (const auto& cell : grid.local_cells)
190 {
191 const auto& cell_mapping = sdm.GetCellMapping(cell);
192 const size_t num_nodes = cell_mapping.NumNodes();
193 const auto cc_nodes = cell_mapping.GetNodeLocations();
194 const auto qp_data = cell_mapping.MakeVolumetricQuadraturePointData();
195
196 const auto imat = cell.material_id_;
197 MatDbl Acell(num_nodes, VecDbl(num_nodes, 0.0));
198 VecDbl cell_rhs(num_nodes, 0.0);
199
200 // Assemble volumetric terms
201 for (size_t i = 0; i < num_nodes; ++i)
202 {
203 const int64_t imap = sdm.MapDOF(cell, i);
204
205 for (size_t j = 0; j < num_nodes; ++j)
206 {
207 const int64_t jmap = sdm.MapDOF(cell, j);
208 double entry_aij = 0.0;
209 for (size_t qp : qp_data.QuadraturePointIndices())
210 {
211 entry_aij += (d_coef_function_->Evaluate(imat, qp_data.QPointXYZ(qp)) *
212 qp_data.ShapeGrad(i, qp).Dot(qp_data.ShapeGrad(j, qp)) +
213 sigma_a_function_->Evaluate(imat, qp_data.QPointXYZ(qp)) *
214 qp_data.ShapeValue(i, qp) * qp_data.ShapeValue(j, qp)) *
215 qp_data.JxW(qp);
216 } // for qp
217 MatSetValue(A_, imap, jmap, entry_aij, ADD_VALUES);
218 } // for j
219 double entry_rhs_i = 0.0;
220 for (size_t qp : qp_data.QuadraturePointIndices())
221 entry_rhs_i += q_ext_function_->Evaluate(imat, qp_data.QPointXYZ(qp)) *
222 qp_data.ShapeValue(i, qp) * qp_data.JxW(qp);
223 VecSetValue(b_, imap, entry_rhs_i, ADD_VALUES);
224 } // for i
225
226 // Assemble face terms
227 const size_t num_faces = cell.faces_.size();
228 for (size_t f = 0; f < num_faces; ++f)
229 {
230 const auto& face = cell.faces_[f];
231 const auto& n_f = face.normal_;
232 const size_t num_face_nodes = cell_mapping.NumFaceNodes(f);
233 const auto fqp_data = cell_mapping.MakeSurfaceQuadraturePointData(f);
234
235 const double hm = HPerpendicular(cell, f);
236
237 typedef MeshContinuum Grid;
238
239 // interior face
240 if (face.has_neighbor_)
241 {
242 const auto& adj_cell = grid.cells[face.neighbor_id_];
243 const auto& adj_cell_mapping = sdm.GetCellMapping(adj_cell);
244 const auto ac_nodes = adj_cell_mapping.GetNodeLocations();
245 const size_t acf = Grid::MapCellFace(cell, adj_cell, f);
246 const double hp_neigh = HPerpendicular(adj_cell, acf);
247
248 const auto imat_neigh = adj_cell.material_id_;
249
250 // Compute Ckappa IP
251 double Ckappa = 1.0;
252 if (cell.Type() == CellType::SLAB) Ckappa = 2.0;
253 if (cell.Type() == CellType::POLYGON) Ckappa = 2.0;
254 if (cell.Type() == CellType::POLYHEDRON) Ckappa = 4.0;
255
256 // Assembly penalty terms
257 for (size_t fi = 0; fi < num_face_nodes; ++fi)
258 {
259 const int i = cell_mapping.MapFaceNode(f, fi);
260 const int64_t imap = sdm.MapDOF(cell, i);
261
262 for (size_t fj = 0; fj < num_face_nodes; ++fj)
263 {
264 const int jm = cell_mapping.MapFaceNode(f, fj); // j-minus
265 const int64_t jmmap = sdm.MapDOF(cell, jm);
266
267 const int jp =
268 MapFaceNodeDisc(cell, adj_cell, cc_nodes, ac_nodes, f, acf, fj); // j-plus
269 const int64_t jpmap = sdm.MapDOF(adj_cell, jp);
270
271 double aij = 0.0;
272 for (size_t qp : fqp_data.QuadraturePointIndices())
273 aij += Ckappa *
274 (d_coef_function_->Evaluate(imat, fqp_data.QPointXYZ(qp)) / hm +
275 d_coef_function_->Evaluate(imat_neigh, fqp_data.QPointXYZ(qp)) / hp_neigh) /
276 2. * fqp_data.ShapeValue(i, qp) * fqp_data.ShapeValue(jm, qp) *
277 fqp_data.JxW(qp);
278
279 MatSetValue(A_, imap, jmmap, aij, ADD_VALUES);
280 MatSetValue(A_, imap, jpmap, -aij, ADD_VALUES);
281 } // for fj
282 } // for fi
283
284 // Assemble gradient terms
285 // For the following comments we use the notation:
286 // Dk = 0.5* n dot nabla bk
287
288 // {{D d_n b_i}}[[Phi]]
289 // 0.5*D* n dot (b_j^+ - b_j^-)*nabla b_i^-
290
291 // loop over node of current cell (gradient of b_i)
292 for (int i = 0; i < num_nodes; ++i)
293 {
294 const int64_t imap = sdm.MapDOF(cell, i);
295
296 // loop over faces
297 for (int fj = 0; fj < num_face_nodes; ++fj)
298 {
299 const int jm = cell_mapping.MapFaceNode(f, fj); // j-minus
300 const int64_t jmmap = sdm.MapDOF(cell, jm);
301 const int jp =
302 MapFaceNodeDisc(cell, adj_cell, cc_nodes, ac_nodes, f, acf, fj); // j-plus
303 const int64_t jpmap = sdm.MapDOF(adj_cell, jp);
304
305 Vector3 vec_aij;
306 for (size_t qp : fqp_data.QuadraturePointIndices())
307 vec_aij += d_coef_function_->Evaluate(imat, fqp_data.QPointXYZ(qp)) *
308 fqp_data.ShapeValue(jm, qp) * fqp_data.ShapeGrad(i, qp) * fqp_data.JxW(qp);
309 const double aij = -0.5 * n_f.Dot(vec_aij);
310
311 MatSetValue(A_, imap, jmmap, aij, ADD_VALUES);
312 MatSetValue(A_, imap, jpmap, -aij, ADD_VALUES);
313 } // for fj
314 } // for i
315
316 // {{D d_n Phi}}[[b_i]]
317 // 0.5*D* n dot (b_i^+ - b_i^-)*nabla b_j^-
318 for (int fi = 0; fi < num_face_nodes; ++fi)
319 {
320 const int im = cell_mapping.MapFaceNode(f, fi); // i-minus
321 const int64_t immap = sdm.MapDOF(cell, im);
322
323 const int ip = MapFaceNodeDisc(cell, adj_cell, cc_nodes, ac_nodes, f, acf, fi); // i-plus
324 const int64_t ipmap = sdm.MapDOF(adj_cell, ip);
325
326 for (int j = 0; j < num_nodes; ++j)
327 {
328 const int64_t jmap = sdm.MapDOF(cell, j);
329
330 Vector3 vec_aij;
331 for (size_t qp : fqp_data.QuadraturePointIndices())
332 vec_aij += d_coef_function_->Evaluate(imat, fqp_data.QPointXYZ(qp)) *
333 fqp_data.ShapeValue(im, qp) * fqp_data.ShapeGrad(j, qp) * fqp_data.JxW(qp);
334 const double aij = -0.5 * n_f.Dot(vec_aij);
335
336 MatSetValue(A_, immap, jmap, aij, ADD_VALUES);
337 MatSetValue(A_, ipmap, jmap, -aij, ADD_VALUES);
338 } // for j
339 } // for fi
340
341 } // internal face
342 else
343 { // boundary face
344 const auto& bndry = boundaries_[face.neighbor_id_];
345 // Robin boundary
346 if (bndry.type_ == BoundaryType::Robin)
347 {
348 const auto qp_face_data = cell_mapping.MakeSurfaceQuadraturePointData(f);
349
350 const auto& aval = bndry.values_[0];
351 const auto& bval = bndry.values_[1];
352 const auto& fval = bndry.values_[2];
353
354 log.Log0Verbose1() << "Boundary set as Robin with a,b,f = (" << aval << "," << bval
355 << "," << fval << ") ";
356 // true Robin when a!=0, otherwise, it is a Neumann:
357 // Assert if b=0
358 if (std::fabs(bval) < 1e-8)
359 throw std::logic_error("if b=0, this is a Dirichlet BC, not a Robin BC");
360
361 for (size_t fi = 0; fi < num_face_nodes; fi++)
362 {
363 const uint i = cell_mapping.MapFaceNode(f, fi);
364 const int64_t ir = sdm.MapDOF(cell, i);
365
366 if (std::fabs(aval) >= 1.0e-12)
367 {
368 for (size_t fj = 0; fj < num_face_nodes; fj++)
369 {
370 const uint j = cell_mapping.MapFaceNode(f, fj);
371 const int64_t jr = sdm.MapDOF(cell, j);
372
373 double aij = 0.0;
374 for (size_t qp : fqp_data.QuadraturePointIndices())
375 aij += fqp_data.ShapeValue(i, qp) * fqp_data.ShapeValue(j, qp) * fqp_data.JxW(qp);
376 aij *= (aval / bval);
377
378 MatSetValue(A_, ir, jr, aij, ADD_VALUES);
379 } // for fj
380 } // if a nonzero
381
382 if (std::fabs(fval) >= 1.0e-12)
383 {
384 double rhs_val = 0.0;
385 for (size_t qp : fqp_data.QuadraturePointIndices())
386 rhs_val += fqp_data.ShapeValue(i, qp) * fqp_data.JxW(qp);
387 rhs_val *= (fval / bval);
388
389 VecSetValue(b_, ir, rhs_val, ADD_VALUES);
390 } // if f nonzero
391 } // for fi
392 } // Robin BC
393 else if (bndry.type_ == BoundaryType::Dirichlet)
394 {
395 const double bc_value = bndry.values_[0];
396 // Compute kappa
397 double Ckappa = 2.0;
398 if (cell.Type() == CellType::SLAB) Ckappa = 4.0; // fmax(4.0*Dg/hm,0.25);
399 if (cell.Type() == CellType::POLYGON) Ckappa = 4.0;
400 if (cell.Type() == CellType::POLYHEDRON) Ckappa = 8.0;
401
402 // Assembly penalty terms
403 for (size_t fi = 0; fi < num_face_nodes; ++fi)
404 {
405 const uint i = cell_mapping.MapFaceNode(f, fi);
406 const int64_t imap = sdm.MapDOF(cell, i);
407
408 for (size_t fj = 0; fj < num_face_nodes; ++fj)
409 {
410 const uint jm = cell_mapping.MapFaceNode(f, fj);
411 const int64_t jmmap = sdm.MapDOF(cell, jm);
412
413 double aij = 0.0;
414 for (size_t qp : fqp_data.QuadraturePointIndices())
415 aij += Ckappa * d_coef_function_->Evaluate(imat, fqp_data.QPointXYZ(qp)) / hm *
416 fqp_data.ShapeValue(i, qp) * fqp_data.ShapeValue(jm, qp) * fqp_data.JxW(qp);
417 double aij_bc_value = aij * bc_value;
418
419 MatSetValue(A_, imap, jmmap, aij, ADD_VALUES);
420 VecSetValue(b_, imap, aij_bc_value, ADD_VALUES);
421 } // for fj
422 } // for fi
423
424 // Assemble gradient terms
425 // For the following comments we use the notation:
426 // Dk = 0.5* n dot nabla bk
427
428 // 0.5*D* n dot (b_j^+ - b_j^-)*nabla b_i^-
429 for (size_t i = 0; i < num_nodes; i++)
430 {
431 const int64_t imap = sdm.MapDOF(cell, i);
432
433 for (size_t j = 0; j < num_nodes; j++)
434 {
435 const int64_t jmap = sdm.MapDOF(cell, j);
436
437 Vector3 vec_aij;
438 for (size_t qp : fqp_data.QuadraturePointIndices())
439 vec_aij += (fqp_data.ShapeValue(j, qp) * fqp_data.ShapeGrad(i, qp) +
440 fqp_data.ShapeValue(i, qp) * fqp_data.ShapeGrad(j, qp)) *
441 fqp_data.JxW(qp) *
442 d_coef_function_->Evaluate(imat, fqp_data.QPointXYZ(qp));
443
444 const double aij = -n_f.Dot(vec_aij);
445 double aij_bc_value = aij * bc_value;
446
447 MatSetValue(A_, imap, jmap, aij, ADD_VALUES);
448 VecSetValue(b_, imap, aij_bc_value, ADD_VALUES);
449 } // for fj
450 } // for i
451 } // Dirichlet BC
452 else {} // else BC
453 } // boundary face
454 } // for face f
455 } // for cell
456
457 log.Log() << "Global assembly";
458
459 MatAssemblyBegin(A_, MAT_FINAL_ASSEMBLY);
460 MatAssemblyEnd(A_, MAT_FINAL_ASSEMBLY);
461 VecAssemblyBegin(b_);
462 VecAssemblyEnd(b_);
463
464 // MatView(A, PETSC_VIEWER_STDERR_WORLD);
465 //
466 // PetscViewer viewer;
467 // PetscViewerASCIIOpen(opensn::mpi_comm,"A.m",&viewer);
468 // PetscViewerPushFormat(viewer, PETSC_VIEWER_ASCII_MATLAB);
469 // MatView(A,viewer);
470 // PetscViewerPopFormat(viewer);
471 // PetscViewerDestroy(&viewer);
472
473 log.Log() << "Done global assembly";
474
475 // Create Krylov Solver
476 log.Log() << "Solving: ";
477 auto petsc_solver =
478 CreateCommonKrylovSolverSetup(A_,
479 TextName(),
480 KSPCG,
481 PCGAMG,
482 basic_options_("residual_tolerance").FloatValue(),
483 basic_options_("max_iters").IntegerValue());
484
485 // Solve
486 KSPSolve(petsc_solver.ksp, b_, x_);
487
488 log.Log() << "Done solving";
489
490 const auto& OneDofPerNode = sdm.UNITARY_UNKNOWN_MANAGER;
491 sdm.LocalizePETScVector(x_, field_, OneDofPerNode);
492
493 field_functions_.front()->UpdateFieldVector(field_);
494}

496double
497Solver::HPerpendicular(const Cell& cell, unsigned int f)
498{
499 const auto& sdm = *sdm_ptr_;
500
501 const auto& cell_mapping = sdm.GetCellMapping(cell);
502 double hp;
503
504 const size_t num_faces = cell.faces_.size();
505 const size_t num_vertices = cell.vertex_ids_.size();
506
507 const double volume = cell_mapping.CellVolume();
508 const double face_area = cell_mapping.FaceArea(f);
509
510 /**Lambda to compute surface area.*/
511 auto ComputeSurfaceArea = [&cell_mapping, &num_faces]()
512 {
513 double surface_area = 0.0;
514 for (size_t fr = 0; fr < num_faces; ++fr)
515 surface_area += cell_mapping.FaceArea(fr);
516
517 return surface_area;
518 };
519
520 //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% SLAB
521 if (cell.Type() == CellType::SLAB) hp = volume / 2.0;
522 //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% POLYGON
523 else if (cell.Type() == CellType::POLYGON)
524 {
525 if (num_faces == 3) hp = 2.0 * volume / face_area;
526 else if (num_faces == 4)
527 hp = volume / face_area;
528 else // Nv > 4
529 {
530 const double surface_area = ComputeSurfaceArea();
531
532 if (num_faces % 2 == 0) hp = 4.0 * volume / surface_area;
533 else
534 {
535 hp = 2.0 * volume / surface_area;
536 hp +=
537 sqrt(2.0 * volume /
538 (static_cast<double>(num_faces) * sin(2.0 * M_PI / static_cast<double>(num_faces))));
539 }
540 }
541 }
542 //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% POLYHEDRON
543 else if (cell.Type() == CellType::POLYHEDRON)
544 {
545 const double surface_area = ComputeSurfaceArea();
546
547 if (num_faces == 4) // Tet
548 hp = 3 * volume / surface_area;
549 else if (num_faces == 6 && num_vertices == 8) // Hex
550 hp = volume / surface_area;
551 else // Polyhedron
552 hp = 6 * volume / surface_area;
553 } // Polyhedron
554 else
555 throw std::logic_error("Solver::HPerpendicular: "
556 "Unsupported cell type in call to HPerpendicular");
557
558 return hp;
559}

561int
562Solver::MapFaceNodeDisc(const Cell& cur_cell,
563 const Cell& adj_cell,
564 const std::vector<Vector3>& cc_node_locs,
565 const std::vector<Vector3>& ac_node_locs,
566 size_t ccf,
567 size_t acf,
568 size_t ccfi,
569 double epsilon)
570{
571 const auto& sdm = *sdm_ptr_;
572
573 const auto& cur_cell_mapping = sdm.GetCellMapping(cur_cell);
574 const auto& adj_cell_mapping = sdm.GetCellMapping(adj_cell);
575
576 const int i = cur_cell_mapping.MapFaceNode(ccf, ccfi);
577 const auto& node_i_loc = cc_node_locs[i];
578
579 const size_t adj_face_num_nodes = adj_cell_mapping.NumFaceNodes(acf);
580
581 for (size_t fj = 0; fj < adj_face_num_nodes; ++fj)
582 {
583 const int j = adj_cell_mapping.MapFaceNode(acf, fj);
584 if ((node_i_loc - ac_node_locs[j]).NormSquare() < epsilon) return j;
585 }
586
587 throw std::logic_error("Solver::MapFaceNodeDisc: Mapping failure.");
588}

590void
591Solver::UpdateFieldFunctions()
