24std::shared_ptr<FiniteVolume>
25FiniteVolume::New(const MeshContinuum& in_grid, CoordinateSystemType in_cs_type)
26{
27 // First try to find an existing spatial discretization that matches the
28 // one requested.
29 for (auto& sdm : sdm_stack)
30 if (sdm->Type() == SpatialDiscretizationType::FINITE_VOLUME and
31 std::addressof(sdm->Grid()) == std::addressof(in_grid) and
32 sdm->GetCoordinateSystemType() == in_cs_type)
33 {
34 auto sdm_ptr = std::dynamic_pointer_cast<FiniteVolume>(sdm);
35
36 ChiLogicalErrorIf(not sdm_ptr, "Casting failure");
37
38 return sdm_ptr;
39 }
40
41 // If no existing discretization was found then go ahead and make a
42 // new one
43 auto new_sdm = std::shared_ptr<FiniteVolume>(new FiniteVolume(in_grid, in_cs_type));
44
45 sdm_stack.push_back(new_sdm);
46
47 return new_sdm;
48}

51FiniteVolume::CreateCellMappings()
52{
53 constexpr std::string_view fname = "SpatialDiscretization_FV::"
54 "CreateCellMappings";
55
56 auto MakeCellMapping = [this, fname](const Cell& cell)
57 {
58 using namespace std;
59 using namespace opensn;
60 std::unique_ptr<CellMapping> mapping;
61
62 switch (cell.Type())
63 {
64 case CellType::SLAB:
65 case CellType::POLYGON:
66 case CellType::POLYHEDRON:
67 {
68 typedef std::vector<std::vector<int>> FaceDofMapping;
69 mapping = make_unique<FiniteVolumeMapping>(
70 ref_grid_, cell, cell.centroid_, FaceDofMapping(cell.faces_.size(), {-1}));
71 break;
72 }
73 default:
74 throw std::logic_error(std::string(fname) +
75 std::string(": Invalid cell type encountered."));
76 }
77 return mapping;
78 };
79
80 for (const auto& cell : ref_grid_.local_cells)
81 cell_mappings_.push_back(MakeCellMapping(cell));
82
83 const auto ghost_ids = ref_grid_.cells.GetGhostGlobalIDs();
84 for (uint64_t ghost_id : ghost_ids)
85 {
86 auto ghost_mapping = MakeCellMapping(ref_grid_.cells[ghost_id]);
87 nb_cell_mappings_.insert(std::make_pair(ghost_id, std::move(ghost_mapping)));
88 }

91void
92FiniteVolume::OrderNodes()
93{
94 // Communicate node counts
95 const uint64_t local_num_nodes = ref_grid_.local_cells.size();
96 mpi_comm.all_gather(local_num_nodes, locJ_block_size_);
97
98 // Build block addresses
99 locJ_block_address_.assign(opensn::mpi_comm.size(), 0);
100 uint64_t global_num_nodes = 0;
101 for (int j = 0; j < opensn::mpi_comm.size(); ++j)
102 {
103 locJ_block_address_[j] = global_num_nodes;
104 global_num_nodes += locJ_block_size_[j];
105 }
106
107 local_block_address_ = locJ_block_address_[opensn::mpi_comm.rank()];
108
109 local_base_block_size_ = local_num_nodes;
110 globl_base_block_size_ = global_num_nodes;
111
112 // Sort neigbor ids
113 const auto neighbor_gids = ref_grid_.cells.GetGhostGlobalIDs();
114 std::map<uint64_t, std::vector<uint64_t>> sorted_nb_gids;
115 for (uint64_t gid : neighbor_gids)
116 {
117 const auto& cell = ref_grid_.cells[gid];
118 sorted_nb_gids[cell.partition_id_].push_back(gid);
119 }
120
121 // Communicate neighbor ids requiring mapping
122 const auto query_nb_gids = MapAllToAll(sorted_nb_gids, mpi_comm);
123
124 // Map the ids
125 std::map<uint64_t, std::vector<uint64_t>> mapped_query_nb_gids;
126 for (const auto& pid_list_pair : query_nb_gids)
127 {
128 const uint64_t pid = pid_list_pair.first;
129 const auto& gids = pid_list_pair.second;
130
131 auto& local_ids = mapped_query_nb_gids[pid];
132 local_ids.reserve(gids.size());
133 for (uint64_t gid : gids)
134 {
135 if (not ref_grid_.IsCellLocal(gid)) throw std::logic_error(mapping_error);
136
137 const auto& local_cell = ref_grid_.cells[gid];
138 local_ids.push_back(local_cell.local_id_);
139 } // for gid
140 } // for pid_list_pair
141
142 // Communicate back the mapped ids
143 const auto mapped_nb_gids = MapAllToAll(mapped_query_nb_gids, mpi_comm);
144
145 // Create the neighbor cell mapping
146 neighbor_cell_local_ids_.clear();
147 for (const auto& pid_list_pair : sorted_nb_gids)
148 {
149 try
150 {
151 const auto& pid = pid_list_pair.first;
152 const auto& gid_list = pid_list_pair.second;
153 const auto& lid_list = mapped_nb_gids.at(pid);
154
155 if (gid_list.size() != lid_list.size())
156 throw std::logic_error(mapping_error + std::string(" Size-mismatch."));
157
158 for (size_t i = 0; i < gid_list.size(); ++i)
159 neighbor_cell_local_ids_.insert(std::make_pair(gid_list[i], lid_list[i]));
160 }
161 catch (const std::out_of_range& oor)
162 {
163 throw std::logic_error(mapping_error + std::string(" OOR."));
164 }
165 } // for pid_list_pair
166
167 local_base_block_size_ = ref_grid_.local_cells.size();
168 globl_base_block_size_ = ref_grid_.GetGlobalNumberOfCells();
169}

171void
172FiniteVolume::BuildSparsityPattern(std::vector<int64_t>& nodal_nnz_in_diag,
173 std::vector<int64_t>& nodal_nnz_off_diag,
174 const UnknownManager& unknown_manager) const
175{
176 unsigned int num_uk = unknown_manager.unknowns_.size(); // Number of unknowns
177 unsigned int N = unknown_manager.GetTotalUnknownStructureSize(); // Total number of unknowns
178
179 nodal_nnz_in_diag.clear();
180 nodal_nnz_off_diag.clear();
181
182 const size_t num_local_cells = ref_grid_.local_cells.size();
183
184 nodal_nnz_in_diag.resize(num_local_cells * N, 0.0);
185 nodal_nnz_off_diag.resize(num_local_cells * N, 0.0);
186
187 for (int uk = 0; uk < num_uk; ++uk)
188 {
189 const unsigned int num_comps = unknown_manager.unknowns_[uk].num_components_;
190 for (int comp = 0; comp < num_comps; ++comp)
191 {
192 for (auto& cell : ref_grid_.local_cells)
193 {
194 const int64_t i = MapDOFLocal(cell, 0, unknown_manager, uk, comp);
195
196 nodal_nnz_in_diag[i] += 1;
197
198 for (auto& face : cell.faces_)
199 {
200 if (not face.has_neighbor_) continue;
201
202 if (face.IsNeighborLocal(ref_grid_)) nodal_nnz_in_diag[i] += 1;
203 else
204 nodal_nnz_off_diag[i] += 1;
205 }
206 } // for cell
207 } // for components
208 } // for unknown
209}

211int64_t
212FiniteVolume::MapDOF(const Cell& cell,
213 const unsigned int,
214 const UnknownManager& unknown_manager,
215 const unsigned int unknown_id,
216 const unsigned int component) const
217{
218 auto storage = unknown_manager.dof_storage_type_;
219
220 const size_t num_unknowns = unknown_manager.GetTotalUnknownStructureSize();
221 const size_t block_id = unknown_manager.MapUnknown(unknown_id, component);
222 const size_t num_local_cells = ref_grid_.local_cells.size();
223
224 if (component >= num_unknowns) return -1;
225
226 int64_t address = -1;
227 if (cell.partition_id_ == opensn::mpi_comm.rank())
228 {
229 if (storage == UnknownStorageType::BLOCK)
230 address = static_cast<int64_t>(local_block_address_) * num_unknowns +
231 num_local_cells * block_id + cell.local_id_;
232 else if (storage == UnknownStorageType::NODAL)
233 address = static_cast<int64_t>(local_block_address_) * num_unknowns +
234 cell.local_id_ * num_unknowns + block_id;
235 }
236 else
237 {
238 const uint64_t ghost_local_id = neighbor_cell_local_ids_.at(cell.global_id_);
239
240 if (storage == UnknownStorageType::BLOCK)
241 address = static_cast<int64_t>(locJ_block_address_[cell.partition_id_]) * num_unknowns +
242 locJ_block_size_[cell.partition_id_] * block_id + ghost_local_id;
243 else if (storage == UnknownStorageType::NODAL)
244 address = static_cast<int64_t>(locJ_block_address_[cell.partition_id_]) * num_unknowns +
245 ghost_local_id * num_unknowns + block_id;
246 }
247
248 return address;
249}

252FiniteVolume::MapDOFLocal(const Cell& cell,
253 const unsigned int,
254 const UnknownManager& unknown_manager,
255 const unsigned int unknown_id,
256 const unsigned int component) const
257{
258 auto storage = unknown_manager.dof_storage_type_;
259
260 const size_t num_unknowns = unknown_manager.GetTotalUnknownStructureSize();
261 const size_t block_id = unknown_manager.MapUnknown(unknown_id, component);
262 const size_t num_local_cells = ref_grid_.local_cells.size();
263
264 if (component >= num_unknowns) return -1;
265
266 int64_t address = -1;
267 if (cell.partition_id_ == opensn::mpi_comm.rank())
268 {
269 if (storage == UnknownStorageType::BLOCK)
270 address = static_cast<int64_t>(num_local_cells) * block_id + cell.local_id_;
271 else if (storage == UnknownStorageType::NODAL)
272 address = static_cast<int64_t>(cell.local_id_) * num_unknowns + block_id;
273 }
274 else
275 {
276 const size_t num_local_dofs = GetNumLocalDOFs(unknown_manager);
277 const size_t num_ghost_nodes = GetNumGhostDOFs(UNITARY_UNKNOWN_MANAGER);
278 const uint64_t ghost_local_id = ref_grid_.cells.GetGhostLocalID(cell.global_id_);
279
280 if (storage == UnknownStorageType::BLOCK)
281 address = static_cast<int64_t>(num_local_dofs) +
282 static_cast<int64_t>(num_ghost_nodes) * block_id + ghost_local_id;
283 else if (storage == UnknownStorageType::NODAL)
284 address = static_cast<int64_t>(num_local_dofs) +
285 num_unknowns * static_cast<int64_t>(ghost_local_id) + block_id;
286 }
