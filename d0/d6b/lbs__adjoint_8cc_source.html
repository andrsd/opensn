20std::array<double, 2>
21MakeExpRepFromP1(const std::array<double, 4>& P1_moments, bool verbose)
22{
23 // Custom function to implement the non-linear equations
24 // that make up the system to solve the a and b coefficients
25 // of an exponential representation of the form exp(a + b*mu).
26 // The coefficients must be such that the following equations
27 // hold:
28 // 2\pi \int_{-1}^{+1} e^{a + b\mu} d\mu = \phi
29 // 2\pi \int_{-1}^{+1} e^{a + b\mu} \mu d\mu = ||J||_2
30 //
31 // These integrals are analytical in \mu and therefore results
32 // in the equations for the F-method. The Jacobian matrix is returned
33 // with the J-method.
34 //
35 // Note that the function only accept the current normalized with
36 // 1/phi. So phi is always 1.0.
37 class CustomF : public NonLinearFunction
38 {
39 private:
40 double J_x, J_y, J_z;
41
42 public:
43 /**Constructor. Parameters are the 3 components of the currents
44 * normalized with 1/phi.*/
45 explicit CustomF(const std::array<double, 3>& input)
46 : J_x(input[0]), J_y(input[1]), J_z(input[2])
47 {
48 }
49
50 /**Function evaluation at vector-x.*/
51 VecDbl F(const VecDbl& x) const override
52 {
53 assert(x.size() == 2);
54 const double a = x[0];
55 const double b = x[1];
56 const double FOUR_PI = 4.0 * M_PI;
57
58 double size_J = Vec2Norm({J_x, J_y, J_z});
59
60 return {(FOUR_PI / b) * exp(a) * sinh(b) - 1.0,
61 (FOUR_PI / b / b) * exp(a) * (b * cosh(b) - sinh(b)) - size_J};
62 }
63 /**Jacobian evaluation at vector-x.*/
64 MatDbl J(const VecDbl& x) const override
65 {
66 assert(x.size() == 2);
67 const double a = x[0];
68 const double b = x[1];
69 const double FOUR_PI = 4.0 * M_PI;
70
71 return {
72 {(FOUR_PI / b) * exp(a) * sinh(b), (FOUR_PI / b / b) * exp(a) * (b * cosh(b) - sinh(b))},
73 {(FOUR_PI / b / b) * exp(a) * (b * cosh(b) - sinh(b)),
74 (FOUR_PI / b / b / b) * exp(a) * ((b * b + 2) * sinh(b) - 2 * b * cosh(b))}};
75 }
76 };
77
78 // Convert P1 moments to phi and J
79 double phi = P1_moments[0];
80 double J_x = P1_moments[1];
81 double J_y = P1_moments[2];
82 double J_z = P1_moments[3];
83
84 // Compute initial ratio size_J/phi
85 double size_J_i = Vec2Norm({J_x, J_y, J_z});
86 double ratio_i = size_J_i / phi;
87
88 if (phi < 1.0e-10 or ratio_i > 0.9)
89 {
90 J_x *= 0.9 / size_J_i;
91 J_y *= 0.9 / size_J_i;
92 J_z *= 0.9 / size_J_i;
93 }
94 else
95 {
96 J_x /= phi;
97 J_y /= phi;
98 J_z /= phi;
99 }
100
101 double size_J_f = Vec2Norm({J_x, J_y, J_z});
102 double ratio_f = size_J_f;
103
104 if (verbose)
105 {
106 std::stringstream outstr;
107 outstr << "P1 moments initial: " << P1_moments[0] << " " << P1_moments[1] << " "
108 << P1_moments[2] << " " << P1_moments[3] << " |J|=";
109 outstr << size_J_i << " ratio=" << ratio_i << "\n";
110 outstr << "P1 moments initial: " << 1.0 << " " << J_x << " " << J_y << " " << J_z << " |J|=";
111 outstr << size_J_f << " ratio=" << ratio_f << "\n";
112
113 log.Log() << outstr.str();
114 }
115
116 if (size_J_f < 1.0e-10)
117 {
118 double a = std::log(phi / 4.0 / M_PI);
119 double b = 0.0;
120
121 if (verbose) { log.Log() << "Solution: " << a << " " << b; }
122
123 return {a, b};
124 }
125 else
126 {
127 CustomF custom_function({J_x, J_y, J_z});
128 auto solution = NewtonIteration(custom_function, {1.0, 0.1}, 100, 1.0e-8, verbose);
129
130 double a = solution[0];
131 double b = solution[1];
132
133 if (verbose) { log.Log() << "Solution: " << a << " " << b; }
134
135 return {a, b};
136 }
137}
