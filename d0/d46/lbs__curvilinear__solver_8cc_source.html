258{

261 auto qorder = QuadratureOrder::INVALID_ORDER;
262 auto system = CoordinateSystemType::UNDEFINED;
263
264 // primary discretisation
265 switch (options_.geometry_type)
266 {
267 case lbs::GeometryType::ONED_SPHERICAL:
268 {
269 qorder = QuadratureOrder::FOURTH;

274 case lbs::GeometryType::TWOD_CYLINDRICAL:
275 {

279 }

290 typedef PieceWiseLinearDiscontinuous SDM_PWLD;
291 discretization_ = SDM_PWLD::New(*grid_ptr_, qorder, system);
292
293 ComputeUnitIntegrals();
294
295 // secondary discretisation
296 // system - manipulated such that the spatial discretisation returns
297 // a cell view of the same type but with weighting of degree one less
298 // than the primary discretisation
299 switch (options_.geometry_type)
300 {
301 case lbs::GeometryType::ONED_SPHERICAL:
302 {
303 qorder = QuadratureOrder::THIRD;
304 system = CoordinateSystemType::CYLINDRICAL;
305 break;

308 case lbs::GeometryType::TWOD_CYLINDRICAL:
309 {

313 }

318 << "invalid geometry, static_cast<int>(type) = "

329void
330DiscreteOrdinatesCurvilinearSolver::ComputeSecondaryUnitIntegrals()
331{
332 log.Log() << "Computing RZ secondary unit integrals.\n";
333 const auto& sdm = *discretization_;
334
335 // Define spatial weighting functions
336 std::function<double(const Vector3&)> swf =
337 SpatialDiscretization::CylindricalRZSpatialWeightFunction;
338
339 // Define lambda for cell-wise comps
340 auto ComputeCellUnitIntegrals = [&sdm, &swf](const Cell& cell)
341 {
342 const auto& cell_mapping = sdm.GetCellMapping(cell);
343 // const size_t cell_num_faces = cell.faces.size();
344 const size_t cell_num_nodes = cell_mapping.NumNodes();
345 const auto vol_qp_data = cell_mapping.MakeVolumetricQuadraturePointData();
346
347 MatDbl IntV_shapeI_shapeJ(cell_num_nodes, VecDbl(cell_num_nodes));
348
349 // Volume integrals
350 for (unsigned int i = 0; i < cell_num_nodes; ++i)
351 {
352 for (unsigned int j = 0; j < cell_num_nodes; ++j)
353 {
354 for (const auto& qp : vol_qp_data.QuadraturePointIndices())
355 {
356 IntV_shapeI_shapeJ[i][j] += swf(vol_qp_data.QPointXYZ(qp)) *
357 vol_qp_data.ShapeValue(i, qp) *
358 vol_qp_data.ShapeValue(j, qp) * vol_qp_data.JxW(qp);
359 } // for qp
360 } // for j
361 } // for i
362
363 return lbs::UnitCellMatrices{{},
364 {},
365 IntV_shapeI_shapeJ,
366 {},
367
368 {},
369 {},
370 {}};
371 };
372
373 const size_t num_local_cells = grid_ptr_->local_cells.size();
374 secondary_unit_cell_matrices_.resize(num_local_cells);
375
376 for (const auto& cell : grid_ptr_->local_cells)
377 secondary_unit_cell_matrices_[cell.local_id_] = ComputeCellUnitIntegrals(cell);
378
379 opensn::mpi_comm.barrier();
380 log.Log() << "Secondary Cell matrices computed. Process memory = " << std::setprecision(3)
381 << GetMemoryUsageInMB() << " MB";
382}
