86SweepWGSContext::SystemSize()
87{
88 const size_t local_node_count = lbs_solver_.LocalNodeCount();

91

94 const size_t local_size =
95 local_node_count * num_moments * groupset_numgrps + num_delayed_psi_info.first;
96 const size_t globl_size =
97 globl_node_count * num_moments * groupset_numgrps + num_delayed_psi_info.second;
98 const size_t num_angles = groupset_.quadrature_->abscissae_.size();
99 const size_t num_psi_global = globl_node_count * num_angles * groupset_.groups_.size();
100 const size_t num_delayed_psi_globl = num_delayed_psi_info.second;
101
102 if (log_info_)
103 {
104 log.Log() << "Total number of angular unknowns: " << num_psi_global << "\n"
105 << "Number of lagged angular unknowns: " << num_delayed_psi_globl << "("
106 << std::setprecision(2)
107 << static_cast<double>(num_delayed_psi_globl) * 100 /
108 static_cast<double>(num_psi_global)
109 << "%)";
110 }
111
112 return {static_cast<int64_t>(local_size), static_cast<int64_t>(globl_size)};

115void
116SweepWGSContext::ApplyInverseTransportOperator(SourceFlags scope)
117{
118 ++counter_applications_of_inv_op_;
119 const bool use_bndry_source_flag =
120 (scope & APPLY_FIXED_SOURCES) and (not lbs_solver_.Options().use_src_moments);
121
122 sweep_scheduler_.SetBoundarySourceActiveFlag(use_bndry_source_flag);
123
124 if (scope & ZERO_INCOMING_DELAYED_PSI) sweep_scheduler_.ZeroIncomingDelayedPsi();
125
126 // Sweep
127 sweep_scheduler_.ZeroOutputFluxDataStructures();
128 sweep_scheduler_.Sweep();

131void
132SweepWGSContext::PostSolveCallback()
133{
134 // Perform final sweep with converged phi and delayed psi dofs
135 if (groupset_.iterative_method_ != IterativeMethod::KRYLOV_RICHARDSON)
136 {
137 lbs_ss_solver_.ZeroOutflowBalanceVars(groupset_);
138
139 const auto scope = lhs_src_scope_ | rhs_src_scope_;
140
141 set_source_function_(groupset_, lbs_solver_.QMomentsLocal(), lbs_solver_.PhiOldLocal(), scope);
142 sweep_scheduler_.SetDestinationPhi(lbs_solver_.PhiNewLocal());
143
144 ApplyInverseTransportOperator(scope);
145
146 lbs_solver_.GSScopedCopyPrimarySTLvectors(
147 groupset_, PhiSTLOption::PHI_NEW, PhiSTLOption::PHI_OLD);
148 }
149
150 // Print solution info
151 {
152 double sweep_time = sweep_scheduler_.GetAverageSweepTime();
153 double chunk_overhead_ratio = 1.0 - sweep_scheduler_.GetAngleSetTimings()[2];
154 double source_time =
155 log.ProcessEvent(lbs_solver_.GetSourceEventTag(), Logger::EventOperation::AVERAGE_DURATION);
156 size_t num_angles = groupset_.quadrature_->abscissae_.size();
157 size_t num_unknowns = lbs_solver_.GlobalNodeCount() * num_angles * groupset_.groups_.size();
158
159 if (log_info_)
160 {
161 log.Log() << "\n\n";
162 log.Log() << " Set Src Time/sweep (s): " << source_time;
163 log.Log() << " Average sweep time (s): " << sweep_time;
164 log.Log() << " Chunk-Overhead-Ratio : " << chunk_overhead_ratio;
165 log.Log() << " Sweep Time/Unknown (ns): "
166 << sweep_time * 1.0e9 * opensn::mpi_comm.size() / static_cast<double>(num_unknowns);
167 log.Log() << " Number of unknowns per sweep: " << num_unknowns;
168 log.Log() << "\n\n";
169
170 std::string sweep_log_file_name =
171 std::string("GS_") + std::to_string(groupset_.id_) + std::string("_SweepLog_") +
172 std::to_string(opensn::mpi_comm.rank()) + std::string(".log");
173 groupset_.PrintSweepInfoFile(sweep_scheduler_.SweepEventTag(), sweep_log_file_name);
174 }
175 }
