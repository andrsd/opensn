40bool
41RCCLogicalVolume::Inside(const Vector3& point) const
42{
43 typedef Vector3 Vec3;
44
45 const auto& pr = point; // reference point
46 const Vec3 p0(x0_, y0_, z0_); // cylinder root
47 const Vec3 cyl_dir_vec(vx_, vy_, vz_); // cylinder direction vector
48 const Vec3 k_hat(0.0, 0.0, 1.0); // k_hat
49
50 const Vec3 p0r = pr - p0;
51 const Vec3 cyl_unit_dir = cyl_dir_vec.Normalized(); // aka cud
52 const double cyl_length = cyl_dir_vec.Norm();
53
54 // Check if point is within normal extents
55 const double p0r_dot_cud = p0r.Dot(cyl_unit_dir);
56 if (p0r_dot_cud < 0.0 or p0r_dot_cud > cyl_length) return false;
57
58 // Building rotation matrix
59 // This rotation matrix must be such that,
60 // when a coordinate system is rotated with it,
61 // the new normal vector points along the
62 Vec3 binorm;
63 Vec3 tangent;
64 if (std::abs(cyl_dir_vec.Dot(k_hat) / cyl_dir_vec.Norm()) > (1.0 - 1.0e-12))
65 {
66 binorm = Vec3(0.0, 1.0, 0.0);
67 tangent = Vec3(1.0, 0.0, 0.0);
68 }
69 else
70 {
71 binorm = k_hat.Cross(cyl_dir_vec);
72 binorm = binorm / binorm.Norm();
73 tangent = binorm.Cross(cyl_dir_vec);
74 tangent = tangent / tangent.Norm();
75 }
76
77 // Project p0r onto the binorm and tangent
78 const Vec3 p0r_projected(p0r.Dot(tangent), p0r.Dot(binorm), 0.0);
79
80 // Determine if point is within cylinder
81 if (p0r_projected.NormSquare() <= r_ * r_) return true;
82 else
83 return false;
84}
