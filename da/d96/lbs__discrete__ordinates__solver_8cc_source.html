495void
496DiscreteOrdinatesSolver::ComputeBalance()
497{
498 opensn::mpi_comm.barrier();
499 log.Log() << "\n********** Computing balance\n";
500
501 // Get material source
502 // This is done using the SetSource routine
503 // because it allows a lot of flexibility.
504 auto mat_src = phi_old_local_;
505 mat_src.assign(mat_src.size(), 0.0);
506 for (auto& groupset : groupsets_)
507 {
508 q_moments_local_.assign(q_moments_local_.size(), 0.0);
509 active_set_source_function_(groupset,
510 q_moments_local_,
511 PhiOldLocal(),
512 APPLY_FIXED_SOURCES | APPLY_AGS_FISSION_SOURCES |
513 APPLY_WGS_FISSION_SOURCES);
514 LBSSolver::GSScopedCopyPrimarySTLvectors(groupset, q_moments_local_, mat_src);
515 }
516
517 // Compute absorption, material-source and in-flow
518 double local_out_flow = 0.0;
519 double local_in_flow = 0.0;
520 double local_absorption = 0.0;
521 double local_production = 0.0;
522 for (const auto& cell : grid_ptr_->local_cells)
523 {
524 const auto& cell_mapping = discretization_->GetCellMapping(cell);
525 const auto& transport_view = cell_transport_views_[cell.local_id_];
526 const auto& fe_intgrl_values = unit_cell_matrices_[cell.local_id_];
527 const size_t num_nodes = transport_view.NumNodes();
528 const auto& IntV_shapeI = fe_intgrl_values.Vi_vectors;
529 const auto& IntS_shapeI = fe_intgrl_values.face_Si_vectors;
530
531 // Inflow
532 // This is essentially an integration over
533 // all faces, all angles, and all groups.
534 // Only the cosines that are negative are
535 // added to the integral.
536 for (int f = 0; f < cell.faces_.size(); ++f)
537 {
538 const auto& face = cell.faces_[f];
539
540 for (const auto& groupset : groupsets_)
541 {
542 for (int n = 0; n < groupset.quadrature_->omegas_.size(); ++n)
543 {
544 const auto& omega = groupset.quadrature_->omegas_[n];
545 const double wt = groupset.quadrature_->weights_[n];
546 const double mu = omega.Dot(face.normal_);
547
548 if (mu < 0.0 and (not face.has_neighbor_)) // mu<0 and bndry
549 {
550 const auto& bndry = sweep_boundaries_[face.neighbor_id_];
551 for (int fi = 0; fi < face.vertex_ids_.size(); ++fi)
552 {
553 const int i = cell_mapping.MapFaceNode(f, fi);
554 const auto& IntFi_shapeI = IntS_shapeI[f][i];
555
556 for (const auto& group : groupset.groups_)
557 {
558 const int g = group.id_;
559 const double psi = *bndry->HeterogeneousPsiIncoming(cell.local_id_, f, fi, n, g, 0);
560 local_in_flow -= mu * wt * psi * IntFi_shapeI;
561 } // for g
562 } // for fi
563 } // if bndry
564 } // for n
565 } // for groupset
566 } // for f
567
568 // Outflow
569 // The group-wise outflow was determined
570 // during a solve so here we just
571 // consolidate it.
572 for (int g = 0; g < num_groups_; ++g)
573 local_out_flow += transport_view.GetOutflow(g);
574
575 // Absorption and Src
576 // Isotropic flux based absorption and source
577 const auto& xs = transport_view.XS();
578 const auto& sigma_a = xs.SigmaAbsorption();
579 for (int i = 0; i < num_nodes; ++i)
580 for (int g = 0; g < num_groups_; ++g)
581 {
582 size_t imap = transport_view.MapDOF(i, 0, g);
583 double phi_0g = phi_old_local_[imap];
584 double q_0g = mat_src[imap];
585
586 local_absorption += sigma_a[g] * phi_0g * IntV_shapeI[i];
587 local_production += q_0g * IntV_shapeI[i];
588 } // for g
589 } // for cell
590
591 // Consolidate local balances
592 double local_balance = local_production + local_in_flow - local_absorption - local_out_flow;
593 double local_gain = local_production + local_in_flow;
594
595 std::vector<double> local_balance_table = {
596 local_absorption, local_production, local_in_flow, local_out_flow, local_balance, local_gain};
597 size_t table_size = local_balance_table.size();
598
599 std::vector<double> globl_balance_table(table_size, 0.0);
600
601 mpi_comm.all_reduce(
602 local_balance_table.data(), table_size, globl_balance_table.data(), mpi::op::sum<double>());
603
604 double globl_absorption = globl_balance_table.at(0);
605 double globl_production = globl_balance_table.at(1);
606 double globl_in_flow = globl_balance_table.at(2);
607 double globl_out_flow = globl_balance_table.at(3);
608 double globl_balance = globl_balance_table.at(4);
609 double globl_gain = globl_balance_table.at(5);
610
611 log.Log() << "Balance table:\n"
612 << std::setprecision(5) << std::scientific
613 << " Absorption rate = " << globl_absorption << "\n"
614 << " Production rate = " << globl_production << "\n"
615 << " In-flow rate = " << globl_in_flow << "\n"
616 << " Out-flow rate = " << globl_out_flow << "\n"
617 << " Net Gain (In-flow + sources) = " << globl_gain << "\n"
618 << " Net Balance = " << globl_balance << "\n"
619 << " (Net Balance)/(Net Gain) = " << globl_balance / globl_gain << "\n";
620
621 log.Log() << "\n********** Done computing balance\n";
622
623 opensn::mpi_comm.barrier();
624}

626std::vector<double>
627DiscreteOrdinatesSolver::ComputeLeakage(const int groupset_id, const uint64_t boundary_id) const
628{
629 const std::string fname = "lbs::SteadySolver::ComputeLeakage";
630
631 // Perform checks
632 if (groupset_id < 0 or groupset_id >= groupsets_.size())
633 throw std::invalid_argument(fname + ": Invalid groupset_id specified.");
634
635 if (not options_.save_angular_flux)
636 throw std::logic_error(fname + ": Requires options.save_angular_flux to be"
637 " true.");
638
639 // Get info
640 const auto& sdm = *discretization_;
641 const auto& groupset = groupsets_.at(groupset_id);
642 const auto& psi_uk_man = groupset.psi_uk_man_;
643 const auto& quadrature = groupset.quadrature_;
644
645 const size_t num_angles = quadrature->omegas_.size();
646
647 const int gsi = groupset.groups_.front().id_;
648 const int gsf = groupset.groups_.back().id_;
649 const int gs_num_groups = gsf + 1 - gsi;
650
651 // Start integration
652 std::vector<double> local_leakage(gs_num_groups, 0.0);
653 for (const auto& cell : grid_ptr_->local_cells)
654 {
655 const auto& cell_mapping = sdm.GetCellMapping(cell);
656 const auto& fe_values = unit_cell_matrices_[cell.local_id_];
657
658 size_t f = 0;
659 for (const auto& face : cell.faces_)
660 {
661 if (not face.has_neighbor_ and face.neighbor_id_ == boundary_id)
662 {
663 const auto& IntF_shapeI = fe_values.face_Si_vectors[f];
664 const size_t num_face_nodes = cell_mapping.NumFaceNodes(f);
665 for (size_t fi = 0; fi < num_face_nodes; ++fi)
666 {
667 const int i = cell_mapping.MapFaceNode(f, fi);
668 for (size_t n = 0; n < num_angles; ++n)
669 {
670 const auto& omega = quadrature->omegas_[n];
671 const auto& weight = quadrature->weights_[n];
672 const double mu = omega.Dot(face.normal_);
673 if (mu > 0.0)
674 {
675 for (int gi = 0; gi < gs_num_groups; ++gi)
676 {
677 const int g = gi + gsi;
678 const int64_t imap = sdm.MapDOFLocal(cell, i, psi_uk_man, n, g);
679
680 const double psi = psi_new_local_[groupset_id][imap];
681
682 local_leakage[gi] += weight * mu * psi * IntF_shapeI[i];
683 } // for g
684 } // outgoing
685 } // for n
686 } // for face node
687 } // if right bndry
688 ++f;
689 } // for face
690 } // for cell
691
692 std::vector<double> global_leakage(gs_num_groups, 0.0);
693 mpi_comm.all_reduce(
694 local_leakage.data(), gs_num_groups, global_leakage.data(), mpi::op::sum<double>());
695
696 return global_leakage;
697}

699void
700DiscreteOrdinatesSolver::InitializeSweepDataStructures()
701{
702 log.Log() << program_timer.GetTimeString() << " Initializing sweep datastructures.\n";
703
704 // Perform checks
705 {
706 auto& mesh_handler = GetCurrentHandler();
707 auto& mesher = mesh_handler.GetVolumeMesher();
708
709 for (const auto& groupset : groupsets_)
710 {
711 bool no_cycles_parmetis_partitioning =
712 ((mesher.options.partition_type == VolumeMesher::PartitionType::PARMETIS) and
713 (not groupset.allow_cycles_));
714
715 bool is_1D_geometry = options_.geometry_type == GeometryType::ONED_SLAB;
716
717 if (no_cycles_parmetis_partitioning and not is_1D_geometry and (opensn::mpi_comm.size() > 1))
718 throw std::logic_error("When using PARMETIS type partitioning then groupset iterative "
719 "method must be NPT_CLASSICRICHARDSON_CYCLES or NPT_GMRES_CYCLES");
720 } // for groupset
721 }
722
723 // Define sweep ordering groups
724 quadrature_unq_so_grouping_map_.clear();
725 std::map<std::shared_ptr<AngularQuadrature>, bool> quadrature_allow_cycles_map_;
726 for (auto& groupset : groupsets_)
727 {
728 if (quadrature_unq_so_grouping_map_.count(groupset.quadrature_) == 0)
729 quadrature_unq_so_grouping_map_[groupset.quadrature_] = AssociateSOsAndDirections(
730 *grid_ptr_, *groupset.quadrature_, groupset.angleagg_method_, options_.geometry_type);
731
732 if (quadrature_allow_cycles_map_.count(groupset.quadrature_) == 0)
733 quadrature_allow_cycles_map_[groupset.quadrature_] = groupset.allow_cycles_;
734 }
735
736 // Build sweep orderings
737 quadrature_spds_map_.clear();
738 for (const auto& [quadrature, info] : quadrature_unq_so_grouping_map_)
739 {
740 const auto& unique_so_groupings = info.first;
741
742 for (const auto& so_grouping : unique_so_groupings)
743 {
744 if (so_grouping.empty()) continue;
745
746 const size_t master_dir_id = so_grouping.front();
747 const auto& omega = quadrature->omegas_[master_dir_id];
748
749 bool verbose = false;
750 if (not verbose_sweep_angles_.empty())
751 for (const size_t dir_id : verbose_sweep_angles_)
752 if (VectorListHas(so_grouping, dir_id))
753 {
754 verbose = true;
755 break;
756 }
757
758 if (sweep_type_ == "AAH")
759 {
760 const auto new_swp_order = std::make_shared<SPDS_AdamsAdamsHawkins>(
761 omega, *this->grid_ptr_, quadrature_allow_cycles_map_[quadrature], verbose);
762 quadrature_spds_map_[quadrature].push_back(new_swp_order);
763 }
764 else if (sweep_type_ == "CBC")
765 {
766 const auto new_swp_order = std::make_shared<CBC_SPDS>(
767 omega, *this->grid_ptr_, quadrature_allow_cycles_map_[quadrature], verbose);
768 quadrature_spds_map_[quadrature].push_back(new_swp_order);
769 }
770 else
771 ChiInvalidArgument("Unsupported sweeptype \"" + sweep_type_ + "\"");
772 }
773 } // quadrature info-pack
774
775 // Build FLUDS templates
776 quadrature_fluds_commondata_map_.clear();
777 for (const auto& [quadrature, spds_list] : quadrature_spds_map_)
778 {
779 for (const auto& spds : spds_list)
780 {
781 if (sweep_type_ == "AAH")
782 {
783 quadrature_fluds_commondata_map_[quadrature].push_back(
784 std::make_unique<AAH_FLUDSCommonData>(
785 grid_nodal_mappings_, *spds, *grid_face_histogram_));
786 }
787 else if (sweep_type_ == "CBC")
788 {
789 quadrature_fluds_commondata_map_[quadrature].push_back(
790 std::make_unique<CBC_FLUDSCommonData>(*spds, grid_nodal_mappings_));
791 }
792 else
793 ChiInvalidArgument("Unsupported sweeptype \"" + sweep_type_ + "\"");
794 }
795 } // for quadrature spds-list pair
796
797 log.Log() << program_timer.GetTimeString() << " Done initializing sweep datastructures.\n";
798}

800std::pair<UniqueSOGroupings, DirIDToSOMap>
801DiscreteOrdinatesSolver::AssociateSOsAndDirections(const MeshContinuum& grid,
802 const AngularQuadrature& quadrature,
803 const AngleAggregationType agg_type,
804 const lbs::GeometryType lbs_geo_type)
805{
806 const std::string fname = __FUNCTION__;
807
808 // Checks
809 if (quadrature.omegas_.empty())
810 throw std::logic_error(fname + ": Quadrature with no omegas cannot be used.");
811 if (quadrature.weights_.empty())
812 throw std::logic_error(fname + ": Quadrature with no weights cannot be used.");
813
814 // Build groupings
815 UniqueSOGroupings unq_so_grps;
816 switch (agg_type)
817 {
818 // Single
819 // The easiest aggregation type. Every direction
820 // either has/is assumed to have a unique sweep
821 // ordering. Hence there is only group holding ALL
822 // the direction indices.
823 case AngleAggregationType::SINGLE:
824 {
825 const size_t num_dirs = quadrature.omegas_.size();
826 for (size_t n = 0; n < num_dirs; ++n)
827 unq_so_grps.push_back({n});
828 break;
829 } // case agg_type SINGLE
830
831 // Polar
832 // The following conditions allow for polar
833 // angle aggregation.
834 case AngleAggregationType::POLAR:
835 {
836 // Check geometry types
837 const auto grid_attribs = grid.Attributes();
838 if (not(grid_attribs & ORTHOGONAL or grid_attribs & DIMENSION_2 or grid_attribs & EXTRUDED))
839 throw std::logic_error(
840 fname + ": The simulation is using polar angle aggregation for which only certain "
841 "geometry types are supported, i.e., ORTHOGONAL, DIMENSION_2 or 3D EXTRUDED.");
842
843 // Check quadrature type
844 const auto quad_type = quadrature.type_;
845 if (quad_type != AngularQuadratureType::ProductQuadrature)
846 throw std::logic_error(fname + ": The simulation is using polar angle aggregation for "
847 "which only Product-type quadratures are supported.");
848
849 // Process Product Quadrature
850 try
851 {
852 typedef ProductQuadrature ProdQuadType;
853 const auto& product_quad = dynamic_cast<const ProdQuadType&>(quadrature);
854
855 const auto num_azi = product_quad.azimu_ang_.size();
856 const auto num_pol = product_quad.polar_ang_.size();
857
858 // Make two separate list of polar angles
859 // One upward-pointing and one downward
860 std::vector<size_t> upward_polar_ids;
861 std::vector<size_t> dnward_polar_ids;
862 for (size_t p = 0; p < num_pol; ++p)
863 if (product_quad.polar_ang_[p] > M_PI_2) upward_polar_ids.push_back(p);
864 else
865 dnward_polar_ids.push_back(p);
866
867 // Define lambda working for both upward and dnward polar-ids
868 /**Lambda to convert indices and push it onto unq_so_grps.*/
869 auto MapPolarAndAzimuthalIDs =
870 [&product_quad, &unq_so_grps](const DirIDs& polar_ids, const size_t azimuthal_id)
871 {
872 DirIDs dir_ids;
873 dir_ids.reserve(polar_ids.size());
874 for (const size_t p : polar_ids)
875 dir_ids.push_back(product_quad.GetAngleNum(p, azimuthal_id));
876 unq_so_grps.push_back(std::move(dir_ids));
877 };
878
879 // Stack id's for all azimuthal angles
880 for (size_t a = 0; a < num_azi; ++a)
881 {
882 if (not upward_polar_ids.empty()) MapPolarAndAzimuthalIDs(upward_polar_ids, a);
883 if (not dnward_polar_ids.empty()) MapPolarAndAzimuthalIDs(dnward_polar_ids, a);
884 } // for azi-id a
885
886 } // try product quadrature
887 catch (const std::bad_cast& bc)
888 {
889 throw std::runtime_error(
890 fname + ": Casting the angular quadrature to the product quadrature base, failed.");
891 }
892
893 break;
894 } // case agg_type POLAR
895
896 // Azimuthal
897 case AngleAggregationType::AZIMUTHAL:
898 {
899 // Check geometry types
900 if (not(lbs_geo_type == GeometryType::ONED_SPHERICAL or
901 lbs_geo_type == GeometryType::TWOD_CYLINDRICAL))
902 throw std::logic_error(
903 fname + ": The simulation is using azimuthal angle aggregation for which only "
904 "ONED_SPHERICAL or TWOD_CYLINDRICAL derived geometry types are supported.");
905
906 // Check quadrature type
907 const auto quad_type = quadrature.type_;
908 if (quad_type != AngularQuadratureType::ProductQuadrature)
909 throw std::logic_error(fname + ": The simulation is using azimuthal angle aggregation for "
910 "which only Product-type quadratures are supported.");
911
912 // Process Product Quadrature
913 try
914 {
915 typedef ProductQuadrature ProdQuadType;
916 const auto& product_quad = dynamic_cast<const ProdQuadType&>(quadrature);
917
918 for (const auto& dir_set : product_quad.GetDirectionMap())
919 {
920 std::vector<unsigned int> group1;
921 std::vector<unsigned int> group2;
922 for (const auto& dir_id : dir_set.second)
923 if (quadrature.abscissae_[dir_id].phi > M_PI_2) group1.push_back(dir_id);
924 else
925 group2.push_back(dir_id);
926
927 DirIDs group1_ids(group1.begin(), group1.end());
928 DirIDs group2_ids(group2.begin(), group2.end());
929
930 unq_so_grps.push_back(std::move(group1_ids));
931 unq_so_grps.push_back(std::move(group2_ids));
932 }
933 } // try product quadrature
934 catch (const std::bad_cast& bc)
935 {
936 throw std::runtime_error(
937 fname + ": Casting the angular quadrature to the product quadrature base, failed.");
938 }
939
940 break;
941 }
942 default:
943 throw std::invalid_argument(fname + ": Called with UNDEFINED angle "
944 "aggregation type.");
945 } // switch angle aggregation type
946
947 // Map directions to sweep orderings
948 DirIDToSOMap dir_id_to_so_map;
949 {
950 size_t so_grouping_id = 0;
951 for (const auto& so_grouping : unq_so_grps)
952 {
953 for (const size_t dir_id : so_grouping)
954 dir_id_to_so_map[dir_id] = so_grouping_id;
955
956 ++so_grouping_id;
957 } // for so_grouping
958 } // map scope
959
960 return {unq_so_grps, dir_id_to_so_map};
961}

963void
964DiscreteOrdinatesSolver::InitFluxDataStructures(LBSGroupset& groupset)
965{
966 typedef AngleSetGroup TAngleSetGroup;
967 typedef AAH_AngleSet TAAH_AngleSet;
968
969 const auto& quadrature_sweep_info = quadrature_unq_so_grouping_map_[groupset.quadrature_];
970
971 const auto& unique_so_groupings = quadrature_sweep_info.first;
972 const auto& dir_id_to_so_map = quadrature_sweep_info.second;
973
974 const size_t gs_num_grps = groupset.groups_.size();
975 const size_t gs_num_ss = groupset.grp_subset_infos_.size();
976
977 // Passing the sweep boundaries
978 // to the angle aggregation
979 typedef AngleAggregation AngleAgg;
980 groupset.angle_agg_ = std::make_shared<AngleAgg>(
981 sweep_boundaries_, gs_num_grps, gs_num_ss, groupset.quadrature_, grid_ptr_);
982
983 TAngleSetGroup angle_set_group;
984 size_t angle_set_id = 0;
985 for (const auto& so_grouping : unique_so_groupings)
986 {
987 const size_t master_dir_id = so_grouping.front();
988 const size_t so_id = dir_id_to_so_map.at(master_dir_id);
989
990 const auto& sweep_ordering = quadrature_spds_map_[groupset.quadrature_][so_id];
991 const auto& fluds_common_data = *quadrature_fluds_commondata_map_[groupset.quadrature_][so_id];
992
993 // Compute direction subsets
994 const auto dir_subsets = MakeSubSets(so_grouping.size(), groupset.master_num_ang_subsets_);
995
996 for (size_t gs_ss = 0; gs_ss < gs_num_ss; gs_ss++)
997 {
998 const size_t gs_ss_size = groupset.grp_subset_infos_[gs_ss].ss_size;
999 for (const auto& dir_ss_info : dir_subsets)
1000 {
1001 const auto& dir_ss_begin = dir_ss_info.ss_begin;
1002 const auto& dir_ss_end = dir_ss_info.ss_end;
1003 const auto& dir_ss_size = dir_ss_info.ss_size;
1004
1005 std::vector<size_t> angle_indices(dir_ss_size, 0);
1006 {
1007 size_t k = 0;
1008 for (size_t n = dir_ss_begin; n <= dir_ss_end; ++n)
1009 angle_indices[k++] = so_grouping[n];
1010 }
1011
1012 if (sweep_type_ == "AAH")
1013 {
1014 std::shared_ptr<FLUDS> fluds = std::make_shared<AAH_FLUDS>(
1015 gs_ss_size,
1016 angle_indices.size(),
1017 dynamic_cast<const AAH_FLUDSCommonData&>(fluds_common_data));
1018
1019 auto angleSet = std::make_shared<TAAH_AngleSet>(angle_set_id++,
1020 gs_ss_size,
1021 gs_ss,
1022 *sweep_ordering,
1023 fluds,
1024 angle_indices,
1025 sweep_boundaries_,
1026 options_.sweep_eager_limit,
1027 *grid_local_comm_set_);
1028
1029 angle_set_group.AngleSets().push_back(angleSet);
1030 }
1031 else if (sweep_type_ == "CBC")
1032 {
1033 ChiLogicalErrorIf(not options_.save_angular_flux,
1034 "When using sweep_type \"CBC\" then "
1035 "\"save_angular_flux\" must be true.");
1036 std::shared_ptr<FLUDS> fluds =
1037 std::make_shared<CBC_FLUDS>(gs_ss_size,
1038 angle_indices.size(),
1039 dynamic_cast<const CBC_FLUDSCommonData&>(fluds_common_data),
1040 psi_new_local_[groupset.id_],
1041 groupset.psi_uk_man_,
1042 *discretization_);
1043
1044 auto angleSet = std::make_shared<CBC_AngleSet>(angle_set_id++,
1045 gs_ss_size,
1046 *sweep_ordering,
1047 fluds,
1048 angle_indices,
1049 sweep_boundaries_,
1050 gs_ss,
1051 *grid_local_comm_set_);
1052
1053 angle_set_group.AngleSets().push_back(angleSet);
1054 }
1055 else
1056 ChiInvalidArgument("Unsupported sweeptype \"" + sweep_type_ + "\"");
1057 } // for an_ss
1058 } // for gs_ss
1059 } // for so_grouping
1060
1061 groupset.angle_agg_->angle_set_groups.push_back(std::move(angle_set_group));
1062
1063 if (options_.verbose_inner_iterations)
1064 log.Log() << program_timer.GetTimeString() << " Initialized Angle Aggregation. "
1065 << " Process memory = " << std::setprecision(3) << GetMemoryUsageInMB()
1066 << " MB.";
1067
1068 opensn::mpi_comm.barrier();
1069}
