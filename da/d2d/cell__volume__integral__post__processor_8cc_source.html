75void
76CellVolumeIntegralPostProcessor::Execute(const Event& event_context)
77{
78 if (not initialized_) Initialize();
79
80 const auto* grid_field_function = GetGridBasedFieldFunction();
81
82 ChiLogicalErrorIf(not grid_field_function,
83 "Attempted to access invalid field"
84 "function");
85
86 const auto& ref_ff = *grid_field_function;
87 const auto& sdm = ref_ff.GetSpatialDiscretization();
88 const auto& grid = sdm.Grid();
89
90 const auto& uk_man = ref_ff.GetUnknownManager();
91 const auto uid = 0;
92 const auto cid = 0;
93
94 const auto field_data = ref_ff.GetGhostedFieldVector();
95
96 auto coord = sdm.GetSpatialWeightingFunction();
97
98 double local_integral = 0.0;
99 double local_volume = 0.0;
100 for (const uint64_t cell_local_id : cell_local_ids_)
101 {
102 const auto& cell = grid.local_cells[cell_local_id];
103 const auto& cell_mapping = sdm.GetCellMapping(cell);
104 const size_t num_nodes = cell_mapping.NumNodes();
105 const auto qp_data = cell_mapping.MakeVolumetricQuadraturePointData();
106
107 std::vector<double> node_dof_values(num_nodes, 0.0);
108 for (size_t i = 0; i < num_nodes; ++i)
109 {
110 const int64_t imap = sdm.MapDOFLocal(cell, i, uk_man, uid, cid);
111 node_dof_values[i] = field_data[imap];
112 } // for i
113
114 for (const size_t qp : qp_data.QuadraturePointIndices())
115 {
116 // phi_h = sum_j b_j phi_j
117 double ff_value = 0.0;
118 for (size_t j = 0; j < num_nodes; ++j)
119 ff_value += qp_data.ShapeValue(j, qp) * node_dof_values[j];
120
121 local_integral += ff_value * coord(qp_data.QPointXYZ(qp)) * qp_data.JxW(qp);
122 local_volume += coord(qp_data.QPointXYZ(qp)) * qp_data.JxW(qp);
123 } // for qp
124 } // for cell-id
125
126 double globl_integral;
127 mpi_comm.all_reduce(local_integral, globl_integral, mpi::op::sum<double>());
128 if (not compute_volume_average_) value_ = ParameterBlock("", globl_integral);
129 else
130 {
131 double globl_volume;
132 mpi_comm.all_reduce(local_volume, globl_volume, mpi::op::sum<double>());
133
134 value_ = ParameterBlock("", globl_integral / globl_volume);
135 }
136
137 const int event_code = event_context.Code();
138 if (event_code == Event::SolverInitialized or event_code == Event::SolverAdvanced)
139 {
140 const auto& event_params = event_context.Parameters();
141
142 if (event_params.Has("timestep_index") and event_params.Has("time"))
143 {
144 const size_t index = event_params.GetParamValue<size_t>("timestep_index");
145 const double time = event_params.GetParamValue<double>("time");
146 TimeHistoryEntry entry{index, time, value_};
147 time_history_.push_back(std::move(entry));
148 }
149 }
150}
