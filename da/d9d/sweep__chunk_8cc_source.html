79{
80 const size_t f = sweep_dependency_interface_.current_face_idx_;
81 const auto& IntF_shapeI = (*IntS_shapeI_)[f];
82 const double mu = face_mu_values_[f];
83 const double wt = direction_qweight_;
84
85 const bool on_boundary = sweep_dependency_interface_.on_boundary_;
86 const bool is_reflecting_boundary = sweep_dependency_interface_.is_reflecting_bndry_;
87
88 const size_t num_face_nodes = cell_mapping_->NumFaceNodes(f);
89 for (int fi = 0; fi < num_face_nodes; ++fi)
90 {
91 const int i = cell_mapping_->MapFaceNode(f, fi);
92
93 double* psi = sweep_dependency_interface_.GetDownwindPsi(fi);
94
95 if (psi != nullptr)
96 if (not on_boundary or is_reflecting_boundary)
97 for (int gsg = 0; gsg < gs_ss_size_; ++gsg)
98 psi[gsg] = b_[gsg][i];
99 if (on_boundary and not is_reflecting_boundary)
100 for (int gsg = 0; gsg < gs_ss_size_; ++gsg)
101 cell_transport_view_->AddOutflow(gs_gi_ + gsg, wt * mu * b_[gsg][i] * IntF_shapeI[i]);
102
103 } // for fi
104}

116void
117SweepChunk::KernelFEMUpwindSurfaceIntegrals()
118{
119 const size_t f = sweep_dependency_interface_.current_face_idx_;
120 const auto& M_surf_f = (*M_surf_)[f];
121 const double mu = face_mu_values_[f];
122 const size_t num_face_nodes = sweep_dependency_interface_.num_face_nodes_;
123 for (int fi = 0; fi < num_face_nodes; ++fi)
124 {
125 const int i = cell_mapping_->MapFaceNode(f, fi);
126 for (int fj = 0; fj < num_face_nodes; ++fj)
127 {
128 const int j = cell_mapping_->MapFaceNode(f, fj);
129
130 const double* psi = sweep_dependency_interface_.GetUpwindPsi(fj);
131
132 const double mu_Nij = -mu * M_surf_f[i][j];
133 Amat_[i][j] += mu_Nij;
134
135 if (psi == nullptr) continue;
136
137 for (int gsg = 0; gsg < gs_ss_size_; ++gsg)
138 b_[gsg][i] += psi[gsg] * mu_Nij;
139 } // for face node j
140 } // for face node i
141}

143void
144SweepChunk::KernelFEMSTDMassTerms()
145{
146 const auto& M = *M_;
147 const auto& m2d_op = groupset_.quadrature_->GetMomentToDiscreteOperator();
148
149 // Contribute source moments q = M_n^T * q_moms
150 for (int i = 0; i < cell_num_nodes_; ++i)
151 {
152 double temp_src = 0.0;
153 for (int m = 0; m < num_moments_; ++m)
154 {
155 const size_t ir = cell_transport_view_->MapDOF(i, m, static_cast<int>(g_));
156 temp_src += m2d_op[m][direction_num_] * q_moments_[ir];
157 } // for m
158 source_[i] = temp_src;
159 } // for i
160
161 // Mass Matrix and Source
162 // Atemp = Amat + sigma_tgr * M
163 // b += M * q
164 for (int i = 0; i < cell_num_nodes_; ++i)
165 {
166 double temp = 0.0;
167 for (int j = 0; j < cell_num_nodes_; ++j)
168 {
169 const double Mij = M[i][j];
170 Atemp_[i][j] = Amat_[i][j] + Mij * sigma_tg_;
171 temp += Mij * source_[j];
172 } // for j
173 b_[gsg_][i] += temp;
174 } // for i
175}

177void
178SweepChunk::KernelPhiUpdate()
179{
180 const auto& d2m_op = groupset_.quadrature_->GetDiscreteToMomentOperator();
181
182 auto& output_phi = GetDestinationPhi();
183
184 for (int m = 0; m < num_moments_; ++m)
185 {
186 const double wn_d2m = d2m_op[m][direction_num_];
187 for (int i = 0; i < cell_num_nodes_; ++i)
188 {
189 const size_t ir = cell_transport_view_->MapDOF(i, m, gs_gi_);
190 for (int gsg = 0; gsg < gs_ss_size_; ++gsg)
191 output_phi[ir + gsg] += wn_d2m * b_[gsg][i];
192 }
193 }
