20 // Check grid available
21 if (field_functions_.empty())

23 "field function interpolator.");

29
30 for (const auto& cell : grid.local_cells)
31 {
32 auto cell_local_index = cell.local_id_;
33
34 if (cell.Type() == CellType::SLAB)
35 throw std::logic_error("FieldFunctionInterpolationSlice "
36 "does not support 1D cells.");
37 if (cell.Type() == CellType::POLYGON) intersecting_cell_indices.push_back(cell_local_index);
38 else if (cell.Type() == CellType::POLYHEDRON)
39 {
40 bool intersects = false;
41
42 size_t num_faces = cell.faces_.size();
43 for (size_t f = 0; f < num_faces; f++)
44 {
45 const auto& face = cell.faces_[f];
46
47 size_t num_edges = face.vertex_ids_.size();
48 for (size_t e = 0; e < num_edges; e++)
49 {
50 size_t ep1 = (e < (num_edges - 1)) ? e + 1 : 0;
51 uint64_t v0_i = face.vertex_ids_[e];
52 uint64_t v1_i = face.vertex_ids_[ep1];
53
54 std::vector<Vector3> tet_points;
55
56 tet_points.push_back(grid.vertices[v0_i]);
57 tet_points.push_back(grid.vertices[v1_i]);
58 tet_points.push_back(cell.faces_[f].centroid_);
59 tet_points.push_back(cell.centroid_);
60
61 if (CheckPlaneTetIntersect(this->normal_, this->plane_point_, tet_points))
62 {
63 intersecting_cell_indices.push_back(cell_local_index);
64 intersects = true;
65 break; // from for e
66 }
67 } // for e
68 if (intersects) break; // from for f
69 } // for f
70 }
71 else
72 throw std::logic_error("Unsupported cell type in call "
73 "to Slice Initialize.");
74 } // for local cell
75
76 // Computing cell intersections
77 for (const uint64_t cell_local_index : intersecting_cell_indices)
78 {
79 const auto& cell = grid.local_cells[cell_local_index];
80
81 if (cell.Type() == CellType::POLYGON)
82 {
83 FFICellIntersection cell_isds;
84 cell_isds.ref_cell_local_id = cell_local_index;
85
86 // Loop over vertices
87 for (uint64_t v0gi : cell.vertex_ids_)
88 {
89 FFIFaceEdgeIntersection face_isds;
90
91 const auto nudge = 1.0e-4 * (grid.vertices[v0gi] - cell.centroid_);
92
93 face_isds.point = grid.vertices[v0gi] - nudge;
94 face_isds.point2d = grid.vertices[v0gi];
95 cell_isds.intersections.push_back(face_isds);
96 }
97
98 // Set intersection center
99 cell_isds.intersection_centre = cell.centroid_;
100
101 // Set straight 2D center
102 // This is normally transformed for the 3D case
103 cell_isds.intersection_2d_centre = cell.centroid_;
104
105 // Same for 2D points
106 size_t num_points = cell_isds.intersections.size();
107 for (size_t p = 0; p < num_points; p++)
108 {
109 Vector3 vref = cell_isds.intersections[p].point - plane_point_;
110
111 cell_isds.intersections[p].point2d = vref;
112 }
113
114 cell_intersections_.push_back(cell_isds);
115 } // polygon
116 //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% POLYHEDRON
117 else if (cell.Type() == CellType::POLYHEDRON)
118 {
119 // Initialize cell intersection data structure
120 FFICellIntersection cell_isds;
121 cell_isds.ref_cell_local_id = cell_local_index;
122
123 // Loop over faces
124 size_t num_faces = cell.faces_.size();
125 for (size_t f = 0; f < num_faces; f++)
126 {
127 const auto& face = cell.faces_[f];
128 // Loop over edges
129 size_t num_edges = face.vertex_ids_.size();
130 for (size_t e = 0; e < num_edges; e++)
131 {
132 size_t ep1 = (e < (num_edges - 1)) ? e + 1 : 0;
133 uint64_t v0gi = face.vertex_ids_[e]; // global index v0
134 uint64_t v1gi = face.vertex_ids_[ep1]; // global index v1
135
136 const auto& v0 = grid.vertices[v0gi];
137 const auto& v1 = grid.vertices[v1gi];
138
139 Vertex interstion_point; // Placeholder
140 std::pair<double, double> weights;
141
142 // Check if intersects plane
143 if (CheckPlaneLineIntersect(
144 this->normal_, this->plane_point_, v0, v1, interstion_point, &weights))
145 {
146 // Check for duplicate
147 bool duplicate_found = false;
148 for (auto& existing_face_isds : cell_isds.intersections)
149 {
150 double dif = (existing_face_isds.point - interstion_point).NormSquare();
151 if (dif < 1.0e-6)
152 {
153 duplicate_found = true;
154 break;
155 }
156 }
157
158 // No duplicate
159 if (!duplicate_found)
160 {
161 FFIFaceEdgeIntersection face_isds;
162
163 face_isds.point = interstion_point;
164 cell_isds.intersections.push_back(face_isds);
165 }
166 } // if intersecting
167 } // for edge
168 } // for face
169
170 // Computing intersection centre
171 size_t num_points = cell_isds.intersections.size();
172 if (num_points > 0)
173 {
174 for (int p = 0; p < num_points; p++)
175 {
176 cell_isds.intersection_centre =
177 cell_isds.intersection_centre + cell_isds.intersections[p].point;
178 }
179 cell_isds.intersection_centre =
180 cell_isds.intersection_centre / static_cast<double>(num_points);
181 }
182 else
183 {
184 log.LogAllWarning() << "No face intersections encountered "
185 "for a cell that is indicated as being "
186 "intersected. Slice FF interp.";
187 }
188
189 // Computing 2D transforms
190 Vector3 vref = cell_isds.intersection_centre - plane_point_;
191
192 cell_isds.intersection_2d_centre.x = vref.Dot(tangent_);
193 cell_isds.intersection_2d_centre.y = vref.Dot(binorm_);
194 cell_isds.intersection_2d_centre.z = vref.Dot(normal_);
195
196 // Points
197 std::vector<FFIFaceEdgeIntersection> unsorted_points;
198 for (int p = 0; p < num_points; p++)
199 {
200 vref = cell_isds.intersections[p].point - plane_point_;
201
202 cell_isds.intersections[p].point2d.x = vref.Dot(tangent_);
203 cell_isds.intersections[p].point2d.y = vref.Dot(binorm_);
204 cell_isds.intersections[p].point2d.z = vref.Dot(normal_);
205
206 unsorted_points.push_back(cell_isds.intersections[p]);
207 }
208 cell_isds.intersections.clear();
209
210 // Sort points clockwise
211 // The first point is retrieved from the unused stack.
212 // Subsequent points are only added if they form a
213 // convex line wrt the right hand rule.
214 cell_isds.intersections.push_back(unsorted_points[0]);
215 unsorted_points.erase(unsorted_points.begin());
216
217 while (!unsorted_points.empty())
218 {
219 for (int p = 0; p < unsorted_points.size(); p++)
220 {
221 Vector3 v1 = unsorted_points[p].point2d - cell_isds.intersections.back().point2d;
222
223 bool illegal_value = false;
224 for (int pr = 0; pr < unsorted_points.size(); pr++)
225 {
226 if (pr != p)
227 {
228 Vector3 vr = unsorted_points[pr].point2d - unsorted_points[p].point2d;
229
230 if (vr.Cross(v1).z < 0.0)
231 {
232 illegal_value = true;
233 break;
234 }
235 } // if not p
236 } // for pr
237
238 if (!illegal_value)
239 {
240 cell_isds.intersections.push_back(unsorted_points[p]);
241 unsorted_points.erase(unsorted_points.begin() + p);
242 break;
243 }
244
245 } // for p
246 }
247 cell_intersections_.push_back(cell_isds);
248 } // polyhedron
249 } // for intersected cell

254void
255FieldFunctionInterpolationSlice::Execute()
256{
257 const auto& ref_ff = *field_functions_.front();
258 const auto& sdm = ref_ff.GetSpatialDiscretization();
259 const auto& grid = sdm.Grid();
260
261 const auto& uk_man = ref_ff.GetUnknownManager();
262 const auto uid = 0;
263 const auto cid = ref_component_;
264
265 const auto field_data = ref_ff.GetGhostedFieldVector();
266
267 for (auto& cell_intersection : cell_intersections_)
268 {
269 const auto& cell = grid.local_cells[cell_intersection.ref_cell_local_id];
270 const auto& cell_mapping = sdm.GetCellMapping(cell);
271 const size_t num_nodes = cell_mapping.NumNodes();
272
273 std::vector<double> dof_values(num_nodes, 0.0);
274 for (size_t i = 0; i < num_nodes; ++i)
275 {
276 const int64_t imap = sdm.MapDOFLocal(cell, i, uk_man, uid, cid);
277 dof_values[i] = field_data[imap];
278 }
279
280 std::vector<double> shape_values(num_nodes, 0.0);
281 for (auto& edge_intersection : cell_intersection.intersections)
282 {
283 cell_mapping.ShapeValues(edge_intersection.point, shape_values);
284 double point_value = 0.0;
285 for (size_t i = 0; i < num_nodes; ++i)
286 point_value += dof_values[i] * shape_values[i];
287
288 edge_intersection.point_value = point_value;
289 } // for edge intersection
290 } // for cell intersection
291}
