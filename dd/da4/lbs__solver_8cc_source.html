68{
69 // Make groups
70 const size_t num_groups = params.GetParamValue<size_t>("num_groups");
71 for (size_t g = 0; g < num_groups; ++g)
72 groups_.push_back(LBSGroup(static_cast<int>(g)));
73
74 // Make groupsets
75 const auto& groupsets_array = params.GetParam("groupsets");
76
77 const size_t num_gs = groupsets_array.NumParameters();
78 for (size_t gs = 0; gs < num_gs; ++gs)

80 const auto& groupset_params = groupsets_array.GetParam(gs);
81
82 InputParameters gs_input_params = LBSGroupset::GetInputParameters();
83 gs_input_params.SetObjectType("LBSSolver:LBSGroupset");
84 gs_input_params.AssignParameters(groupset_params);
85
86 groupsets_.emplace_back(gs_input_params, gs, *this);
87 } // for gs
88
89 // Options
90 if (params.ParametersAtAssignment().Has("options"))
91 {
92 auto options_params = LBSSolver::OptionsBlock();
93 options_params.AssignParameters(params.GetParam("options"));

95 this->SetOptions(options_params);

626void
627LBSSolver::SetBoundaryOptions(const InputParameters& params)
628{
629 const std::string fname = __FUNCTION__;
630 const auto& user_params = params.ParametersAtAssignment();
631 const auto boundary_name = user_params.GetParamValue<std::string>("name");
632 const auto bndry_type = user_params.GetParamValue<std::string>("type");
633
634 const std::map<std::string, uint64_t> supported_bndry_names = {
635 {"xmin", 1}, {"xmax", 0}, {"ymin", 3}, {"ymax", 2}, {"zmin", 5}, {"zmax", 4}};
636 const auto bid = supported_bndry_names.at(boundary_name);
637 const std::map<std::string, lbs::BoundaryType> type_list = {
638 {"vacuum", BoundaryType::VACUUM},
639 {"incident_isotropic", BoundaryType::INCIDENT_ISOTROPIC},
640 {"reflecting", BoundaryType::REFLECTING},
641 {"incident_anisotropic_heterogeneous", BoundaryType::INCIDENT_ANISTROPIC_HETEROGENEOUS}};
642
643 const auto type = type_list.at(bndry_type);
644 switch (type)
645 {
646 case BoundaryType::VACUUM:
647 case BoundaryType::REFLECTING:
648 {
649 BoundaryPreferences()[bid] = {type};
650 break;
651 }
652 case BoundaryType::INCIDENT_ISOTROPIC:
653 {
654 if (not user_params.Has("group_strength"))
655 {
656 std::string message = fname;
657 message += ":boundary_conditions:"
658 "type=\"incident_isotropic\" requires parameter "
659 "\"group_strength\".";
660
661 throw std::invalid_argument(message);
662 }
663 user_params.RequireParameterBlockTypeIs("group_strength", ParameterBlockType::ARRAY);
664
665 const auto group_strength = user_params.GetParamVectorValue<double>("group_strength");
666 BoundaryPreferences()[bid] = {type, group_strength};
667 break;
668 }
669 case BoundaryType::INCIDENT_ANISTROPIC_HETEROGENEOUS:
670 {
671 if (not user_params.Has("function_name"))
672 {
673 std::string message = fname;
674 message += ":boundary_conditions:"
675 "type=\"incident_anisotropic_heterogeneous\" requires "
676 "parameter \"function_name\".";
677
678 throw std::invalid_argument(message);
679 }
680 const auto bndry_function_name = user_params.GetParamValue<std::string>("function_name");
681
682 BoundaryPreferences()[bid] = {type, {}, bndry_function_name};
683 break;
684 }
685 }
686}

707void
708LBSSolver::PerformInputChecks()
709{
710 if (groups_.empty())
711 {
712 log.LogAllError() << "LinearBoltzmann::SteadyStateSolver: No groups added to solver.";
713 Exit(EXIT_FAILURE);
714 }
715
716 num_groups_ = groups_.size();
717
718 if (groupsets_.empty())
719 {
720 log.LogAllError() << "LinearBoltzmann::SteadyStateSolver: No group-sets added to solver.";
721 Exit(EXIT_FAILURE);
722 }
723 int grpset_counter = 0;
724 for (auto& group_set : groupsets_)
725 {
726 if (group_set.groups_.empty())
727 {
728 log.LogAllError() << "LinearBoltzmann::SteadyStateSolver: No groups added to groupset "
729 << grpset_counter << ".";
730 Exit(EXIT_FAILURE);
731 }
732 ++grpset_counter;
733 }
734 if (options_.sd_type == SpatialDiscretizationType::UNDEFINED)
735 {
736 log.LogAllError() << "LinearBoltzmann::SteadyStateSolver: No discretization_ method set.";

797LBSSolver::InitMaterials()
798{
799 const std::string fname = "lbs::SteadyStateSolver::InitMaterials";
800 log.Log0Verbose1() << "Initializing Materials";
801
802 // Create set of material ids locally relevant
803 std::set<int> unique_material_ids;
804 int invalid_mat_cell_count = 0;
805 for (auto& cell : grid_ptr_->local_cells)
806 {
807 unique_material_ids.insert(cell.material_id_);
808 if (cell.material_id_ < 0) ++invalid_mat_cell_count;
809 }
810 const auto& ghost_cell_ids = grid_ptr_->cells.GetGhostGlobalIDs();
811 for (uint64_t cell_id : ghost_cell_ids)
812 {
813 const auto& cell = grid_ptr_->cells[cell_id];
814 unique_material_ids.insert(cell.material_id_);
815 if (cell.material_id_ < 0) ++invalid_mat_cell_count;
816 }
817
818 if (invalid_mat_cell_count > 0)
819 {
820 log.LogAllWarning() << "Number of invalid material cells: " << invalid_mat_cell_count;
821 }
822
823 // Get ready for processing
824 std::stringstream materials_list;
825 matid_to_xs_map_.clear();
826 matid_to_src_map_.clear();
827
828 //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% Process materials found
829 const size_t num_physics_mats = material_stack.size();
830
831 for (const int& mat_id : unique_material_ids)
832 {
833 materials_list << "Material id " << mat_id;
834
835 // Check valid ids
836 if (mat_id < 0)
837 throw std::logic_error(fname + ": Cells encountered with no assigned "
838 "material.");
839 if (static_cast<size_t>(mat_id) >= num_physics_mats)
840 throw std::logic_error(fname + ": Cells encountered with material id that"
841 " matches no material in physics material "
842 "library.");
843
844 auto current_material = GetStackItemPtr(material_stack, mat_id, fname);
845
846 // Extract properties
847 using MatProperty = PropertyType;
848 bool found_transport_xs = false;
849 for (const auto& property : current_material->properties_)
850 {
851 if (property->Type() == MatProperty::TRANSPORT_XSECTIONS)
852 {
853 auto transp_xs = std::static_pointer_cast<MultiGroupXS>(property);
854 matid_to_xs_map_[mat_id] = transp_xs;
855 found_transport_xs = true;
856 } // transport xs
857 if (property->Type() == MatProperty::ISOTROPIC_MG_SOURCE)
858 {
859 auto mg_source = std::static_pointer_cast<IsotropicMultiGrpSource>(property);
860
861 if (mg_source->source_value_g_.size() < groups_.size())
862 {
863 log.LogAllWarning() << fname + ": Isotropic Multigroup source specified in "
864 << "material \"" << current_material->name_ << "\" has fewer "
865 << "energy groups than called for in the simulation. "
866 << "Source will be ignored.";
867 }
868 else { matid_to_src_map_[mat_id] = mg_source; }
869 } // P0 source
870 } // for property
871
872 // Check valid property
873 if (!found_transport_xs)
874 {
875 log.LogAllError() << fname + ": Found no transport cross-section property for "
876 << "material \"" << current_material->name_ << "\".";
877 Exit(EXIT_FAILURE);
878 }
879 // Check number of groups legal
880 if (matid_to_xs_map_[mat_id]->NumGroups() < groups_.size())
881 {
882 log.LogAllError() << fname + ": Found material \"" << current_material->name_ << "\" has "
883 << matid_to_xs_map_[mat_id]->NumGroups() << " groups and"
884 << " the simulation has " << groups_.size() << " groups."
885 << " The material must have a greater or equal amount of groups.";
886 Exit(EXIT_FAILURE);
887 }
888
889 // Check number of moments
890 if (matid_to_xs_map_[mat_id]->ScatteringOrder() < options_.scattering_order)
891 {
892 log.Log0Warning() << fname + ": Found material \"" << current_material->name_
893 << "\" has a scattering order of "
894 << matid_to_xs_map_[mat_id]->ScatteringOrder()
895 << " and the simulation has a scattering order of "
896 << options_.scattering_order
897 << ". The higher moments will therefore not be used.";
898 }
899
900 materials_list << " number of moments " << matid_to_xs_map_[mat_id]->ScatteringOrder() + 1
901 << "\n";
902 } // for material id
903
904 //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% Initialize precursor
905 // properties
906
907 num_precursors_ = 0;
908 max_precursors_per_material_ = 0;
909 for (const auto& mat_id_xs : matid_to_xs_map_)
910 {
911 const auto& xs = mat_id_xs.second;
912 num_precursors_ += xs->NumPrecursors();
913 if (xs->NumPrecursors() > max_precursors_per_material_)
914 max_precursors_per_material_ = xs->NumPrecursors();
915 }
916
917 // if no precursors, turn off precursors
918 if (num_precursors_ == 0) options_.use_precursors = false;
919
920 // check compatibility when precursors are on
921 if (options_.use_precursors)
922 {
923 for (const auto& mat_id_xs : matid_to_xs_map_)
924 {
925 const auto& xs = mat_id_xs.second;
926 if (xs->IsFissionable() && num_precursors_ == 0)
927 throw std::logic_error("Incompatible cross section data encountered."
928 "When delayed neutron data is present for one "
929 "fissionable material, it must be present for "
930 "all fissionable materials.");
931 }
932 }
933
934 // Update transport views if available
935 if (grid_ptr_->local_cells.size() == cell_transport_views_.size())
936 for (const auto& cell : grid_ptr_->local_cells)
937 {
938 const auto& xs_ptr = matid_to_xs_map_[cell.material_id_];
939 auto& transport_view = cell_transport_views_[cell.local_id_];
940
941 transport_view.ReassingXS(*xs_ptr);
942 }
943
944 log.Log0Verbose1() << "Materials Initialized:\n" << materials_list.str() << "\n";
945
946 mpi_comm.barrier();
947}

958void
959LBSSolver::ComputeUnitIntegrals()
960{
961 log.Log() << "Computing unit integrals.\n";
962 const auto& sdm = *discretization_;
963
964 // Define spatial weighting functions
965 struct SpatialWeightFunction // SWF
966 {
967 virtual double operator()(const Vector3& pt) const { return 1.0; }
968 virtual ~SpatialWeightFunction() = default;
969 };
970
971 struct SphericalSWF : public SpatialWeightFunction
972 {
973 double operator()(const Vector3& pt) const override { return pt[2] * pt[2]; }
974 };
975
976 struct CylindricalSWF : public SpatialWeightFunction
977 {
978 double operator()(const Vector3& pt) const override { return pt[0]; }
979 };
980
981 auto swf_ptr = std::make_shared<SpatialWeightFunction>();
982 if (options_.geometry_type == lbs::GeometryType::ONED_SPHERICAL)
983 swf_ptr = std::make_shared<SphericalSWF>();
984 if (options_.geometry_type == lbs::GeometryType::TWOD_CYLINDRICAL)
985 swf_ptr = std::make_shared<CylindricalSWF>();
986
987 auto ComputeCellUnitIntegrals = [&sdm](const Cell& cell, const SpatialWeightFunction& swf)
988 {
989 const auto& cell_mapping = sdm.GetCellMapping(cell);
990 const size_t cell_num_faces = cell.faces_.size();
991 const size_t cell_num_nodes = cell_mapping.NumNodes();
992 const auto vol_qp_data = cell_mapping.MakeVolumetricQuadraturePointData();
993
994 MatDbl IntV_gradshapeI_gradshapeJ(cell_num_nodes, VecDbl(cell_num_nodes));
995 MatVec3 IntV_shapeI_gradshapeJ(cell_num_nodes, VecVec3(cell_num_nodes));
996 MatDbl IntV_shapeI_shapeJ(cell_num_nodes, VecDbl(cell_num_nodes));
997 VecDbl IntV_shapeI(cell_num_nodes);
998
999 std::vector<MatDbl> IntS_shapeI_shapeJ(cell_num_faces);
1000 std::vector<MatVec3> IntS_shapeI_gradshapeJ(cell_num_faces);
1001 std::vector<VecDbl> IntS_shapeI(cell_num_faces);
1002
1003 // Volume integrals
1004 for (unsigned int i = 0; i < cell_num_nodes; ++i)
1005 {
1006 for (unsigned int j = 0; j < cell_num_nodes; ++j)
1007 {
1008 for (const auto& qp : vol_qp_data.QuadraturePointIndices())
1009 {
1010 IntV_gradshapeI_gradshapeJ[i][j] +=
1011 swf(vol_qp_data.QPointXYZ(qp)) *
1012 vol_qp_data.ShapeGrad(i, qp).Dot(vol_qp_data.ShapeGrad(j, qp)) *
1013 vol_qp_data.JxW(qp); // K-matrix
1014
1015 IntV_shapeI_gradshapeJ[i][j] +=
1016 swf(vol_qp_data.QPointXYZ(qp)) * vol_qp_data.ShapeValue(i, qp) *
1017 vol_qp_data.ShapeGrad(j, qp) * vol_qp_data.JxW(qp); // G-matrix
1018
1019 IntV_shapeI_shapeJ[i][j] +=
1020 swf(vol_qp_data.QPointXYZ(qp)) * vol_qp_data.ShapeValue(i, qp) *
1021 vol_qp_data.ShapeValue(j, qp) * vol_qp_data.JxW(qp); // M-matrix
1022 } // for qp
1023 } // for j
1024
1025 for (const auto& qp : vol_qp_data.QuadraturePointIndices())
1026 {
1027 IntV_shapeI[i] +=
1028 swf(vol_qp_data.QPointXYZ(qp)) * vol_qp_data.ShapeValue(i, qp) * vol_qp_data.JxW(qp);
1029 } // for qp
1030 } // for i
1031
1032 // surface integrals
1033 for (size_t f = 0; f < cell_num_faces; ++f)
1034 {
1035 const auto faces_qp_data = cell_mapping.MakeSurfaceQuadraturePointData(f);
1036 IntS_shapeI_shapeJ[f].resize(cell_num_nodes, VecDbl(cell_num_nodes));
1037 IntS_shapeI[f].resize(cell_num_nodes);
1038 IntS_shapeI_gradshapeJ[f].resize(cell_num_nodes, VecVec3(cell_num_nodes));
1039
1040 for (unsigned int i = 0; i < cell_num_nodes; ++i)
1041 {
1042 for (unsigned int j = 0; j < cell_num_nodes; ++j)
1043 {
1044 for (const auto& qp : faces_qp_data.QuadraturePointIndices())
1045 {
1046 IntS_shapeI_shapeJ[f][i][j] += swf(faces_qp_data.QPointXYZ(qp)) *
1047 faces_qp_data.ShapeValue(i, qp) *
1048 faces_qp_data.ShapeValue(j, qp) * faces_qp_data.JxW(qp);
1049 IntS_shapeI_gradshapeJ[f][i][j] +=
1050 swf(faces_qp_data.QPointXYZ(qp)) * faces_qp_data.ShapeValue(i, qp) *
1051 faces_qp_data.ShapeGrad(j, qp) * faces_qp_data.JxW(qp);
1052 } // for qp
1053 } // for j
1054
1055 for (const auto& qp : faces_qp_data.QuadraturePointIndices())
1056 {
1057 IntS_shapeI[f][i] += swf(faces_qp_data.QPointXYZ(qp)) * faces_qp_data.ShapeValue(i, qp) *
1058 faces_qp_data.JxW(qp);
1059 } // for qp
1060 } // for i
1061 } // for f
1062
1063 return UnitCellMatrices{IntV_gradshapeI_gradshapeJ,
1064 IntV_shapeI_gradshapeJ,
1065 IntV_shapeI_shapeJ,
1066 IntV_shapeI,
1067
1068 IntS_shapeI_shapeJ,
1069 IntS_shapeI_gradshapeJ,
1070 IntS_shapeI};
1071 };
1072
1073 const size_t num_local_cells = grid_ptr_->local_cells.size();
1074 unit_cell_matrices_.resize(num_local_cells);
1075
1076 for (const auto& cell : grid_ptr_->local_cells)
1077 unit_cell_matrices_[cell.local_id_] = ComputeCellUnitIntegrals(cell, *swf_ptr);
1078
1079 const auto ghost_ids = grid_ptr_->cells.GetGhostGlobalIDs();
1080 for (uint64_t ghost_id : ghost_ids)
1081 unit_ghost_cell_matrices_[ghost_id] =
1082 ComputeCellUnitIntegrals(grid_ptr_->cells[ghost_id], *swf_ptr);
1083
1084 // Assessing global unit cell matrix storage
1085 std::array<size_t, 2> num_local_ucms = {unit_cell_matrices_.size(),
1086 unit_ghost_cell_matrices_.size()};
1087 std::array<size_t, 2> num_globl_ucms = {0, 0};
1088
1089 mpi_comm.all_reduce(num_local_ucms.data(), 2, num_globl_ucms.data(), mpi::op::sum<size_t>());
1090
1091 opensn::mpi_comm.barrier();
1092 log.Log() << "Ghost cell unit cell-matrix ratio: "
1093 << (double)num_globl_ucms[1] * 100 / (double)num_globl_ucms[0] << "%";
1094 log.Log() << "Cell matrices computed. Process memory = " << std::setprecision(3)
1095 << GetMemoryUsageInMB() << " MB";
1096}

1137void
1138LBSSolver::InitializeParrays()
1139{
1140 log.Log() << "Initializing parallel arrays."
1141 << " G=" << num_groups_ << " M=" << num_moments_ << std::endl;
1142
1143 // Initialize unknown
1144 // structure
1145 flux_moments_uk_man_.unknowns_.clear();
1146 for (size_t m = 0; m < num_moments_; m++)
1147 {
1148 flux_moments_uk_man_.AddUnknown(UnknownType::VECTOR_N, groups_.size());
1149 flux_moments_uk_man_.unknowns_.back().text_name_ = "m" + std::to_string(m);
1150 }
1151
1152 // Compute local # of dof
1153 auto per_node = UnknownManager::GetUnitaryUnknownManager();
1154 local_node_count_ = discretization_->GetNumLocalDOFs(per_node);
1155 glob_node_count_ = discretization_->GetNumGlobalDOFs(per_node);
1156
1157 // Compute num of unknowns
1158 size_t num_grps = groups_.size();
1159 size_t local_unknown_count = local_node_count_ * num_grps * num_moments_;
1160
1161 log.LogAllVerbose1() << "LBS Number of phi unknowns: " << local_unknown_count;
1162
1163 // Size local vectors
1164 q_moments_local_.assign(local_unknown_count, 0.0);
1165 phi_old_local_.assign(local_unknown_count, 0.0);
1166 phi_new_local_.assign(local_unknown_count, 0.0);
1167
1168 // Setup groupset psi vectors
1169 psi_new_local_.clear();
1170 for (auto& groupset : groupsets_)
1171 {
1172 psi_new_local_.emplace_back();
1173 if (options_.save_angular_flux)
1174 {
1175 size_t num_ang_unknowns = discretization_->GetNumLocalDOFs(groupset.psi_uk_man_);
1176 psi_new_local_.back().assign(num_ang_unknowns, 0.0);
1177 }
1178 }
1179
1180 // Setup precursor vector
1181 if (options_.use_precursors)
1182 {
1183 size_t num_precursor_dofs = grid_ptr_->local_cells.size() * max_precursors_per_material_;
1184 precursor_new_local_.assign(num_precursor_dofs, 0.0);
1185 }
1186
1187 // Read Restart data
1188 if (options_.read_restart_data)
1189 ReadRestartData(options_.read_restart_folder_name, options_.read_restart_file_base);
1190 opensn::mpi_comm.barrier();
1191
1192 // Initialize transport views
1193 // Transport views act as a data structure to store information
1194 // related to the transport simulation. The most prominent function
1195 // here is that it holds the means to know where a given cell's
1196 // transport quantities are located in the unknown vectors (i.e. phi)
1197 //
1198 // Also, for a given cell, within a given sweep chunk,
1199 // we need to solve a matrix which square size is the
1200 // amount of nodes on the cell. max_cell_dof_count is
1201 // initialized here.
1202 //
1203 size_t block_MG_counter = 0; // Counts the strides of moment and group
1204
1205 const Vector3 ihat(1.0, 0.0, 0.0);
1206 const Vector3 jhat(0.0, 1.0, 0.0);
1207 const Vector3 khat(0.0, 0.0, 1.0);
1208
1209 max_cell_dof_count_ = 0;
1210 cell_transport_views_.clear();
1211 cell_transport_views_.reserve(grid_ptr_->local_cells.size());
1212 for (auto& cell : grid_ptr_->local_cells)
1213 {
1214 size_t num_nodes = discretization_->GetCellNumNodes(cell);
1215 int mat_id = cell.material_id_;
1216
1217 // compute cell volumes
1218 double cell_volume = 0.0;
1219 const auto& IntV_shapeI = unit_cell_matrices_[cell.local_id_].Vi_vectors;
1220 for (size_t i = 0; i < num_nodes; ++i)
1221 cell_volume += IntV_shapeI[i];
1222
1223 size_t cell_phi_address = block_MG_counter;
1224
1225 const size_t num_faces = cell.faces_.size();
1226 std::vector<bool> face_local_flags(num_faces, true);
1227 std::vector<int> face_locality(num_faces, opensn::mpi_comm.rank());
1228 std::vector<const Cell*> neighbor_cell_ptrs(num_faces, nullptr);
1229 bool cell_on_boundary = false;
1230 int f = 0;
1231 for (auto& face : cell.faces_)
1232 {
1233 if (not face.has_neighbor_)
1234 {
1235 Vector3& n = face.normal_;
1236
1237 int boundary_id = -1;
1238 if (n.Dot(ihat) > 0.999) boundary_id = 0;
1239 else if (n.Dot(ihat) < -0.999)
1240 boundary_id = 1;
1241 else if (n.Dot(jhat) > 0.999)
1242 boundary_id = 2;
1243 else if (n.Dot(jhat) < -0.999)
1244 boundary_id = 3;
1245 else if (n.Dot(khat) > 0.999)
1246 boundary_id = 4;
1247 else if (n.Dot(khat) < -0.999)
1248 boundary_id = 5;
1249
1250 if (boundary_id >= 0) face.neighbor_id_ = boundary_id;
1251 cell_on_boundary = true;
1252
1253 face_local_flags[f] = false;
1254 face_locality[f] = -1;
1255 } // if bndry
1256 else
1257 {
1258 const int neighbor_partition = face.GetNeighborPartitionID(*grid_ptr_);
1259 face_local_flags[f] = (neighbor_partition == opensn::mpi_comm.rank());
1260 face_locality[f] = neighbor_partition;
1261 neighbor_cell_ptrs[f] = &grid_ptr_->cells[face.neighbor_id_];
1262 }
1263
1264 ++f;
1265 } // for f
1266
1267 if (num_nodes > max_cell_dof_count_) max_cell_dof_count_ = num_nodes;
1268
1269 cell_transport_views_.emplace_back(cell_phi_address,
1270 num_nodes,
1271 num_grps,
1272 num_moments_,
1273 *matid_to_xs_map_[mat_id],
1274 cell_volume,
1275 face_local_flags,
1276 face_locality,
1277 neighbor_cell_ptrs,
1278 cell_on_boundary);
1279 block_MG_counter += num_nodes * num_grps * num_moments_;
1280 } // for local cell
1281
1282 // Populate grid nodal mappings
1283 // This is used in the Flux Data Structures (FLUDS)
1284 grid_nodal_mappings_.clear();
1285 grid_nodal_mappings_.reserve(grid_ptr_->local_cells.size());
1286 for (auto& cell : grid_ptr_->local_cells)
1287 {
1288 CellFaceNodalMapping cell_nodal_mapping;
1289 cell_nodal_mapping.reserve(cell.faces_.size());
1290
1291 for (auto& face : cell.faces_)
1292 {
1293 std::vector<short> face_node_mapping;
1294 std::vector<short> cell_node_mapping;
1295 int ass_face = -1;
1296
1297 if (face.has_neighbor_)
1298 {
1299 grid_ptr_->FindAssociatedVertices(face, face_node_mapping);
1300 grid_ptr_->FindAssociatedCellVertices(face, cell_node_mapping);
1301 ass_face = face.GetNeighborAssociatedFace(*grid_ptr_);
1302 }
1303
1304 cell_nodal_mapping.emplace_back(ass_face, face_node_mapping, cell_node_mapping);
1305 } // for f
1306
1307 grid_nodal_mappings_.push_back(cell_nodal_mapping);
1308 } // for local cell
1309
1310 // Get grid localized communicator set
1311 grid_local_comm_set_ = grid_ptr_->MakeMPILocalCommunicatorSet();
1312
1313 // Make face histogram
1314 grid_face_histogram_ = grid_ptr_->MakeGridFaceHistogram();
1315
1316 // Initialize Field Functions
1317 InitializeFieldFunctions();
1318
1319 opensn::mpi_comm.barrier();
1320 log.Log() << "Done with parallel arrays. Process memory = " << std::setprecision(3)
1321 << GetMemoryUsageInMB() << " MB" << std::endl;
1322}

1324void
1325LBSSolver::InitializeFieldFunctions()
1326{
1327 if (not field_functions_.empty()) return;
1328
1329 // Initialize Field Functions
1330 // for flux moments

1381void
1382LBSSolver::InitializeBoundaries()
1383{
1384 const std::string fname = "lbs::LBSSolver::InitializeBoundaries";
1385 // Determine boundary-ids involved in the problem
1386 std::set<uint64_t> globl_unique_bids_set;
1387 {
1388 std::set<uint64_t> local_unique_bids_set;
1389 for (const auto& cell : grid_ptr_->local_cells)
1390 for (const auto& face : cell.faces_)
1391 if (not face.has_neighbor_) local_unique_bids_set.insert(face.neighbor_id_);
1392
1393 std::vector<uint64_t> local_unique_bids(local_unique_bids_set.begin(),
1394 local_unique_bids_set.end());
1395 std::vector<uint64_t> recvbuf;
1396 mpi_comm.all_gather(local_unique_bids, recvbuf);
1397
1398 globl_unique_bids_set = local_unique_bids_set; // give it a head start
1399
1400 for (uint64_t bid : recvbuf)
1401 globl_unique_bids_set.insert(bid);
1402 }
1403
1404 // Initialize default incident boundary
1405 const size_t G = num_groups_;
1406
1407 sweep_boundaries_.clear();
1408 for (uint64_t bid : globl_unique_bids_set)
1409 {
1410 const bool has_no_preference = boundary_preferences_.count(bid) == 0;
1411 const bool has_not_been_set = sweep_boundaries_.count(bid) == 0;
1412 if (has_no_preference and has_not_been_set)
1413 {
1414 sweep_boundaries_[bid] = std::make_shared<BoundaryVaccuum>(G);
1415 } // defaulted
1416 else if (has_not_been_set)
1417 {
1418 const auto& bndry_pref = boundary_preferences_.at(bid);
1419 const auto& mg_q = bndry_pref.isotropic_mg_source;
1420
1421 if (bndry_pref.type == lbs::BoundaryType::VACUUM)
1422 sweep_boundaries_[bid] = std::make_shared<BoundaryVaccuum>(G);
1423 else if (bndry_pref.type == lbs::BoundaryType::INCIDENT_ISOTROPIC)
1424 sweep_boundaries_[bid] = std::make_shared<BoundaryIsotropicHomogenous>(G, mg_q);
1425 else if (bndry_pref.type == BoundaryType::INCIDENT_ANISTROPIC_HETEROGENEOUS)
1426 {
1427 // FIXME:
1428#if 0
1429 sweep_boundaries_[bid] = std::make_shared<BoundaryIncidentHeterogeneous>(
1430 G, std::make_unique<BoundaryFunctionToLua>(bndry_pref.source_function), bid);
1431#endif
1432 }
1433 else if (bndry_pref.type == lbs::BoundaryType::REFLECTING)
1434 {
1435 // Locally check all faces, that subscribe to this boundary,
1436 // have the same normal
1437 typedef Vector3 Vec3;
1438 const double EPSILON = 1.0e-12;
1439 std::unique_ptr<Vec3> n_ptr = nullptr;
1440 for (const auto& cell : grid_ptr_->local_cells)
1441 for (const auto& face : cell.faces_)
1442 if (not face.has_neighbor_ and face.neighbor_id_ == bid)
1443 {
1444 if (not n_ptr) n_ptr = std::make_unique<Vec3>(face.normal_);
1445 if (std::fabs(face.normal_.Dot(*n_ptr) - 1.0) > EPSILON)
1446 throw std::logic_error(fname +
1447 ": Not all face normals are, within tolerance, locally the "
1448 "same for the reflecting boundary condition requested.");
1449 }
1450
1451 // Now check globally
1452 const int local_has_bid = n_ptr != nullptr ? 1 : 0;
1453 const Vec3 local_normal = local_has_bid ? *n_ptr : Vec3(0.0, 0.0, 0.0);
1454
1455 std::vector<int> locJ_has_bid(opensn::mpi_comm.size(), 1);
1456 std::vector<double> locJ_n_val(opensn::mpi_comm.size() * 3, 0.0);
1457
1458 mpi_comm.all_gather(local_has_bid, locJ_has_bid);
1459 std::vector<double> lnv = {local_normal.x, local_normal.y, local_normal.z};
1460 mpi_comm.all_gather(lnv.data(), 3, locJ_n_val.data(), 3);
1461
1462 Vec3 global_normal;
1463 for (int j = 0; j < opensn::mpi_comm.size(); ++j)
1464 {
1465 if (locJ_has_bid[j])
1466 {
1467 int offset = 3 * j;
1468 const double* n = &locJ_n_val[offset];
1469 const Vec3 locJ_normal(n[0], n[1], n[2]);
1470
1471 if (local_has_bid)
1472 if (std::fabs(local_normal.Dot(locJ_normal) - 1.0) > EPSILON)
1473 throw std::logic_error(fname +
1474 ": Not all face normals are, within tolerance, globally the "
1475 "same for the reflecting boundary condition requested.");
1476
1477 global_normal = locJ_normal;
1478 }
1479 }
1480
1481 sweep_boundaries_[bid] = std::make_shared<BoundaryReflecting>(
1482 G, global_normal, MapGeometryTypeToCoordSys(options_.geometry_type));
1483 }
1484 } // non-defaulted
1485 } // for bndry id
1486}

1488void
1489LBSSolver::InitializePointSources()
1490{
1491 const std::string fname = "InitializePointSources";
1492
1493 // Loop over point sources
1494 for (auto& point_source : point_sources_)
1495 {
1496 if (point_source.Strength().size() != num_groups_)
1497 throw std::logic_error(fname +
1498 ": Point source multigroup strength vector "
1499 "is not compatible with the number of "
1500 "groups in the simulation. Expected " +
1501 std::to_string(num_groups_) + " found " +
1502 std::to_string(point_source.Strength().size()));
1503
1504 const auto& p = point_source.Location();
1505 double v_total = 0.0; // Total volume of all cells sharing
1506 // this source
1507 std::vector<PointSource::ContainingCellInfo> temp_list;
1508 for (const auto& cell : grid_ptr_->local_cells)
1509 {
1510 if (grid_ptr_->CheckPointInsideCell(cell, p))
1511 {
1512 const auto& cell_view = discretization_->GetCellMapping(cell);
1513 const auto& cell_matrices = unit_cell_matrices_[cell.local_id_];
1514 const auto& M = cell_matrices.M_matrix;
1515 const auto& I = cell_matrices.Vi_vectors;
1516
1517 std::vector<double> shape_values;
1518 cell_view.ShapeValues(point_source.Location(), shape_values);
1519
1520 const auto M_inv = Inverse(M);
1521
1522 const auto q_p_weights = MatMul(M_inv, shape_values);
1523
1524 double v_cell = 0.0;
1525 for (double val : I)
1526 v_cell += val;
1527 v_total += v_cell;
1528
1529 temp_list.push_back(
1530 PointSource::ContainingCellInfo{v_cell, cell.local_id_, shape_values, q_p_weights});
1531 } // if inside
1532 } // for local cell
1533
1534 auto ghost_global_ids = grid_ptr_->cells.GetGhostGlobalIDs();
1535 for (uint64_t ghost_global_id : ghost_global_ids)
1536 {
1537 const auto& neighbor_cell = grid_ptr_->cells[ghost_global_id];
1538 if (grid_ptr_->CheckPointInsideCell(neighbor_cell, p))
1539 {
1540 const auto& cell_matrices = unit_ghost_cell_matrices_[neighbor_cell.global_id_];
1541 for (double val : cell_matrices.Vi_vectors)
1542 v_total += val;
1543 } // if point inside
1544 } // for ghost cell
1545
1546 point_source.ClearInitializedInfo();
1547 for (const auto& info : temp_list)
1548 {
1549 point_source.AddContainingCellInfo(
1550 info.volume_weight / v_total, info.cell_local_id, info.shape_values, info.node_weights);
1551 const auto& cell = grid_ptr_->local_cells[info.cell_local_id];
1552 // Output message
1553 {
1554 std::stringstream output;
1555 output << "Point source at " << p.PrintStr() << " assigned to cell " << cell.global_id_
1556 << " with shape values ";
1557 for (double val : info.shape_values)
1558 output << val << " ";
1559 output << "volume_weight=" << info.volume_weight / v_total;
1560
1561 log.LogAll() << output.str();
1562 }
1563 } // for info in temp list
1564 } // for point_source
1565}

2688{

2695 {
