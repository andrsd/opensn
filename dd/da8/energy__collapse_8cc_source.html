36 for (int g = 0; g < num_groups; g++)

38 if (scheme == EnergyCollapseScheme::JFULL)

40 A[g][g] = sigma_t[g] - S[g][g];
41 for (int gp = 0; gp < g; gp++)
42 B[g][gp] = S[g][gp];

45 B[g][gp] = S[g][gp];

56 // Some cross-sections developed from monte-carlo
57 // methods can result in some of the groups

60 // initial guess of 1.0. Here we reset them

74
75 // Compute two-grid diffusion quantities
76 double sum = 0.0;
77 for (int g = 0; g < num_groups; g++)
78 sum += std::fabs(E[g]);
79
80 for (int g = 0; g < num_groups; g++)
81 spectrum[g] = std::fabs(E[g]) / sum;
82
83 for (int g = 0; g < num_groups; ++g)
84 {
85 collapsed_D += diffusion_coeff[g] * spectrum[g];
86
87 collapsed_sig_a += sigma_t[g] * spectrum[g];
88
89 for (int gp = 0; gp < num_groups; ++gp)
90 collapsed_sig_a -= S[g][gp] * spectrum[gp];
91 }
92
93 // Verbose output the spectrum
94 log.Log0Verbose1() << "Fundamental eigen-value: " << rho;
95 std::stringstream outstr;
96 for (auto& xi : spectrum)
97 outstr << xi << '\n';
98 log.Log0Verbose1() << outstr.str();
99
100 return {collapsed_D, collapsed_sig_a, spectrum};
101}
