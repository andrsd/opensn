118std::vector<double>
119QuadratureGaussLegendre::FindRoots(unsigned int N, unsigned int max_iters, double tol)
120{
121 // Populate init guess
122 // This initial guess proved to be quite important
123 // at higher N since the roots start to get
124 // squeezed to -1 and 1.
125 int num_search_intvls = 1000;
126 if (N > 64) num_search_intvls *= 10;
127 if (N > 256) num_search_intvls *= 10;
128 if (N > 768) num_search_intvls *= 10;
129
130 if (N > 2056)
131 {
132 num_search_intvls *= 10;
133 log.Log0Warning() << "QuadratureGaussLegendre::FindRoots: "
134 << "The order of the polynomial for which to find the roots is "
135 << "greater than 2056. Accuracy of the root finder will be diminished "
136 << "along with a reduction in stability.";
137 }
138
139 // For this code we simply check to see where the
140 // polynomial changes sign.
141 double delta = 2.0 / num_search_intvls;
142 std::vector<double> xk(N, 0.0);
143 int counter = -1;
144 for (size_t i = 0; i < num_search_intvls; i++)
145 {
146 double x_i = -1.0 + i * delta;
147 double x_ip1 = x_i + delta;
148
149 if (Legendre(N, x_i) * Legendre(N, x_ip1) < 0.0) xk[++counter] = (x_ip1 + x_i) / 2.0;
150 }
151
152 // Apply algorithm
153 // Refer to equation 4.3 in [1]. Sum 1 (S1) is used in the
154 // computation of B at x_k. Sum 2 (S2) is used in equation 4.3.
155 // Equation 4.3 is broken up into pieces as follows:
156 // - a = block bracket containing the second derivative
157 // - b = denominator
158 // - c = everything but xold
159 for (int k = 0; k < N; k++)
160 {
161 for (size_t iteration = 0; iteration < max_iters; iteration++)
162 {
163 double xold = xk[k];
164 double f = Legendre(N, xold); // Function evaluation
165 double fp = dLegendredx(N, xold); // First derivative
166 double fpp = d2Legendredx2(N, xold); // Second derivative
167
168 // Compute sum 1
169 double S1 = 0.0;
170 for (int i = 0; i <= (k - 1); i++)
171 S1 += 1.0 / (xk[k] - xk[i]);
172
173 // Compute B at x_k
174 double B_xk = fp - f * S1;
175
176 // Compute sum 2
177 double S2 = 0.0;
178 for (int i = 0; i <= (k - 1); i++)
179 S2 += 1.0 / (xk[k] - xk[i]) / (xk[k] - xk[i]);
180
181 // Compute final formula
182 double a = fpp + f * S2;
183 double b = B_xk * B_xk + fp * fp - f * a;
184 double c = 2.0 * f * B_xk / b;
185
186 xk[k] = xold - c;
187
188 if (std::fabs(xk[k] - xold) < tol) break;
189 } // for iteration
190 } // for k
191
192 std::stable_sort(xk.begin(), xk.end());
193
194 return xk;
195}
