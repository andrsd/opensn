43CBC_ASynchronousCommunicator::SendData()
44{
45 typedef int MPIRank;
46
47 // First we convert any new outgoing messages from the queue into
48 // buffer messages. We aggregate these messages per location-id
49 // they need to be sent to
50 if (not outgoing_message_queue_.empty())
51 {
52 std::map<MPIRank, BufferItem> locI_buffer_map;
53
54 for (const auto& [msg_key, data] : outgoing_message_queue_)
55 {
56 const MPIRank locI = std::get<0>(msg_key);
57 const uint64_t cell_global_id = std::get<1>(msg_key);
58 const unsigned int face_id = std::get<2>(msg_key);
59 const size_t data_size = data.size();
60
61 BufferItem& buffer_item = locI_buffer_map[locI];
62
63 buffer_item.destination_ = locI;
64 auto& buffer_array = buffer_item.data_array_;
65
66 buffer_array.Write(cell_global_id);
67 buffer_array.Write(face_id);
68 buffer_array.Write(data_size);
69
70 for (const double value : data) // actual psi_data
71 buffer_array.Write(value);
72 } // for item in queue
73
74 for (auto& [locI, buffer] : locI_buffer_map)
75 send_buffer_.push_back(std::move(buffer));
76
77 outgoing_message_queue_.clear();
78 } // if there are outgoing messages
79
80 // Now we attempt to flush items in the send buffer
81 bool all_messages_sent = true;
82 for (auto& buffer_item : send_buffer_)
83 {
84 if (not buffer_item.send_initiated_)
85 {
86 const int locJ = buffer_item.destination_;
87 auto& comm = comm_set_.LocICommunicator(locJ);
88 auto dest = comm_set_.MapIonJ(locJ, locJ);
89 auto tag = static_cast<int>(angle_set_id_);
90 buffer_item.mpi_request_ = comm.isend(dest, tag, buffer_item.data_array_.Data());
91 buffer_item.send_initiated_ = true;
92 }
93
94 if (not buffer_item.completed_)
95 {
96 if (mpi::test(buffer_item.mpi_request_)) buffer_item.completed_ = true;
97 else
98 all_messages_sent = false;
99 }
100 } // for item in buffer
101
102 return all_messages_sent;
103}

105std::vector<uint64_t>
106CBC_ASynchronousCommunicator::ReceiveData()
107{
108 typedef std::pair<uint64_t, unsigned int> CellFaceKey; // cell_gid + face_id
109
110 std::map<CellFaceKey, std::vector<double>> received_messages;
111 std::vector<uint64_t> cells_who_received_data;
112 auto& location_dependencies = fluds_.GetSPDS().GetLocationDependencies();
113 for (int locJ : location_dependencies)
114 {
115 auto& comm = comm_set_.LocICommunicator(opensn::mpi_comm.rank());
116 auto source_rank = comm_set_.MapIonJ(locJ, opensn::mpi_comm.rank());
117 auto tag = static_cast<int>(angle_set_id_);
118 mpi::Status status;
119 if (comm.iprobe(source_rank, tag, status))
120 {
121 int num_items = status.get_count<std::byte>();
122 std::vector<std::byte> recv_buffer(num_items);
123 comm.recv(source_rank, status.tag(), recv_buffer.data(), num_items);
124
125 ByteArray data_array(recv_buffer);
126
127 while (not data_array.EndOfBuffer())
128 {
129 const uint64_t cell_global_id = data_array.Read<uint64_t>();
130 const auto face_id = data_array.Read<unsigned int>();
131 const size_t data_size = data_array.Read<size_t>();
132
133 std::vector<double> psi_data;
134 psi_data.reserve(data_size);
135 for (size_t k = 0; k < data_size; ++k)
136 psi_data.push_back(data_array.Read<double>());
137
138 received_messages[{cell_global_id, face_id}] = std::move(psi_data);
139 cells_who_received_data.push_back(
140 fluds_.GetSPDS().Grid().MapCellGlobalID2LocalID(cell_global_id));
141 } // while not at end of buffer
142 } // Process each message embedded in buffer
143 }
144
145 cbc_fluds_.DeplocsOutgoingMessages().merge(received_messages);
146
147 return cells_who_received_data;
148}
