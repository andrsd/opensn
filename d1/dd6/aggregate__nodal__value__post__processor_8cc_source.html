69void
70AggregateNodalValuePostProcessor::Execute(const Event& event_context)
71{
72 if (not initialized_) Initialize();
73
74 const auto* grid_field_function = GetGridBasedFieldFunction();
75
76 ChiLogicalErrorIf(not grid_field_function,
77 "Attempted to access invalid field"
78 "function");
79
80 const auto& ref_ff = *grid_field_function;
81 const auto& sdm = ref_ff.GetSpatialDiscretization();
82 const auto& grid = sdm.Grid();
83
84 const auto& uk_man = ref_ff.GetUnknownManager();
85 const auto uid = 0;
86 const auto cid = 0;
87
88 const auto field_data = ref_ff.GetGhostedFieldVector();
89 const size_t num_local_dofs =
90 ref_ff.GetSpatialDiscretization().GetNumLocalDOFs(ref_ff.GetUnknownManager());
91
92 double local_max_value = 0.0;
93 double local_min_value = 0.0;
94 double local_accumulation = 0.0;
95 bool first_local = true;
96 for (const uint64_t cell_local_id : cell_local_ids_)
97 {
98 const auto& cell = grid.local_cells[cell_local_id];
99 const auto& cell_mapping = sdm.GetCellMapping(cell);
100 const size_t num_nodes = cell_mapping.NumNodes();
101
102 for (size_t i = 0; i < num_nodes; ++i)
103 {
104 const int64_t imap = sdm.MapDOFLocal(cell, i, uk_man, uid, cid);
105 if (imap >= 0 and imap < num_local_dofs)
106 {
107 const double field_value = field_data[imap];
108 if (first_local)
109 {
110 local_max_value = field_value;
111 local_min_value = field_value;
112 first_local = false;
113 }
114
115 local_max_value = std::max(local_max_value, field_value);
116 local_min_value = std::min(local_min_value, field_value);
117 local_accumulation += field_value;
118 }
119 } // for i
120 } // for cell-id
121
122 if (operation_ == "max")
123 {
124 double globl_max_value;
125 mpi_comm.all_reduce(local_max_value, globl_max_value, mpi::op::sum<double>());
126
127 value_ = ParameterBlock("", globl_max_value);
128 }
129 else if (operation_ == "min")
130 {
131 double globl_min_value;
132 mpi_comm.all_reduce(local_min_value, globl_min_value, mpi::op::min<double>());
133
134 value_ = ParameterBlock("", globl_min_value);
135 }
136 else if (operation_ == "avg")
137 {
138 double globl_accumulation;
139 mpi_comm.all_reduce(local_accumulation, globl_accumulation, mpi::op::sum<double>());
140
141 const size_t num_globl_dofs =
142 ref_ff.GetSpatialDiscretization().GetNumGlobalDOFs(ref_ff.GetUnknownManager());
143 value_ = ParameterBlock("", globl_accumulation / double(num_globl_dofs));
144 }
145 else
146 ChiLogicalError("Unsupported operation type \"" + operation_ + "\".");
147
148 const int event_code = event_context.Code();
149 if (event_code == Event::SolverInitialized or event_code == Event::SolverAdvanced)
150 {
151 const auto& event_params = event_context.Parameters();
152
153 if (event_params.Has("timestep_index") and event_params.Has("time"))
154 {
155 const size_t index = event_params.GetParamValue<size_t>("timestep_index");
156 const double time = event_params.GetParamValue<double>("time");
157 TimeHistoryEntry entry{index, time, value_};
158 time_history_.push_back(std::move(entry));
159 }
160 }
161}
