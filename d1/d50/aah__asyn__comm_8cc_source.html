243bool
244AAH_ASynchronousCommunicator::ReceiveDelayedData(int angle_set_num)
245{
246 const auto& spds = fluds_.GetSPDS();
247
248 const auto& delayed_location_dependencies = spds.GetDelayedLocationDependencies();
249 const size_t num_delayed_loc_deps = delayed_location_dependencies.size();
250
251 // Receive delayed data
252 bool all_messages_received = true;
253 for (size_t prelocI = 0; prelocI < num_delayed_loc_deps; prelocI++)
254 {
255 int locJ = delayed_location_dependencies[prelocI];
256
257 int num_mess = delayed_prelocI_message_count[prelocI];
258 for (int m = 0; m < num_mess; m++)
259 {
260 if (not delayed_prelocI_message_received[prelocI][m])
261 {
262 auto& comm = comm_set_.LocICommunicator(opensn::mpi_comm.rank());
263 auto source_rank = comm_set_.MapIonJ(locJ, opensn::mpi_comm.rank());
264 auto tag = max_num_mess * angle_set_num + m;
265 auto message_available = comm.iprobe(source_rank, tag);
266 if (not message_available)
267 {
268 all_messages_received = false;
269 continue;
270 }
271
272 // Receive upstream data
273 auto& upstream_psi = fluds_.DelayedPrelocIOutgoingPsi()[prelocI];
274
275 u_ll_int block_addr = delayed_prelocI_message_blockpos[prelocI][m];
276 u_ll_int message_size = delayed_prelocI_message_size[prelocI][m];
277 comm.recv(source_rank, tag, &upstream_psi[block_addr], message_size);
278
279 delayed_prelocI_message_received[prelocI][m] = true;
280 } // if not message already received
281 } // for message
282 } // for delayed predecessor
283
284 if (not all_messages_received) return false;
285
286 return true;
287}

289AngleSetStatus
290AAH_ASynchronousCommunicator::ReceiveUpstreamPsi(int angle_set_num)
291{
292 const auto& spds = fluds_.GetSPDS();
293
294 // Resize FLUDS non-local incoming Data
295 const size_t num_loc_deps = spds.GetLocationDependencies().size();
296 if (!upstream_data_initialized)
297 {
298 fluds_.AllocatePrelocIOutgoingPsi(num_groups_, num_angles_, num_loc_deps);
299
300 upstream_data_initialized = true;
301 }
302
303 // Assume all data is available and now try to receive all of it
304 bool ready_to_execute = true;
305 for (size_t prelocI = 0; prelocI < num_loc_deps; prelocI++)
306 {
307 int locJ = spds.GetLocationDependencies()[prelocI];
308
309 size_t num_mess = prelocI_message_count[prelocI];
310 for (int m = 0; m < num_mess; m++)
311 {
312 if (!prelocI_message_received[prelocI][m])
313 {
314 auto& comm = comm_set_.LocICommunicator(opensn::mpi_comm.rank());
315 auto source = comm_set_.MapIonJ(locJ, opensn::mpi_comm.rank());
316 auto tag = max_num_mess * angle_set_num + m;
317 if (not comm.iprobe(source, tag))
318 {
319 ready_to_execute = false;
320 continue;
321 } // if message is not available
322
323 // Receive upstream data
324 auto& upstream_psi = fluds_.PrelocIOutgoingPsi()[prelocI];
325
326 u_ll_int block_addr = prelocI_message_blockpos[prelocI][m];
327 u_ll_int message_size = prelocI_message_size[prelocI][m];
328
329 comm.recv(source, tag, &upstream_psi[block_addr], message_size);
330 prelocI_message_received[prelocI][m] = true;
331 } // if not message already received
332 } // for message
333
334 if (!ready_to_execute) break;
335 } // for predecessor
336
337 if (!ready_to_execute) return AngleSetStatus::RECEIVING;
338 else
339 return AngleSetStatus::READY_TO_EXECUTE;
340}

342void
343AAH_ASynchronousCommunicator::SendDownstreamPsi(int angle_set_num)
344{
345 const auto& spds = fluds_.GetSPDS();
346
347 const auto& location_successors = spds.GetLocationSuccessors();
348
349 const size_t num_successors = location_successors.size();
350 for (size_t deplocI = 0; deplocI < num_successors; deplocI++)
351 {
352 int locJ = location_successors[deplocI];
353
354 int num_mess = deplocI_message_count[deplocI];
355 for (int m = 0; m < num_mess; m++)
356 {
357 u_ll_int block_addr = deplocI_message_blockpos[deplocI][m];
358 u_ll_int message_size = deplocI_message_size[deplocI][m];
359
360 const auto& outgoing_psi = fluds_.DeplocIOutgoingPsi()[deplocI];
361
362 auto& comm = comm_set_.LocICommunicator(locJ);
363 auto dest = comm_set_.MapIonJ(locJ, locJ);
364 auto tag = max_num_mess * angle_set_num + m;
365 deplocI_message_request[deplocI][m] =
366 comm.isend(dest, tag, &outgoing_psi[block_addr], message_size);
367 } // for message
368 } // for deplocI
