269 VectorNX<3, NumberFormat> Cross(const VectorNX<2, NumberFormat>& rhs)
270 {
271 static_assert(N == 2 or N == 3, "VectorNX::Cross only defined for dimension 2 or 3 vectors.");
272
273 VectorNX<3, NumberFormat> newVector;
274 if (dimension == 3)
275 {
276 newVector(0) = -elements[2] * rhs.elements[1];
277 newVector(1) = elements[2] * rhs.elements[0];
278 newVector(2) = elements[0] * rhs.elements[1] - elements[1] * rhs.elements[0];

286 /**Vector cross-product.

288 VectorNX<3, NumberFormat> Cross(const VectorNX<3, NumberFormat>& rhs)

291

296 newVector(1) = elements[2] * rhs.elements[0] - elements[0] * rhs.elements[2];
297 newVector(2) = elements[0] * rhs.elements[1] - elements[1] * rhs.elements[0];
