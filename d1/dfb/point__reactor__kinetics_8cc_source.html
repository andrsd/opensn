84void
85TransientSolver::Initialize()
86{
87 // Check size
88 if (lambdas_.size() != betas_.size())
89 throw std::logic_error(TextName() + ": Number of precursors cannot be "
90 "deduced from precursor data because "
91 "the data lists are of different size.");
92
93 beta_ = std::accumulate(betas_.begin(), betas_.end(), 0.0);
94
95 // Initializing linalg items
96 const auto& J = num_precursors_;
97 A_ = DynamicMatrix<double>(J + 1, J + 1, 0.0);
98 I_ = A_;
99 I_.SetDiagonal(1.0);
100
101 x_t_ = DynamicVector<double>(J + 1, 0.0);
102
103 // Assembling system
104 A_[0][0] = beta_ * (rho_ - 1.0) / gen_time_;
105 for (size_t j = 1; j <= J; ++j)
106 {
107 A_[0][j] = lambdas_[j - 1];
108 A_[j][j] = -lambdas_[j - 1];
109 A_[j][0] = betas_[j - 1] / gen_time_;
110 }
111
112 q_.resize(J + 1, 0.0);
113 q_[0] = source_strength_;
114
115 // Initializing x
116 // If there is a source and the reactivity is < 0 then
117 // there exists a unique solution.
118 if (source_strength_ > 0.0 and rho_ < 0.0)
119 {
120 const auto b_theta = -1.0 * q_;
121
122 x_t_ = A_.Inverse() * b_theta;
123 }
124 // Otherwise we initialize the system as a critical system with
125 // no source.
126 else
127 {
128 auto A_temp = A_;
129 auto b_temp = x_t_;
130 b_temp.Set(0.0);
131 b_temp[0] = 1.0;
132 for (auto& val : A_temp[0])
133 val = 0.0;
134 A_temp[0][0] = 1.0;
135
136 x_t_ = A_temp.Inverse() * b_temp;
137 }
138
139 log.Log() << "Final: " << x_t_.PrintStr();
140}

154void
155TransientSolver::Step()
156{
157 log.Log() << "Solver \"" + TextName() + "\" " + timestepper_->StringTimeInfo();
158
159 const double dt = timestepper_->TimeStepSize();
160
161 A_[0][0] = beta_ * (rho_ - 1.0) / gen_time_;
162
163 if (time_integration_ == "implicit_euler" or time_integration_ == "crank_nicolson")
164 {
165 double theta = 1.0;
166
167 if (time_integration_ == "implicit_euler") theta = 1.0;
168 else if (time_integration_ == "crank_nicolson")
169 theta = 0.5;
170
171 const double inv_tau = theta * dt;
172
173 auto A_theta = I_ - inv_tau * A_;
174 auto b_theta = x_t_ + inv_tau * q_;
175
176 auto x_theta = A_theta.Inverse() * b_theta;
177
178 x_tp1_ = x_t_ + (1.0 / theta) * (x_theta - x_t_);
179 }
180 else if (time_integration_ == "explicit_euler") { x_tp1_ = x_t_ + dt * A_ * x_t_ + dt * q_; }
181 else
182 ChiLogicalError("Unsupported time integration scheme.");
183
184 period_tph_ = dt / std::log(x_tp1_[0] / x_t_[0]);
185
186 if (period_tph_ > 0.0 and period_tph_ > 1.0e6) period_tph_ = 1.0e6;
187 if (period_tph_ < 0.0 and period_tph_ < -1.0e6) period_tph_ = -1.0e6;
188}
