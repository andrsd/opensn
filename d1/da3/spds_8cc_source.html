87
88 switch (orientation)

90 case FOPARALLEL: adj_face_ori = FOPARALLEL; break;

94 }
95 // clang-format on
96 } // if face owned

100 const auto ass_face = face.GetNeighborAssociatedFace(grid_);
101 const auto& adj_face = adj_cell.faces_[ass_face];
102
103 auto& cur_face_ori = cell_face_orientations_[cell.local_id_][f];
104
105 const double adj_mu = omega.Dot(adj_face.normal_);
106 if (adj_mu > tolerance) orientation = FOOUTGOING;
107 else if (adj_mu < tolerance)
108 orientation = FOINCOMING;
109
110 switch (orientation)
111 {
112 case FOPARALLEL:
113 cur_face_ori = FOPARALLEL;
114 break;
115 case FOINCOMING:
116 cur_face_ori = FOOUTGOING;
117 break;
118 case FOOUTGOING:
119 cur_face_ori = FOINCOMING;
120 break;
121 }
122 } // if not face owned locally at all
123
124 ++f;
125 } // for face
126 }
127
128 // Make directed connections
129 for (auto& cell : grid_.local_cells)
130 {
131 const uint64_t c = cell.local_id_;
132 size_t f = 0;
133 for (auto& face : cell.faces_)
134 {
135 const double mu = omega.Dot(face.normal_);
136 // If outgoing determine if it is to a local cell
137 if (cell_face_orientations_[cell.local_id_][f] == FOOUTGOING)
138 {
139 // If it is a cell and not bndry
140 if (face.has_neighbor_)
141 {
142 // If it is in the current location
143 if (face.IsNeighborLocal(grid_))
144 {
145 double weight = mu * face.ComputeFaceArea(grid_);
146 cell_successors[c].insert(std::make_pair(face.GetNeighborLocalID(grid_), weight));
147 }
148 else

150 }
151 }
152 // If not outgoing determine what it is dependent on
153 else
154 {
155 // if it is a cell and not bndry
156 if (face.has_neighbor_ and not face.IsNeighborLocal(grid_))

158 }
