119std::shared_ptr<SpatialDiscretization>
120FieldFunctionGridBased::MakeSpatialDiscretization(const InputParameters& params)
121{
122 const auto& user_params = params.ParametersAtAssignment();
123 const auto& grid_ptr = GetCurrentHandler().GetGrid();
124 const auto sdm_type = params.GetParamValue<std::string>("sdm_type");
125
126 typedef FiniteVolume FV;
127 typedef PieceWiseLinearContinuous PWLC;
128 typedef PieceWiseLinearDiscontinuous PWLD;
129 typedef LagrangeContinuous LagC;
130 typedef LagrangeDiscontinuous LagD;
131
132 if (sdm_type == "FV") return FV::New(*grid_ptr);
133
134 CoordinateSystemType cs_type = CoordinateSystemType::CARTESIAN;
135 std::string cs = "cartesian";
136 if (user_params.Has("coordinate_system"))
137 {
138 cs = params.GetParamValue<std::string>("coordinate_system");
139
140 if (cs == "cartesian") cs_type = CoordinateSystemType::CARTESIAN;
141 if (cs == "cylindrical") cs_type = CoordinateSystemType::CYLINDRICAL;
142 if (cs == "spherical") cs_type = CoordinateSystemType::SPHERICAL;
143 }
144
145 QuadratureOrder q_order = QuadratureOrder::SECOND;
146
147 if (user_params.Has("quadrature_order"))
148 {
149 const uint32_t max_order = static_cast<uint32_t>(QuadratureOrder::FORTYTHIRD);
150 const uint32_t q_order_int = params.GetParamValue<uint32_t>("quadrature_order");
151 ChiInvalidArgumentIf(q_order_int > max_order,
152 "Invalid quadrature point order " + std::to_string(q_order_int));
153 q_order = static_cast<QuadratureOrder>(q_order_int);
154 }
155 else // Defaulted
156 {
157 if (cs == "cartesian") q_order = QuadratureOrder::SECOND;
158 if (cs == "cylindrical") q_order = QuadratureOrder::THIRD;
159 if (cs == "spherical") q_order = QuadratureOrder::FOURTH;
160 }
161
162 if (sdm_type == "PWLC") return PWLC::New(*grid_ptr, q_order, cs_type);
163 else if (sdm_type == "PWLD")
164 return PWLD::New(*grid_ptr, q_order, cs_type);
165 else if (sdm_type == "LagrangeC")
166 return LagC::New(*grid_ptr, q_order, cs_type);
167 else if (sdm_type == "LagrangeD")
168 return LagD::New(*grid_ptr, q_order, cs_type);
169
170 // If not returned by now
171 ChiInvalidArgument("Unsupported sdm_type \"" + sdm_type + "\"");
172}

205void
206FieldFunctionGridBased::ExportMultipleToVTK(
207 const std::string& file_base_name,
208 const std::vector<std::shared_ptr<const FieldFunctionGridBased>>& ff_list)
209{
210 const std::string fname = "chi_physics::FieldFunction::ExportMultipleToVTK";
211 log.Log() << "Exporting field functions to VTK with file base \"" << file_base_name << "\"";
212
213 if (ff_list.empty())
214 throw std::logic_error(fname + ": Cannot be used with empty field-function"
215 " list");
216
217 // Setup master and check slaves
218 const auto& master_ff_ptr = ff_list.front();
219 const auto& master_ff = *master_ff_ptr;
220
221 for (const auto& ff_ptr : ff_list)
222 if (ff_ptr != master_ff_ptr)
223 if (&ff_ptr->sdm_->Grid() != &master_ff_ptr->sdm_->Grid())
224 throw std::logic_error(fname +
225 ": Cannot be used with field functions based on different grids.");
226
227 // Get grid
228 const auto& grid = master_ff.sdm_->Grid();
229
230 auto ugrid = PrepareVtkUnstructuredGrid(grid);
231
232 // Upload cell/point data
233 auto cell_data = ugrid->GetCellData();
234 auto point_data = ugrid->GetPointData();
235 for (const auto& ff_ptr : ff_list)
236 {
237 const auto field_vector = ff_ptr->GetGhostedFieldVector();
238
239 const auto& uk_man = ff_ptr->GetUnknownManager();
240 const auto& unknown = ff_ptr->Unknown();
241 const auto& sdm = ff_ptr->sdm_;
242 const size_t num_comps = unknown.NumComponents();
243
244 for (uint c = 0; c < num_comps; ++c)
245 {
246 std::string component_name = ff_ptr->TextName() + unknown.text_name_;
247 if (num_comps > 1) component_name += unknown.component_text_names_[c];
248
249 vtkNew<vtkDoubleArray> point_array;
250 vtkNew<vtkDoubleArray> cell_array;
251
252 point_array->SetName(component_name.c_str());
253 cell_array->SetName(component_name.c_str());
254
255 // Populate the array here
256 for (const auto& cell : grid.local_cells)
257 {
258 const size_t num_nodes = sdm->GetCellNumNodes(cell);
259
260 if (num_nodes == cell.vertex_ids_.size())
261 {
262 double node_average = 0.0;
263 for (int n = 0; n < num_nodes; ++n)
264 {
265 const int64_t nmap = sdm->MapDOFLocal(cell, n, uk_man, 0, c);
266
267 const double field_value = field_vector[nmap];
268
269 point_array->InsertNextValue(field_value);
270 node_average += field_value;
271 } // for node
272 node_average /= static_cast<double>(num_nodes);
273 cell_array->InsertNextValue(node_average);
274 }
275 else
276 {
277 double node_average = 0.0;
278 for (int n = 0; n < num_nodes; ++n)
279 {
280 const int64_t nmap = sdm->MapDOFLocal(cell, n, uk_man, 0, c);
281
282 const double field_value = field_vector[nmap];
283 node_average += field_value;
284 } // for node
285 node_average /= static_cast<double>(num_nodes);
286 cell_array->InsertNextValue(node_average);
287 for (int n = 0; n < cell.vertex_ids_.size(); ++n)
288 {
289 point_array->InsertNextValue(node_average);
290 } // for vertex
291 }
292
293 } // for cell
294
295 point_data->AddArray(point_array);
296 cell_data->AddArray(cell_array);
297 } // for component
298 } // for ff_ptr
299
300 WritePVTUFiles(ugrid, file_base_name);
301
302 log.Log() << "Done exporting field functions to VTK.";
303}

311std::vector<double>
312FieldFunctionGridBased::GetPointValue(const Vector3& point) const
313{
314 typedef const int64_t cint64_t;
315 const auto& uk_man = GetUnknownManager();
316 const size_t num_components = uk_man.GetTotalUnknownStructureSize();
317
318 size_t local_num_point_hits = 0;
319 std::vector<double> local_point_value(num_components, 0.0);
320
321 const auto& xyz_min = local_grid_bounding_box_.first;
322 const auto& xyz_max = local_grid_bounding_box_.second;
323
324 const double xmin = xyz_min.x;
325 const double ymin = xyz_min.y;
326 const double zmin = xyz_min.z;
327
328 const double xmax = xyz_max.x;
329 const double ymax = xyz_max.y;
330 const double zmax = xyz_max.z;
331
332 const auto& field_vector = *ghosted_field_vector_;
333
334 if (point.x >= xmin and point.x <= xmax and point.y >= ymin and point.y <= ymax and
335 point.z >= zmin and point.z <= zmax)
336 {
337 const auto& grid = sdm_->Grid();
338 for (const auto& cell : grid.local_cells)
339 {
340 if (grid.CheckPointInsideCell(cell, point))
341 {
342 const auto& cell_mapping = sdm_->GetCellMapping(cell);
343 std::vector<double> shape_values;
344 cell_mapping.ShapeValues(point, shape_values);
345
346 local_num_point_hits += 1;
347
348 const size_t num_nodes = cell_mapping.NumNodes();
349 for (size_t c = 0; c < num_components; ++c)
350 {
351 for (size_t j = 0; j < num_nodes; ++j)
352 {
353 cint64_t dof_map_j = sdm_->MapDOFLocal(cell, j, uk_man, 0, c);
354 const double dof_value_j = field_vector[dof_map_j];
355
356 local_point_value[c] += dof_value_j * shape_values[j];
357 } // for node i
358 } // for component c
359 } // if inside cell
360 } // for cell
361 } // if in bounding box
362
363 // Communicate number of point hits
364 size_t globl_num_point_hits;
365 mpi_comm.all_reduce(local_num_point_hits, globl_num_point_hits, mpi::op::sum<size_t>());
366
367 std::vector<double> globl_point_value(num_components, 0.0);
368 mpi_comm.all_reduce(
369 local_point_value.data(), 1, globl_point_value.data(), mpi::op::sum<double>());
370
371 Scale(globl_point_value, 1.0 / static_cast<double>(globl_num_point_hits));
372
373 return globl_point_value;
374}

376double
377FieldFunctionGridBased::Evaluate(const Cell& cell,
378 const Vector3& position,
379 unsigned int component) const
380{
381 const auto& field_vector = *ghosted_field_vector_;
382
383 typedef const int64_t cint64_t;
384 const auto& cell_mapping = sdm_->GetCellMapping(cell);
385
386 std::vector<double> shape_values;
387 cell_mapping.ShapeValues(position, shape_values);
388
389 double value = 0.0;
390 const size_t num_nodes = cell_mapping.NumNodes();
391 for (size_t j = 0; j < num_nodes; ++j)
392 {
393 cint64_t dof_map = sdm_->MapDOFLocal(cell, j, GetUnknownManager(), 0, component);
394
395 value += field_vector[dof_map] * shape_values[j];
396 }
397
398 return value;
399}
