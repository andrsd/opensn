38 for (const auto& ang : azimu_ang_)

42 for (const auto& ang : polar_ang_)
43 log.Log() << ang;

45
46 // Create angle pairs
47 map_directions_.clear();
48 for (unsigned int j = 0; j < Np; ++j)
49 map_directions_.emplace(j, std::vector<unsigned int>());
50
51 abscissae_.clear();
52 weights_.clear();

55 for (unsigned int i = 0; i < Na; ++i)
56 {
57 for (unsigned int j = 0; j < Np; ++j)
58 {
59 map_directions_[j].emplace_back(i * Np + j);
60

62
63 abscissae_.emplace_back(abscissa);
64

67 weight_sum += weight;
68
69 if (verbose)
70 {
71 char buf[200];
72 snprintf(buf,
73 200,
74 "Varphi=%.2f Theta=%.2f Weight=%.3e\n",
75 abscissa.phi * 180.0 / M_PI,

81 }

89 new_omega.y = sin(qpoint.theta) * sin(qpoint.phi);

104void
105ProductQuadrature::OptimizeForPolarSymmetry(const double normalization)
106{
107 std::vector<QuadraturePointPhiTheta> new_abscissae;
108 std::vector<double> new_weights;
109 std::vector<Vector3> new_omegas;
110 std::vector<double> new_polar_ang;
111 std::vector<double> new_azimu_ang;
112
113 const size_t num_pol = polar_ang_.size();
114 const size_t num_azi = azimu_ang_.size();
115
116 std::vector<unsigned int> new_polar_map;
117 for (size_t p = 0; p < num_pol; ++p)
118 if (polar_ang_[p] < M_PI_2)
119 {
120 new_polar_ang.push_back(polar_ang_[p]);
121 new_polar_map.push_back(p);
122 }
123 new_azimu_ang = azimu_ang_;
124
125 const size_t new_num_pol = new_polar_ang.size();
126 double weight_sum = 0.0;
127 for (size_t a = 0; a < num_azi; ++a)
128 for (size_t p = 0; p < new_num_pol; ++p)
129 {
130 const auto pmap = new_polar_map[p];
131 const auto dmap = GetAngleNum(pmap, a);
132 new_weights.push_back(weights_[dmap]);
133 weight_sum += weights_[dmap];
134 }
135
136 if (normalization > 0.0)
137 for (double& w : new_weights)
138 w *= normalization / weight_sum;
139
140 AssembleCosines(new_azimu_ang, new_polar_ang, new_weights, false);
141 polar_ang_ = new_polar_ang;
142 azimu_ang_ = new_azimu_ang;
143}

165AngularQuadratureProdGLL::AngularQuadratureProdGLL(int Na, int Np, bool verbose)
166{
167 QuadratureGaussLegendre gl_polar(Np * 2);
168 QuadratureGaussLegendre gl_azimu(Na * 4);
169
170 // Create azimuthal angles
171 azimu_ang_.clear();
172 for (unsigned int i = 0; i < (Na * 4); ++i)
173 azimu_ang_.emplace_back(M_PI * gl_azimu.qpoints_[i][0] + M_PI);
174
175 // Create polar angles
176 polar_ang_.clear();
177 for (unsigned int j = 0; j < (Np * 2); ++j)
178 polar_ang_.emplace_back(M_PI - acos(gl_polar.qpoints_[j][0]));
179
180 // Create combined weights
181 std::vector<double> weights;
182 for (unsigned int i = 0; i < azimu_ang_.size(); ++i)
183 for (unsigned int j = 0; j < polar_ang_.size(); ++j)
184 weights.emplace_back(M_PI * gl_azimu.weights_[i] * gl_polar.weights_[j]);
185
186 // Initialize
187 AssembleCosines(azimu_ang_, polar_ang_, weights, verbose);
