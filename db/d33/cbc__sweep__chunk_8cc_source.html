116void
117CBC_SweepChunk::Sweep(AngleSet& angle_set)
118{
119 using FaceOrientation = FaceOrientation;
120 const auto& face_orientations = angle_set.GetSPDS().CellFaceOrientations()[cell_local_id_];
121 const auto& sigma_t = xs_.at(cell_->material_id_)->SigmaTotal();
122
123 // as = angle set
124 // ss = subset
125 const std::vector<size_t>& as_angle_indices = angle_set.GetAngleIndices();
126 const size_t as_num_angles = as_angle_indices.size();
127 for (size_t as_ss_idx = 0; as_ss_idx < as_num_angles; ++as_ss_idx)
128 {
129 direction_num_ = as_angle_indices[as_ss_idx];
130 omega_ = groupset_.quadrature_->omegas_[direction_num_];
131 direction_qweight_ = groupset_.quadrature_->weights_[direction_num_];
132
133 sweep_dependency_interface_.angle_set_index_ = as_ss_idx;
134 sweep_dependency_interface_.angle_num_ = direction_num_;
135
136 // Reset right-handside
137 for (int gsg = 0; gsg < gs_ss_size_; ++gsg)
138 b_[gsg].assign(cell_num_nodes_, 0.0);
139
140 ExecuteKernels(direction_data_callbacks_and_kernels_);
141
142 // Update face orientations
143 face_mu_values_.assign(cell_num_faces_, 0.0);
144 for (int f = 0; f < cell_num_faces_; ++f)
145 face_mu_values_[f] = omega_.Dot(cell_->faces_[f].normal_);
146
147 // Surface integrals
148 for (int f = 0; f < cell_num_faces_; ++f)
149 {
150 const auto& face = cell_->faces_[f];
151
152 if (face_orientations[f] != FaceOrientation::INCOMING) continue;
153
154 const bool local = cell_transport_view_->IsFaceLocal(f);
155 const bool boundary = not face.has_neighbor_;
156
157 sweep_dependency_interface_.SetupIncomingFace(
158 f, cell_mapping_->NumFaceNodes(f), face.neighbor_id_, local, boundary);
159
160 // IntSf_mu_psi_Mij_dA
161 ExecuteKernels(surface_integral_kernels_);
162 } // for f
163
164 // Looping over groups, assembling mass terms
165 for (int gsg = 0; gsg < gs_ss_size_; ++gsg)
166 {
167 g_ = gs_gi_ + gsg;
168 gsg_ = gsg;
169 sigma_tg_ = sigma_t[g_];
170
171 ExecuteKernels(mass_term_kernels_);
172
173 // Solve system
174 GaussElimination(Atemp_, b_[gsg], static_cast<int>(cell_num_nodes_));
175 }
176
177 // Flux updates
178 ExecuteKernels(flux_update_kernels_);
179
180 // Perform outgoing surface operations
181 for (int f = 0; f < cell_num_faces_; ++f)
182 {
183 if (face_orientations[f] != FaceOrientation::OUTGOING) continue;
184
185 // Set flags and counters
186 const auto& face = cell_->faces_[f];
187 const bool local = cell_transport_view_->IsFaceLocal(f);
188 const bool boundary = not face.has_neighbor_;
189 const int locality = cell_transport_view_->FaceLocality(f);
190
191 sweep_dependency_interface_.SetupOutgoingFace(
192 f, cell_mapping_->NumFaceNodes(f), face.neighbor_id_, local, boundary, locality);
193
194 OutgoingSurfaceOperations();
195 } // for face
196
197 ExecuteKernels(post_cell_dir_sweep_callbacks_);
198 } // for n
199}

264CBC_SweepDependencyInterface::GetUpwindPsi(int face_node_local_idx) const
265{
266 const double* psi;
267 if (on_local_face_)
268 {
269 const unsigned int adj_cell_node = face_nodal_mapping_->cell_node_mapping_[face_node_local_idx];
270

274 else if (not on_boundary_)

277

279 }
280 else
281 psi = angle_set_->PsiBndry(neighbor_id_,
282 angle_num_,
283 cell_local_id_,
284 current_face_idx_,
285 face_node_local_idx,
286 gs_gi_,

288 surface_source_active_);

293double*
294CBC_SweepDependencyInterface::GetDownwindPsi(int face_node_local_idx) const
295{
296 double* psi = nullptr;
297
298 if (on_local_face_) psi = nullptr; // We don't write local face outputs
299 else if (not on_boundary_)
300 {
301 const size_t addr_offset =
302 face_node_local_idx * group_angle_stride_ + angle_set_index_ * group_stride_;
303
304 psi = &(*psi_dnwnd_data_)[addr_offset];
305 }
306 else if (is_reflecting_bndry_)
307 psi = angle_set_->ReflectingPsiOutBoundBndry(neighbor_id_,
308 angle_num_,
309 cell_local_id_,
310 current_face_idx_,
311 face_node_local_idx,
312 gs_ss_begin_);
313
314 return psi;
