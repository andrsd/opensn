60Solver::InitTimeStepper(const InputParameters& params)
61{
62 const auto& user_params = params.ParametersAtAssignment();
63
64 if (user_params.Has("timestepper"))
65 {
66 auto stepper = GetStackItemPtrAsType<TimeStepper>(
67 object_stack, params.GetParamValue<size_t>("timestepper"), __FUNCTION__);
68
69 stepper->SetTimeStepSize(params.GetParamValue<double>("dt"));
70 stepper->SetTime(params.GetParamValue<double>("time"));
71 stepper->SetStartTime(params.GetParamValue<double>("start_time"));
72 stepper->SetEndTime(params.GetParamValue<double>("end_time"));
73 stepper->SetMaxTimeSteps(params.GetParamValue<int>("max_time_steps"));
74
75 return stepper;
76 }
77 else
78 {
79 auto& factory = ObjectFactory::GetInstance();
80
81 const std::string obj_type = "physics::ConstantTimeStepper";
82 auto valid_params = factory.GetRegisteredObjectParameters(obj_type);
83 ParameterBlock custom_params;
84
85 if (params.NumParameters() != 0)
86 {
87 custom_params.AddParameter(params.GetParam("dt"));
88 custom_params.AddParameter(params.GetParam("time"));
89 custom_params.AddParameter(params.GetParam("start_time"));
90 custom_params.AddParameter(params.GetParam("end_time"));
91 custom_params.AddParameter(params.GetParam("max_time_steps"));
92 }
93
94 valid_params.AssignParameters(custom_params);
95
96 auto stepper = std::make_shared<ConstantTimeStepper>(valid_params);
97 object_stack.push_back(stepper);
98 stepper->SetStackID(object_stack.size() - 1);
