48void
49UnpartitionedMesh::ComputeCentroidsAndCheckQuality()
50{
51 const Vector3 khat(0.0, 0.0, 1.0);
52
53 log.Log0Verbose1() << "Computing cell-centroids.";
54 for (auto cell : raw_cells_)
55 {
56 cell->centroid = Vertex(0.0, 0.0, 0.0);
57 for (auto vid : cell->vertex_ids)
58 cell->centroid += vertices_[vid];
59
60 cell->centroid = cell->centroid / static_cast<double>(cell->vertex_ids.size());
61 }
62 log.Log0Verbose1() << "Done computing cell-centroids.";
63
64 log.Log0Verbose1() << "Checking cell-center-to-face orientations";
65 size_t num_negative_volume_elements = 0;
66 for (auto cell : raw_cells_)
67 {
68 if (cell->type == CellType::POLYGON)
69 {
70 // Form triangles
71 size_t num_verts = cell->vertex_ids.size();
72 for (size_t v = 0; v < num_verts; ++v)
73 {
74 size_t vp1 = (v < (num_verts - 1)) ? v + 1 : 0;
75
76 const auto& v0 = vertices_[cell->vertex_ids[v]];
77 const auto& v1 = vertices_[cell->vertex_ids[vp1]];
78
79 auto E01 = v1 - v0;
80 auto n = E01.Cross(khat).Normalized();
81
82 if (n.Dot(v0 - cell->centroid) < 0.0) ++num_negative_volume_elements;
83 } // for v
84 }
85 else if (cell->type == CellType::POLYHEDRON)
86 {
87 for (auto& face : cell->faces)
88 {
89 if (face.vertex_ids.size() < 2)
90 throw std::logic_error(std::string(__PRETTY_FUNCTION__) +
91 ": cell-center-to-face check encountered face "
92 "with less than 2 vertices on a face, making "
93 "normal computation impossible.");
94
95 // Compute centroid
96 Vector3 face_centroid;
97 for (uint64_t vid : face.vertex_ids)
98 face_centroid += vertices_[vid];
99 face_centroid /= static_cast<double>(face.vertex_ids.size());
100
101 // Form tets for each face edge
102 size_t num_face_verts = face.vertex_ids.size();
103 for (size_t fv = 0; fv < face.vertex_ids.size(); ++fv)
104 {
105 size_t fvp1 = (fv < (num_face_verts - 1)) ? fv + 1 : 0;
106
107 const auto& fv1 = vertices_[face.vertex_ids[fv]];
108 const auto& fv2 = vertices_[face.vertex_ids[fvp1]];
109
110 auto E0 = fv1 - face_centroid;
111 auto E1 = fv2 - face_centroid;
112 auto n = E0.Cross(E1).Normalized();
113
114 if (n.Dot(fv1 - cell->centroid) < 0.0) ++num_negative_volume_elements;
115 }
116 } // for face
117 } // if polyhedron
118 } // for cell in raw_cells
119
120 log.Log0Verbose1() << "Checking face sizes";
121 size_t cell_id = 0;
122 for (auto cell : raw_cells_)
123 {
124 if (cell->type == CellType::POLYGON)
125 {
126 size_t f = 0;
127 for (const auto& face : cell->faces)
128 {
129 const auto& v0 = vertices_.at(face.vertex_ids[0]);
130 const auto& v1 = vertices_.at(face.vertex_ids[1]);
131 ChiLogicalErrorIf((v1 - v0).Norm() < 1.0e-12,
132 "Cell " + std::to_string(cell_id) +
133 " (centroid=" + cell->centroid.PrintStr() + ") face " +
134 std::to_string(f) + ": Face has length < 1.0e-12.");
135 ++f;
136 }
137 } // if polygon
138 if (cell->type == CellType::POLYHEDRON)
139 {
140 size_t f = 0;
141 for (const auto& face : cell->faces)
142 {
143 size_t num_face_verts = face.vertex_ids.size();
144 for (size_t s = 0; s < face.vertex_ids.size(); ++s)
145 {
146 size_t fvp1 = (s < (num_face_verts - 1)) ? s + 1 : 0;
147
148 const auto& v0 = vertices_.at(face.vertex_ids[s]);
149 const auto& v1 = vertices_.at(face.vertex_ids[fvp1]);
150
151 ChiLogicalErrorIf((v1 - v0).Norm() < 1.0e-12,
152 "Cell " + std::to_string(cell_id) + " (centroid=" +
153 cell->centroid.PrintStr() + ") face " + std::to_string(f) + " side " +
154 std::to_string(s) + ": Side has length < 1.0e-12.");
155 }
156
157 ++f;
158 }
159 } // if polyhedron
160 ++cell_id;
161 } // for cell in raw_cells
162
163 if (num_negative_volume_elements > 0)
164 log.LogAllWarning() << "Cell quality checks detected " << num_negative_volume_elements
165 << " negative volume sub-elements (sub-triangle or sub-tetrahedron)."
166 << " This issue could result in incorrect quantities"
167 << " under some circumstances.";
168 log.Log() << "Done checking cell-center-to-face orientations";
169}

196void
197UnpartitionedMesh::BuildMeshConnectivity()
198{
199 const size_t num_raw_cells = raw_cells_.size();
200 const size_t num_raw_vertices = vertices_.size();
201
202 // Reset all cell neighbors
203 int num_bndry_faces = 0;
204 for (auto& cell : raw_cells_)
205 for (auto& face : cell->faces)
206 if (not face.has_neighbor) ++num_bndry_faces;
207
208 log.Log0Verbose1() << program_timer.GetTimeString()
209 << " Number of unconnected faces "
210 "before connectivity: "
211 << num_bndry_faces;
212
213 log.Log() << program_timer.GetTimeString() << " Establishing cell connectivity.";
214
215 // Establish internal connectivity
216 // Populate vertex subscriptions to internal cells
217 vertex_cell_subscriptions_.resize(num_raw_vertices);
218 {
219 uint64_t cur_cell_id = 0;
220 for (const auto& cell : raw_cells_)
221 {
222 for (auto vid : cell->vertex_ids)
223 vertex_cell_subscriptions_.at(vid).insert(cur_cell_id);
224 ++cur_cell_id;
225 }
226 }
227
228 log.Log() << program_timer.GetTimeString() << " Vertex cell subscriptions complete.";
229
230 // Process raw cells
231 {
232 uint64_t aux_counter = 0;
233 uint64_t cur_cell_id = 0;
234 for (auto& cell : raw_cells_)
235 {
236 for (auto& cur_cell_face : cell->faces)
237 {
238 if (cur_cell_face.has_neighbor) { continue; }
239 const std::set<uint64_t> cfvids(cur_cell_face.vertex_ids.begin(),
240 cur_cell_face.vertex_ids.end());
241
242 std::set<size_t> cells_to_search;
243 for (uint64_t vid : cfvids)
244 for (uint64_t cell_id : vertex_cell_subscriptions_.at(vid))
245 if (cell_id != cur_cell_id) cells_to_search.insert(cell_id);
246
247 for (uint64_t adj_cell_id : cells_to_search)
248 {
249 auto adj_cell = raw_cells_.at(adj_cell_id);
250
251 for (auto& adj_cell_face : adj_cell->faces)
252 {
253 if (adj_cell_face.has_neighbor) { continue; }
254 const std::set<uint64_t> afvids(adj_cell_face.vertex_ids.begin(),
255 adj_cell_face.vertex_ids.end());
256
257 if (cfvids == afvids)
258 {
259 cur_cell_face.neighbor = adj_cell_id;
260 adj_cell_face.neighbor = cur_cell_id;
261
262 cur_cell_face.has_neighbor = true;
263 adj_cell_face.has_neighbor = true;
264
265 goto face_neighbor_found;
266 }
267 } // for adjacent cell face
268 }
269 face_neighbor_found:;
270 } // for face
271
272 ++cur_cell_id;
273 const double fraction_complete =
274 static_cast<double>(cur_cell_id) / static_cast<double>(num_raw_cells);
275 if (fraction_complete >= static_cast<double>(aux_counter + 1) * 0.1)
276 {
277 log.Log() << program_timer.GetTimeString() << " Surpassing cell " << cur_cell_id << " of "
278 << num_raw_cells << " (" << (aux_counter + 1) * 10 << "%)";
279 ++aux_counter;
280 }
281 } // for cell
282 }
283
284 log.Log() << program_timer.GetTimeString() << " Establishing cell boundary connectivity.";
285
286 // Establish boundary connectivity
287 // Make list of internal cells on the boundary
288 std::vector<LightWeightCell*> internal_cells_on_boundary;
289 for (auto& cell : raw_cells_)
290 {
291 bool cell_on_boundary = false;
292 for (auto& face : cell->faces)
293 if (not face.has_neighbor)
294 {
295 cell_on_boundary = true;
296 break;
297 }
298
299 if (cell_on_boundary) internal_cells_on_boundary.push_back(cell);
300 }
301
302 // Populate vertex subscriptions to boundary cells
303 std::vector<std::set<uint64_t>> vertex_bndry_cell_subscriptions(vertices_.size());
304 {
305 uint64_t cur_cell_id = 0;
306 for (auto& cell : raw_boundary_cells_)
307 {
308 for (auto vid : cell->vertex_ids)
309 vertex_bndry_cell_subscriptions.at(vid).insert(cur_cell_id);
310 ++cur_cell_id;
311 }
312 }
313
314 // Process boundary cells
315 for (auto& cell : internal_cells_on_boundary)
316 for (auto& face : cell->faces)
317 {
318 if (face.has_neighbor) continue;
319 std::set<uint64_t> cfvids(face.vertex_ids.begin(), face.vertex_ids.end());
320
321 std::set<size_t> cells_to_search;
322 for (uint64_t vid : face.vertex_ids)
323 for (uint64_t cell_id : vertex_bndry_cell_subscriptions[vid])
324 cells_to_search.insert(cell_id);
325
326 for (uint64_t adj_cell_id : cells_to_search)
327 {
328 auto& adj_cell = raw_boundary_cells_[adj_cell_id];
329
330 std::set<uint64_t> afvids(adj_cell->vertex_ids.begin(), adj_cell->vertex_ids.end());
331
332 if (cfvids == afvids)
333 {
334 face.neighbor = adj_cell->material_id;
335 break;
336 }
337 } // for adj_cell_id
338 } // for face
339
340 num_bndry_faces = 0;
341 for (auto cell : raw_cells_)
342 for (auto& face : cell->faces)
343 if (not face.has_neighbor) ++num_bndry_faces;
344
345 log.Log0Verbose1() << program_timer.GetTimeString()
346 << " Number of boundary faces "
347 "after connectivity: "
348 << num_bndry_faces;
349
350 log.Log() << program_timer.GetTimeString() << " Done establishing cell connectivity.";
351}

353UnpartitionedMesh::LightWeightCell*
354UnpartitionedMesh::CreateCellFromVTKPolyhedron(vtkCell* vtk_cell)
355{
356 const std::string fname = "UnpartitionedMesh::CreateCellFromVTKPolyhedron";
357
358 CellType sub_type;
359 switch (vtk_cell->GetCellType())
360 {
361 case VTK_HEXAGONAL_PRISM:
362 case VTK_PENTAGONAL_PRISM:
363 case VTK_POLYHEDRON:
364 sub_type = CellType::POLYHEDRON;
365 break;
366 case VTK_PYRAMID:
367 sub_type = CellType::PYRAMID;
368 break;
369 case VTK_WEDGE:
370 sub_type = CellType::WEDGE;
371 break;
372 case VTK_HEXAHEDRON:
373 case VTK_VOXEL:
374 sub_type = CellType::HEXAHEDRON;
375 break;
376 case VTK_TETRA:
377 sub_type = CellType::TETRAHEDRON;
378 break;
379 default:
380 throw std::logic_error(fname + ": Unsupported 3D cell type encountered.");
381 }
382 auto polyh_cell = new LightWeightCell(CellType::POLYHEDRON, sub_type);
383
384 auto num_cpoints = vtk_cell->GetNumberOfPoints();
385 auto num_cfaces = vtk_cell->GetNumberOfFaces();
386
387 polyh_cell->vertex_ids.reserve(num_cpoints);
388 auto point_ids = vtk_cell->GetPointIds();
389 for (int p = 0; p < num_cpoints; ++p)
390 {
391 uint64_t point_id = point_ids->GetId(p);
392 polyh_cell->vertex_ids.push_back(point_id);
393 } // for p
394
395 switch (sub_type)
396 {
397 // The cell vertex ids in VTK is the same as in ChiTech so we don't
398 // need to remap the vertices. We do however need to remap the faces.
399 case CellType::HEXAHEDRON:
400 {
401 std::vector<std::vector<uint64_t>> face_vids = {
402 {1, 2, 6, 5}, {3, 0, 4, 7}, {2, 3, 7, 6}, {0, 1, 5, 4}, {4, 5, 6, 7}, {3, 2, 1, 0}};
403 for (int f = 0; f < 6; ++f)
404 {
405 LightWeightFace face;
406
407 face.vertex_ids.reserve(4);
408 for (int p = 0; p < 4; ++p)
409 face.vertex_ids.push_back(polyh_cell->vertex_ids[face_vids[f][p]]);
410
411 polyh_cell->faces.push_back(face);
412 }
413 break;
414 }
415 // For wedges we need to remap cell vertices and faces.
416 case CellType::WEDGE:
417 {
418 std::vector<uint64_t> remapping = {0, 2, 1, 3, 5, 4};
419 std::vector<uint64_t> remapped(6, 0);
420 for (int i = 0; i < 6; ++i)
421 remapped[i] = polyh_cell->vertex_ids[remapping[i]];
422 polyh_cell->vertex_ids = remapped;
423
424 std::vector<std::vector<uint64_t>> face_vids = {
425 {0, 1, 4, 3}, {1, 2, 5, 4}, {2, 0, 3, 5}, {3, 4, 5}, {0, 2, 1}};
426 for (int f = 0; f < 6; ++f)
427 {
428 LightWeightFace face;
429
430 face.vertex_ids.reserve(4);
431 for (int p = 0; p < face_vids[f].size(); ++p)
432 face.vertex_ids.push_back(polyh_cell->vertex_ids[face_vids[f][p]]);
433
434 polyh_cell->faces.push_back(face);
435 }
436 break;
437 }
438 // We dont need cell vertex id remapping but we need to impose our own
439 // face orientation
440 case CellType::TETRAHEDRON:
441 {
442 std::vector<std::vector<uint64_t>> face_vids = {{0, 2, 1}, {0, 1, 3}, {0, 3, 2}, {3, 1, 2}};
443 for (int f = 0; f < 4; ++f)
444 {
445 LightWeightFace face;
446
447 face.vertex_ids.reserve(3);
448 for (int p = 0; p < 3; ++p)
449 face.vertex_ids.push_back(polyh_cell->vertex_ids[face_vids[f][p]]);
450
451 polyh_cell->faces.push_back(face);
452 }
453 break;
454 }
455 default:
456 {
457 polyh_cell->faces.reserve(num_cfaces);
458 for (int f = 0; f < num_cfaces; ++f)
459 {
460 LightWeightFace face;
461 auto vtk_face = vtk_cell->GetFace(f);
462 auto num_face_points = vtk_face->GetNumberOfPoints();
463
464 face.vertex_ids.reserve(num_face_points);
465 auto face_point_ids = vtk_face->GetPointIds();
466 for (int p = 0; p < num_face_points; ++p)
467 {
468 uint64_t point_id = face_point_ids->GetId(p);
469 face.vertex_ids.push_back(point_id);
470 }
471
472 polyh_cell->faces.push_back(face);
473 }
474 break;
475 } // default
476 } // switch on sub_type
477
478 return polyh_cell;
479}

598void
599UnpartitionedMesh::CopyUGridCellsAndPoints(vtkUnstructuredGrid& ugrid,
600 const double scale,
601 int dimension_to_copy)
602{
603 const std::string fname = "UnpartitionedMesh::CopyUGridCellsAndPoints";
604
605 const vtkIdType total_cell_count = ugrid.GetNumberOfCells();
606 const vtkIdType total_point_count = ugrid.GetNumberOfPoints();
607
608 bool has_cell_gids = ugrid.GetCellData()->GetGlobalIds();
609 bool has_pnts_gids = ugrid.GetPointData()->GetGlobalIds();
610 bool has_global_ids = has_cell_gids and has_pnts_gids;
611
612 const auto& block_id_array_name = mesh_options_.material_id_fieldname;
613
614 if (not ugrid.GetCellData()->GetArray(block_id_array_name.c_str()))
615 throw std::logic_error(fname + ": grid has no \"" + block_id_array_name + "\" array.");
616
617 auto block_id_array =
618 vtkIntArray::SafeDownCast(ugrid.GetCellData()->GetArray(block_id_array_name.c_str()));
619
620 ChiLogicalErrorIf(not block_id_array, "Failed to cast BlockID array to vtkInt");
621
622 if (has_global_ids)
623 {
624 std::vector<LightWeightCell*> cells(total_cell_count, nullptr);
625 std::vector<Vector3*> vertices(total_point_count, nullptr);
626
627 auto cell_gids_ptr = ugrid.GetCellData()->GetGlobalIds();
628 auto pnts_gids_ptr = ugrid.GetPointData()->GetGlobalIds();
629
630 auto cell_gids = vtkIdTypeArray::SafeDownCast(cell_gids_ptr);
631 auto pnts_gids = vtkIdTypeArray::SafeDownCast(pnts_gids_ptr);
632
633 // Determine id offset
634 // We do this because some mesh formats (like ExodusII)
635 // are indexed with a 1 base instead of 0
636 int cid_offset = 0, pid_offset = 0;
637 {
638 vtkIdType min_cid = total_point_count; // Minimum cell-id
639 vtkIdType min_pid = total_point_count; // Minimum point-id
640
641 for (vtkIdType c = 0; c < total_cell_count; ++c)
642 min_cid = std::min(min_cid, cell_gids->GetValue(c));
643
644 for (vtkIdType p = 0; p < total_point_count; ++p)
645 min_pid = std::min(min_pid, pnts_gids->GetValue(p));
646
647 cid_offset -= static_cast<int>(min_cid);
648 pid_offset -= static_cast<int>(min_pid);
649 } // build offset
650
651 // Build node map
652 std::vector<vtkIdType> node_map(total_point_count, 0);
653 for (vtkIdType p = 0; p < total_point_count; ++p)
654 node_map[p] = pnts_gids->GetValue(p) + pid_offset;
655
656 // Load cells
657 for (vtkIdType c = 0; c < total_cell_count; ++c)
658 {
659 auto vtk_cell = ugrid.GetCell(static_cast<vtkIdType>(c));
660 auto vtk_celldim = vtk_cell->GetCellDimension();
661 const vtkIdType cell_gid = cell_gids->GetValue(c) + cid_offset;
662
663 if (vtk_celldim != dimension_to_copy) continue;
664
665 LightWeightCell* raw_cell;
666 if (vtk_celldim == 3) raw_cell = CreateCellFromVTKPolyhedron(vtk_cell);
667 else if (vtk_celldim == 2)
668 raw_cell = CreateCellFromVTKPolygon(vtk_cell);
669 else if (vtk_celldim == 1)
670 raw_cell = CreateCellFromVTKLine(vtk_cell);
671 else if (vtk_celldim == 0)
672 raw_cell = CreateCellFromVTKVertex(vtk_cell);
673 else
674 throw std::logic_error(fname + ": Unsupported cell dimension ." +
675 std::to_string(vtk_celldim));
676
677 // Map the cell vertex-ids
678 for (uint64_t& vid : raw_cell->vertex_ids)
679 vid = node_map[vid];
680
681 // Map face vertex-ids
682 for (auto& face : raw_cell->faces)
683 for (uint64_t& vid : face.vertex_ids)
684 vid = node_map[vid];
685
686 raw_cell->material_id = block_id_array->GetValue(c);
687
688 cells[cell_gid] = raw_cell;
689 } // for cell c
690
691 // Load points
692 for (vtkIdType p = 0; p < total_point_count; ++p)
693 {
694 auto point = ugrid.GetPoint(static_cast<vtkIdType>(p));
695 const vtkIdType point_gid = pnts_gids->GetValue(p) + pid_offset;
696
697 auto vertex = new Vector3(point[0], point[1], point[2]);
698
699 *vertex *= scale;
700
701 vertices.at(point_gid) = vertex;
702 } // for point p
703
704 // Check all cells assigned
705 for (vtkIdType c = 0; c < total_cell_count; ++c)
706 if (cells[c] == nullptr) throw std::logic_error(fname + ": Cell pointer not assigned ");
707
708 // Check all points assigned
709 for (vtkIdType p = 0; p < total_point_count; ++p)
710 if (vertices[p] == nullptr) throw std::logic_error(fname + ": Vertex pointer not assigned");
711
712 raw_cells_ = cells;
713 vertices_.reserve(total_point_count);
714 for (auto& vertex_ptr : vertices)
715 vertices_.push_back(*vertex_ptr);
716 } // If global-ids available
717 else
718 {
719 // Push cells
720 for (vtkIdType c = 0; c < total_cell_count; ++c)
721 {
722 auto vtk_cell = ugrid.GetCell(static_cast<vtkIdType>(c));
723 auto vtk_celldim = vtk_cell->GetCellDimension();
724
725 if (vtk_celldim != dimension_to_copy) continue;
726
727 if (vtk_celldim == 3) raw_cells_.push_back(CreateCellFromVTKPolyhedron(vtk_cell));
728 else if (vtk_celldim == 2)
729 raw_cells_.push_back(CreateCellFromVTKPolygon(vtk_cell));
730 else if (vtk_celldim == 1)
731 raw_cells_.push_back(CreateCellFromVTKLine(vtk_cell));
732 else if (vtk_celldim == 0)
733 raw_cells_.push_back(CreateCellFromVTKVertex(vtk_cell));
734 else
735 throw std::logic_error(fname + ": Unsupported cell dimension.");
736
737 raw_cells_.back()->material_id = block_id_array->GetValue(c);
738 } // for c
739
740 // Push points
741 for (size_t p = 0; p < total_point_count; ++p)
742 {
743 auto point = ugrid.GetPoint(static_cast<vtkIdType>(p));
744
745 Vector3 vertex(point[0], point[1], point[2]);
746
747 vertex *= scale;
748
749 vertices_.emplace_back(point[0], point[1], point[2]);
750 }
751 } // if no global-ids
752
753 // Determine bound box
754 for (size_t p = 0; p < total_point_count; ++p)
755 {
756 const auto& vertex = vertices_[p];
757 if (not bound_box_)
758 bound_box_ = std::shared_ptr<BoundBox>(
759 new BoundBox{vertex.x, vertex.x, vertex.y, vertex.y, vertex.z, vertex.z});
760
761 bound_box_->xmin = std::min(bound_box_->xmin, vertex.x);
762 bound_box_->xmax = std::max(bound_box_->xmax, vertex.x);
763 bound_box_->ymin = std::min(bound_box_->ymin, vertex.y);
764 bound_box_->ymax = std::max(bound_box_->ymax, vertex.y);
765 bound_box_->zmin = std::min(bound_box_->zmin, vertex.z);
766 bound_box_->zmax = std::max(bound_box_->zmax, vertex.z);
767 }
768
769 log.Log() << fname + ": Done";
770}

781void
782UnpartitionedMesh::SetBoundaryIDsFromBlocks(std::vector<vtkUGridPtrAndName>& bndry_grid_blocks)
783{
784 const double EPSILON = 1.0e-12;
785 // Build boundary faces
786 std::vector<LightWeightFace*> bndry_faces;
787 for (auto& cell_ptr : raw_cells_)
788 for (auto& face : cell_ptr->faces)
789 if (not face.has_neighbor) bndry_faces.push_back(&face);
790
791 log.Log() << "Number of boundary faces: " << bndry_faces.size();
792
793 // Build boundary vertex ids
794 std::set<uint64_t> bndry_vids_set;
795 for (const auto& face_ptr : bndry_faces)
796 for (const auto vid : face_ptr->vertex_ids)
797 bndry_vids_set.insert(vid);
798
799 // Process each boundary block
800 uint64_t bndry_id = 0;
801 for (const auto& ugrid_name : bndry_grid_blocks)
802 {
803 auto ugrid = ugrid_name.first;
804
805 mesh_options_.boundary_id_map[bndry_id] = ugrid_name.second;
806
807 // Build vertex map
808 bool mapping_failed = false;
809 std::vector<size_t> vertex_map(ugrid->GetNumberOfPoints(), 0);
810 for (size_t p = 0; p < ugrid->GetNumberOfPoints(); ++p)
811 {
812 Vector3 point;
813 ugrid->GetPoint(static_cast<vtkIdType>(p), &point.x);
814
815 bool map_found = false;
816 for (const auto vid : bndry_vids_set)
817 if ((point - vertices_[vid]).NormSquare() < EPSILON)
818 {
819 vertex_map[p] = vid;
820 map_found = true;
821 break;
822 }
823
824 if (not map_found)
825 {
826 log.Log0Warning() << "UnpartitionedMesh::"
827 "SetBoundaryIDsFromBlocks: Failed to map a vertex. " +
828 point.PrintStr() + " for boundary " + ugrid_name.second +
829 " therefore the boundary assignment was skipped.";
830 mapping_failed = true;
831 break;
832 }
833 } // for point in boundary block
834
835 if (mapping_failed) continue;
836
837 // Build vertex subscriptions
838 std::map<uint64_t, std::set<size_t>> vertex_face_subs;
839 for (size_t f = 0; f < bndry_faces.size(); ++f)
840 for (const auto vid : bndry_faces[f]->vertex_ids)
841 vertex_face_subs[vid].insert(f);
842
843 // Process each cell in bndry block
844 size_t num_faces_boundarified = 0;
845 auto& bndry_block = ugrid_name.first;
846 size_t num_bndry_block_cells = bndry_block->GetNumberOfCells();
847 for (size_t bc = 0; bc < num_bndry_block_cells; ++bc)
848 {
849 auto bndry_cell = bndry_block->GetCell(static_cast<vtkIdType>(bc));
850
851 // Build list of face candidates and vertex set
852 std::set<size_t> face_ids_short_list;
853 std::set<uint64_t> bndry_cell_id_set;
854 size_t num_points = bndry_cell->GetNumberOfPoints();
855 for (size_t p = 0; p < num_points; ++p)
856 {
857 auto point_id = bndry_cell->GetPointId(static_cast<int>(p));
858 bndry_cell_id_set.insert(vertex_map[point_id]);
859 for (size_t face_id : vertex_face_subs[vertex_map[point_id]])
860 face_ids_short_list.insert(face_id);
861 } // for point p
862
863 for (size_t face_id : face_ids_short_list)
864 {
865 auto& face = bndry_faces[face_id];
866 const auto& face_vids = face->vertex_ids;
867 std::set<uint64_t> face_id_set(face_vids.begin(), face_vids.end());
868
869 if (face_id_set == bndry_cell_id_set)
870 {
871 face->neighbor = bndry_id;
872 ++num_faces_boundarified;
873 }
874 } // for face_id
875 } // for boundary cell bc
876
877 log.Log() << "UnpartitionedMesh: assigned " << num_faces_boundarified << " to boundary id "
878 << bndry_id << " with name " << ugrid_name.second;
879
880 ++bndry_id;
881 } // for boundary_block
882}

1121void
1122UnpartitionedMesh::ReadFromWavefrontOBJ(const Options& options)
1123{
1124 const std::string fname = "UnpartitionedMesh::ReadFromWavefrontOBJ";
1125
1126 // Opening the file
1127 std::ifstream file;
1128 file.open(options.file_name);
1129 if (!file.is_open())
1130 {
1131 log.LogAllError() << "Failed to open file: " << options.file_name << " in call "
1132 << "to ImportFromOBJFile \n";
1133 Exit(EXIT_FAILURE);
1134 }
1135
1136 opensn::mpi_comm.barrier();
1137 log.Log() << "Making Unpartitioned mesh from wavefront file " << options.file_name;
1138
1139 typedef std::pair<uint64_t, uint64_t> Edge;
1140 struct BlockData
1141 {
1142 std::string name;
1143 std::vector<LightWeightCell*> cells;
1144 std::vector<Edge> edges;
1145 };
1146
1147 std::vector<BlockData> block_data;
1148 std::vector<Vertex> file_vertices;
1149
1150 // Reading every line
1151 std::string file_line;
1152 std::string delimiter = " ";
1153 int material_id = -1;
1154 while (std::getline(file, file_line))
1155 {
1156 // Get the first word
1157 size_t beg_of_word = file_line.find_first_not_of(delimiter);
1158 size_t end_of_word = file_line.find(delimiter, beg_of_word - beg_of_word);
1159 std::string first_word = file_line.substr(beg_of_word, end_of_word);
1160 std::string sub_word;
1161
1162 if (first_word == "o")
1163 {
1164 beg_of_word = file_line.find_first_not_of(delimiter, end_of_word);
1165 end_of_word = file_line.find(delimiter, beg_of_word);
1166 sub_word = file_line.substr(beg_of_word, end_of_word - beg_of_word);
1167
1168 std::string block_name = sub_word;
1169 block_data.push_back({block_name, {}});
1170 }
1171
1172 if (first_word == "usemtl")
1173 {
1174 log.Log0Verbose1() << "New material at cell count: " << block_data.back().cells.size();
1175 ++material_id;
1176 }
1177
1178 // Keyword "v" for Vertex
1179 if (first_word == "v")
1180 {
1181 Vertex newVertex;
1182 for (int k = 1; k <= 3; k++)
1183 {
1184 // Extract sub word
1185 beg_of_word = file_line.find_first_not_of(delimiter, end_of_word);
1186 end_of_word = file_line.find(delimiter, beg_of_word);
1187 sub_word = file_line.substr(beg_of_word, end_of_word - beg_of_word);
1188
1189 // Convert word to number
1190 try
1191 {
1192 double numValue = std::stod(sub_word);
1193
1194 if (k == 1) newVertex.x = numValue;
1195 else if (k == 2)
1196 newVertex.y = numValue;
1197 else if (k == 3)
1198 newVertex.z = numValue;
1199 }
1200
1201 // Catch conversion error
1202 catch (const std::invalid_argument& ia)
1203 {
1204 log.Log0Warning() << "Failed to convert vertex in line " << file_line << std::endl;
1205 }
1206
1207 // Stop word extraction on line end
1208 if (end_of_word == std::string::npos) { break; }
1209 }
1210 file_vertices.push_back(newVertex);
1211 } // if (first_word == "v")
1212
1213 // Keyword "f" for face
1214 if (first_word == "f")
1215 {
1216 size_t number_of_verts = std::count(file_line.begin(), file_line.end(), '/') / 2;
1217
1218 CellType sub_type = CellType::POLYGON;
1219 if (number_of_verts == 3) sub_type = CellType::TRIANGLE;
1220 else if (number_of_verts == 4)
1221 sub_type = CellType::QUADRILATERAL;
1222
1223 auto cell = new LightWeightCell(CellType::POLYGON, sub_type);
1224 cell->material_id = material_id;
1225
1226 // Populate vertex-ids
1227 for (size_t k = 1; k <= number_of_verts; k++)
1228 {
1229 // Extract sub word
1230 beg_of_word = file_line.find_first_not_of(delimiter, end_of_word);
1231 end_of_word = file_line.find(delimiter, beg_of_word);
1232 sub_word = file_line.substr(beg_of_word, end_of_word - beg_of_word);
1233
1234 // Extract locations of hiphens
1235 size_t first_dash = sub_word.find('/');
1236 size_t last_dash = sub_word.find_last_of('/');
1237
1238 // Extract the words ass. vertex and normal
1239 std::string vert_word = sub_word.substr(0, first_dash - 0);
1240 std::string norm_word = sub_word.substr(last_dash + 1, sub_word.length() - last_dash - 1);
1241
1242 // Convert word to number (Vertex)
1243 try
1244 {
1245 int numValue = std::stoi(vert_word);
1246 cell->vertex_ids.push_back(numValue - 1);
1247 }
1248 catch (const std::invalid_argument& ia)
1249 {
1250 log.Log0Warning() << "Failed converting work to number in line " << file_line
1251 << std::endl;
1252 }
1253
1254 // Stop word extraction on line end
1255 if (end_of_word == std::string::npos) { break; }
1256 }
1257
1258 // Build faces
1259 const size_t num_verts = cell->vertex_ids.size();
1260 for (uint64_t v = 0; v < num_verts; ++v)
1261 {
1262 LightWeightFace face;
1263
1264 face.vertex_ids.resize(2);
1265 face.vertex_ids[0] = cell->vertex_ids[v];
1266 face.vertex_ids[1] = (v < (num_verts - 1)) ? cell->vertex_ids[v + 1] : cell->vertex_ids[0];
1267
1268 cell->faces.push_back(std::move(face));
1269 } // for v
1270
1271 if (block_data.empty())
1272 throw std::logic_error(fname + ": Could not add cell to block-data. "
1273 "This normally indicates that the file does not have the "
1274 "\"o Object Name\" entry.");
1275
1276 block_data.back().cells.push_back(cell);
1277 } // if (first_word == "f")
1278
1279 // Keyword "l" for edge
1280 if (first_word == "l")
1281 {
1282 Edge edge;
1283 for (int k = 1; k <= 2; ++k)
1284 {
1285 // Extract sub word
1286 beg_of_word = file_line.find_first_not_of(delimiter, end_of_word);
1287 end_of_word = file_line.find(delimiter, beg_of_word);
1288 sub_word = file_line.substr(beg_of_word, end_of_word - beg_of_word);
1289
1290 // Convert word to number
1291 try
1292 {
1293 int vertex_id = std::stoi(sub_word);
1294 if (k == 1) edge.first = vertex_id - 1;
1295 if (k == 2) edge.second = vertex_id - 1;
1296 }
1297
1298 // Catch conversion error
1299 catch (const std::invalid_argument& ia)
1300 {
1301 log.Log0Warning() << "Failed to text to integer in line " << file_line << std::endl;
1302 }
1303 } // for k
1304
1305 if (block_data.empty())
1306 throw std::logic_error(fname + ": Could not add edge to block-data. "
1307 "This normally indicates that the file does not have the "
1308 "\"o Object Name\" entry.");
1309
1310 block_data.back().edges.push_back(edge);
1311 } // if (first_word == "l")
1312 }
1313 file.close();
1314 log.Log0Verbose0() << "Max material id: " << material_id;
1315
1316 // Error checks
1317 for (const auto& block : block_data)
1318 for (const auto& cell : block.cells)
1319 for (const auto vid : cell->vertex_ids)
1320 {
1321 ChiLogicalErrorIf(
1322 vid >= file_vertices.size(),
1323 "Cell vertex id " + std::to_string(vid) +
1324 " not in list of vertices read (size=" + std::to_string(file_vertices.size()) + ").");
1325 }
1326
1327 // Filter blocks
1328 std::vector<size_t> bndry_block_ids;
1329 size_t num_cell_blocks = 0;
1330 size_t main_block_id = 0;
1331 for (size_t block_id = 0; block_id < block_data.size(); ++block_id)
1332 {
1333 if (not block_data[block_id].edges.empty()) bndry_block_ids.push_back(block_id);
1334 if (not block_data[block_id].cells.empty())
1335 {
1336 ++num_cell_blocks;
1337 main_block_id = block_id;
1338 }
1339 } // for block_id
1340
1341 if (num_cell_blocks != 1)
1342 throw std::logic_error(fname +
1343 ": More than one cell-block has been read "
1344 "from the file. Only a single face-containing object is supported. "
1345 "If you exported this mesh from blender, be sure to export "
1346 "\"selection only\"");
1347
1348 // Process blocks
1349 std::vector<Vertex> cell_vertices;
1350 {
1351 // Initial map is straight
1352 std::vector<size_t> vertex_map;
1353 vertex_map.reserve(file_vertices.size());
1354 for (size_t m = 0; m < file_vertices.size(); ++m)
1355 vertex_map.push_back(m);
1356
1357 // Build set of cell vertices
1358 std::set<size_t> cell_vertex_id_set;
1359 for (const auto& cell_ptr : block_data.at(main_block_id).cells)
1360 for (size_t vid : cell_ptr->vertex_ids)
1361 cell_vertex_id_set.insert(vid);
1362
1363 // Make cell_vertices and edit map
1364 {
1365 size_t new_id = 0;
1366 for (size_t vid : cell_vertex_id_set)
1367 {
1368 cell_vertices.push_back(file_vertices[vid]);
1369 vertex_map[vid] = new_id;
1370 ++new_id;
1371 }
1372 }
1373
1374 // Build set of bndry vertices
1375 std::set<size_t> bndry_vertex_id_set;
1376 for (size_t block_id : bndry_block_ids)
1377 for (const auto& edge : block_data[block_id].edges)
1378 {
1379 bndry_vertex_id_set.insert(edge.first);
1380 bndry_vertex_id_set.insert(edge.second);
1381 }
1382
1383 // Find a match for each boundary vertex and
1384 // place it in the map
1385 for (size_t bvid : bndry_vertex_id_set)
1386 {
1387 const auto& bndry_vertex = file_vertices[bvid];
1388
1389 bool match_found = false;
1390 for (size_t cvid = 0; cvid < cell_vertices.size(); ++cvid)
1391 {
1392 const auto& cell_vertex = cell_vertices[cvid];
1393
1394 if ((bndry_vertex - cell_vertex).NormSquare() < 1.0e-12)
1395 {
1396 vertex_map[bvid] = cvid;
1397 match_found = true;
1398 break;
1399 }
1400 } // for cvid
1401 if (not match_found)
1402 throw std::logic_error(fname +
1403 ": Failed to map a boundary vertex to"
1404 "any cell vertex. Check that the edges are conformal with the "
1405 "object containing the faces.");
1406 } // for bvid
1407
1408 // Change cell and face vertex ids to cell vertex ids
1409 // using vertex map
1410 for (auto& cell_ptr : block_data.at(main_block_id).cells)
1411 {
1412 for (uint64_t& vid : cell_ptr->vertex_ids)
1413 vid = vertex_map[vid];
1414
1415 for (auto& face : cell_ptr->faces)
1416 for (uint64_t& vid : face.vertex_ids)
1417 vid = vertex_map[vid];
1418 }
1419
1420 // Change edge vertex ids to cell vertex ids using
1421 // the vertex map
1422 for (size_t block_id : bndry_block_ids)
1423 for (auto& edge : block_data[block_id].edges)
1424 {
1425 edge.first = static_cast<int>(vertex_map[edge.first]);
1426 edge.second = static_cast<int>(vertex_map[edge.second]);
1427 }
1428 }
1429 this->vertices_ = cell_vertices;
1430 this->raw_cells_ = block_data[main_block_id].cells;
1431
1432 // Always do this
1433 attributes_ = DIMENSION_2 | UNSTRUCTURED;
1434
1435 ComputeCentroidsAndCheckQuality();
1436 BuildMeshConnectivity();
1437
1438 // Set boundary ids
1439 if (bndry_block_ids.empty()) { mesh_options_.boundary_id_map[0] = "Default Boundary"; }
1440 else
1441 {
1442 std::vector<LightWeightFace*> bndry_faces;
1443 for (auto& cell_ptr : raw_cells_)
1444 for (auto& face : cell_ptr->faces)
1445 if (not face.has_neighbor) bndry_faces.push_back(&face);
1446
1447 size_t bndry_id = 0;
1448 for (size_t bid : bndry_block_ids)
1449 {
1450 const auto& bndry_edges = block_data[bid].edges;
1451
1452 size_t num_faces_boundarified = 0;
1453 for (const auto& edge : bndry_edges)
1454 {
1455 std::set<size_t> edge_vert_id_set({edge.first, edge.second});
1456
1457 for (auto& face_ptr : bndry_faces)
1458 {
1459 const auto& vert_ids = face_ptr->vertex_ids;
1460 std::set<size_t> face_vert_id_set(vert_ids.begin(), vert_ids.end());
1461
1462 if (face_vert_id_set == edge_vert_id_set)
1463 {
1464 face_ptr->neighbor = bndry_id;
1465 ++num_faces_boundarified;
1466 break;
1467 }
1468 } // for face
1469 } // for edge
1470
1471 log.Log() << "UnpartitionedMesh: assigned " << num_faces_boundarified
1472 << " faces to boundary id " << bndry_id << " with name " << block_data[bid].name;
1473
1474 mesh_options_.boundary_id_map[bndry_id] = block_data[bid].name;
1475
1476 ++bndry_id;
1477 } // for boundary block
1478 }
1479}

1481void
1482UnpartitionedMesh::ReadFromMsh(const Options& options)
1483{
1484 const std::string fname = "UnpartitionedMesh::ReadFromMsh";
1485
1486 // Opening the file
1487 std::ifstream file;
1488 file.open(options.file_name);
1489 if (!file.is_open())
1490 {
1491 log.LogAllError() << "Failed to open file: " << options.file_name << " in call "
1492 << "to ReadFromMsh \n";
1493 Exit(EXIT_FAILURE);
1494 }
1495
1496 log.Log() << "Making Unpartitioned mesh from msh format file " << options.file_name;
1497 opensn::mpi_comm.barrier();
1498
1499 // Declarations
1500 std::string file_line;
1501 std::istringstream iss;
1502 const std::string node_section_name = "$Nodes";
1503 const std::string elements_section_name = "$Elements";
1504 const std::string format_section_name = "$MeshFormat";
1505
1506 // Check the format of this input
1507 // Role file forward until "$MeshFormat" line is encountered.
1508 while (std::getline(file, file_line))
1509 if (format_section_name == file_line) break;
1510
1511 std::getline(file, file_line);
1512 iss = std::istringstream(file_line);
1513 double format;
1514 if (!(iss >> format)) throw std::logic_error(fname + ": Failed to read the file format.");
1515 else if (format != 2.2)
1516 throw std::logic_error(fname + ": Currently, only msh format 2.2 is supported.");
1517
1518 // Find section with node information and then read the nodes
1519 file.seekg(0);
1520 while (std::getline(file, file_line))
1521 if (node_section_name == file_line) break;
1522
1523 std::getline(file, file_line);
1524 iss = std::istringstream(file_line);
1525 int num_nodes;
1526 if (!(iss >> num_nodes))
1527 throw std::logic_error(fname + ": Failed while trying to read "
1528 "the number of nodes.");
1529
1530 vertices_.clear();
1531 vertices_.resize(num_nodes);
1532
1533 for (int n = 0; n < num_nodes; n++)
1534 {
1535 std::getline(file, file_line);
1536 iss = std::istringstream(file_line);
1537
1538 int vert_index;
1539 if (!(iss >> vert_index)) throw std::logic_error(fname + ": Failed to read vertex index.");
1540
1541 if (!(iss >> vertices_[vert_index - 1].x >> vertices_[vert_index - 1].y >>
1542 vertices_[vert_index - 1].z))
1543 throw std::logic_error(fname + ": Failed while reading the vertex "
1544 "coordinates.");
1545 }
1546
1547 // Define utility lambdas
1548 /**Lambda for reading nodes.*/
1549 auto ReadNodes = [&iss, &fname](int num_nodes)
1550 {
1551 std::vector<int> raw_nodes(num_nodes, 0);
1552 for (int i = 0; i < num_nodes; ++i)
1553 if (!(iss >> raw_nodes[i]))
1554 throw std::logic_error(fname + ": Failed when reading element "
1555 "node index.");
1556
1557 std::vector<uint64_t> nodes(num_nodes, 0);
1558 for (int i = 0; i < num_nodes; ++i)
1559 if ((raw_nodes[i] - 1) >= 0) nodes[i] = raw_nodes[i] - 1;
1560 return nodes;
1561 };
1562
1563 /**Lamda for checking if an element is 1D.*/
1564 auto IsElementType1D = [](int element_type)
1565 {
1566 if (element_type == 1) return true;
1567
1568 return false;
1569 };
1570
1571 /**Lamda for checking if an element is 2D.*/
1572 auto IsElementType2D = [](int element_type)
1573 {
1574 if (element_type == 2 or element_type == 3) return true;
1575
1576 return false;
1577 };
1578
1579 /**Lamda for checking if an element is 2D.*/
1580 auto IsElementType3D = [](int element_type)
1581 {
1582 if (element_type >= 4 and element_type <= 7) return true;
1583
1584 return false;
1585 };
1586
1587 /**Lambda for checking supported elements.*/
1588 auto IsElementSupported = [](int element_type)
1589 {
1590 if (element_type >= 1 and element_type <= 7) return true;
1591
1592 return false;
1593 };
1594
1595 /**Lambda giving the cell subtype, given the MSH cell type.*/
1596 auto CellTypeFromMSHTypeID = [](int element_type)
1597 {
1598 CellType cell_type = CellType::GHOST;
1599
1600 if (element_type == 1) cell_type = CellType::SLAB;
1601 else if (element_type == 2)
1602 cell_type = CellType::TRIANGLE;
1603 else if (element_type == 3)
1604 cell_type = CellType::QUADRILATERAL;
1605 else if (element_type == 4)
1606 cell_type = CellType::TETRAHEDRON;
1607 else if (element_type == 5)
1608 cell_type = CellType::HEXAHEDRON;
1609 else if (element_type == 6 or // Prism
1610 element_type == 7)
1611 cell_type = CellType::POLYHEDRON; // Pyramid
1612
1613 return cell_type;
1614 };
1615
1616 // Determine mesh type 2D/3D
1617 // Only 2D and 3D meshes are supported. If the mesh
1618 // is 1D then no elements will be read but the state
1619 // would still be safe.
1620 // This section will run through all the elements
1621 // looking for a 3D element. It will not process
1622 // any elements.
1623 bool mesh_is_2D_assumption = true;
1624 file.seekg(0);
1625 while (std::getline(file, file_line))
1626 if (elements_section_name == file_line) break;
1627
1628 std::getline(file, file_line);
1629 iss = std::istringstream(file_line);
1630 int num_elems;
1631 if (!(iss >> num_elems)) throw std::logic_error(fname + ": Failed to read number of elements.");
1632
1633 for (int n = 0; n < num_elems; n++)
1634 {
1635 int elem_type, num_tags, physical_reg, tag, element_index;
1636
1637 std::getline(file, file_line);
1638 iss = std::istringstream(file_line);
1639
1640 if (!(iss >> element_index >> elem_type >> num_tags))
1641 throw std::logic_error(fname + ": Failed while reading element index, "
1642 "element type, and number of tags.");
1643
1644 if (!(iss >> physical_reg))
1645 throw std::logic_error(fname + ": Failed while reading physical region.");
1646
1647 for (int i = 1; i < num_tags; i++)
1648 if (!(iss >> tag)) throw std::logic_error(fname + ": Failed when reading tags.");
1649
1650 if (IsElementType3D(elem_type))
1651 {
1652 mesh_is_2D_assumption = false;
1653 log.Log() << "Mesh identified as 3D.";
1654 break; // have the answer now leave loop
1655 }
1656
1657 if (elem_type == 15) // skip point type element
1658 continue;
1659
1660 if (not IsElementSupported(elem_type))
1661 throw std::logic_error(fname + ": Unsupported element encountered.");
1662 } // for n
1663
1664 // Return to the element listing section
1665 // Now we will actually read the elements.
1666 file.seekg(0);
1667 while (std::getline(file, file_line))
1668 if (elements_section_name == file_line) break;
1669
1670 std::getline(file, file_line);
1671 iss = std::istringstream(file_line);
1672 if (!(iss >> num_elems)) throw std::logic_error(fname + ": Failed to read number of elements.");
1673
1674 for (int n = 0; n < num_elems; n++)
1675 {
1676 int elem_type, num_tags, physical_reg, tag, element_index;
1677
1678 std::getline(file, file_line);
1679 iss = std::istringstream(file_line);
1680
1681 if (!(iss >> element_index >> elem_type >> num_tags))
1682 throw std::logic_error(fname + ": Failed while reading element index, "
1683 "element type, and number of tags.");
1684
1685 if (!(iss >> physical_reg))
1686 throw std::logic_error(fname + ": Failed while reading physical region.");
1687
1688 for (int i = 1; i < num_tags; i++)
1689 if (!(iss >> tag)) throw std::logic_error(fname + ": Failed when reading tags.");
1690
1691 if (elem_type == 15) // skip point type elements
1692 continue;
1693
1694 if (not IsElementSupported(elem_type))
1695 throw std::logic_error(fname + ": Unsupported element encountered.");
1696
1697 log.Log0Verbose2() << "Reading element: " << file_line << " type: " << elem_type;
1698
1699 int num_cell_nodes;
1700 if (elem_type == 1) num_cell_nodes = 2;
1701 else if (elem_type == 2) // 3-node triangle
1702 num_cell_nodes = 3;
1703 else if (elem_type == 3 or elem_type == 4) // 4-node quadrangle or tet
1704 num_cell_nodes = 4;
1705 else if (elem_type == 5) // 8-node hexahedron
1706 num_cell_nodes = 8;
1707 else
1708 continue;
1709
1710 // Make the cell on either the volume or the boundary
1711 LightWeightCell* raw_cell = nullptr;
1712 if (mesh_is_2D_assumption)
1713 {
1714 if (IsElementType1D(elem_type))
1715 {
1716 raw_cell = new LightWeightCell(CellType::SLAB, CellType::SLAB);
1717 raw_boundary_cells_.push_back(raw_cell);
1718 log.Log0Verbose2() << "Added to raw_boundary_cells.";
1719 }
1720 else if (IsElementType2D(elem_type))
1721 {
1722 raw_cell = new LightWeightCell(CellType::POLYGON, CellTypeFromMSHTypeID(elem_type));
1723 raw_cells_.push_back(raw_cell);
1724 log.Log0Verbose2() << "Added to raw_cells.";
1725 }
1726 }
1727 else
1728 {
1729 if (IsElementType2D(elem_type))
1730 {
1731 raw_cell = new LightWeightCell(CellType::POLYGON, CellTypeFromMSHTypeID(elem_type));
1732 raw_boundary_cells_.push_back(raw_cell);
1733 log.Log0Verbose2() << "Added to raw_boundary_cells.";
1734 }
1735 else if (IsElementType3D(elem_type))
1736 {
1737 raw_cell = new LightWeightCell(CellType::POLYHEDRON, CellTypeFromMSHTypeID(elem_type));
1738 raw_cells_.push_back(raw_cell);
1739 log.Log0Verbose2() << "Added to raw_cells.";
1740 }
1741 }
1742
1743 if (raw_cell == nullptr) continue;
1744
1745 auto& cell = *raw_cell;
1746 cell.material_id = physical_reg;
1747 cell.vertex_ids = ReadNodes(num_cell_nodes);
1748
1749 // Populate faces
1750 if (elem_type == 1) // 2-node edge
1751 {
1752 LightWeightFace face0;
1753 LightWeightFace face1;
1754
1755 face0.vertex_ids = {cell.vertex_ids.at(0)};
1756 face1.vertex_ids = {cell.vertex_ids.at(1)};
1757
1758 cell.faces.push_back(face0);
1759 cell.faces.push_back(face1);
1760 }
1761 else if (elem_type == 2 or elem_type == 3) // 3-node triangle or 4-node quadrangle
1762 {
1763 size_t num_verts = cell.vertex_ids.size();
1764 for (size_t e = 0; e < num_verts; e++)
1765 {
1766 size_t ep1 = (e < (num_verts - 1)) ? e + 1 : 0;
1767 LightWeightFace face;
1768
1769 face.vertex_ids = {cell.vertex_ids[e], cell.vertex_ids[ep1]};
1770
1771 cell.faces.push_back(std::move(face));
1772 } // for e
1773 } // if 2D elements
1774 else if (elem_type == 4) // 4-node tetrahedron
1775 {
1776 auto& v = cell.vertex_ids;
1777 std::vector<LightWeightFace> lw_faces(4);
1778 lw_faces[0].vertex_ids = {v[0], v[2], v[1]}; // base-face
1779 lw_faces[1].vertex_ids = {v[0], v[3], v[2]};
1780 lw_faces[2].vertex_ids = {v[3], v[1], v[2]};
1781 lw_faces[3].vertex_ids = {v[3], v[0], v[1]};
1782
1783 for (auto& lw_face : lw_faces)
1784 cell.faces.push_back(lw_face);
1785 }
1786 else if (elem_type == 5) // 8-node hexahedron
1787 {
1788 auto& v = cell.vertex_ids;
1789 std::vector<LightWeightFace> lw_faces(6);
1790 lw_faces[0].vertex_ids = {v[5], v[1], v[2], v[6]}; // East face
1791 lw_faces[1].vertex_ids = {v[0], v[4], v[7], v[3]}; // West face
1792 lw_faces[2].vertex_ids = {v[0], v[3], v[2], v[1]}; // North face
1793 lw_faces[3].vertex_ids = {v[4], v[5], v[6], v[7]}; // South face
1794 lw_faces[4].vertex_ids = {v[2], v[3], v[7], v[6]}; // Top face
1795 lw_faces[5].vertex_ids = {v[0], v[1], v[5], v[4]}; // Bottom face
1796
1797 for (auto& lw_face : lw_faces)
1798 cell.faces.push_back(lw_face);
1799 }
1800 else
1801 throw std::runtime_error(fname + ": Unsupported cell type");
1802
1803 } // for elements
1804
1805 file.close();
1806
1807 // Remap material-ids
1808 std::set<int> material_ids_set_as_read;
1809 std::map<int, int> material_mapping;
1810
1811 for (auto& cell : raw_cells_)
1812 material_ids_set_as_read.insert(cell->material_id);
1813
1814 std::set<int> boundary_ids_set_as_read;
1815 std::map<int, int> boundary_mapping;
1816
1817 for (auto& cell : raw_boundary_cells_)
1818 boundary_ids_set_as_read.insert(cell->material_id);
1819
1820 {
1821 int m = 0;
1822 for (const auto& mat_id : material_ids_set_as_read)
1823 material_mapping.insert(std::make_pair(mat_id, m++));
1824
1825 int b = 0;
1826 for (const auto& bndry_id : boundary_ids_set_as_read)
1827 boundary_mapping.insert(std::make_pair(bndry_id, b++));
1828 }
1829
1830 for (auto& cell : raw_cells_)
1831 cell->material_id = material_mapping[cell->material_id];
1832
1833 for (auto& cell : raw_boundary_cells_)
1834 cell->material_id = boundary_mapping[cell->material_id];
1835
1836 // Always do this
1837 MeshAttributes dimension = DIMENSION_2;
1838 if (not mesh_is_2D_assumption) dimension = DIMENSION_3;
1839
1840 attributes_ = dimension | UNSTRUCTURED;
1841
1842 ComputeCentroidsAndCheckQuality();
1843 BuildMeshConnectivity();
1844
1845 log.Log() << "Done processing " << options.file_name << ".\n"
1846 << "Number of nodes read: " << vertices_.size() << "\n"
1847 << "Number of cells read: " << raw_cells_.size();
1848}

1850void
1851UnpartitionedMesh::ReadFromExodus(const UnpartitionedMesh::Options& options)
1852{
1853 log.Log() << "Reading Exodus file: " << options.file_name << ".";
1854
1855 // Attempt to open file
1856 std::ifstream file;
1857 file.open(options.file_name);
1858 if (!file.is_open())
1859 throw std::runtime_error(ErrorReadingFileStr(options.file_name, "ReadFromExodus"));
1860 file.close();
1861
1862 // Read the file
1863 mesh_options_ = options;
1864 auto reader = vtkSmartPointer<vtkExodusIIReader>::New();
1865 reader->SetFileName(options.file_name.c_str());
1866
1867 if (not reader->CanReadFile(options.file_name.c_str()))
1868 throw std::logic_error("Unable to read file-type with this routine");
1869
1870 reader->UpdateInformation();
1871 // Exodus ships boundary-ids via SideSets and NodeSets. This allows
1872 // it to be read from the file. Here we have to enable the reader
1873 // to process this because it does not do it by default.
1874 reader->SetAllArrayStatus(reader->NODE_SET, 1);
1875 reader->SetAllArrayStatus(reader->NODE_SET_CONN, 1);
1876 reader->SetAllArrayStatus(reader->SIDE_SET, 1);
1877 reader->SetAllArrayStatus(reader->SIDE_SET_CONN, 1);
1878
1879 // The exodusII file format ships blocks of elements
1880 // together with their points/vertices as self-contained (localized)
1881 // unstructured meshes. To relate these localized blocks to the original
1882 // mesh, where are the blocks formed a whole, we need to know the mapping
1883 // from block-local ids to the original ids. This information can
1884 // be derived from the GlobalNodeID arrays loaded onto point-data and
1885 // cell-data. Again, this information is not read by default so we have to
1886 // turn this on.
1887 reader->SetGenerateGlobalNodeIdArray(true);
1888 reader->SetGenerateGlobalElementIdArray(true);
1889 reader->Update();
1890
1891 // Get all the grid blocks
1892 // This part was quite difficult. I eventually found how to do
1893 // this from this post:
1894 // https://public.kitware.com/pipermail/vtkusers/2010-December/064921.html
1895 // It indicated that all exodus formats are read with a single
1896 // top level vtkMultiBlockDataSet (Level 1). Each of the blocks in
1897 // level 2 is also of type vtkMultiBlockDataSet. The level 2 block that has
1898 // the actual elements is also split into blocks but these, level 3,
1899 // blocks each contain a structure castable to vtkUnstructuredGrid.
1900 auto multiblock = reader->GetOutput();
1901
1902 std::vector<vtkUGridPtrAndName> grid_blocks;
1903 auto iter_a = multiblock->NewIterator();
1904 iter_a->GoToFirstItem();
1905 while (not iter_a->IsDoneWithTraversal())
1906 {
1907 auto block_a = iter_a->GetCurrentDataObject();
1908
1909 const std::string block_name =
1910 StringTrim(iter_a->GetCurrentMetaData()->Get(vtkCompositeDataSet::NAME()));
1911
1912 if (block_a->GetDataObjectType() == VTK_UNSTRUCTURED_GRID)
1913 {
1914 grid_blocks.emplace_back(vtkUnstructuredGrid::SafeDownCast(block_a), block_name);
