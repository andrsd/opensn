24std::shared_ptr<MPICommunicatorSet>
25MeshContinuum::MakeMPILocalCommunicatorSet() const
26{
27 // Build the communicator
28 log.Log0Verbose1() << "Building communicator.";
29 std::set<int> local_graph_edges;
30
31 // Loop over local cells
32 // Populate local_graph_edges
33 local_graph_edges.insert(opensn::mpi_comm.rank()); // add current location
34 for (auto& cell : local_cells)
35 {
36 for (auto& face : cell.faces_)
37 {
38 if (face.has_neighbor_)
39 if (not face.IsNeighborLocal(*this))
40 local_graph_edges.insert(face.GetNeighborPartitionID(*this));
41 } // for f
42 } // for local cells
43
44 // Convert set to vector
45 // This is just done for convenience because MPI
46 // needs a contiguous array
47 std::vector<int> local_connections(local_graph_edges.begin(), local_graph_edges.end());
48
49 // Broadcast local connection size
50 log.Log0Verbose1() << "Communicating local connections.";
51
52 std::vector<std::vector<int>> global_graph(opensn::mpi_comm.size(), std::vector<int>());
53 for (int locI = 0; locI < opensn::mpi_comm.size(); locI++)
54 {
55 int locI_num_connections = static_cast<int>(local_connections.size());
56
57 // If chi::mpi.location_id == locI then this call will
58 // act like a send instead of receive. Otherwise
59 // It receives the count.
60 mpi_comm.broadcast(locI_num_connections, locI);
61
62 if (opensn::mpi_comm.rank() != locI) { global_graph[locI].resize(locI_num_connections, -1); }
63 else
64 {
65 std::copy(
66 local_connections.begin(), local_connections.end(), std::back_inserter(global_graph[locI]));
67 }
68 }
69
70 // Broadcast local connections
71 for (int locI = 0; locI < opensn::mpi_comm.size(); locI++)
72 {
73 // If chi::mpi.location_id == locI then this call will
74 // act like a send instead of receive. Otherwise
75 // It receives the count.
76 mpi_comm.broadcast(global_graph[locI].data(), global_graph[locI].size(), locI);
77 }
78
79 log.Log0Verbose1() << "Done communicating local connections.";
80
81 // Build groups
82 mpi::Group world_group = mpi_comm.group();
83
84 std::vector<mpi::Group> location_groups;
85 location_groups.resize(opensn::mpi_comm.size());
86
87 for (int locI = 0; locI < opensn::mpi_comm.size(); locI++)
88 location_groups[locI] = world_group.include(global_graph[locI]);
89
90 // Build communicators
91 std::vector<mpi::Communicator> communicators;
92 log.Log0Verbose1() << "Building communicators.";
93 communicators.resize(opensn::mpi_comm.size());
94
95 for (int locI = 0; locI < opensn::mpi_comm.size(); locI++)
96 communicators[locI] = mpi_comm.create(location_groups[locI], 0);
97
98 log.Log0Verbose1() << "Done building communicators.";
99
100 return std::make_shared<MPICommunicatorSet>(communicators, location_groups, world_group);
101}

103void
104MeshContinuum::ExportCellsToExodus(const std::string& file_base_name,
105 bool suppress_node_sets,
106 bool suppress_side_sets) const
107{
108 const std::string fname = "MeshContinuum::ExportCellsToExodus";
109 log.Log() << "Exporting mesh to Exodus file with base " << file_base_name;
110
111 if (opensn::mpi_comm.size() != 1)
112 throw std::logic_error(fname + ": Currently this routine is only allowed "
113 "in serial.");
114
115 const auto& grid = *this;
116
117 // Check block consistency
118 std::map<int, CellType> block_id_map;
119 for (const auto& cell : local_cells)
120 {
121 const int mat_id = cell.material_id_;
122 if (block_id_map.count(mat_id) == 0) block_id_map[mat_id] = cell.SubType();
123 else
124 {
125 if (cell.SubType() != block_id_map.at(mat_id))
126 throw std::logic_error(fname + ": Material id " + std::to_string(mat_id) +
127 " appearing for more "
128 "than one cell type.");
129 }
130 }
131
132 // Create unstructured meshes for each material-type pair
133 vtkNew<vtkMultiBlockDataSet> grid_blocks;
134 int max_dimension = 0;
135 {
136 vtkNew<vtkUnstructuredGrid> ugrid;
137 vtkNew<vtkPoints> points;
138
139 points->SetDataType(VTK_DOUBLE);
140
141 vtkNew<vtkIdTypeArray> global_node_id_list;
142 global_node_id_list->SetName("GlobalNodeId");
143
144 vtkNew<vtkIdTypeArray> global_elem_id_list;
145 global_elem_id_list->SetName("GlobalElementId");
146
147 vtkNew<vtkIntArray> block_id_list;
148 block_id_list->SetName("BlockID");
149
150 // Load vertices
151 std::vector<uint64_t> vertex_map(grid.GetGlobalVertexCount(), 0);
152 const size_t num_verts = grid.GetGlobalVertexCount();
153 for (size_t v = 0; v < num_verts; ++v)
154 {
155 vertex_map[v] = v;
156 const auto& vertex = grid.vertices[v];
157 points->InsertNextPoint(vertex.x, vertex.y, vertex.z);
158
159 // Exodus node- and cell indices are 1-based
160 // therefore we add a 1 here.
161 global_node_id_list->InsertNextValue(static_cast<vtkIdType>(v + 1));
162 }
163
164 // Load cells
165 for (const auto& cell : grid.local_cells)
166 {
167 if (cell.SubType() == CellType::POLYGON or cell.SubType() == CellType::POLYHEDRON)
168 throw std::logic_error(fname + ": Cell-subtype \"" + CellTypeName(cell.SubType()) +
169 "\" encountered that is not"
170 "supported by exodus.");
171 UploadCellGeometryContinuous(cell, vertex_map, ugrid);
172 block_id_list->InsertNextValue(cell.material_id_);
173 max_dimension = std::max(max_dimension, MeshContinuum::GetCellDimension(cell));
174
175 // Exodus node- and cell indices are 1-based
176 // therefore we add a 1 here.
177 global_elem_id_list->InsertNextValue(static_cast<vtkIdType>(cell.global_id_ + 1));
178 } // for local cells
179
180 // Set arrays
181 ugrid->SetPoints(points);
182 ugrid->GetPointData()->AddArray(global_node_id_list);
183 ugrid->GetCellData()->AddArray(global_elem_id_list);
184 ugrid->GetCellData()->AddArray(block_id_list);
185
186 ugrid->GetPointData()->SetActiveGlobalIds("GlobalNodeId");
187 ugrid->GetCellData()->SetActiveGlobalIds("GlobalElementId");
188
189 // Set block
190 grid_blocks->SetBlock(0, ugrid);
191
192 log.Log() << "Writing grid block "
193 << " Number of cells: " << ugrid->GetNumberOfCells()
194 << " Number of points: " << ugrid->GetNumberOfPoints();
195 } // end of grid_blocks assignment
196
197 // Separate faces by boundary id
198 struct FaceInfo
199 {
200 CellFace const* face_ptr;
201 uint64_t source_cell_id;
202 int source_face_id;
203 };
204 typedef std::vector<FaceInfo> ListOfFaces;
205 std::map<uint64_t, ListOfFaces> boundary_id_faces_map;
206 for (const auto& cell : grid.local_cells)
207 {
208 // Here we build a face mapping because ChiTech's face orientation
209 // for prisms (wedges) and hexahedrons differ from that of VTK.
210 // ChiTech's orientation for prisms and hexes actually matches that
211 // of Exodus but VTK assumes the incoming mesh to be conformant to
212 // VTK and therefore, internally performs a mapping. Fortunately,
213 // the only relevant cell-types, for which a special mapping is
214 // required, are the prisms and hexes.
215 const size_t num_faces = cell.faces_.size();
216 std::vector<int> face_mapping(num_faces, 0);
217 if (cell.SubType() == CellType::WEDGE) face_mapping = {2, 3, 4, 0, 1};
218 else if (cell.SubType() == CellType::HEXAHEDRON)
219 face_mapping = {2, 1, 3, 0, 4, 5};
220 else
221 {
222 for (size_t f = 0; f < cell.faces_.size(); ++f)
223 face_mapping[f] = static_cast<int>(f);
224 }
225
226 // Here we store face information as a triplet, i.e., a face pointer,
227 // the id of the cell owning it, and the local face index (relative to
228 // the cell) of the face.
229 int f = 0;
230 for (const auto& face : cell.faces_)
231 {
232 if (not face.has_neighbor_)
233 boundary_id_faces_map[face.neighbor_id_].push_back(
234 {&face, cell.global_id_, face_mapping[f]});
235 ++f;
236 }
237 }
238
239 // Make NodeSets and/or SideSets
240 vtkNew<vtkMultiBlockDataSet> nodesets_blocks;
241 vtkNew<vtkMultiBlockDataSet> sidesets_blocks;
242 for (const auto& [bndry_id, face_list] : boundary_id_faces_map)
243 {
244 const std::string block_name = grid.GetBoundaryIDMap().at(bndry_id);
245 log.Log0Verbose1() << "bid: " + std::to_string(bndry_id) + " name=\"" + block_name + "\"";
246
247 // NodeSet
248 {
249 vtkNew<vtkUnstructuredGrid> ugrid;
250 vtkNew<vtkPoints> points;
251
252 points->SetDataType(VTK_DOUBLE);
253
254 vtkNew<vtkIdTypeArray> node_global_ids;
255 node_global_ids->SetName("GlobalNodeId");
256
257 // Build vertex set
258 std::set<uint64_t> vid_set;
259 for (const auto& face_info : face_list)
260 for (uint64_t vid : face_info.face_ptr->vertex_ids_)
261 vid_set.insert(vid);
262
263 // Build vertex map
264 std::vector<uint64_t> vertex_map(grid.GetGlobalVertexCount(), 0);
265 {
266 uint64_t mapped_id = 0;
267 for (uint64_t vid : vid_set)
268 vertex_map[vid] = mapped_id++;
269 }
270
271 // Load vertices
272 for (uint64_t vid : vid_set)
273 {
274 const auto& vertex = grid.vertices[vid];
275 points->InsertNextPoint(vertex.x, vertex.y, vertex.z);
276
277 // Exodus node- and cell indices are 1-based
278 // therefore we add a 1 here.
279 node_global_ids->InsertNextValue(static_cast<vtkIdType>(vid + 1));
280 }
281
282 // Load cells
283 for (uint64_t vid : vid_set)
284 {
285 std::vector<vtkIdType> cell_vids = {static_cast<vtkIdType>(vertex_map[vid])};
286 ugrid->InsertNextCell(VTK_VERTEX, static_cast<vtkIdType>(1), cell_vids.data());
287 }
288
289 ugrid->SetPoints(points);
290 ugrid->GetPointData()->AddArray(node_global_ids);
291
292 ugrid->GetPointData()->SetActiveGlobalIds("GlobalNodeId");
293
294 nodesets_blocks->SetBlock(bndry_id, ugrid);
295 nodesets_blocks->GetMetaData(bndry_id)->Set(vtkCompositeDataSet::NAME(), block_name);
296
297 log.Log() << "Writing nodeset block " << block_name
298 << " Number of cells: " << ugrid->GetNumberOfCells()
299 << " Number of points: " << ugrid->GetNumberOfPoints();
300 }
301
302 // SideSet
303 {
304 vtkNew<vtkUnstructuredGrid> ugrid;
305 vtkNew<vtkPoints> points;
306
307 points->SetDataType(VTK_DOUBLE);
308
309 vtkNew<vtkIdTypeArray> src_cell_global_ids;
310 vtkNew<vtkIntArray> src_cell_face_id;
311 src_cell_global_ids->SetName("SourceElementId");
312 src_cell_face_id->SetName("SourceElementSide");
313
314 // Build vertex set
315 std::set<uint64_t> vid_set;
316 for (const auto& face_info : face_list)
317 for (uint64_t vid : face_info.face_ptr->vertex_ids_)
318 vid_set.insert(vid);
319
320 // Build vertex map
321 std::vector<uint64_t> vertex_map(grid.GetGlobalVertexCount(), 0);
322 {
323 uint64_t mapped_id = 0;
324 for (uint64_t vid : vid_set)
325 vertex_map[vid] = mapped_id++;
326 }
327
328 // Load vertices
329 for (uint64_t vid : vid_set)
330 {
331 const auto& vertex = grid.vertices[vid];
332 points->InsertNextPoint(vertex.x, vertex.y, vertex.z);
333 }
334
335 // Load faces
336 for (const auto& face_info : face_list)
337 {
338 UploadFaceGeometry(*face_info.face_ptr, vertex_map, ugrid);
339 src_cell_global_ids->InsertNextValue(static_cast<vtkIdType>(face_info.source_cell_id));
340 src_cell_face_id->InsertNextValue(face_info.source_face_id);
341 }
342
343 ugrid->SetPoints(points);
344 ugrid->GetCellData()->AddArray(src_cell_global_ids);
345 ugrid->GetCellData()->AddArray(src_cell_face_id);
346
347 sidesets_blocks->SetBlock(bndry_id, ugrid);
348 sidesets_blocks->GetMetaData(bndry_id)->Set(vtkCompositeDataSet::NAME(), block_name);
349
350 log.Log() << "Writing sideset block " << block_name
351 << " Number of cells: " << ugrid->GetNumberOfCells()
352 << " Number of points: " << ugrid->GetNumberOfPoints();
353 } // End of side-set
354 }
355
356 // Write the file
357 unsigned int next_block = 0;
358 vtkNew<vtkMultiBlockDataSet> main_block;
359 main_block->SetBlock(next_block++, grid_blocks);
360 if (not suppress_node_sets)
361 {
362 log.Log0Verbose1() << "Exporting nodeset";
363 main_block->SetBlock(next_block, nodesets_blocks);
364 main_block->GetMetaData(next_block++)->Set(vtkCompositeDataSet::NAME(), "Node Sets");
365 }
366 if (not suppress_side_sets)
367 {
368 log.Log0Verbose1() << "Exporting sideset";
369 main_block->SetBlock(next_block, sidesets_blocks);
370 main_block->GetMetaData(next_block++)->Set(vtkCompositeDataSet::NAME(), "Side Sets");
371 }
372
373 vtkNew<vtkExodusIIWriter> writer;
374 writer->SetBlockIdArrayName("BlockID");
375
376 writer->SetFileName((file_base_name + ".e").c_str());
377 writer->SetStoreDoubles(1);
378
379 writer->SetInputData(main_block);
380
381 writer->WriteOutGlobalNodeIdArrayOff();
382 writer->WriteOutGlobalElementIdArrayOff();
383 writer->WriteOutBlockIdArrayOff();
384
385 // The code below requires a VTK patch
386 //
387 // {
388 // auto em_in = vtkModelMetadata::New();
389 //
390 // char **dimNames = new char *[3];
391 // dimNames[0] = new char[]{"X"};
392 // dimNames[1] = new char[]{"Y"};
393 // dimNames[2] = new char[]{"Z"};
394 //
395 // max_dimension = std::min(max_dimension, 3);
396 // chi::log.Log() << "Max dimension set to " << max_dimension;
397 //
398 // em_in->SetCoordinateNames(max_dimension, dimNames);
399 //
400 // writer->SetModelMetadata(em_in);
401 // }
402
403 writer->Write();
404
405 auto em = writer->GetModelMetadata();
406
407 log.Log() << "Num Blocks : " << em->GetNumberOfBlocks();
408 log.Log() << "Num Node Sets: " << em->GetNumberOfNodeSets();
409 log.Log() << "Num Side Sets: " << em->GetNumberOfSideSets();
410 log.Log() << "Dimension : " << em->GetDimension();
411
412 // writer->PrintSelf(std::cout, vtkIndent());
413
414 log.Log() << "Done exporting mesh to exodus.";
415 opensn::mpi_comm.barrier();
416}

420{
421 if (!per_material)
422 {
423 FILE* of = fopen(fileName, "w");
424

427 log.LogAllError() << "Could not open file: " << std::string(fileName);

430

567 node_counter++;

616std::vector<uint64_t>
617MeshContinuum::GetDomainUniqueBoundaryIDs() const
618{
619 opensn::mpi_comm.barrier();
620 log.Log() << "Identifying unique boundary-ids.";
621
622 // Develop local bndry-id set
623 std::set<uint64_t> local_bndry_ids_set;
624 for (auto& cell : local_cells)
625 for (auto& face : cell.faces_)
626 if (not face.has_neighbor_) local_bndry_ids_set.insert(face.neighbor_id_);
627
628 // Vectorify it
629 std::vector<uint64_t> local_bndry_ids(local_bndry_ids_set.begin(), local_bndry_ids_set.end());
630 std::vector<uint64_t> globl_bndry_ids;
631 mpi_comm.all_gather(local_bndry_ids, globl_bndry_ids);
632
633 std::set<uint64_t> globl_bndry_ids_set(globl_bndry_ids.begin(), globl_bndry_ids.end());
634

636 return unique_bdnry_ids;

639std::shared_ptr<GridFaceHistogram>
640MeshContinuum::MakeGridFaceHistogram(double master_tolerance, double slave_tolerance) const
641{
642 std::vector<std::pair<size_t, size_t>> face_categories_list;
643 // Fill histogram
644 std::vector<size_t> face_size_histogram;
645 for (const auto& cell : local_cells)
646 for (const auto& face : cell.faces_)
647 face_size_histogram.push_back(face.vertex_ids_.size());
648
649 std::stable_sort(face_size_histogram.begin(), face_size_histogram.end());
650
651 // Determine total face dofs
652 size_t total_face_dofs_count = 0;
653 for (auto face_size : face_size_histogram)
654 total_face_dofs_count += face_size;
655
656 // Compute average and ratio
657 size_t smallest_face = face_size_histogram.front();
658 size_t largest_face = face_size_histogram.back();
659 size_t total_num_faces = face_size_histogram.size();
660 double average_dofs_per_face = (double)total_face_dofs_count / (double)total_num_faces;
661
662 std::stringstream outstr;
663 outstr << "\nSmallest face = " << smallest_face;
664 outstr << "\nLargest face = " << largest_face;
665 outstr << "\nTotal face dofs = " << total_face_dofs_count;
666 outstr << "\nTotal faces = " << face_size_histogram.size();
667 outstr << "\nAverage dofs/face = " << average_dofs_per_face;
668 outstr << "\nMax to avg ratio = " << (double)largest_face / average_dofs_per_face;
669 log.LogAllVerbose2() << outstr.str();
670
671 // Determine number of bins
672 size_t last_bin_num_faces = total_num_faces;
673 if (((double)largest_face / average_dofs_per_face) > master_tolerance)
674 {
675 log.LogAllVerbose2() << "The ratio of max face dofs to average face dofs "
676 << "is larger than " << master_tolerance
677 << ", therefore a binned histogram "
678 << "will be constructed.";
679
680 // Build categories
681 size_t running_total_face_dofs = 0;
682 size_t running_face_count = 0;
683 size_t running_face_size = face_size_histogram[0];
684
685 double running_average = (double)face_size_histogram[0];
686
687 for (size_t f = 0; f < total_num_faces; ++f)
688 {
689 if (((double)face_size_histogram[f] / running_average) > slave_tolerance)
690 {
691 face_categories_list.emplace_back(running_face_size, running_face_count);
692 running_total_face_dofs = 0;
693 running_face_count = 0;
694 }
695
696 running_face_size = face_size_histogram[f];
697 running_total_face_dofs += face_size_histogram[f];
698 running_face_count++;
699 running_average = (double)running_total_face_dofs / double(running_face_count);
700 last_bin_num_faces = running_face_count;
701 }
702 }
703 face_categories_list.emplace_back(largest_face, last_bin_num_faces);
704
705 // Verbose print bins
706 outstr.str(std::string());
707 outstr << "A total of " << face_categories_list.size() << " bins were created:\n";
708
709 size_t bin_counter = -1;
710 for (auto bins : face_categories_list)
711 {
712 outstr << "Bin " << ++bin_counter << ": " << bins.second << " faces with max face dofs "
713 << bins.first << "\n";
714 }
715
716 log.LogAllVerbose2() << outstr.str();
717
718 return std::make_shared<GridFaceHistogram>(face_categories_list);
719}

903bool
904MeshContinuum::CheckPointInsideCell(const Cell& cell, const Vector3& point) const
905{
906 const auto& grid_ref = *this;
907 typedef Vector3 Vec3;
908 auto InsideTet = [](const Vec3& point, const Vec3& v0, const Vec3& v1, const Vec3& v2)
909 {
910 const auto& v01 = v1 - v0;
911 const auto& v02 = v2 - v0;
912
913 const auto n = v01.Cross(v02).Normalized();
914 const auto c = (v0 + v1 + v2) / 3.0;
915
916 const auto pc = point - c;
917
918 if (pc.Dot(n) > 0.0) return true;
919 else
920 return false;
921 };
922
923 bool inside = true;
924 if (cell.Type() == CellType::SLAB)
925 {
926 const auto& v0 = grid_ref.vertices[cell.vertex_ids_[0]];
927 const auto& v1 = grid_ref.vertices[cell.vertex_ids_[1]];
928
929 const auto v01 = v1 - v0;
930 const auto v0p = point - v0;
931
932 const double v0p_dot_v01 = v0p.Dot(v01);
933
934 if (not(v0p_dot_v01 >= 0 and v0p_dot_v01 < v01.Norm())) inside = false;
935 } // slab
936
937 else if (cell.Type() == CellType::POLYGON)
938 {
939 for (const auto& face : cell.faces_)
940 {
941 const auto& vcp = point - face.centroid_;
942
943 if (vcp.Dot(face.normal_) > 0)
944 {
945 inside = false;
946 break;
947 }
948 } // for face
949 } // polygon
950
951 else if (cell.Type() == CellType::POLYHEDRON)
952 {
953 inside = false;
954 // form tetra hedrons
955 const auto& vcc = cell.centroid_;
956 for (const auto& face : cell.faces_)
957 {
958 const auto& vfc = face.centroid_;
959
960 const size_t num_sides = face.vertex_ids_.size();
961 for (size_t s = 0; s < num_sides; ++s)
962 {
963 const size_t sp1 = (s < (num_sides - 1)) ? s + 1 : 0;
964 const auto& v0 = grid_ref.vertices[face.vertex_ids_[s]];
965 const auto& v1 = vfc;
966 const auto& v2 = grid_ref.vertices[face.vertex_ids_[sp1]];
967 const auto& v3 = vcc;
968
969 typedef std::tuple<Vec3, Vec3, Vec3> TetFace;
970
971 std::vector<TetFace> tet_faces;
972 tet_faces.emplace_back(v0, v1, v2);
973 tet_faces.emplace_back(v0, v2, v3);
974 tet_faces.emplace_back(v1, v3, v2);
975 tet_faces.emplace_back(v0, v3, v1);
976
977 bool inside_tet = true;
978 for (const auto& tet_face : tet_faces)
979 {
980 if (not InsideTet(
981 point, std::get<0>(tet_face), std::get<1>(tet_face), std::get<2>(tet_face)))
982 {
983 inside_tet = false;
984 break;
985 }
986 } // for triangular tet_face
987 if (inside_tet)
988 {
989 inside = true;
990 break;
991 }
992 } // for side
993 if (inside) break;
994 } // for face
995 } // polyhedron
996 else
997 throw std::logic_error("MeshContinuum::CheckPointInsideCell: "
998 "Unsupported cell-type encountered.");
999
1000 return inside;
1001}
