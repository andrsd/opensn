78void
79MeshGenerator::Execute()
80{
81 // Execute all input generators
82 // Note these could be empty
83 std::unique_ptr<UnpartitionedMesh> current_umesh = nullptr;
84 for (auto mesh_generator_ptr : inputs_)
85 {
86 auto new_umesh = mesh_generator_ptr->GenerateUnpartitionedMesh(std::move(current_umesh));
87 current_umesh = std::move(new_umesh);
88 }
89
90 // Generate final umesh and convert it
91 current_umesh = GenerateUnpartitionedMesh(std::move(current_umesh));
92
93 std::vector<int64_t> cell_pids;
94 if (opensn::mpi_comm.rank() == 0)
95 cell_pids = PartitionMesh(*current_umesh, opensn::mpi_comm.size());
96
97 BroadcastPIDs(cell_pids, 0, mpi_comm);
98
99 auto grid_ptr = SetupMesh(std::move(current_umesh), cell_pids);
100
101 // Assign the mesh to a VolumeMesher
102 auto new_mesher = std::make_shared<VolumeMesher>(VolumeMesherType::UNPARTITIONED);
103 new_mesher->SetContinuum(grid_ptr);
104
105 if (current_mesh_handler < 0) PushNewHandlerAndGetIndex();
106
107 auto& cur_hndlr = GetCurrentHandler();
108 cur_hndlr.SetVolumeMesher(new_mesher);
109
110 opensn::mpi_comm.barrier();
111}

121void
122MeshGenerator::ComputeAndPrintStats(const MeshContinuum& grid)
123{
124 const size_t num_local_cells = grid.local_cells.size();
125 size_t num_global_cells = 0;
126
127 mpi_comm.all_reduce(num_local_cells, num_global_cells, mpi::op::sum<size_t>());
128
129 size_t max_num_local_cells;
130 mpi_comm.all_reduce(num_local_cells, max_num_local_cells, mpi::op::max<size_t>());
131
132 size_t min_num_local_cells;
133 mpi_comm.all_reduce(num_local_cells, min_num_local_cells, mpi::op::min<size_t>());
134
135 const size_t avg_num_local_cells = num_global_cells / opensn::mpi_comm.size();
136 const size_t num_local_ghosts = grid.cells.GetNumGhosts();
137 const double local_ghost_to_local_cell_ratio = double(num_local_ghosts) / double(num_local_cells);
138
139 double average_ghost_ratio;
140 mpi_comm.all_reduce(local_ghost_to_local_cell_ratio, average_ghost_ratio, mpi::op::sum<double>());
141
142 average_ghost_ratio /= opensn::mpi_comm.size();
143
144 std::stringstream outstr;
145 outstr << "Mesh statistics:\n";
146 outstr << " Global cell count : " << num_global_cells << "\n";
147 outstr << " Local cell count (avg,max,min): ";
148 outstr << avg_num_local_cells << ",";
149 outstr << max_num_local_cells << ",";
150 outstr << min_num_local_cells << "\n";
151 outstr << " Ghost-to-local ratio (avg) : " << average_ghost_ratio;
152
153 log.Log() << "\n" << outstr.str() << "\n\n";
154
155 log.LogAllVerbose2() << opensn::mpi_comm.rank() << "Local cells=" << num_local_cells;
156}

158std::vector<int64_t>
159MeshGenerator::PartitionMesh(const UnpartitionedMesh& input_umesh, int num_partitions)
160{
161 const auto& raw_cells = input_umesh.GetRawCells();
162 const size_t num_raw_cells = raw_cells.size();
163
164 ChiLogicalErrorIf(num_raw_cells == 0, "No cells in final input mesh");
165
166 // Build cell graph and centroids
167 typedef std::vector<uint64_t> CellGraphNode;
168 typedef std::vector<CellGraphNode> CellGraph;
169 CellGraph cell_graph;
170 std::vector<Vector3> cell_centroids;
171
172 cell_graph.reserve(num_raw_cells);
173 cell_centroids.reserve(num_raw_cells);
174 {
175 for (const auto& raw_cell_ptr : raw_cells)
176 {
177 CellGraphNode cell_graph_node; // <-- Note A
178 for (auto& face : raw_cell_ptr->faces)
179 if (face.has_neighbor) cell_graph_node.push_back(face.neighbor);
180
181 cell_graph.push_back(cell_graph_node);
182 cell_centroids.push_back(raw_cell_ptr->centroid);
183 }
184 }
185
186 // Note A: We do not add the diagonal here. If we do it, ParMETIS seems
187 // to produce sub-optimal partitions
188
189 // Execute partitioner
190 std::vector<int64_t> cell_pids =
191 partitioner_->Partition(cell_graph, cell_centroids, num_partitions);
192
193 std::vector<size_t> partI_num_cells(num_partitions, 0);
194 for (int64_t pid : cell_pids)
195 partI_num_cells[pid] += 1;
196
197 size_t max_num_cells = partI_num_cells.front();
198 size_t min_num_cells = partI_num_cells.front();
199 size_t avg_num_cells = 0;
200 for (size_t count : partI_num_cells)
201 {
202 max_num_cells = std::max(max_num_cells, count);
203 min_num_cells = std::min(min_num_cells, count);
204 avg_num_cells += count;
205 }
206 avg_num_cells /= num_partitions;
207
208 log.Log() << "Partitioner num_cells allocated max,min,avg = " << max_num_cells << ","
209 << min_num_cells << "," << avg_num_cells;
210
211 return cell_pids;
212}

214std::shared_ptr<MeshContinuum>
215MeshGenerator::SetupMesh(std::unique_ptr<UnpartitionedMesh> input_umesh_ptr,
216 const std::vector<int64_t>& cell_pids)
217{
218 // Convert mesh
219 auto grid_ptr = MeshContinuum::New();
220
221 grid_ptr->GetBoundaryIDMap() = input_umesh_ptr->GetMeshOptions().boundary_id_map;
222
223 auto& vertex_subs = input_umesh_ptr->GetVertextCellSubscriptions();
224 size_t cell_globl_id = 0;
225 for (auto& raw_cell : input_umesh_ptr->GetRawCells())
226 {
227 if (CellHasLocalScope(
228 opensn::mpi_comm.rank(), *raw_cell, cell_globl_id, vertex_subs, cell_pids))
229 {
230 auto cell = SetupCell(*raw_cell,
231 cell_globl_id,
232 cell_pids[cell_globl_id],
233 STLVertexListHelper(input_umesh_ptr->GetVertices()));
234
235 for (uint64_t vid : cell->vertex_ids_)
236 grid_ptr->vertices.Insert(vid, input_umesh_ptr->GetVertices()[vid]);
237
238 grid_ptr->cells.push_back(std::move(cell));
239 }
240
241 delete raw_cell;
242 raw_cell = nullptr;
243
244 ++cell_globl_id;
245 } // for raw_cell
246
247 SetGridAttributes(*grid_ptr,
248 input_umesh_ptr->GetMeshAttributes(),
249 {input_umesh_ptr->GetMeshOptions().ortho_Nx,
250 input_umesh_ptr->GetMeshOptions().ortho_Ny,
251 input_umesh_ptr->GetMeshOptions().ortho_Nz});
252
253 grid_ptr->SetGlobalVertexCount(input_umesh_ptr->GetVertices().size());
254
255 ComputeAndPrintStats(*grid_ptr);
256
257 return grid_ptr;
258}

269bool
270MeshGenerator::CellHasLocalScope(int location_id,
271 const UnpartitionedMesh::LightWeightCell& lwcell,
272 uint64_t cell_global_id,
273 const std::vector<std::set<uint64_t>>& vertex_subscriptions,
274 const std::vector<int64_t>& cell_partition_ids) const
275{
276 if (replicated_) return true;
277 // First determine if the cell is a local cell
278 int cell_pid = static_cast<int>(cell_partition_ids[cell_global_id]);
279 if (cell_pid == location_id) return true;
280
281 // Now determine if the cell is a ghost cell
282 for (uint64_t vid : lwcell.vertex_ids)
283 for (uint64_t cid : vertex_subscriptions[vid])
284 {
285 if (cid == cell_global_id) continue;
286 int adj_pid = static_cast<int>(cell_partition_ids[cid]);
287 if (adj_pid == location_id) return true;
288 }
289
290 return false;
291}

293std::unique_ptr<Cell>
294MeshGenerator::SetupCell(const UnpartitionedMesh::LightWeightCell& raw_cell,
295 uint64_t global_id,
296 uint64_t partition_id,
297 const VertexListHelper& vertices)
298{
299 auto cell = std::make_unique<Cell>(raw_cell.type, raw_cell.sub_type);
300 cell->centroid_ = raw_cell.centroid;
301 cell->global_id_ = global_id;
302 cell->partition_id_ = partition_id;
303 cell->material_id_ = raw_cell.material_id;
304
305 cell->vertex_ids_ = raw_cell.vertex_ids;
306
307 size_t face_counter = 0;
308 for (auto& raw_face : raw_cell.faces)
309 {
310 CellFace newFace;
311
312 newFace.has_neighbor_ = raw_face.has_neighbor;
313 newFace.neighbor_id_ = raw_face.neighbor;
314
315 newFace.vertex_ids_ = raw_face.vertex_ids;
316 auto vfc = Vertex(0.0, 0.0, 0.0);
317 for (auto fvid : newFace.vertex_ids_)
318 vfc = vfc + vertices.at(fvid);
319 newFace.centroid_ = vfc / static_cast<double>(newFace.vertex_ids_.size());
320
321 if (cell->Type() == CellType::SLAB)
322 {
323 // A slab face is very easy. If it is the first face
324 // the normal is -khat. If it is the second face then
325 // it is +khat.
326 if (face_counter == 0) newFace.normal_ = Vector3(0.0, 0.0, -1.0);
327 else
328 newFace.normal_ = Vector3(0.0, 0.0, 1.0);
329 }
330 else if (cell->Type() == CellType::POLYGON)
331 {
332 // A polygon face is just a line so we can just grab
333 // the first vertex and form a vector with the face
334 // centroid. The normal is then just khat
335 // cross-product with this vector.
336 uint64_t fvid = newFace.vertex_ids_[0];
337 auto vec_vvc = vertices.at(fvid) - newFace.centroid_;
338
339 newFace.normal_ = Vector3(0.0, 0.0, 1.0).Cross(vec_vvc);
340 newFace.normal_.Normalize();
341 }
342 else if (cell->Type() == CellType::POLYHEDRON)
343 {
344 // A face of a polyhedron can itself be a polygon
345 // which can be multifaceted. Here we need the
346 // average normal over all the facets computed
347 // using an area-weighted average.
348 const size_t num_face_verts = newFace.vertex_ids_.size();
349 double total_area = 0.0;
350 for (size_t fv = 0; fv < num_face_verts; ++fv)
351 {
352 size_t fvp1 = (fv < (num_face_verts - 1)) ? fv + 1 : 0;
353
354 uint64_t fvid_m = newFace.vertex_ids_[fv];
355 uint64_t fvid_p = newFace.vertex_ids_[fvp1];
356
357 auto leg_m = vertices.at(fvid_m) - newFace.centroid_;
358 auto leg_p = vertices.at(fvid_p) - newFace.centroid_;
359
360 auto vn = leg_m.Cross(leg_p);
361
362 double area = 0.5 * vn.Norm();
363 total_area += area;
364
365 newFace.normal_ = newFace.normal_ + area * vn.Normalized();
366 }
367 newFace.normal_ = newFace.normal_ / total_area;
368 newFace.normal_.Normalize();
369 }
370 ++face_counter;
371
372 cell->faces_.push_back(newFace);
373 }
374
375 return cell;
376}
