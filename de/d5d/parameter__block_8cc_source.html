163{

165 if (parameters_.empty())

167 type_ = ParameterBlockType::ARRAY;

224void
225ParameterBlock::SortParameters()
226{
227 struct AlphabeticFunctor
228 {
229 bool operator()(const ParameterBlock& paramA, const ParameterBlock& paramB)
230 {
231 return paramA.Name() < paramB.Name();
232 }
233 };
234
235 struct AlphabeticNumericFunctor
236 {
237 bool operator()(const ParameterBlock& paramA, const ParameterBlock& paramB)
238 {
239 return std::stoi(paramA.Name()) < std::stoi(paramB.Name());
240 }
241 };
242
243 // The different functor here is necessary when the parameters are guaranteed
244 // to have integer names that were converted to strings. It never showed up
245 // because we were not testing with enough values. Essentially "11" < "2" in
246 // the realm of strings but not in integer world.
247 if (this->Type() != ParameterBlockType::ARRAY)
248 std::sort(parameters_.begin(), parameters_.end(), AlphabeticFunctor());
249 else
250 std::sort(parameters_.begin(), parameters_.end(), AlphabeticNumericFunctor());
251}

314ParameterBlock::RecursiveDumpToString(std::string& outstr, const std::string& offset) const
315{
316 outstr += offset + this->Name() + " = \n";
317 outstr += offset + "{\n";
318
319 if (HasValue()) outstr += value_ptr_->PrintStr();
320
321 for (const auto& param : parameters_)
322 {
323
324 switch (param.Type())
325 {
326 case ParameterBlockType::BOOLEAN:
327 {
328 outstr += offset + " " + param.Name() + " = ";
329 const bool value = param.Value().BoolValue();
330 outstr += std::string(value ? "true" : "false") + ",\n";
331 break;
332 }
333 case ParameterBlockType::FLOAT:
334 {
335 outstr += offset + " " + param.Name() + " = ";
336 const double value = param.Value().FloatValue();
337 outstr += std::to_string(value) + ",\n";
338 break;
339 }
340 case ParameterBlockType::STRING:
341 {
342 outstr += offset + " " + param.Name() + " = ";
343 const auto& value = param.Value().StringValue();
344 outstr += "\"" + value + "\",\n";
345 break;
346 }
347 case ParameterBlockType::INTEGER:
348 {
349 outstr += offset + " " + param.Name() + " = ";
