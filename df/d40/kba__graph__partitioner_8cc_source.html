73 } // for each coordinate

77KBAGraphPartitioner::Partition(const std::vector<std::vector<uint64_t>>& graph,
78 const std::vector<Vector3>& centroids,
79 int number_of_parts)
80{
81 log.Log0Verbose1() << "Partitioning with KBAGraphPartitioner";
82
83 ChiLogicalErrorIf(centroids.size() != graph.size(),
84 "Graph number of entries not equal to centroids' number of entries.");
85 const size_t num_cells = graph.size();
86 std::vector<int64_t> pids(num_cells, 0);
87 for (size_t c = 0; c < num_cells; ++c)
88 {
89 const auto& point = centroids[c];
90 // Partitions the point
91 std::array<size_t, 3> p_vals = {0, 0, 0};
92 for (size_t i = 0; i < 3; ++i)
93 {
94 const auto& cuts = *coordinate_infos_[i].cuts_;
95 const size_t num_cuts = cuts.size();
96
97 size_t p_val;
98 bool home_found = false;
99 for (size_t j = 0; j < num_cuts; ++j)
100 if (cuts[j] > point[i])
101 {
102 p_val = j;
103 home_found = true;
104 break;
105 }
106
107 p_vals[i] = home_found ? p_val : (coordinate_infos_[i].n_ - 1);
108 }
109
110 const int64_t nx = static_cast<int64_t>(coordinate_infos_[0].n_);
111 const int64_t ny = static_cast<int64_t>(coordinate_infos_[1].n_);
112
113 const int64_t i = static_cast<int64_t>(p_vals[0]);
114 const int64_t j = static_cast<int64_t>(p_vals[1]);
115 const int64_t k = static_cast<int64_t>(p_vals[2]);
116
117 pids[c] = nx * ny * k + nx * j + i;
118 } // for cell c
119
120 if ((nx_ * ny_ * nz_) != number_of_parts)
121 log.Log0Warning() << "KBAGraphPartitioner::Partition nx_*ny_*nz_ != number_of_parts";
122
123 const auto pid_subsets = MakeSubSets(nx_ * ny_ * nz_, number_of_parts);
124
125 std::vector<int64_t> real_pids(num_cells, 0);
126 for (size_t c = 0; c < num_cells; ++c)
127 {
128 for (size_t p = 0; p < number_of_parts; ++p)
129 {
130 if (pids[c] >= pid_subsets[p].ss_begin and pids[c] <= pid_subsets[p].ss_end)
131 real_pids[c] = static_cast<int64_t>(p);
132 }
133 }
134
135 log.Log0Verbose1() << "Done partitioning with KBAGraphPartitioner";
136
137 return real_pids;
138}
