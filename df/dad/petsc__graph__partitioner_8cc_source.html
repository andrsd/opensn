33std::vector<int64_t>
34PETScGraphPartitioner::Partition(const std::vector<std::vector<uint64_t>>& graph,
35 const std::vector<Vector3>&,
36 int number_of_parts)
37{
38 log.Log0Verbose1() << "Partitioning with PETScGraphPartitioner";
39 // Determine avg num faces per cell
40 // This is done so we can reserve size better
41 const size_t num_raw_cells = graph.size();
42 size_t num_raw_faces = 0;
43 for (auto& cell_row : graph)
44 num_raw_faces += cell_row.size();
45 size_t avg_num_face_per_cell =
46 std::ceil(static_cast<double>(num_raw_faces) / static_cast<double>(num_raw_cells));
47
48 // Start building indices
49 std::vector<int64_t> cell_pids(num_raw_cells, 0);
50 if (num_raw_cells > 1)
51 {
52 // Build indices
53 std::vector<int64_t> i_indices(num_raw_cells + 1, 0);
54 std::vector<int64_t> j_indices;
55 j_indices.reserve(num_raw_cells * avg_num_face_per_cell);
56 {
57 int64_t i = 0;
58 int64_t icount = 0;
59 for (const auto& cell : graph)
60 {
61 i_indices[i] = icount;
62
63 for (const uint64_t neighbor_id : cell)
64 {
65 j_indices.push_back(static_cast<int64_t>(neighbor_id));
66 ++icount;
67 }
68 ++i;
69 }
70 i_indices[i] = icount;
71 }
72
73 log.Log0Verbose1() << "Done building indices.";
74
75 // Copy to raw arrays
76 int64_t* i_indices_raw;
77 int64_t* j_indices_raw;
78 PetscMalloc(i_indices.size() * sizeof(int64_t), &i_indices_raw);
79 PetscMalloc(j_indices.size() * sizeof(int64_t), &j_indices_raw);
80
81 for (int64_t j = 0; j < static_cast<int64_t>(i_indices.size()); ++j)
82 i_indices_raw[j] = i_indices[j];
83
84 for (int64_t j = 0; j < static_cast<int64_t>(j_indices.size()); ++j)
85 j_indices_raw[j] = j_indices[j];
86
87 log.Log0Verbose1() << "Done copying to raw indices.";
88
89 // Create adjacency matrix
90 Mat Adj; // Adjacency matrix
91 MatCreateMPIAdj(PETSC_COMM_SELF,
92 (int64_t)num_raw_cells,
93 (int64_t)num_raw_cells,
94 i_indices_raw,
95 j_indices_raw,
96 nullptr,
97 &Adj);
98
99 log.Log0Verbose1() << "Done creating adjacency matrix.";
100
101 // Create partitioning
102 MatPartitioning part;
103 IS is, isg;
104 MatPartitioningCreate(MPI_COMM_SELF, &part);
105 MatPartitioningSetAdjacency(part, Adj);
106 MatPartitioningSetType(part, type_.c_str());
107 MatPartitioningSetNParts(part, number_of_parts);
108 MatPartitioningApply(part, &is);
109 MatPartitioningDestroy(&part);
110 MatDestroy(&Adj);
111 ISPartitioningToNumbering(is, &isg);
112 log.Log0Verbose1() << "Done building paritioned index set.";
113
114 // Get cell global indices
115 const int64_t* cell_pids_raw;
116 ISGetIndices(is, &cell_pids_raw);
117 for (size_t i = 0; i < num_raw_cells; ++i)
118 cell_pids[i] = cell_pids_raw[i];
119 ISRestoreIndices(is, &cell_pids_raw);
120
121 log.Log0Verbose1() << "Done retrieving cell global indices.";
122 } // if more than 1 cell
123
124 log.Log0Verbose1() << "Done partitioning with PETScGraphPartitioner";
125 return cell_pids;
126}
