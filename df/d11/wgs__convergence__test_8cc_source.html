20GSConvergenceTest(KSP ksp, PetscInt n, PetscReal rnorm, KSPConvergedReason* convergedReason, void*)
21{
22 // Get data context
23 WGSContext* context;
24 KSPGetApplicationContext(ksp, &context);
25
26 // Set rhs norm
27 double residual_scale = 1.0;
28 switch (context->residual_scale_type)
29 {
30 case ResidualScaleType::NONE:
31 residual_scale = 1.0;
32 break;
33 case ResidualScaleType::RHS_NORM:
34 if (context->rhs_norm > 1.0e-25) residual_scale = 1.0 / context->rhs_norm;
35 break;
36 case ResidualScaleType::RHS_PRECONDITIONED_NORM:
37 if (context->rhs_preconditioned_norm > 1.0e-25)
38 residual_scale = 1.0 / context->rhs_preconditioned_norm;
39 break;
40 case ResidualScaleType::CUSTOM_SCALE:
41 if (context->custom_residual_scale > 1.0e-25)
42 residual_scale = 1.0 / context->custom_residual_scale;
43 break;
44 }
45
46 // Compute test criterion
47 double tol;
48 int64_t maxIts;
49 KSPGetTolerances(ksp, nullptr, &tol, nullptr, &maxIts);
50
51 double scaled_residual = rnorm * residual_scale;
52
53 // Print iteration information
54 std::string offset;
55 if (context->groupset_.apply_wgdsa_ || context->groupset_.apply_tgdsa_)
56 offset = std::string(" ");
57
58 std::stringstream iter_info;
59 iter_info << program_timer.GetTimeString() << " " << offset << "WGS groups ["
60 << context->groupset_.groups_.front().id_ << "-"

62 << " Iteration " << std::setw(5) << n << " Residual " << std::setw(9)
63 << scaled_residual;
64
65 if (scaled_residual < tol)
66 {
67 *convergedReason = KSP_CONVERGED_RTOL;
68 iter_info << " CONVERGED\n";
69 }
70
71 if (context->log_info_) log.Log() << iter_info.str() << std::endl;
72
73 return KSP_CONVERGED_ITERATING;
74}
