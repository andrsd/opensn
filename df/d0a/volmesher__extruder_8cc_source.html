15void
16VolumeMesherExtruder::CreateLocalNodes(MeshContinuum& template_grid, MeshContinuum& grid)
17{
18 // For each layer
19 std::set<uint64_t> vertex_ids_with_local_scope;
20 for (size_t iz = 0; iz < (vertex_layers_.size() - 1); iz++)
21 {
22 for (const auto& template_cell : template_grid.local_cells)
23 {
24 // Check template cell type
25 if (template_cell.Type() != CellType::POLYGON)
26 throw std::logic_error("Extruder::CreateLocalNodes: "
27 "Template cell error. Not of base type POLYGON");
28
29 bool has_local_scope = HasLocalScope(template_cell, template_grid, iz);
30
31 if (has_local_scope)
32 {
33 auto& vertex_set = vertex_ids_with_local_scope;
34 for (auto tc_vid : template_cell.vertex_ids_)
35 vertex_set.insert(tc_vid + iz * node_z_index_incr_);
36
37 for (auto tc_vid : template_cell.vertex_ids_)
38 vertex_set.insert(tc_vid + (iz + 1) * node_z_index_incr_);
39 }
40 } // for template cell
41 } // for layer
42
43 // Now add all nodes that are local or neighboring
44 uint64_t vid = 0;
45 for (auto layer_z_level : vertex_layers_)
46 {
47 for (auto& id_vertex : template_grid.vertices)
48 {
49 const auto& vertex = id_vertex.second;
50 auto local_scope = vertex_ids_with_local_scope.find(vid);
51
52 if (local_scope != vertex_ids_with_local_scope.end())
53 grid.vertices.Insert(vid, Vector3(vertex.x, vertex.y, layer_z_level));
54
55 ++vid;
56 } // for vertex
57 } // for layer
58
59 grid.SetGlobalVertexCount(vid);
60}

63VolumeMesherExtruder::Execute()
64{
65 log.Log() << program_timer.GetTimeString()
66 << " VolumeMesherExtruder executed. Memory in use = " << GetMemoryUsageInMB() << " MB"
67 << std::endl;

70 log.Log0Verbose1() << "VolumeMesherExtruder: Processing Region" << std::endl;
71
72 // Create new continuum

74 auto temp_grid = MeshContinuum::New();

76 SetContinuum(grid);

80 // populates vertex-layers

147 // Print info

149 << "] amount of local cells=" << grid->local_cells.size();
150
151 log.Log() << "VolumeMesherExtruder: Number of cells in region = " << total_global_cells

157void
158VolumeMesherExtruder::ExtrudeCells(MeshContinuum& template_grid, MeshContinuum& grid)
159{
160 const Vector3 khat(0.0, 0.0, 1.0);
161 // Start extrusion
162 size_t num_global_cells = 0;
163 for (size_t iz = 0; iz < (vertex_layers_.size() - 1); iz++)
164 {
165 for (const auto& template_cell : template_grid.local_cells)
166 {
167 // Check template cell type
168 if (template_cell.Type() != CellType::POLYGON)
169 throw std::logic_error("Extruder::ExtrudeCells: "
170 "Template cell error. Not of base type POLYGON");
171
172 // Check cell not inverted
173 {
174 const auto& v0 = template_cell.centroid_;
175 const auto& v1 = template_grid.vertices[template_cell.vertex_ids_[0]];
176 const auto& v2 = template_grid.vertices[template_cell.vertex_ids_[1]];
177
178 auto v01 = v1 - v0;
179 auto v02 = v2 - v0;
180
181 if (v01.Cross(v02).Dot(khat) < 0.0)
182 throw std::logic_error("Extruder attempting to extrude a template"
183 " cell with a normal pointing downward. This"
184 " causes erratic behavior and needs to be"
185 " corrected.");
186 }
187
188 auto projected_centroid = ProjectCentroidToLevel(template_cell.centroid_, iz);
189 int pid = GetCellKBAPartitionIDFromCentroid(projected_centroid);
190
191 bool has_local_scope = HasLocalScope(template_cell, template_grid, iz);
192
193 if (has_local_scope)
194 {
195 auto cell = MakeExtrudedCell(
196 template_cell, grid, iz, num_global_cells, pid, template_grid.local_cells.size());
197
198 cell->material_id_ = template_cell.material_id_;
199
200 grid.cells.push_back(std::move(cell));
201 }
202 ++num_global_cells;
203
204 } // for template cell
205
206 } // for iz
207}

209void
210VolumeMesherExtruder::SetupLayers(int default_layer_count)
211{
212 // Create default layers if no input layers are provided
213 if (input_layers_.empty())
214 {
215 log.Log0Warning() << "VolumeMesherExtruder: No extrusion layers have been specified. "
216 << "A default single layer will be used with height 1.0 and a single "
217 << "subdivision.";
218 double dz = 1.0 / default_layer_count;
219 for (int k = 0; k <= default_layer_count; k++)
220 {
221 vertex_layers_.push_back(k * dz);
222 }
223 }
224 else
225 {
226 double last_z = 0.0;
227 vertex_layers_.push_back(last_z);
228
229 for (const auto& input_layer : input_layers_)
230 {
231 double dz = input_layer.height / input_layer.sub_divisions;
232
233 for (int k = 0; k < input_layer.sub_divisions; k++)
234 {
235 last_z += dz;
236 vertex_layers_.push_back(last_z);
237 }
238 }
239 }
240
241 log.Log() << "VolumeMesherExtruder: Total number of cell layers is " << vertex_layers_.size() - 1;
242}

255int
256VolumeMesherExtruder::GetCellKBAPartitionIDFromCentroid(Vector3& centroid)
257{
258 int px = options.partition_x;
259 int py = options.partition_y;
260
261 Cell n_gcell(CellType::GHOST, CellType::GHOST);
262 n_gcell.centroid_ = centroid;
263
264 auto xyz_partition_indices = GetCellXYZPartitionID(&n_gcell);
265
266 int nxi = std::get<0>(xyz_partition_indices);
267 int nyi = std::get<1>(xyz_partition_indices);
268 int nzi = std::get<2>(xyz_partition_indices);
269
270 return nzi * px * py + nyi * px + nxi;
271}

273bool
274VolumeMesherExtruder::HasLocalScope(const Cell& template_cell,
275 const MeshContinuum& template_continuum,
276 size_t z_level)
277{
278 // Check if the template cell is in the current partition
279 {
280 auto& centroid = template_cell.centroid_;
281 auto projected_centroid = ProjectCentroidToLevel(centroid, z_level);
282 int pid = GetCellKBAPartitionIDFromCentroid(projected_centroid);
283 if (pid == opensn::mpi_comm.rank()) return true;
284 }
285
286 const size_t last_z_level = vertex_layers_.size() - 1;
287 const size_t z_level_below = z_level - 1;
288 const size_t z_level_above = z_level + 1;
289
290 // Build z-levels to search
291 std::vector<size_t> z_levels_to_search;
292 z_levels_to_search.reserve(3);
293
294 z_levels_to_search.push_back(z_level);
295 if (z_level != 0) z_levels_to_search.push_back(z_level_below);
296 if (z_level != (last_z_level - 1)) z_levels_to_search.push_back(z_level_above);
297
298 // Search template cell's lateral neighbors
299 const auto& vertex_subs = template_unpartitioned_mesh_->GetVertextCellSubscriptions();
300 for (uint64_t vid : template_cell.vertex_ids_)
301 for (uint64_t cid : vertex_subs[vid])
302 {
303 if (cid == template_cell.local_id_) continue;
304
305 auto& candidate_cell = template_continuum.local_cells[cid];
306 auto& cc_centroid = candidate_cell.centroid_;
307
308 for (size_t z : z_levels_to_search)
309 {
310 auto projected_centroid = ProjectCentroidToLevel(cc_centroid, z);
311 int pid = GetCellKBAPartitionIDFromCentroid(projected_centroid);
312 if (pid == opensn::mpi_comm.rank()) return true;
313 }
314 } // for cid
315
316 // Search template cell's longitudinal neighbors
317 for (size_t z : z_levels_to_search)
318 {
319 if (z == z_level) continue;
320
321 auto projected_centroid = ProjectCentroidToLevel(template_cell.centroid_, z);
322 int pid = GetCellKBAPartitionIDFromCentroid(projected_centroid);
323 if (pid == opensn::mpi_comm.rank()) return true;
324 } // for z
325
326 return false;
327}

329std::unique_ptr<Cell>
330VolumeMesherExtruder::MakeExtrudedCell(const Cell& template_cell,
331 const MeshContinuum& grid,
332 size_t z_level,
333 uint64_t cell_global_id,
334 int partition_id,
335 size_t num_template_cells)
336{
337 const size_t tc_num_verts = template_cell.vertex_ids_.size();
338
339 // Determine cell sub-type
340 CellType extruded_subtype;
341 switch (template_cell.SubType())
342 {
343 case CellType::TRIANGLE:
344 extruded_subtype = CellType::WEDGE;
345 break;
346 case CellType::QUADRILATERAL:
347 extruded_subtype = CellType::HEXAHEDRON;
348 break;
349 default:
350 extruded_subtype = CellType::POLYHEDRON;
351 }
352
353 // Create polyhedron
354 auto cell = std::make_unique<Cell>(CellType::POLYHEDRON, extruded_subtype);
355 cell->global_id_ = cell_global_id;
356 // cell->local_id set when added to mesh
357 cell->partition_id_ = partition_id;
358 cell->centroid_ = ProjectCentroidToLevel(template_cell.centroid_, z_level);
359
360 // Populate cell v-indices
361 cell->vertex_ids_.reserve(2 * tc_num_verts);
362 for (auto tc_vid : template_cell.vertex_ids_)
363 cell->vertex_ids_.push_back(tc_vid + z_level * node_z_index_incr_);
364
365 for (auto tc_vid : template_cell.vertex_ids_)
366 cell->vertex_ids_.push_back(tc_vid + (z_level + 1) * node_z_index_incr_);
367
368 // Create side faces
369 for (auto& face : template_cell.faces_)
370 {
371 CellFace newFace;
372
373 newFace.vertex_ids_.resize(4, -1);
374 newFace.vertex_ids_[0] = face.vertex_ids_[0] + z_level * node_z_index_incr_;
375 newFace.vertex_ids_[1] = face.vertex_ids_[1] + z_level * node_z_index_incr_;
376 newFace.vertex_ids_[2] = face.vertex_ids_[1] + (z_level + 1) * node_z_index_incr_;
377 newFace.vertex_ids_[3] = face.vertex_ids_[0] + (z_level + 1) * node_z_index_incr_;
378
379 // Compute centroid
380 const auto& v0 = grid.vertices[newFace.vertex_ids_[0]];
381 const auto& v1 = grid.vertices[newFace.vertex_ids_[1]];
382 const auto& v2 = grid.vertices[newFace.vertex_ids_[2]];
383 const auto& v3 = grid.vertices[newFace.vertex_ids_[3]];
384
385 Vertex vfc = (v0 + v1 + v2 + v3) / 4.0;
386 newFace.centroid_ = vfc;
387
388 // Compute normal
389 Vector3 va = v0 - vfc;
390 Vector3 vb = v1 - vfc;
391
392 Vector3 vn = va.Cross(vb);
393
394 newFace.normal_ = (vn / vn.Norm());
395
396 // Set neighbor
397 // The side connections have the same connections as the
398 // template cell + the z_level specifiers of the layer.
399 if (face.has_neighbor_)
400 {
401 newFace.neighbor_id_ = face.neighbor_id_ + z_level * num_template_cells;
402 newFace.has_neighbor_ = true;
403 }
404 else
405 newFace.neighbor_id_ = face.neighbor_id_;
406
407 cell->faces_.push_back(newFace);
408 } // for side faces
409
410 // Create top and bottom faces
411 // Bottom face
412 {
413 CellFace newFace;
414 // Vertices
415 auto vfc = Vertex(0.0, 0.0, 0.0);
416 newFace.vertex_ids_.reserve(tc_num_verts);
417 for (int tv = (static_cast<int>(tc_num_verts) - 1); tv >= 0; tv--)
418 {
419 newFace.vertex_ids_.push_back(template_cell.vertex_ids_[tv] + z_level * node_z_index_incr_);
420 const auto& v = grid.vertices[newFace.vertex_ids_.back()];
421 vfc = vfc + v;
422 }
423
424 // Compute centroid
425 vfc = vfc / static_cast<double>(tc_num_verts);
426 newFace.centroid_ = vfc;
427
428 // Compute normal
429 auto va = grid.vertices[newFace.vertex_ids_[0]] - vfc;
430 auto vb = grid.vertices[newFace.vertex_ids_[1]] - vfc;
431
432 auto vn = va.Cross(vb);
433 newFace.normal_ = vn / vn.Norm();
434
435 // Set neighbor
436 if (z_level == 0)
437 {
438 newFace.neighbor_id_ = zmin_bndry_id;
439 newFace.has_neighbor_ = false;
440 }
441 else
442 {
443 newFace.neighbor_id_ = template_cell.local_id_ + (z_level - 1) * num_template_cells;
444 newFace.has_neighbor_ = true;
445 }
446
447 cell->faces_.push_back(newFace);
448 }
449
450 // Top face
451 {
452 CellFace newFace;
453 // Vertices
454 auto vfc = Vertex(0.0, 0.0, 0.0);
455 newFace.vertex_ids_.reserve(tc_num_verts);
456 for (auto tc_vid : template_cell.vertex_ids_)
457 {
458 newFace.vertex_ids_.push_back(tc_vid + (z_level + 1) * node_z_index_incr_);
459 const auto& v = grid.vertices[newFace.vertex_ids_.back()];
460 vfc = vfc + v;
461 }
462
463 // Compute centroid
464 vfc = vfc / static_cast<double>(tc_num_verts);
465 newFace.centroid_ = vfc;
466
467 // Compute normal
468 auto va = grid.vertices[newFace.vertex_ids_[0]] - vfc;
469 auto vb = grid.vertices[newFace.vertex_ids_[1]] - vfc;
470
471 auto vn = va.Cross(vb);
472 newFace.normal_ = vn / vn.Norm();
473
474 // Set neighbor
475 if (z_level == (vertex_layers_.size() - 2))
476 {
477 newFace.neighbor_id_ = zmax_bndry_id;
478 newFace.has_neighbor_ = false;
479 }
480 else
481 {
482 newFace.neighbor_id_ = template_cell.local_id_ + (z_level + 1) * num_template_cells;
483 newFace.has_neighbor_ = true;
484 }
485
486 cell->faces_.push_back(newFace);
487 }
488
489 return cell;
490}
