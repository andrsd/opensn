162void
163XXPowerIterationKEigenSCDSA::Initialize()
164{
165 XXPowerIterationKEigen::Initialize();
166
167 // Make UnknownManager
168 const size_t num_gs_groups = front_gs_.groups_.size();
169 UnknownManager uk_man;
170 uk_man.AddUnknown(UnknownType::VECTOR_N, num_gs_groups);
171
172 // Make boundary conditions
173 auto bcs = TranslateBCs(lbs_solver_.SweepBoundaries(), true);
174
175 // Make xs map
176 auto matid_2_mgxs_map = PackGroupsetXS(
177 lbs_solver_.GetMatID2XSMap(), front_gs_.groups_.front().id_, front_gs_.groups_.back().id_);
178
179 // Create solver
180 const auto& sdm = lbs_solver_.SpatialDiscretization();
181 const auto& unit_cell_matrices = lbs_solver_.GetUnitCellMatrices();
182
183 if (diffusion_solver_sdm_ == "pwld")
184 diffusion_solver_ = std::make_shared<DiffusionMIPSolver>(std::string(TextName() + "_WGDSA"),
185 sdm,
186 uk_man,
187 bcs,
188 matid_2_mgxs_map,
189 unit_cell_matrices,
190 true); // verbosity
191 else
192 {
193 continuous_sdm_ptr_ = PieceWiseLinearContinuous::New(sdm.Grid());
194 diffusion_solver_ = std::make_shared<DiffusionPWLCSolver>(std::string(TextName() + "_WGDSA"),
195 *continuous_sdm_ptr_,
196 uk_man,
197 bcs,
198 matid_2_mgxs_map,
199 unit_cell_matrices,
200 true); // verbosity
201 requires_ghosts_ = true;
202 lbs_pwld_ghost_info_ =
203 MakePWLDVecGhostCommInfo(lbs_solver_.SpatialDiscretization(), lbs_solver_.UnknownManager());
204
205 const auto& cfem_sdm = *continuous_sdm_ptr_;
206 const auto ghost_dof_ids = cfem_sdm.GetGhostDOFIndices(lbs_solver_.UnknownManager());
207 }
208
209 {
210 auto& ds = diffusion_solver_;
211
212 ds->options.residual_tolerance = diff_accel_diffusion_l_abs_tol_;
213 ds->options.max_iters = diff_accel_diffusion_max_iters_;
214 ds->options.verbose = diff_accel_diffusion_verbose_;
215 ds->options.additional_options_string = diff_accel_diffusion_petsc_options_;
216 }
217
218 log.Log() << "Initializing diffusion solver";
219 diffusion_solver_->Initialize();
220 opensn::mpi_comm.barrier();
221 log.Log() << "Done Initializing diffusion solver";
222
223 log.Log() << "Assembling A and b";
224 std::vector<double> dummy_rhs;
225 if (diffusion_solver_sdm_ == "pwld") dummy_rhs.assign(sdm.GetNumLocalDOFs(uk_man), 0.0);
226 else
227 dummy_rhs.assign(continuous_sdm_ptr_->GetNumLocalAndGhostDOFs(uk_man), 0.0);
228
229 diffusion_solver_->AssembleAand_b(dummy_rhs);
230 log.Log() << "Done Assembling A and b";
231}

233void
234XXPowerIterationKEigenSCDSA::Execute()
235{
236 auto phi_temp = phi_old_local_;
237
238 /**Lambda for the creation of scattering sources but the
239 * input vector is only the zeroth moment*/
240 auto SetLBSScatterSourcePhi0 =
241 [this, &phi_temp](const VecDbl& input, const bool additive, const bool suppress_wg_scat = false)
242 {
243 ProjectBackPhi0(front_gs_, input, phi_temp);
244 SetLBSScatterSource(phi_temp, additive, suppress_wg_scat);
245 };
246
247 const size_t tag_SCDSA_solve_time = log.GetRepeatingEventTag("SCDSA_solve_time");
248 const size_t tag_sweep_timing = log.GetRepeatingEventTag("Sweep Timing");
249
250 k_eff_ = 1.0;
251 double k_eff_prev = 1.0;
252 double k_eff_change = 1.0;
253
254 // Start power iterations
255 int nit = 0;
256 bool converged = false;
257 while (nit < max_iters_)
258 {
259 // Set the fission source
260 SetLBSFissionSource(phi_old_local_, false);
261 Scale(q_moments_local_, 1.0 / k_eff_);
262
263 auto Sf_ell = q_moments_local_;
264 auto Sf0_ell = CopyOnlyPhi0(front_gs_, q_moments_local_);
265
266 // This solves the inners for transport
267 primary_ags_solver_->Setup();
268 primary_ags_solver_->Solve();
269
270 // lph_i = l + 1/2,i
271 auto phi0_lph_i = CopyOnlyPhi0(front_gs_, phi_new_local_);
272
273 // Now we produce lph_ip1 = l + 1/2, i+1
274 q_moments_local_ = Sf_ell; // Restore 1/k F phi_l
275 SetLBSScatterSource(phi_new_local_, true);
276
277 front_wgs_context_->ApplyInverseTransportOperator(SourceFlags()); // Sweep
278
279 auto phi0_lph_ip1 = CopyOnlyPhi0(front_gs_, phi_new_local_);
280
281 // Power Iteration Acceleration
282 SetLBSScatterSourcePhi0(phi0_lph_ip1 - phi0_lph_i, false);
283 auto Ss_res = CopyOnlyPhi0(front_gs_, q_moments_local_);
284
285 double production_k = lbs_solver_.ComputeFissionProduction(phi_new_local_);
286
287 VecDbl epsilon_k(phi0_lph_ip1.size(), 0.0);
288 auto epsilon_kp1 = epsilon_k;
289
290 double lambda_k = k_eff_;
291 double lambda_kp1 = lambda_k;
292
293 for (size_t k = 0; k < accel_pi_max_its_; ++k)
294 {
295 ProjectBackPhi0(front_gs_, epsilon_k + phi0_lph_ip1, phi_temp);
296
297 // double production_k = lbs_solver_.ComputeFissionProduction(phi_temp);
298
299 SetLBSFissionSource(phi_temp, false);
300 Scale(q_moments_local_, 1.0 / lambda_k);
301
302 auto Sfaux = CopyOnlyPhi0(front_gs_, q_moments_local_);
303
304 // Inner iterations seems extremely wasteful therefore I
305 // am leaving this at 1 iteration here for further investigation.
306 for (int i = 0; i < 1; ++i)
307 {
308 SetLBSScatterSourcePhi0(epsilon_k, false, true);
309
310 auto Ss = CopyOnlyPhi0(front_gs_, q_moments_local_);
311
312 // Solve the diffusion system
313 log.LogEvent(tag_SCDSA_solve_time, Logger::EventType::EVENT_BEGIN);
314 diffusion_solver_->Assemble_b(Ss + Sfaux + Ss_res - Sf0_ell);
315 diffusion_solver_->Solve(epsilon_kp1, true);
316 log.LogEvent(tag_SCDSA_solve_time, Logger::EventType::EVENT_END);
317
318 epsilon_k = epsilon_kp1;
319 }
320
321 ProjectBackPhi0(front_gs_, epsilon_kp1 + phi0_lph_ip1, phi_old_local_);
322
323 double production_kp1 = lbs_solver_.ComputeFissionProduction(phi_old_local_);
324
325 lambda_kp1 = production_kp1 / (production_k / lambda_k);
326
327 const double lambda_change = std::fabs(1.0 - lambda_kp1 / lambda_k);
328 if (accel_pi_verbose_ >= 1)
329 log.Log() << "PISCDSA iteration " << k << " lambda " << lambda_kp1 << " lambda change "
330 << lambda_change;
331
332 if (lambda_change < accel_pi_k_tol_) break;
333
334 lambda_k = lambda_kp1;
335 epsilon_k = epsilon_kp1;
336 production_k = production_kp1;
337 } // acceleration
338
339 ProjectBackPhi0(front_gs_, epsilon_kp1 + phi0_lph_ip1, phi_new_local_);
340 lbs_solver_.GSScopedCopyPrimarySTLvectors(front_gs_, phi_new_local_, phi_old_local_);
341
342 const double production = lbs_solver_.ComputeFissionProduction(phi_old_local_);
343 lbs_solver_.ScalePhiVector(PhiSTLOption::PHI_OLD, lambda_kp1 / production);
344
345 // Recompute k-eigenvalue
346 k_eff_ = lambda_kp1;
347 double reactivity = (k_eff_ - 1.0) / k_eff_;
348
349 // Check convergence, bookkeeping
350 k_eff_change = fabs(k_eff_ - k_eff_prev) / k_eff_;
351 k_eff_prev = k_eff_;
352 nit += 1;
353
354 if (k_eff_change < std::max(k_tolerance_, 1.0e-12)) converged = true;
355
356 // Print iteration summary
357 if (lbs_solver_.Options().verbose_outer_iterations)
358 {
359 std::stringstream k_iter_info;
360 k_iter_info << program_timer.GetTimeString() << " "
361 << " Iteration " << std::setw(5) << nit << " k_eff " << std::setw(11)
362 << std::setprecision(7) << k_eff_ << " k_eff change " << std::setw(12)
363 << k_eff_change << " reactivity " << std::setw(10) << reactivity * 1e5;
364 if (converged) k_iter_info << " CONVERGED\n";
365
366 log.Log() << k_iter_info.str();
367 }
368
369 if (converged) break;
370 } // for k iterations
371
372 // Print summary
373 log.Log() << "\n";
374 log.Log() << " Final k-eigenvalue : " << std::setprecision(7) << k_eff_;
375 log.Log() << " Final change : " << std::setprecision(6) << k_eff_change
376 << " (num_TrOps:" << front_wgs_context_->counter_applications_of_inv_op_ << ")"
377 << "\n"
378 << " Diffusion solve time : "
379 << log.ProcessEvent(tag_SCDSA_solve_time, Logger::EventOperation::TOTAL_DURATION) *
380 1.0e-6
381 << "s\n"
382 << " Total sweep time : "
383 << log.ProcessEvent(tag_sweep_timing, Logger::EventOperation::TOTAL_DURATION);
384 log.Log() << "\n";
385
386 if (lbs_solver_.Options().use_precursors)
387 {
388 lbs_solver_.ComputePrecursors();
389 Scale(lbs_solver_.PrecursorsNewLocal(), 1.0 / k_eff_);
390 }
391
392 lbs_solver_.UpdateFieldFunctions();
393
394 log.Log() << "LinearBoltzmann::KEigenvalueSolver execution completed\n\n";
395}

397std::vector<double>
398XXPowerIterationKEigenSCDSA::CopyOnlyPhi0(const LBSGroupset& groupset,
399 const std::vector<double>& phi_in)
400{
401 typedef const int64_t cint64;
402
403 const auto& lbs_sdm = lbs_solver_.SpatialDiscretization();
404 const auto& diff_sdm = diffusion_solver_->SpatialDiscretization();
405 const auto& diff_uk_man = diffusion_solver_->UnknownStructure();
406 const auto& phi_uk_man = lbs_solver_.UnknownManager();
407
408 const int gsi = groupset.groups_.front().id_;
409 const size_t gss = groupset.groups_.size();
410
411 const size_t diff_num_local_dofs = requires_ghosts_
412 ? diff_sdm.GetNumLocalAndGhostDOFs(diff_uk_man)
413 : diff_sdm.GetNumLocalDOFs(diff_uk_man);
414
415 std::vector<double> phi_data;
416 if (continuous_sdm_ptr_)
417 phi_data =
418 NodallyAveragedPWLDVector(phi_in, lbs_sdm, diff_sdm, phi_uk_man, lbs_pwld_ghost_info_);
419 else
420 phi_data = phi_in;
421
422 VecDbl output_phi_local(diff_num_local_dofs, 0.0);
423
424 for (const auto& cell : lbs_solver_.Grid().local_cells)
425 {
426 const auto& cell_mapping = lbs_sdm.GetCellMapping(cell);
427 const size_t num_nodes = cell_mapping.NumNodes();
428
429 for (size_t i = 0; i < num_nodes; i++)
430 {
431 cint64 diff_phi_map = diff_sdm.MapDOFLocal(cell, i, diff_uk_man, 0, 0);
432 cint64 lbs_phi_map = lbs_sdm.MapDOFLocal(cell, i, phi_uk_man, 0, gsi);
433
434 double* output_mapped = &output_phi_local[diff_phi_map];
435 const double* phi_in_mapped = &phi_data[lbs_phi_map];
436
437 for (size_t g = 0; g < gss; g++)
438 {
439 output_mapped[g] = phi_in_mapped[g];
440 } // for g
441 } // for node
442 } // for cell
443
444 return output_phi_local;
445}

487XXPowerIterationKEigenSCDSA::GhostInfo
488XXPowerIterationKEigenSCDSA::MakePWLDVecGhostCommInfo(const SpatialDiscretization& sdm,
489 const UnknownManager& uk_man)
490{
491 log.Log() << "Making PWLD ghost communicator";
492
493 const size_t num_local_dofs = sdm.GetNumLocalDOFs(uk_man);
494 const size_t num_globl_dofs = sdm.GetNumGlobalDOFs(uk_man);
495
496 log.Log() << "Number of global dofs" << num_globl_dofs;
497
498 const size_t num_unknowns = uk_man.unknowns_.size();
499
500 // Build a list of global ids
501 std::set<int64_t> global_dof_ids_set;
502
503 const auto& grid = lbs_solver_.Grid();
504 const auto ghost_cell_ids = grid.cells.GetGhostGlobalIDs();
505 for (const auto global_id : ghost_cell_ids)
506 {
507 const auto& cell = grid.cells[global_id];
508 const auto& cell_mapping = sdm.GetCellMapping(cell);
509 const size_t num_nodes = cell_mapping.NumNodes();
510
511 for (size_t i = 0; i < num_nodes; ++i)
512 {
513 for (size_t u = 0; u < num_unknowns; ++u)
514 {
515 const size_t num_comps = uk_man.unknowns_[u].num_components_;
516 for (size_t c = 0; c < num_comps; ++c)
517 {
518 const int64_t dof_map = sdm.MapDOF(cell, i, uk_man, u, c);
519 global_dof_ids_set.insert(dof_map);
520 } // for component
521 } // for unknown
522 } // for node i
523 } // for ghost cell
524
525 // Convert the list to a vector
526 std::vector<int64_t> global_indices(global_dof_ids_set.begin(), global_dof_ids_set.end());
527
528 // Create the vector ghost communicator
529 auto vgc = std::make_shared<VectorGhostCommunicator>(
530 num_local_dofs, num_globl_dofs, global_indices, mpi_comm);
531
532 // Create the map
533 std::map<int64_t, int64_t> ghost_global_id_2_local_map;
534 {
535 int64_t k = 0;
536 for (const auto ghost_id : global_indices)
537 {
538 ghost_global_id_2_local_map[ghost_id] = static_cast<int64_t>(num_local_dofs + k++);
539 }
540 }
541
542 log.Log() << "Done making PWLD ghost communicator";
543 return {vgc, ghost_global_id_2_local_map};
544}

546std::vector<double>
547XXPowerIterationKEigenSCDSA::NodallyAveragedPWLDVector(
548 const std::vector<double>& input,
549 const SpatialDiscretization& pwld_sdm,
550 const SpatialDiscretization& pwlc_sdm,
551 const UnknownManager& uk_man,
552 const XXPowerIterationKEigenSCDSA::GhostInfo& ghost_info)
553{
554 const auto& vgc = ghost_info.vector_ghost_communicator;
555 const auto& dfem_dof_global2local_map = ghost_info.ghost_global_id_2_local_map;
556
557 auto input_with_ghosts = vgc->MakeGhostedVector(input);
558 vgc->CommunicateGhostEntries(input_with_ghosts);
559
560 typedef const int64_t cint64_t;
561
562 const auto& grid = pwld_sdm.Grid();
563
564 const size_t num_unknowns = uk_man.unknowns_.size();
565
566 const size_t num_cfem_local_dofs = pwlc_sdm.GetNumLocalAndGhostDOFs(uk_man);
567
568 std::vector<double> cont_input(num_cfem_local_dofs, 0.0);
569 std::vector<double> cont_input_ctr(num_cfem_local_dofs, 0.0);
570
571 std::map<int64_t, int64_t> cfem_dof_global2local_map;
572
573 // Local cells first
574 std::set<uint64_t> partition_bndry_vertex_id_set;
575 for (const auto& cell : grid.local_cells)
576 {
577 const auto& cell_mapping = pwld_sdm.GetCellMapping(cell);
578 const size_t num_nodes = cell_mapping.NumNodes();

581 {
582 for (size_t u = 0; u < num_unknowns; ++u)
583 {
584 const size_t num_components = uk_man.unknowns_[u].num_components_;
585 for (size_t c = 0; c < num_components; ++c)

592

596 cont_input_ctr[dof_cfem_map] += 1.0;
597 } // for component c
598 } // for unknown u
599 } // for node i
600
601 for (const auto& face : cell.faces_)
602 if (face.has_neighbor_)
603 if (not grid.IsCellLocal(face.neighbor_id_))
604 for (const uint64_t vid : face.vertex_ids_)
605 partition_bndry_vertex_id_set.insert(vid);
606 } // for local cell
607
608 // Ghost cells
609 const auto ghost_cell_ids = grid.cells.GetGhostGlobalIDs();
610 const auto& vid_set = partition_bndry_vertex_id_set;
611 for (const auto global_id : ghost_cell_ids)
612 {
613 const auto& cell = grid.cells[global_id];
614 const auto& cell_mapping = pwld_sdm.GetCellMapping(cell);
615 const size_t num_nodes = cell_mapping.NumNodes();
616
617 for (size_t i = 0; i < num_nodes; ++i)
618 {
619 if (vid_set.find(cell.vertex_ids_[i]) == vid_set.end()) continue;
620
621 for (size_t u = 0; u < num_unknowns; ++u)
622 {
623 const size_t num_components = uk_man.unknowns_[u].num_components_;
624 for (size_t c = 0; c < num_components; ++c)
625 {
626 cint64_t dof_dfem_map_globl = pwld_sdm.MapDOF(cell, i, uk_man, u, c);
627 cint64_t dof_cfem_map_globl = pwlc_sdm.MapDOF(cell, i, uk_man, u, c);
628 if (cfem_dof_global2local_map.count(dof_cfem_map_globl) > 0)
629 {
630 cint64_t dof_dfem_map = dfem_dof_global2local_map.at(dof_dfem_map_globl);
631 cint64_t dof_cfem_map = cfem_dof_global2local_map[dof_cfem_map_globl];
632
633 const double phi_value = input_with_ghosts[dof_dfem_map];
634
635 cont_input[dof_cfem_map] += phi_value;
636 cont_input_ctr[dof_cfem_map] += 1.0;
637 }
638 } // for component
639 } // for unknown
640 } // for node i
641 } // for ghost cell
642
643 // Compute nodal averages
644 {
645 const size_t num_vals = cont_input.size();
646 for (size_t k = 0; k < num_vals; ++k)
647 cont_input[k] /= cont_input_ctr[k];
648 }
649
650 // Project back to dfem
651 auto output = input;
652 for (const auto& cell : grid.local_cells)
653 {
654 const auto& cell_mapping = pwld_sdm.GetCellMapping(cell);
655 const size_t num_nodes = cell_mapping.NumNodes();
656
657 for (size_t i = 0; i < num_nodes; ++i)
658 {
659 for (size_t u = 0; u < num_unknowns; ++u)
660 {
661 const size_t num_components = uk_man.unknowns_[u].num_components_;
662 for (size_t c = 0; c < num_components; ++c)
663 {
664 cint64_t dof_dfem_map = pwld_sdm.MapDOFLocal(cell, i, uk_man, u, c);
665 cint64_t dof_cfem_map = pwlc_sdm.MapDOFLocal(cell, i, uk_man, u, c);
666
667 const double phi_value = cont_input[dof_cfem_map];
668
669 output[dof_dfem_map] = phi_value;
670 } // for component c
671 } // for unknown u
672 } // for node i
673 } // for local cell
674
675 return output;
676}
