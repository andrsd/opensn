13{
14 double x = 0.0, y = 0.0, z = 0.0;
15
16 switch (order)

18 case QuadratureOrder::FIRST:
19 {
20 x = 0.25;
21 y = 0.25;
22 z = 0.25;
23 qpoints_.emplace_back(x, y, z);
24 weights_.push_back(1.0 / 6.0);

26 }
27 case QuadratureOrder::SECOND:
28 {

30 double b = (5.0 - sqrt(5)) / 20.0;
31
32 x = a;
33 y = b;
34 z = b;
35 qpoints_.emplace_back(x, y, z);
36 weights_.push_back(1.0 / 24.0);
37

51 y = b;
52 z = b;
53 qpoints_.emplace_back(x, y, z);
54 weights_.push_back(1.0 / 24.0);
55 break;
56 }
57 case QuadratureOrder::THIRD:
58 {

60 conical.Initialize_Conical_Product_Tet();

62 weights_.swap(conical.weights_);

64 }
65 case QuadratureOrder::FOURTH:
66
67 // Walkington's fifth-order 14-point rule from
68 // "Quadrature on Simplices of Arbitrary Dimension"
69 case QuadratureOrder::FIFTH:
70 {
71 qpoints_.resize(14);
72 weights_.resize(14);
73
74 // permutations of these points and suitably-modified versions of
75 // these points are the quadrature point locations
76 const double a[3] = {0.31088591926330060980, // a1 from the paper
77 0.092735250310891226402, // a2 from the paper
78 0.045503704125649649492}; // a3 from the paper
79
80 // weights. a[] and wt[] are the only floating-point inputs required
81 // for this rule.
82 const double wt[3] = {0.018781320953002641800, // w1 from the paper
83 0.012248840519393658257, // w2 from the paper
84 0.0070910034628469110730}; // w3 from the paper
85
86 // The first two sets of 4 points are formed in a similar manner
87 for (unsigned int i = 0; i < 2; ++i)
88 {
89 // Where we will insert values into qpoints and weights
90 const unsigned int offset = 4 * i;
91
92 // Stuff points and weights values into their arrays
93 const double b = 1. - 3. * a[i];
94
95 // Here are the permutations. Order of these is not important,
96 // all have the same weight
97 qpoints_[offset + 0] = Vector3(a[i], a[i], a[i]);
98 qpoints_[offset + 1] = Vector3(a[i], b, a[i]);
99 qpoints_[offset + 2] = Vector3(b, a[i], a[i]);
100 qpoints_[offset + 3] = Vector3(a[i], a[i], b);
101
102 // These 4 points all have the same weights
103 for (unsigned int j = 0; j < 4; ++j)
104 weights_[offset + j] = wt[i];
105 } // end for
106
107 {
108 // The third set contains 6 points and is formed a little differently
109 const unsigned int offset = 8;
110 const double b = 0.5 * (1. - 2. * a[2]);
111
112 // Here are the permutations. Order of these is not important,
113 // all have the same weight
114 qpoints_[offset + 0] = Vector3(b, b, a[2]);
115 qpoints_[offset + 1] = Vector3(b, a[2], a[2]);
116 qpoints_[offset + 2] = Vector3(a[2], a[2], b);
117 qpoints_[offset + 3] = Vector3(a[2], b, a[2]);
118 qpoints_[offset + 4] = Vector3(b, a[2], b);
119 qpoints_[offset + 5] = Vector3(a[2], b, b);
120
121 // These 6 points all have the same weights
122 for (unsigned int j = 0; j < 6; ++j)
123 weights_[offset + j] = wt[2];
124 }
125 break;
126 }
127
128 // This rule is originally from Keast:
129 // Patrick Keast,
130 // Moderate Degree Tetrahedral Quadrature Formulas,
131 // Computer Methods in Applied Mechanics and Engineering,
132 // Volume 55, Number 3, May 1986, pages 339-348.
133 //
134 // It is accurate on 6th-degree polynomials and has 24 points
135 // vs. 64 for the comparable conical product rule.
136 //
137 // Values copied 24th June 2008 from:
138 // http://people.scs.fsu.edu/~burkardt/f_src/keast/keast.f90
139 case QuadratureOrder::SIXTH:
140 {
141 qpoints_.resize(24);
142 weights_.resize(24);
143
144 // The raw data for the quadrature rule.
145 const std::vector<std::vector<double>> rule_data = {
146 {0.356191386222544953e+00, 0.214602871259151684e+00, 0., 0.00665379170969464506e+00}, // 4
147 {0.877978124396165982e+00, 0.0406739585346113397e+00, 0., 0.00167953517588677620e+00}, // 4
148 {0.0329863295731730594e+00, 0.322337890142275646e+00, 0., 0.00922619692394239843e+00}, // 4
149 {0.0636610018750175299e+00,
150 0.269672331458315867e+00,
151 0.603005664791649076e+00,
152 0.00803571428571428248e+00} // 12
153 };
154
155 // Now call the keast routine to generate _points and _weights
156 KeastRule(rule_data, 4);
157 break;
158 }
159 default:

163 qpoints_.swap(conical.qpoints_);
164 weights_.swap(conical.weights_);
165 }

169void
170QuadratureTetrahedron::KeastRule(const std::vector<std::vector<double>>& rule_data,
171 const unsigned int n_pts)
172{
173 auto& _points = qpoints_;
174 auto& _weights = weights_;
175
176 typedef Vector3 Point;
177
178 // clang-format off
179 // Like the Dunavant rule, the input data should have 4 columns. These
180 // columns have the following format and implied permutations (w=weight).
181 // {a, 0, 0, w} = 1-permutation (a,a,a)
182 // {a, b, 0, w} = 4-permutation (a,b,b), (b,a,b), (b,b,a), (b,b,b)
183 // {a, 0, b, w} = 6-permutation (a,a,b), (a,b,b), (b,b,a), (b,a,b), (b,a,a), (a,b,a)
184 // {a, b, c, w} = 12-permutation (a,a,b), (a,a,c), (b,a,a), (c,a,a), (a,b,a), (a,c,a)
185 // (a,b,c), (a,c,b), (b,a,c), (b,c,a), (c,a,b), (c,b,a)
186 // clang-format on
187
188 // Always insert into the points & weights vector relative to the offset
189 unsigned int offset = 0;
190
191 for (unsigned int p = 0; p < n_pts; ++p)
192 {
193
194 // There must always be a non-zero entry to start the row
195 assert(rule_data[p][0] != static_cast<double>(0.0));
196
197 // A zero weight may imply you did not set up the raw data correctly
198 assert(rule_data[p][3] != static_cast<double>(0.0));
199
200 // What kind of point is this?
201 // One non-zero entry in first 3 cols ? 1-perm (centroid) point = 1
202 // Two non-zero entries in first 3 cols ? 3-perm point = 3
203 // Three non-zero entries ? 6-perm point = 6
204 unsigned int pointtype = 1;
205
206 if (rule_data[p][1] != static_cast<double>(0.0))
207 {
208 if (rule_data[p][2] != static_cast<double>(0.0)) pointtype = 12;
209 else
210 pointtype = 4;
211 }
212 else
213 {
214 // The second entry is zero. What about the third?
215 if (rule_data[p][2] != static_cast<double>(0.0)) pointtype = 6;
216 }
217
218 switch (pointtype)
219 {
220 case 1:
221 {
222 // Be sure we have enough space to insert this point
223 assert(offset + 0 < _points.size());
224
225 const double a = rule_data[p][0];
226
227 // The point has only a single permutation (the centroid!)
228 _points[offset + 0] = Point(a, a, a);
229
230 // The weight is always the last entry in the row.
231 _weights[offset + 0] = rule_data[p][3];
232
233 offset += pointtype;
234 break;
235 }
236
237 case 4:
238 {
239 // Be sure we have enough space to insert these points
240 assert(offset + 3 < _points.size());
241
242 const double a = rule_data[p][0];
243 const double b = rule_data[p][1];
244 const double wt = rule_data[p][3];
245
246 // Here it's understood the second entry is to be used twice, and
247 // thus there are three possible permutations.
248 _points[offset + 0] = Point(a, b, b);
249 _points[offset + 1] = Point(b, a, b);
250 _points[offset + 2] = Point(b, b, a);
251 _points[offset + 3] = Point(b, b, b);
252
253 for (unsigned int j = 0; j < pointtype; ++j)
254 _weights[offset + j] = wt;
255
256 offset += pointtype;
257 break;
258 }
259
260 case 6:
261 {
262 // Be sure we have enough space to insert these points
263 assert(offset + 5 < _points.size());
264
265 const double a = rule_data[p][0];
266 const double b = rule_data[p][2];
267 const double wt = rule_data[p][3];
268
269 // Three individual entries with six permutations.
270 _points[offset + 0] = Point(a, a, b);
271 _points[offset + 1] = Point(a, b, b);
272 _points[offset + 2] = Point(b, b, a);
273 _points[offset + 3] = Point(b, a, b);
274 _points[offset + 4] = Point(b, a, a);
275 _points[offset + 5] = Point(a, b, a);
276
277 for (unsigned int j = 0; j < pointtype; ++j)
278 _weights[offset + j] = wt;
279
280 offset += pointtype;
281 break;
282 }
283
284 case 12:
285 {
286 // Be sure we have enough space to insert these points
287 assert(offset + 11 < _points.size());
288
289 const double a = rule_data[p][0];
290 const double b = rule_data[p][1];
291 const double c = rule_data[p][2];
292 const double wt = rule_data[p][3];
293
294 // Three individual entries with six permutations.
295 _points[offset + 0] = Point(a, a, b);
296 _points[offset + 6] = Point(a, b, c);
297 _points[offset + 1] = Point(a, a, c);
298 _points[offset + 7] = Point(a, c, b);
299 _points[offset + 2] = Point(b, a, a);
300 _points[offset + 8] = Point(b, a, c);
301 _points[offset + 3] = Point(c, a, a);
302 _points[offset + 9] = Point(b, c, a);
303 _points[offset + 4] = Point(a, b, a);
304 _points[offset + 10] = Point(c, a, b);
305 _points[offset + 5] = Point(a, c, a);
306 _points[offset + 11] = Point(c, b, a);
307
308 for (unsigned int j = 0; j < pointtype; ++j)
309 _weights[offset + j] = wt;
310
311 offset += pointtype;
312 break;
313 }
314
315 default:
316 throw std::invalid_argument(std::string(__FUNCTION__) +
317 ": Don't know what to do with this many permutation points!");
318 }
319 }
