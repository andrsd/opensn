95/**
96 * SourceFlags is a combination of `SourceType`s
97 */
98struct SourceFlags
99{
100 /**
101 * Create an empty `SourceFlags`
102 */
103 SourceFlags() : flags_(0) {}
104
105 /**
106 * Create `SourceFlags` from a `SourceType`
107 *
108 * \param type Type to set in the `SourceFlags`
109 */
110 SourceFlags(SourceType type) : flags_(type) {}
111
112 /**
113 * Add flags using the `|=` operator
114 *
115 * \param type Type to set
116 * \return Combination of our types and new `type`
117 */
118 SourceFlags& operator|=(SourceType type)
119 {
120 flags_ |= type;
121 return *this;
122 }
123
124 /**
125 * Add source using the `|=` operator
126 *
127 * \param src Source to add
128 * \return Combination of our types and new types from `src`
129 */
130 SourceFlags& operator|=(const SourceFlags& src)
131 {
132 flags_ |= src.flags_;
133 return *this;
134 }
135
136 /**
137 * Test is there are any types set
138 * \return `true` if there are no types set, `false` otherwise
139 */
140 bool Empty() const { return flags_ == 0; }
141
142 /**
143 * Unset a type
144 *
145 * \param type Type to unset
146 */
147 void Unset(SourceType type) { flags_ &= ~type; }
148
149 /**
150 * Test if type is set using the `&` operator
151 *
152 * \param type Type to test
153 * \return `true` if `type` is set, `false` otherwise
154 */
155 bool operator&(const SourceType& type) const { return flags_ & type; }
156
157private:
158 /// Combination of `SourceFlags`
159 int flags_;
160};
161
162/**
163 * Operator to join individual `SourceFlags`s using the `|` operator
164 *
165 * \param s1 First source
166 * \param s2 Second source
167 * \return Source with types from both sources
168 */
169inline SourceFlags
170operator|(SourceFlags s1, SourceFlags s2)
171{
172 SourceFlags src = s1;
173 src |= s2;
174 return src;
175}
176
177/**
178 * Two `SourceType`s get combined into `SourceFlags`
179 *
180 * \param f1 First source type
181 * \param f2 Second source type
182 * \return Source with both types set
183 */
184inline SourceFlags
185operator|(SourceType f1, SourceType f2)
186{
187 SourceFlags src;
188 src |= f1;
189 src |= f2;
190 return src;
191}
