216 {
217 }
218
219 void Advance()
220 {
221 ref_col_++;
222 if (ref_col_ >= sp_matrix.rowI_indices_[ref_row_].size())
223 {
224 ref_row_++;
225 ref_col_ = 0;
226 while ((ref_row_ < sp_matrix.row_size_) and (sp_matrix.rowI_indices_[ref_row_].empty()))
227 ref_row_++;
228 }
229 }
230
231 EIt operator++()
232 {
233 EIt i = *this;
234 Advance();
235 return i;
236 }
237 EIt operator++(int)
238 {
239 Advance();
240 return *this;
241 }
242
243 EntryReference operator*()
244 {
245 return {ref_row_,
246 sp_matrix.rowI_indices_[ref_row_][ref_col_],
247 sp_matrix.rowI_values_[ref_row_][ref_col_]};
248 }
249 bool operator==(const EIt& rhs) const

252 }
