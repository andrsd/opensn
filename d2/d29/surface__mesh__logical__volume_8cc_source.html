65SurfaceMeshLogicalVolume::Inside(const Vector3& point) const
66{
67 double tolerance = 1.0e-5;
68
69 // Boundbox check
70 double x = point.x;
71 double y = point.y;
72 double z = point.z;
73
74 if (not((x >= xbounds_[0]) and (x <= xbounds_[1]))) return false;
75 if (not((y >= ybounds_[0]) and (y <= ybounds_[1]))) return false;
76 if (not((z >= zbounds_[0]) and (z <= zbounds_[1]))) return false;
77
78 // Cheapshot pass
79 // This pass purely checks if the point have a
80 // negative sense with all the faces of the surface.
81 // If it does then .. bonus .. we don't need to do
82 // anything more because the surface is probably convex.
83 bool cheap_pass = true; // now try to disprove
84 for (auto& face : surf_mesh->GetTriangles())
85 {
86 Vector3 fc = face.face_centroid;
87 Vector3 p_to_fc = fc - point;
88
89 p_to_fc = p_to_fc / p_to_fc.Norm();
90
91 double sense = p_to_fc.Dot(face.geometric_normal);
92
93 if (sense < (0.0 - tolerance))
94 {
95 cheap_pass = false;
96 break;
97 }
98 } // for f
99
100 // if (!cheap_pass) return false;
101 if (cheap_pass) return true;
102
103 // Expensive pass
104 // Getting to here means the cheap pass produced
105 // a negative and now we need to do more work.
106 for (size_t f = 0; f < surf_mesh->GetTriangles().size(); f++)
107 {
108 Vector3 fc = surf_mesh->GetTriangles()[f].face_centroid;
109 Vector3 p_to_fc = fc - point;
110 double distance_to_face = p_to_fc.Norm();
111 double closest_distance = 1.0e16;
112 bool closest_sense_pos = false;
113
114 p_to_fc = p_to_fc / p_to_fc.Norm();
115
116 double sense = p_to_fc.Dot(surf_mesh->GetTriangles()[f].geometric_normal);
117
118 bool good_to_go = true;
119 if (sense < (0.0 - tolerance))
120 {
121 good_to_go = false;
122 for (size_t fi = 0; fi < surf_mesh->GetTriangles().size(); fi++)
123 {
124 if (fi == f) continue; // Skip same face
125
126 // Get all the vertices
127 int v0_i = surf_mesh->GetTriangles()[fi].v_index[0];
128 int v1_i = surf_mesh->GetTriangles()[fi].v_index[1];
129 int v2_i = surf_mesh->GetTriangles()[fi].v_index[2];
130 Vertex v0 = surf_mesh->GetVertices()[v0_i];
131 Vertex v1 = surf_mesh->GetVertices()[v1_i];
132 Vertex v2 = surf_mesh->GetVertices()[v2_i];
133
134 // Check if the line intersects plane
135 Vertex intp; // Intersection point
136 std::pair<double, double> weights;
137 bool intersects_plane = CheckPlaneLineIntersect(
138 surf_mesh->GetTriangles()[fi].geometric_normal, v0, point, fc, intp, &weights);
139 if (!intersects_plane) continue;
140
141 // Check if the line intersects the triangle
142 bool intersects_triangle = true;
143
144 // Compute the legs
145 Vector3 v01 = v1 - v0;
146 Vector3 v12 = v2 - v1;
147 Vector3 v20 = v0 - v2;
148
149 // Compute the vertices to the point
150 Vector3 v0p = intp - v0;
151 Vector3 v1p = intp - v1;
152 Vector3 v2p = intp - v2;
153
154 // Compute the cross products
155 Vector3 x0p = v01.Cross(v0p);
156 Vector3 x1p = v12.Cross(v1p);
157 Vector3 x2p = v20.Cross(v2p);
158
159 // Normalize them
160 x0p = x0p / x0p.Norm();
161 x1p = x1p / x1p.Norm();
162 x2p = x2p / x2p.Norm();
163
164 Vector3 face_norm = surf_mesh->GetTriangles()[fi].geometric_normal /
165 surf_mesh->GetTriangles()[fi].geometric_normal.Norm();
166
167 if (x0p.Dot(face_norm) < 0.0) intersects_triangle = false;
168 if (x1p.Dot(face_norm) < 0.0) intersects_triangle = false;
169 if (x2p.Dot(face_norm) < 0.0) intersects_triangle = false;
170
171 if (!intersects_triangle) continue;
172
173 // Determine the sense with the triangle
174 double sense_with_this_tri = p_to_fc.Dot(surf_mesh->GetTriangles()[fi].geometric_normal);
175 double distance_to_triangle = weights.second * distance_to_face;
176
177 if (distance_to_triangle < closest_distance)
178 {
179 closest_distance = distance_to_triangle;
180
181 if (sense_with_this_tri > 0.0) closest_sense_pos = true;
182 else
183 closest_sense_pos = false;
184 } //
185
186 } // for inner iter face
187 } // if sense negative
188
189 if ((closest_distance < distance_to_face) && closest_sense_pos) good_to_go = true;
190
191 if (!good_to_go) return false;
192 } // for f
193
194 return true;
195}
