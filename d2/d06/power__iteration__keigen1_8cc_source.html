52 front_gs.apply_wgdsa_ = true;
53 front_gs.wgdsa_tol_ = basic_options("PISA_MIP_L_ABS_TOL").FloatValue();
54 front_gs.wgdsa_max_iters_ = static_cast<int>(basic_options("PISA_MIP_L_MAX_ITS").IntegerValue());

59 double pisa_pi_k_tol = basic_options("PISA_PI_K_TOL").FloatValue();

65 double k_eff_prev = 1.0;

68 /**Lambda for the creation of fission sources.*/

70 const VecDbl& input, const bool additive)

72 if (not additive) Set(q_moments_local, 0.0);
73 active_set_source_function(
74 front_gs, q_moments_local, input, APPLY_AGS_FISSION_SOURCES | APPLY_WGS_FISSION_SOURCES);
75 };

77 /**Lambda for the creation of scattering sources*/

81 {
82 if (not additive) Set(q_moments_local, 0.0);
83 SourceFlags source_flags = APPLY_AGS_SCATTER_SOURCES | APPLY_WGS_SCATTER_SOURCES;
84 if (suppress_wgs) source_flags |= SUPPRESS_WG_SCATTER;
85 active_set_source_function(front_gs, q_moments_local, input, source_flags);
86 };

89
90 /**Lambda for the creation of scattering sources but the

103 bool converged = false;

107
108 SetLBSFissionSource(phi_old_local, false);
109 Scale(q_moments_local, 1.0 / k_eff);
110
111 auto Sf_all_moments = q_moments_local;

113
114 // This solves the inners for transport
115 // produces phi at l+1/2

118 wgs_solver->Setup();
119 wgs_solver->Solve();

124
125 // Now we produce lph_ip1 = l + 1/2, i+1
126 q_moments_local = Sf_all_moments; // Restore 1/k F phi_l
127 SetLBSScatterSource(phi_new_local, true);
128
129 frons_wgs_context->ApplyInverseTransportOperator(SourceFlags()); // Sweep

131 auto phi0_lph_ip1 = lbs_solver.WGSCopyOnlyPhi0(front_gs, phi_new_local);
132
133 // Power Iteration Acceleration solve
134 VecDbl epsilon_k(phi0_l.size(), 0.0);
135 auto epsilon_kp1 = epsilon_k;
136
137 double lambda_k = k_eff;
138 double lambda_kp1 = lambda_k;
139
140 for (size_t k = 0; k < pisa_pi_max_its; ++k)
141 {
142 lbs_solver.GSProjectBackPhi0(front_gs, epsilon_k + phi0_lph_ip1, phi_temp);
143
144 double production_k = lbs_solver.ComputeFissionProduction(phi_temp);
145
146 SetLBSFissionSource(phi_temp, false);
147 Scale(q_moments_local, 1.0 / lambda_k);
148
149 auto Sfaux = lbs_solver.WGSCopyOnlyPhi0(front_gs, q_moments_local);
150
151 SetLBSScatterSourcePhi0(phi0_lph_ip1 - phi0_lph_i, false);
152
153 auto Ss_res = lbs_solver.WGSCopyOnlyPhi0(front_gs, q_moments_local);
154
155 // Inner iterations seems extremely wasteful therefore I
156 // am leaving this at 1 iteration here for further investigation.
157 for (int i = 0; i < 1; ++i)
158 {

160
161 auto Ss = lbs_solver.WGSCopyOnlyPhi0(front_gs, q_moments_local);
162
163 // Solve the diffusion system

165 diff_solver.Solve(epsilon_kp1, true);

168 }

171 double production_kp1 = lbs_solver.ComputeFissionProduction(phi_old_local);
172
173 lambda_kp1 = production_kp1 / (production_k / lambda_k);
174
175 const double lambda_change = std::fabs(1.0 - lambda_kp1 / lambda_k);
176 if (pisa_verbose_level >= 1)
177 log.Log() << "PISA iteration " << k << " lambda " << lambda_kp1 << " lambda change "
178 << lambda_change;

180 if (lambda_change < pisa_pi_k_tol) break;
181
182 lambda_k = lambda_kp1;
183 epsilon_k = epsilon_kp1;
184 } // acceleration
185
186 lbs_solver.GSProjectBackPhi0(front_gs, epsilon_kp1 + phi0_lph_ip1, phi_new_local);
187 lbs_solver.GSScopedCopyPrimarySTLvectors(front_gs, phi_new_local, phi_old_local);
188
189 const double production = lbs_solver.ComputeFissionProduction(phi_old_local);
190 lbs_solver.ScalePhiVector(PhiSTLOption::PHI_OLD, lambda_kp1 / production);
191
192 // Recompute k-eigenvalue
193 k_eff = lambda_kp1;
194 double reactivity = (k_eff - 1.0) / k_eff;
195
196 // Check convergence, bookkeeping
197 k_eff_change = fabs(k_eff - k_eff_prev) / k_eff;
198 k_eff_prev = k_eff;
199 nit += 1;
200
201 if (k_eff_change < std::max(tolerance, 1.0e-12)) converged = true;
202
203 // Print iteration summary
204 if (lbs_solver.Options().verbose_outer_iterations)
205 {
206 std::stringstream k_iter_info;
207 k_iter_info << program_timer.GetTimeString() << " "
208 << " Iteration " << std::setw(5) << nit << " k_eff " << std::setw(11)
209 << std::setprecision(7) << k_eff << " k_eff change " << std::setw(12)
210 << k_eff_change << " reactivity " << std::setw(10) << reactivity * 1e5;
211 if (converged) k_iter_info << " CONVERGED\n";
212
213 log.Log() << k_iter_info.str();
214 }
215
216 if (converged) break;
217 } // for k iterations
218
219 // Print summary
220 log.Log() << "\n";
221 log.Log() << " Final k-eigenvalue : " << std::setprecision(7) << k_eff;
222 log.Log() << " Final change : " << std::setprecision(6) << k_eff_change

224 << "\n";
225 log.Log() << "\n";
226}
