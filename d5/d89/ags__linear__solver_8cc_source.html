75AGSLinearSolver::Solve()

78 auto& lbs_solver = ags_context_ptr->lbs_solver_;
79
80 const int gid_i = GroupSpanFirstID();
81 const int gid_f = GroupSpanLastID();
82 const auto& phi = lbs_solver.PhiOldLocal();
83
84 Vec x_old;
85 VecDuplicate(x_, &x_old);
86
87 // Save qmoms to be restored after each iteration.
88 // This is necessary for multiple ags iterations to function
89 // and for keigen-value problems
90 const auto saved_qmoms = lbs_solver.QMomentsLocal();
91
92 for (int iter = 0; iter < tolerance_options_.maximum_iterations; ++iter)
93 {
94
95 lbs_solver.SetGroupScopedPETScVecFromPrimarySTLvector(gid_i, gid_f, x_old, phi);
96
97 for (auto& solver : ags_context_ptr->sub_solvers_list_)
98 {
99 solver->Setup();
100 solver->Solve();
101 }
102
103 lbs_solver.SetGroupScopedPETScVecFromPrimarySTLvector(gid_i, gid_f, x_, phi);
104
105 VecAXPY(x_old, -1.0, x_);
106 PetscReal error_norm;
107 VecNorm(x_old, NORM_2, &error_norm);
108 PetscReal sol_norm;
109 VecNorm(x_, NORM_2, &sol_norm);
110
111 if (verbose_)
112 log.Log() << "********** AGS solver iteration " << std::setw(3) << iter << " "
113 << " Relative change " << std::setw(10) << std::setprecision(4)
114 << error_norm / sol_norm;
115
116 lbs_solver.QMomentsLocal() = saved_qmoms; // Restore qmoms
117
118 if (error_norm < tolerance_options_.residual_absolute) break;
119 } // for iteration
120
121 VecDestroy(&x_old);
122}
