122std::string
123TimingBlock::MakeGraphString()
124{
125 if (name_ == opensn::name) TimeSectionEnd();
126
127 std::vector<std::vector<std::string>> string_matrix;
128
129 AppendGraphEntry(string_matrix, nullptr, "");
130
131 const size_t J = 5;
132 const size_t I = string_matrix.size();
133
134 std::vector<size_t> max_col_widths(J, 0);
135 for (size_t i = 0; i < I; ++i)
136 for (size_t j = 0; j < J; ++j)
137 max_col_widths[j] = std::max(max_col_widths[j], string_matrix[i][j].size());
138
139 std::vector<std::string> headers = {
140 "Section Name", "#calls", "Total time[s]", "Average time[s]", "% of parent"};
141
142 for (size_t j = 0; j < J; ++j)
143 max_col_widths[j] = std::max(max_col_widths[j], headers[j].size());
144
145 /**Lambda to left pad an entry.*/
146 auto LeftPad = [](std::string& entry, size_t width)
147 {
148 const size_t pad_size = width - entry.size();
149 entry = std::string(pad_size, ' ').append(entry);
150 };
151 /**Lambda to right pad an entry.*/
152 auto RightPad = [](std::string& entry, size_t width)
153 {
154 const size_t pad_size = width - entry.size();
155 entry.append(std::string(pad_size, ' '));
156 };
157
158 std::stringstream outstr;
159
160 auto HDIV = [&outstr, &max_col_widths]()
161 {
162 outstr << "*-";
163 for (size_t j = 0; j < J; ++j)
164 {
165 outstr << std::string(max_col_widths[j] + 1, '-');
166 if (j < (J - 1)) outstr << "*-";
167 }
168 outstr << "*\n";
169 };
170
171 HDIV();
172
173 outstr << "| ";
174 for (size_t j = 0; j < J; ++j)
175 {
176 if (j == 0) RightPad(headers[j], max_col_widths[j]);
177 else
178 LeftPad(headers[j], max_col_widths[j] + 1);
179 outstr << headers[j] << " |";
180 }
181 outstr << "\n";
182
183 HDIV();
184
185 for (size_t i = 0; i < I; ++i)
186 {
187 outstr << "| ";
188 for (size_t j = 0; j < J; ++j)
189 {
190 if (j == 0) RightPad(string_matrix[i][j], max_col_widths[j]);
191 else
192 LeftPad(string_matrix[i][j], max_col_widths[j] + 1);
193 outstr << string_matrix[i][j] << " |";
194 }
195 outstr << "\n";
196 }
197
198 HDIV();
199
200 return outstr.str();
201}

204void
205TimingBlock::AppendGraphEntry(std::vector<std::vector<std::string>>& string_matrix,
206 const TimingBlock* parent,
207 const std::string& indent) const
208{
209 std::vector<std::string> entry;
210
211 const double parent_total_time = parent ? parent->TotalTime() : 0.0;
212 const double relative_time =
213 parent_total_time > 1.0e-12 ? total_time_ * 100.0 / parent_total_time : 0.0;
214
215 entry.push_back(indent + name_);
216 entry.push_back(std::to_string(num_occurences_));
217
218 {
219 char buffer[20];
220 ChiLogicalErrorIf(snprintf(buffer, 20, "%.5g", total_time_ / 1000.0) < 0,
221 "Failed to convert total_time = " + std::to_string(total_time_));
222 entry.push_back(buffer);
223 }
224
225 {
226 char buffer[20];
227 ChiLogicalErrorIf(snprintf(buffer, 20, "%.5g", AverageTime() / 1000.0) < 0,
228 "Failed to convert AverageTime = " + std::to_string(AverageTime()));
229 entry.push_back(buffer);
230 }
231
232 {
233 char buffer[10];
234 ChiLogicalErrorIf(snprintf(buffer, 10, "%.2f%%", relative_time) < 0,
235 "Failed to convert relative_time = " + std::to_string(relative_time));
236 entry.push_back(parent ? buffer : "--");
237 }
238
239 string_matrix.push_back(std::move(entry));
240
241 for (auto& child_ptr : children_)
242 child_ptr->AppendGraphEntry(string_matrix, this, indent + " ");
243}
