59AAH_SweepChunk::Sweep(AngleSet& angle_set)
60{
61 const SubSetInfo& grp_ss_info = groupset_.grp_subset_infos_[angle_set.GetRefGroupSubset()];
62
63 gs_ss_size_ = grp_ss_info.ss_size;
64 gs_ss_begin_ = grp_ss_info.ss_begin;
65 gs_gi_ = groupset_.groups_[gs_ss_begin_].id_;
66
67 int deploc_face_counter = -1;
68 int preloc_face_counter = -1;
69
70 sweep_dependency_interface_.angle_set_ = &angle_set;
71 sweep_dependency_interface_.surface_source_active_ = IsSurfaceSourceActive();
72 sweep_dependency_interface_.gs_ss_begin_ = gs_ss_begin_;
73 sweep_dependency_interface_.gs_gi_ = gs_gi_;
74
75 auto& aah_sweep_depinterf =
76 dynamic_cast<AAH_SweepDependencyInterface&>(sweep_dependency_interface_);
77 aah_sweep_depinterf.fluds_ = &dynamic_cast<AAH_FLUDS&>(angle_set.GetFLUDS());
78
79 // Loop over each cell
80 const auto& spds = angle_set.GetSPDS();
81 const auto& spls = spds.GetSPLS().item_id;
82 const size_t num_spls = spls.size();
83 for (size_t spls_index = 0; spls_index < num_spls; ++spls_index)
84 {
85 cell_local_id_ = spls[spls_index];
86 cell_ = &grid_.local_cells[cell_local_id_];
87 sweep_dependency_interface_.cell_ptr_ = cell_;
88 sweep_dependency_interface_.cell_local_id_ = cell_local_id_;
89 cell_mapping_ = &grid_fe_view_.GetCellMapping(*cell_);
90 cell_transport_view_ = &grid_transport_view_[cell_->local_id_];
91
92 const auto& face_orientations = spds.CellFaceOrientations()[cell_local_id_];
93
94 cell_num_faces_ = cell_->faces_.size();
95 cell_num_nodes_ = cell_mapping_->NumNodes();
96 const auto& sigma_t = xs_.at(cell_->material_id_)->SigmaTotal();
97
98 aah_sweep_depinterf.spls_index = spls_index;
99
100 // Get Cell matrices
101 const auto& fe_intgrl_values = unit_cell_matrices_[cell_local_id_];
102 G_ = &fe_intgrl_values.G_matrix;
103 M_ = &fe_intgrl_values.M_matrix;
104 M_surf_ = &fe_intgrl_values.face_M_matrices;
105 IntS_shapeI_ = &fe_intgrl_values.face_Si_vectors;
106
107 for (auto& callback : cell_data_callbacks_)
108 callback();
109
110 // Loop over angles in set
111 const int ni_deploc_face_counter = deploc_face_counter;
112 const int ni_preloc_face_counter = preloc_face_counter;
113
114 // as = angle set
115 // ss = subset
116 const std::vector<size_t>& as_angle_indices = angle_set.GetAngleIndices();
117 const size_t as_num_angles = as_angle_indices.size();
118 for (size_t as_ss_idx = 0; as_ss_idx < as_num_angles; ++as_ss_idx)
119 {
120 direction_num_ = as_angle_indices[as_ss_idx];
121 omega_ = groupset_.quadrature_->omegas_[direction_num_];
122 direction_qweight_ = groupset_.quadrature_->weights_[direction_num_];
123
124 sweep_dependency_interface_.angle_set_index_ = as_ss_idx;
125 sweep_dependency_interface_.angle_num_ = direction_num_;
126
127 deploc_face_counter = ni_deploc_face_counter;
128 preloc_face_counter = ni_preloc_face_counter;
129
130 // Reset right-handside
131 for (int gsg = 0; gsg < gs_ss_size_; ++gsg)
132 b_[gsg].assign(cell_num_nodes_, 0.0);
133
134 ExecuteKernels(direction_data_callbacks_and_kernels_);
135
136 // Upwinding structure
137 aah_sweep_depinterf.in_face_counter = 0;
138 aah_sweep_depinterf.preloc_face_counter = 0;
139 aah_sweep_depinterf.out_face_counter = 0;
140 aah_sweep_depinterf.deploc_face_counter = 0;
141
142 // Update face orientations
143 face_mu_values_.assign(cell_num_faces_, 0.0);
144 for (int f = 0; f < cell_num_faces_; ++f)
145 face_mu_values_[f] = omega_.Dot(cell_->faces_[f].normal_);
146
147 // Surface integrals
148 int in_face_counter = -1;
149 for (int f = 0; f < cell_num_faces_; ++f)
150 {
151 const auto& face = cell_->faces_[f];
152
153 if (face_orientations[f] != FaceOrientation::INCOMING) continue;
154
155 const bool local = cell_transport_view_->IsFaceLocal(f);
156 const bool boundary = not face.has_neighbor_;
157
158 if (local) ++in_face_counter;
159 else if (not boundary)
160 ++preloc_face_counter;
161
162 sweep_dependency_interface_.SetupIncomingFace(
163 f, cell_mapping_->NumFaceNodes(f), face.neighbor_id_, local, boundary);
164
165 aah_sweep_depinterf.in_face_counter = in_face_counter;
166 aah_sweep_depinterf.preloc_face_counter = preloc_face_counter;
167
168 // IntSf_mu_psi_Mij_dA
169 ExecuteKernels(surface_integral_kernels_);
170 } // for f
171
172 // Looping over groups, assembling mass terms
173 for (int gsg = 0; gsg < gs_ss_size_; ++gsg)
174 {
175 g_ = gs_gi_ + gsg;
176 gsg_ = gsg;
177 sigma_tg_ = sigma_t[g_];
178
179 ExecuteKernels(mass_term_kernels_);
180
181 // Solve system
182 GaussElimination(Atemp_, b_[gsg], static_cast<int>(cell_num_nodes_));
183 }
184
185 // Flux updates
186 ExecuteKernels(flux_update_kernels_);
187
188 // Perform outgoing surface operations
189 int out_face_counter = -1;
190 for (int f = 0; f < cell_num_faces_; ++f)
191 {
192 if (face_orientations[f] != FaceOrientation::OUTGOING) continue;
193
194 // Set flags and counters
195 out_face_counter++;
196 const auto& face = cell_->faces_[f];
197 const bool local = cell_transport_view_->IsFaceLocal(f);
198 const bool boundary = not face.has_neighbor_;
199 const int locality = cell_transport_view_->FaceLocality(f);
200
201 if (not boundary and not local) ++deploc_face_counter;
202
203 sweep_dependency_interface_.SetupOutgoingFace(
204 f, cell_mapping_->NumFaceNodes(f), face.neighbor_id_, local, boundary, locality);
205
206 aah_sweep_depinterf.out_face_counter = out_face_counter;
207 aah_sweep_depinterf.deploc_face_counter = deploc_face_counter;
208
209 OutgoingSurfaceOperations();
210 } // for face
211
212 ExecuteKernels(post_cell_dir_sweep_callbacks_);
213 } // for n
214 } // for cell
215}

250 face_node_local_idx,
