293std::string
294Varying::PrintStr(bool with_type) const
295{
296 std::stringstream outstr;
297
298 if (this->Type() == VaryingDataType::STRING) outstr << "\"" << this->StringValue() << "\"";
299 else if (this->Type() == VaryingDataType::FLOAT)
300 outstr << this->FloatValue() << (with_type ? "(double)" : "");
301 else if (this->Type() == VaryingDataType::INTEGER)
302 outstr << this->IntegerValue();
303 else if (this->Type() == VaryingDataType::BOOL)
304 outstr << (this->BoolValue() ? "true" : "false");
305
306 return outstr.str();
307}
