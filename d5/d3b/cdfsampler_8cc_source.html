108CDFSampler::Sample(double x)
109{
110 int ret_val = -1;
111 int cdf_size = ref_cdf_.size();
112
113 // Check bracket lo and hi
114 if (x <= ref_cdf_[0]) ret_val = 0;
115 else if (x >= ref_cdf_[cdf_size - 1])
116 ret_val = cdf_size - 1;
117 // Check internal
118 else
119 {
120 std::pair<int, int> range(0, cdf_size - 1);
121
122 // Sample sub-intvls for range
123 int num_sub_intvls = sub_intvls_.size();
124 for (int s = 0; s < num_sub_intvls; s++)
125 {
126 if (sub_intvls_[s]->Sample(x, range)) break;
127 }
128
129 // Sample range
130 // chi::log.Log()
131 // << "Sampling " << x
132 // << " in range " << range.first << "-" << range.second;
133 for (int k = range.first; k <= range.second; k++)
134 {
135 if (k == 0)
136 {
137 if (x < ref_cdf_[k])
138 {
139 ret_val = k;
140 break;
141 }
142 }
143 else if ((x >= ref_cdf_[k - 1]) and (x < ref_cdf_[k]))
144 {
145 ret_val = k;
146 break;
147 }
148 } // for k
149
150 } // if internal
151
152 if (ret_val < 0)
153 {
154 log.LogAllError() << "CDFSampler::Sample. Error in CDF sampling routine. "
155 << "A bin was not found.";
