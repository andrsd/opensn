26void
27DirectedGraph::VertexAccessor::RemoveVertex(size_t v)
28{
29 ChiLogicalErrorIf(v >= vertices_.size(), "Error removing vertex.");
30
31 auto& vertex = vertices_[v];
32
33 // Get adjacent vertices
34 auto num_us = vertex.us_edge.size();
35 auto num_ds = vertex.ds_edge.size();
36 std::vector<size_t> adj_verts;
37 adj_verts.reserve(num_us + num_ds);
38
39 for (size_t u : vertex.us_edge)
40 adj_verts.push_back(u);
41
42 for (size_t u : vertex.ds_edge)
43 adj_verts.push_back(u);
44
45 // Remove v from all u
46 for (size_t u : adj_verts)
47 {
48 vertices_[u].us_edge.erase(v);
49 vertices_[u].ds_edge.erase(v);
50 }
51
52 vertex_valid_flags_[v] = false;
53}

113void
114DirectedGraph::SCCAlgorithm(size_t u,
115 int& time,
116 std::vector<int>& disc,
117 std::vector<int>& low,
118 std::vector<bool>& on_stack,
119 std::stack<size_t>& stack,
120 std::vector<std::vector<size_t>>& SCCs)
121{
122 // Init discovery time and low value
123 disc[u] = low[u] = ++time;
124 stack.push(u);
125 on_stack[u] = true;
126
127 for (auto v : vertices[u].ds_edge)
128 {
129 if (disc[v] == -1)
130 {
131 SCCAlgorithm(v, time, disc, low, on_stack, stack, SCCs);
132 low[u] = std::min(low[u], low[v]);
133 }
134 else if (on_stack[v])
135 low[u] = std::min(low[u], disc[v]);
136 }
137
138 size_t w = 0;
139 if (low[u] == disc[u])
140 {
141 std::vector<size_t> sub_SCC;
142 while (stack.top() != u)
143 {
144 w = stack.top();
145 sub_SCC.push_back(w);
146 on_stack[w] = false;
147 stack.pop();
148 }
149 w = stack.top();
150 sub_SCC.push_back(w);
151 if (sub_SCC.size() > 1) SCCs.push_back(sub_SCC);
152 on_stack[w] = false;
153 stack.pop();
154 }
155}

157std::vector<std::vector<size_t>>
158DirectedGraph::FindStronglyConnectedComponents()
159{
160 size_t V = vertices.size();
161
162 std::vector<int> disc(V, -1); // Discovery times
163 std::vector<int> low(V, -1); // Earliest visited vertex
164 std::vector<bool> on_stack(V, false); // On stack flags
165 std::stack<size_t> stack; // Stack
166
167 std::vector<std::vector<size_t>> SCCs; // Collection of SCCs
168
169 int time = 0;
170
171 for (size_t v = 0; v < V; ++v)
172 if (disc[v] == -1) SCCAlgorithm(v, time, disc, low, on_stack, stack, SCCs);
173
174 return SCCs;
175}

177std::vector<size_t>
178DirectedGraph::GenerateTopologicalSort()
179{
180 bool has_cycles = false;
181 std::vector<size_t> L;
182 std::vector<GraphVertex*> S;
183
184 L.reserve(vertices.size());
185 S.reserve(vertices.size());
186
187 // Make a copy of the graph
188 auto cur_vertices = vertices;
189
190 // Identify vertices that have no incoming edge
191 for (auto& vertex : cur_vertices)
192 if (vertex.us_edge.empty()) S.push_back(&vertex);
193
194 if (S.empty())
195 {
196 has_cycles = true;
197 goto endofalgo;
198 }
199
200 // Repeatedly remove vertices
201 while (not S.empty())
202 {
203 GraphVertex* node_n = S.back();
204 size_t n = node_n->id;
205 S.erase(S.end() - 1);
206
207 L.push_back(n);
208 auto nodes_m = node_n->ds_edge;
209 for (size_t m : nodes_m)
210 {
211 GraphVertex* node_m = &cur_vertices[m];
212
213 // remove edge from graph
214 node_n->ds_edge.erase(m);
215 node_m->us_edge.erase(n);
216
217 if (node_m->us_edge.empty()) S.push_back(node_m);
218 }
219 }
220
221endofalgo:
222{
223 if (has_cycles) return {};
224 if (L.size() != vertices.size()) return {};
225}
226
227 return L;
228}

230std::vector<size_t>
231DirectedGraph::FindApproxMinimumFAS()
232{
233 auto GetVertexDelta = [](GraphVertex& vertex)
234 {
235 double delta = 0.0;
236 for (size_t ds : vertex.ds_edge)
237 delta += 1.0 * vertex.ds_weights[ds];
238
239 for (size_t us : vertex.us_edge)
240 delta -= 1.0 * vertex.us_weights[us];
241
242 return delta;
243 };
244
245 auto& TG = *this;
246
247 // Execute GR-algorithm
248 std::vector<size_t> s1, s2, s;
249 while (TG.vertices.GetNumValid() > 0)
250 {
251 // Remove sinks
252 while (TG.GetNumSinks() > 0)
253 {
254 for (auto& u : TG.vertices)
255 if (u.ds_edge.empty())
256 {
257 TG.RemoveVertex(u.id);
258 s2.push_back(u.id);
259 break;
260 }
261 } // G contains sinks
262
263 // Remove sources
264 while (TG.GetNumSources() > 0)
265 {
266 for (auto& u : TG.vertices)
267 if (u.us_edge.empty())
268 {
269 TG.RemoveVertex(u.id);
270 s1.push_back(u.id);
271 break;
272 }
273 } // G contains sinks
274
275 // Get max delta
276 std::pair<int, double> max_delta(-1, -100.0);
277 for (auto& u : TG.vertices)
278 {
279 double delta = GetVertexDelta(u);
280 if (delta > max_delta.second) max_delta = std::make_pair(u.id, delta);
281 }
282
283 // Remove max delta
284 TG.RemoveVertex(max_delta.first);
285 s1.push_back(max_delta.first);
286 }
287
288 // Make appr. minimum FAS sequence
289 s.reserve(s1.size() + s2.size());
290 for (size_t u : s1)
291 s.push_back(u);
292 for (size_t u : s2)
293 s.push_back(u);
294
295 return s;
296}

355std::vector<std::pair<size_t, size_t>>
356DirectedGraph::RemoveCyclicDependencies()
357{
358 std::vector<std::pair<size_t, size_t>> edges_to_remove;
359
360 // Utility lambdas
361 auto IsInList = [](std::vector<size_t>& list, size_t val)
362 { return std::find(list.begin(), list.end(), val) != list.end(); };
363
364 // Find initial SCCs
365 auto SCCs = FindStronglyConnectedComponents();
366
367 int iter = 0;
368 while (not SCCs.empty())
369 {
370 if (log.GetVerbosity() >= Logger::LOG_LVL::LOG_0VERBOSE_2)
371 log.LogAll() << "Inter cell cyclic dependency removal. Iteration " << ++iter;
372
373 // Remove bi-connected then tri-connected SCCs then n-connected
374 for (auto& subDG : SCCs)
375 {
376 // If bi-connected
377 if (subDG.size() == 2)
378 {
379 RemoveEdge(subDG.front(), subDG.back());
380 edges_to_remove.emplace_back(subDG.front(), subDG.back());
381 } // bi-connected
382 // If tri-connected
383 else if (subDG.size() == 3)
384 {
385 bool found = false;
386 for (size_t u : subDG)
387 {
388 for (size_t v : vertices[u].ds_edge)
389 if (IsInList(subDG, v))
390 {
391 found = true;
392 RemoveEdge(u, v);
393 edges_to_remove.emplace_back(u, v);
394 break;
395 }
396 if (found) break;
397 } // for u
398 } // tri-connected
399 // If n-connected
400 else
401 {
402 // Add vertices to temporary graph
403 DirectedGraph TG; // Temp Graph
404 for (size_t k = 0; k < subDG.size(); ++k)
405 TG.AddVertex();
406
407 // Add local connectivity
408 int mapping_u = 0;
409 for (auto u : subDG)
410 {
411 for (auto v : vertices[u].ds_edge)
412 {
413 auto mapv = std::find(subDG.begin(), subDG.end(), v);
414 if (mapv != subDG.end())
415 {
416 size_t mapping_v = mapv - subDG.begin();
417 TG.AddEdge(mapping_u, mapping_v, vertices[u].ds_weights[v]);
418 }
419 } // for v
420
421 ++mapping_u;
422 } // for u
423
424 // Make a copy of the graph verts
425 std::vector<GraphVertex> verts_copy;
426 verts_copy.reserve(TG.vertices.size());
427 for (auto& v : TG.vertices)
428 verts_copy.push_back(v);
429
430 // Solve the minimum Feedback Arc Set (FAS) problem
431 auto s = TG.FindApproxMinimumFAS();
432
433 // Build a sequence map
434 // This maps original sequence
435 // to minFAS sequence. i.e. originally
436 // we had v=0,1,2,3... and afterwards we
437 // something like s=7,3,1,5,0,....
438 // smap[v] then gives the position of v in s
439 std::vector<int> smap(s.size(), -1);
440 int count = 0;
441 for (size_t u : s)
442 smap[u] = count++;
443
444 // Build edges to remove
445 std::vector<std::pair<int, int>> edges_to_rem;
446 for (auto& u : verts_copy)
447 {
448 int cur_map = smap[u.id];
449 for (size_t v : u.ds_edge)
450 {
451 int adj_map = smap[v];
452 if (adj_map < cur_map) edges_to_rem.emplace_back(u.id, v);
453 }
454 }
455
456 for (auto& edge : edges_to_rem)
457 {
458 size_t u = subDG[edge.first];
459 size_t v = subDG[edge.second];
460 RemoveEdge(u, v);
461 edges_to_remove.emplace_back(u, v);
462 }
463
464 } // n-connected
465 } // for sub-DG
466
467 // Find SSCs again
468 // This step is like an insurance policy for if
469 // something came through. There should be no SSCs
470 // after the minFAS process, however, we just look
471 // again in-case.
472 SCCs = FindStronglyConnectedComponents();
473 }
474
475 return edges_to_remove;
476}
