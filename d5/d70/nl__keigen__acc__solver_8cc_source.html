97
98 auto& lbs_solver = nl_context_ptr->lbs_solver_;
99 auto& groupsets = lbs_solver.Groupsets();
100 auto& front_gs = groupsets.front();
101
102 auto& phi_old_local = lbs_solver.PhiOldLocal();
103 auto& phi_new_local = lbs_solver.PhiNewLocal();
104
105 auto delta_phi = nl_context_ptr->PhiVecToSTLVec(x_);
106 auto& phi_lph_ip1 = nl_context_ptr->phi_lph_ip1_;
107
108 auto phi_lp1_temp = phi_lph_ip1 + delta_phi;
109 lbs_solver.GSProjectBackPhi0(front_gs, phi_lp1_temp, phi_new_local);
110 lbs_solver.GSScopedCopyPrimarySTLvectors(front_gs, phi_new_local, phi_old_local);
111
112 // Compute final k_eff
113 double k_eff = nl_context_ptr->kresid_func_context_.k_eff;
114
115 const double production = lbs_solver.ComputeFissionProduction(phi_old_local);
116 lbs_solver.ScalePhiVector(PhiSTLOption::PHI_OLD, k_eff / production);
117
118 PetscInt number_of_func_evals;
119 SNESGetNumberFunctionEvals(nl_solver_, &number_of_func_evals);
120
121 // Print summary
122 if (nl_context_ptr->verbosity_level_ >= 1)
123 log.Log() << " Final lambda-eigenvalue : " << std::fixed << std::setw(10)
124 << std::setprecision(7) << k_eff << " (num_DOps:" << number_of_func_evals << ")"
