29VectorGhostCommunicator::CachedParallelData
30VectorGhostCommunicator::MakeCachedParallelData()
31{
32 // Construct a mapping between processes and the ghost indices
33 // that belong to them. This information yields what this process
34 // needs to receive from other processes.
35 std::map<int, std::vector<int64_t>> recv_map;
36 for (int64_t ghost_id : ghost_ids_)
37 recv_map[FindOwnerPID(ghost_id)].push_back(ghost_id);
38
39 // This process will receive data in process-contiguous manner,
40 // so a mapping needs to be developed to map each ghost id to
41 // its respective ordering in the received data.
42 std::map<int64_t, size_t> ghost_to_recv_map;
43 size_t count = 0;
44 for (const auto& [pid, gids] : recv_map)
45 for (const int64_t gid : gids)
46 ghost_to_recv_map[gid] = count++;
47
48 std::vector<int> sendcounts;
49 std::vector<int> senddispls;
50 std::vector<int> recvcounts;
51 std::vector<int> recvdispls;
52
53 // Now, the structure of the data being received from communication
54 // is developed. This involves determining the amount of data
55 // being sent per process and the starting position of the data in
56 // the receive buffer per process.
57 int total_recvcounts = 0;
58 recvcounts.assign(process_count_, 0);
59 recvdispls.assign(process_count_, 0);
60 for (const auto& [pid, gids] : recv_map)
61 {
62 recvcounts[pid] = static_cast<int>(gids.size());
63 recvdispls[pid] = total_recvcounts;
64 total_recvcounts += static_cast<int>(gids.size());
65 }
66
67 // For communication, each process must also know what it is
68 // sending to other processors. If each processor sends each
69 // other process the global ids it needs to receive, then each
70 // process will know what other processes need from it. The
71 // MPI utility MapAllToAll in Chi-Tech accomplishes this task,
72 // returning a mapping of processes to the global ids that this
73 // process needs to send.
74 std::map<int, std::vector<int64_t>> send_map = MapAllToAll(recv_map, comm_);
75
76 // With this information, the amount of information that needs
77 // to be sent can be determined.
78 size_t send_size = 0;
79 for (const auto& [pid, gids] : send_map)
80 send_size += gids.size();
81
82 // Next, the local ids on this process that need to be
83 // communicated to other processes can be determined and stored.
84 std::vector<int64_t> local_ids_to_send;
85 local_ids_to_send.reserve(send_size);
86 for (const auto& [pid, gids] : send_map)
87 for (const int64_t gid : gids)
88 {
89 ChiLogicalErrorIf(gid < extents_[location_id_] or gid >= extents_[location_id_ + 1],
90 std::string(__FUNCTION__) + ": " +
91 "Problem determining communication pattern. Process " +
92 std::to_string(pid) + " determined that process " +
93 std::to_string(location_id_) + " needs to communicate global id " +
94 std::to_string(gid) + " to it, but this id is not locally owned.");
95
96 local_ids_to_send.push_back(gid - static_cast<int64_t>(extents_[location_id_]));
97 }
98
99 // Finally, the communication pattern for the data being sent
100 // can be constructed similarly to that for the received data.
101 int total_sendcounts = 0;
102 sendcounts.assign(process_count_, 0);
103 senddispls.assign(process_count_, 0);
104 for (const auto& [pid, gids] : send_map)
105 {
106 sendcounts[pid] = static_cast<int>(gids.size());
107 senddispls[pid] = total_sendcounts;
108 total_sendcounts += static_cast<int>(gids.size());
109 }
110
111 return CachedParallelData{std::move(sendcounts),
112 std::move(senddispls),
113 std::move(recvcounts),
114 std::move(recvdispls),
115 std::move(local_ids_to_send),
116 std::move(ghost_to_recv_map)};
117}

143int64_t
144VectorGhostCommunicator::MapGhostToLocal(const int64_t ghost_id) const
145{
146 ChiInvalidArgumentIf(cached_parallel_data_.ghost_to_recv_map_.count(ghost_id) == 0,
147 "The given ghost id does not belong to this communicator.");
148
149 // Get the position within the ghost id vector of the given ghost id
150 const auto k = std::find(ghost_ids_.begin(), ghost_ids_.end(), ghost_id) - ghost_ids_.begin();
151
152 // Local index is local size plus the position in the ghost id vector
153 return static_cast<int64_t>(local_size_) + k;
154}

156void
157VectorGhostCommunicator::CommunicateGhostEntries(std::vector<double>& ghosted_vector) const
158{
159 ChiInvalidArgumentIf(ghosted_vector.size() != local_size_ + ghost_ids_.size(),
160 std::string(__FUNCTION__) +
161 ": Vector size mismatch. "
162 "input size = " +
163 std::to_string(ghosted_vector.size()) + " requirement " +
164 std::to_string(local_size_ + ghost_ids_.size()));
165
166 // Serialize the data that needs to be sent
167 const size_t send_size = cached_parallel_data_.local_ids_to_send_.size();
168 std::vector<double> send_data;
169 send_data.reserve(send_size);
170 for (const int64_t local_id : cached_parallel_data_.local_ids_to_send_)
171 send_data.push_back(ghosted_vector[local_id]);
172
173 // Create serialized storage for the data to be received
174 const size_t recv_size = ghost_ids_.size();
175 std::vector<double> recv_data(recv_size, 0.0);
176
177 // Communicate the ghost data
178 comm_.all_to_all(send_data,
179 cached_parallel_data_.sendcounts_,
180 cached_parallel_data_.senddispls_,
181 recv_data,
182 cached_parallel_data_.recvcounts_,
183 cached_parallel_data_.recvdispls_);
184
185 // Lastly, populate the local vector with ghost data. All ghost data is
186 // appended to the back of the local vector. Using the mapping between
187 // ghost indices and the relative ghost index position along with the
188 // ordering of the ghost indices, this can be accomplished.
189 for (size_t k = 0; k < recv_size; ++k)
190 ghosted_vector[local_size_ + k] =
191 recv_data[cached_parallel_data_.ghost_to_recv_map_.at(ghost_ids_[k])];
192}

202VectorGhostCommunicator::MakeGhostedVector(const std::vector<double>& local_vector) const

204 ChiInvalidArgumentIf(local_vector.size() != local_size_,

207
208 // Add ghost indices to the back of the unghosted vector

217{

222
