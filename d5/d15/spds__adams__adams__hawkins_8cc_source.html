111SPDS_AdamsAdamsHawkins::BuildTaskDependencyGraph(
112 const std::vector<std::vector<int>>& global_dependencies, bool cycle_allowance_flag)
113{
114
115 std::vector<std::pair<int, int>> edges_to_remove;
116 std::vector<int> raw_edges_to_remove;
117 DirectedGraph TDG;
118
119 // Build graph on home location
120 if (opensn::mpi_comm.rank() == 0)
121 {
122 log.Log0Verbose1() << program_timer.GetTimeString() << " Building Task Dependency Graphs.";
123
124 // Add vertices to the graph
125 for (int loc = 0; loc < opensn::mpi_comm.size(); loc++)
126 TDG.AddVertex();
127
128 // Add dependencies
129 for (int loc = 0; loc < opensn::mpi_comm.size(); loc++)
130 for (int dep = 0; dep < global_dependencies[loc].size(); dep++)
131 TDG.AddEdge(global_dependencies[loc][dep], loc);
132
133 // Remove cyclic dependencies
134 if (cycle_allowance_flag)
135 {
136 log.Log0Verbose1() << program_timer.GetTimeString() << " Removing intra-cellset cycles.";
137 auto edges_to_remove_temp = TDG.RemoveCyclicDependencies();
138 for (const auto& [v0, v1] : edges_to_remove_temp)
139 edges_to_remove.emplace_back(v0, v1);
140 }
141
142 // Serialize edges to be removed
143 raw_edges_to_remove.resize(edges_to_remove.size() * 2, 0);
144 int i = 0;
145 for (const auto& edge : edges_to_remove)
146 {
147 raw_edges_to_remove[i++] = edge.first;
148 raw_edges_to_remove[i++] = edge.second;
149 }
150 } // if home
151
152 // Broadcast edges
153 mpi_comm.broadcast(raw_edges_to_remove, 0);
154 int edge_buffer_size = raw_edges_to_remove.size();
155
156 // De-serialize edges
157 if (opensn::mpi_comm.rank() != 0)
158 {
159 edges_to_remove.resize(edge_buffer_size / 2, std::pair<int, int>(0, 0));
160 int i = 0;
161 for (auto& edge : edges_to_remove)
162 {
163 edge.first = raw_edges_to_remove[i++];
164 edge.second = raw_edges_to_remove[i++];
165 }
166 }
167
168 // Remove edges
169 for (auto& edge_to_remove : edges_to_remove)
170 {
171 int rlocI = edge_to_remove.first;
172 int locI = edge_to_remove.second;
173
174 if (opensn::mpi_comm.rank() == 0) TDG.RemoveEdge(rlocI, locI);
175
176 if (locI == opensn::mpi_comm.rank())
177 {
178 auto dependent_location =
179 std::find(location_dependencies_.begin(), location_dependencies_.end(), rlocI);
180 location_dependencies_.erase(dependent_location);
181 delayed_location_dependencies_.push_back(rlocI);
182 }
183
184 if (rlocI == opensn::mpi_comm.rank()) delayed_location_successors_.push_back(locI);
185 }
186
187 // Generate topological sort
188 std::vector<int> glob_linear_sweep_order;
189 if (opensn::mpi_comm.rank() == 0)
190 {
191 log.LogAllVerbose2() << program_timer.GetTimeString() << " - Generating topological sort.";
192 auto so_temp = TDG.GenerateTopologicalSort();
193 for (auto v : so_temp)
194 glob_linear_sweep_order.emplace_back(v);
195
196 if (glob_linear_sweep_order.empty())

199 << "Cyclic dependencies detected. Cycles need to be allowed"
200 << " by calling application.";
201 Exit(EXIT_FAILURE);
202 }
203 }
204
205 // Broadcast topological sort
206 mpi_comm.broadcast(glob_linear_sweep_order, 0);
207
208 // Compute reorder mapping
209 // This mapping allows us to punch in
210 // the location id and find what its
211 // id is in the TDG
212 std::vector<int> glob_order_mapping(opensn::mpi_comm.size(), -1);
213
214 for (int k = 0; k < opensn::mpi_comm.size(); k++)
215 {
216 int loc = glob_linear_sweep_order[k];
217 glob_order_mapping[loc] = k;
218 }
219
220 // Determine sweep order ranks
221 log.Log0Verbose1() << program_timer.GetTimeString() << " Determining sweep order ranks.";
222
223 std::vector<int> glob_sweep_order_rank(opensn::mpi_comm.size(), -1);
224
225 int abs_max_rank = 0;
226 for (int k = 0; k < opensn::mpi_comm.size(); k++)
227 {
228 int loc = glob_linear_sweep_order[k];
229 if (global_dependencies[loc].empty()) glob_sweep_order_rank[k] = 0;
230 else
231 {
232 int max_rank = -1;
233 for (auto dep_loc : global_dependencies[loc])
234 {
235 if (dep_loc < 0) continue;
236 int dep_mapped_index = glob_order_mapping[dep_loc];
237
238 if (glob_sweep_order_rank[dep_mapped_index] > max_rank)
239 max_rank = glob_sweep_order_rank[dep_mapped_index];
240 }
241 glob_sweep_order_rank[k] = max_rank + 1;
242 if ((max_rank + 1) > abs_max_rank) abs_max_rank = max_rank + 1;
243 }
244 }
245
246 // Generate TDG structure
247 log.Log0Verbose1() << program_timer.GetTimeString() << " Generating TDG structure.";
248 for (int r = 0; r <= abs_max_rank; r++)
249 {
250 STDG new_stdg;
251
252 for (int k = 0; k < opensn::mpi_comm.size(); k++)
253 {
254 if (glob_sweep_order_rank[k] == r) new_stdg.item_id.push_back(glob_linear_sweep_order[k]);
255 }
256 global_sweep_planes_.push_back(new_stdg);
257 }
