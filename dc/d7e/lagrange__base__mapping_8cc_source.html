30std::vector<std::vector<int>>
31LagrangeBaseMapping::MakeFaceNodeMapping(const Cell& cell)
32{
33 const size_t num_faces = cell.faces_.size();
34 std::vector<std::vector<int>> mappings;
35 mappings.reserve(num_faces);
36 for (auto& face : cell.faces_)
37 {
38 std::vector<int> face_dof_mapping;
39 face_dof_mapping.reserve(face.vertex_ids_.size());
40 for (uint64_t fvid : face.vertex_ids_)
41 {
42 int mapping = -1;
43 for (size_t ci = 0; ci < cell.vertex_ids_.size(); ci++)
44 {
45 if (fvid == cell.vertex_ids_[ci])
46 {
47 mapping = static_cast<int>(ci);
48 break;
49 }
50 } // for cell i
51 if (mapping < 0)
52 {
53 log.LogAllError() << "Unknown face mapping encountered. "
54 "pwl_polyhedron.h";
55 Exit(EXIT_FAILURE);
56 }
57 face_dof_mapping.push_back(mapping);
58 } // for face i
59
60 mappings.push_back(face_dof_mapping);
61 }
62 return mappings;
63}

122VolumetricQuadraturePointData
123LagrangeBaseMapping::MakeVolumetricQuadraturePointData() const
124{
125 typedef std::vector<Vec3> VecVec3;
126 const size_t num_qpoints = volume_quadrature_.qpoints_.size();
127
128 // Declare necessary vars
129 std::vector<unsigned int> quadrature_point_indices;
130 VecVec3 qpoints_xyz;
131 std::vector<VecDbl> shape_value;
132 std::vector<VecVec3> shape_grad;
133 VecDbl JxW;
134
135 quadrature_point_indices.reserve(num_qpoints);
136 for (unsigned int qp = 0; qp < num_qpoints; ++qp)
137 quadrature_point_indices.push_back(qp);
138
139 JxW.assign(num_qpoints, 0.0);
140 shape_value.assign(num_nodes_, VecDbl(num_qpoints, 0.0));
141 shape_grad.assign(num_nodes_, VecVec3(num_qpoints));
142 qpoints_xyz.assign(num_qpoints, Vec3{0.0, 0.0, 0.0});
143
144 for (uint32_t qp : quadrature_point_indices)
145 {
146 const Vec3& qpoint = volume_quadrature_.qpoints_[qp];
147 const auto J = RefJacobian(qpoint);
148 const auto JT = Transpose(J);
149 const auto JTinv = Inverse(JT);
150 const double detJ = Determinant(J);
151
152 JxW[qp] = detJ * volume_quadrature_.weights_[qp];
153
154 for (size_t i = 0; i < num_nodes_; ++i)
155 {
156 const double ref_shape_i = RefShape(i, qpoint);
157 shape_value[i][qp] = ref_shape_i;
158 const Vec3 ref_shape_grad = RefGradShape(i, qpoint);
159 const VecDbl b = {ref_shape_grad.x, ref_shape_grad.y, ref_shape_grad.z};
160 const VecDbl JTInv_b = MatMul(JTinv, b);
161
162 shape_grad[i][qp] = {JTInv_b[0], JTInv_b[1], JTInv_b[2]};
163
164 const auto& node_xyz = node_locations_[i];
165
166 qpoints_xyz[qp] += ref_shape_i * node_xyz;
167 } // for i
168 } // for qp
169
170 return VolumetricQuadraturePointData(quadrature_point_indices,
171 qpoints_xyz,
172 shape_value,
173 shape_grad,
174 JxW,
175 face_node_mappings_,
176 num_nodes_);
177}

274VecDbl
275WorldXYZToNaturalMappingHelper::F(const VecDbl& x) const
276{
277 const size_t num_nodes = cell_mapping_.NumNodes();
278 const auto& node_x = cell_mapping_.GetNodeLocations();
279
280 Vec3 result_vec3 = world_x_;
281 if (dimension_ == 1)
282 for (uint32_t i = 0; i < num_nodes; ++i)
283 {
284 const double shape_i_val = cell_mapping_.RefShape(i, {x[0], x[1], x[2]});
285 result_vec3 -= Vec3(0.0, 0.0, shape_i_val * node_x[i].z);
286 }
287 else if (dimension_ == 2)
288 for (uint32_t i = 0; i < num_nodes; ++i)
289 {
290 const double shape_i_val = cell_mapping_.RefShape(i, {x[0], x[1], x[2]});
291 result_vec3 -= Vec3(shape_i_val * node_x[i].x, shape_i_val * node_x[i].y, 0.0);
292 }
293 else if (dimension_ == 3)
294 for (uint32_t i = 0; i < num_nodes; ++i)
295 result_vec3 -= cell_mapping_.RefShape(i, {x[0], x[1], x[2]}) * node_x[i];
296
297 return {result_vec3.x, result_vec3.y, result_vec3.z};
298}
