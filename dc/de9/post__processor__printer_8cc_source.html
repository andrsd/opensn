133{

326void
327PostProcessorPrinter::PrintPPsTimeHistory(const std::string& pps_typename,
328 const std::vector<const PostProcessor*>& pp_list,
329 const Event& event,
330 bool per_column_sizes) const
331{
332 if (pp_list.empty()) return;
333 // Establish unique time history sizes
334 std::set<size_t> unq_time_histsizes;
335
336 for (const auto& pp : pp_list)
337 {
338 const size_t time_histsize = pp->GetTimeHistory().size();
339 unq_time_histsizes.insert(time_histsize);
340 }
341
342 // Subscribe pps to unique time hist sizes
343 std::map<size_t, std::vector<const PostProcessor*>> pp_timehist_size_subs;
344 for (size_t time_histsize : unq_time_histsizes)
345 {
346 auto& subs = pp_timehist_size_subs[time_histsize];
347 for (const auto& pp : pp_list)
348 if (pp->GetTimeHistory().size() == time_histsize) subs.push_back(pp);
349 }
350
351 // For each timeline. Build the table
352 for (const auto& [timehistsize, pp_sub_list] : pp_timehist_size_subs)
353 {
354 if (pp_sub_list.empty()) continue;
355
356 //+2 top header + bottom header
357 const size_t num_rows = std::min(size_t(time_history_limit_ + 2), timehistsize + 2);
358 const size_t num_cols = pp_sub_list.size() + 1; //+1 time column
359
360 auto value_matrix = BuildPPHistoryMatrix(timehistsize, time_history_limit_, pp_sub_list);
361
362 // Find largest column
363 size_t max_column_width = 0;
364 for (const auto& row : value_matrix)
365 for (const auto& entry : row)
366 max_column_width = std::max(max_column_width, entry.size());
367 max_column_width = std::max(max_column_width, size_t(15)); // minimum size 15
368
369 std::vector<size_t> col_sizes;
370 if (not per_column_sizes) col_sizes.assign(num_cols, max_column_width);
371 else
372 {
373 col_sizes.assign(num_cols, 0);
374 for (size_t j = 0; j < num_cols; ++j)
375 {
376 col_sizes[j] = value_matrix[0][j].size();
377 for (size_t t = 1; t < num_rows; ++t)
378 col_sizes[j] = std::max(col_sizes[j], value_matrix[t][j].size());
379 }
380 }
381
382 /**Lambda to left pad an entry.*/
383 auto LeftPad = [](std::string& entry, size_t width)
384 {
385 const size_t pad_size = width - entry.size();
386 entry = std::string(pad_size, ' ').append(entry);
387 };
388 /**Lambda to right pad an entry.*/
389 auto RightPad = [](std::string& entry, size_t width)
390 {
391 const size_t pad_size = width - entry.size();
392 entry.append(std::string(pad_size, ' '));
393 };
394
395 for (size_t j = 0; j < num_cols; ++j)
396 RightPad(value_matrix[0][j], col_sizes[j]);
397 for (size_t i = 1; i < num_rows; ++i)
398 for (size_t j = 0; j < num_cols; ++j)
399 LeftPad(value_matrix[i][j], col_sizes[j]);
400
401 // Build sub matrix structure
402 std::vector<size_t> sub_mat_sizes;
403 size_t total_width = 0;
404 size_t col_counter = 0;
405 for (size_t c = 0; c < num_cols; ++c)
406 {
407 //[0] | 0.000000 | 1.000000 | 0.000000e+00 |
408 // 5 chars for log, 2 for '| ', 1 for ' ', 2 for ' |' at end
409 const size_t projected_total_width = total_width + col_sizes[c] + 5 + 2 + 1 + 2;
410 if (projected_total_width > table_column_limit_)
411 {
412 total_width = col_sizes[c] + 5 + 2 + 1; // the time column
413 sub_mat_sizes.push_back(col_counter);
414 }
415 col_counter += 1;
416 total_width += col_sizes[c] + 2 + 1;
417 }
418 sub_mat_sizes.push_back(col_counter);
419
420 // Now actually build the sub-matrices
421 typedef std::vector<std::string> VecStr;
422 typedef std::vector<VecStr> MatStr;
423 std::vector<MatStr> sub_matrices;
424 size_t last_k = 1;
425 for (const size_t k : sub_mat_sizes)
426 {
427 MatStr sub_matrix(num_rows, VecStr(k - last_k + 1, ""));
428 // Copy time col
429 for (size_t i = 0; i < num_rows; ++i)
430 sub_matrix[i][0] = value_matrix[i][0];
431
432 // Copy other
433 for (size_t i = 0; i < num_rows; ++i)
434 {
435 size_t j_star = 1;
436 for (size_t j = last_k; j < k; ++j, ++j_star)
437 sub_matrix[i][j_star] = value_matrix[i][j];
438 }
439 last_k = k;
440 sub_matrices.push_back(std::move(sub_matrix));
441 }
442
443 std::stringstream outstr;
444 for (const auto& sub_matrix : sub_matrices)
445 outstr << PrintPPsSubTimeHistory(sub_matrix);
446
447 log.Log() << "\n"
448 << pps_typename << " post-processors history at event \"" << event.Name() << "\"\n"
449 << outstr.str();
450 } // for each thing in pp_timehist_size_subs
451}

454PostProcessorPrinter::PrintPPsSubTimeHistory(
455 const std::vector<std::vector<std::string>>& sub_history)
456{
457 const size_t num_rows = sub_history.size();
458 const size_t num_cols = sub_history.front().size();
459
460 std::stringstream output;
461 std::stringstream hline;
462 for (size_t k = 0; k < num_cols; ++k)
463 {
464 const size_t col_str_size = sub_history.front()[k].size();
465 hline << "*" << std::string(col_str_size + 2, '-');
466 }
467 hline << "*\n";
468
469 output << hline.str();
470 for (size_t i = 0; i < num_rows; ++i)

472 if (i == 1) output << hline.str();
473 std::stringstream line;
474 for (size_t k = 0; k < num_cols; ++k)

476 line << "|\n";
477 output << line.str();
478 if (i == (num_rows - 1)) output << hline.str();
479 }
480
481 return output.str();

642std::vector<std::vector<std::string>>
643PostProcessorPrinter::BuildPPHistoryMatrix(size_t timehistsize,
644 size_t time_history_limit,
645 const std::vector<const PostProcessor*>& pp_sub_list)
646{
647 if (pp_sub_list.empty()) return {};
648
649 //+2 top header + bottom header
650 const size_t num_rows = std::min(size_t(time_history_limit + 2), timehistsize + 2);
651 const size_t num_cols = pp_sub_list.size() + 1; //+1 time column
652 const size_t offset = std::max(0, int(timehistsize) - int(time_history_limit));
653
654 const auto& front_time_hist = pp_sub_list.front()->GetTimeHistory();
655
656 typedef std::vector<std::string> VecStr;
657 typedef std::vector<VecStr> MatStr;
658 MatStr value_matrix(num_rows, VecStr(num_cols, ""));
659
660 // Do the header first
661 value_matrix[0][0] = "Time";
662 for (size_t j = 1; j <= pp_sub_list.size(); ++j)
663 value_matrix[0][j] = pp_sub_list.at(j - 1)->Name();
664
665 // Now the time values
666 for (size_t t = 0; t < (num_rows - 2); ++t)
667 {
668 for (size_t j = 0; j <= pp_sub_list.size(); ++j)
669 {
670 if (j == 0) value_matrix[t + 1][j] = std::to_string(front_time_hist[t + offset].time_);
671 else
672 {
673 const auto& pp = pp_sub_list.at(j - 1);
674 value_matrix[t + 1][j] =
675 pp->ConvertValueToString(pp->GetTimeHistory().at(t + offset).value_);
676 }
677 } // for j
678 } // for t
679
