31AngleSetStatus
32AAH_AngleSet::AngleSetAdvance(SweepChunk& sweep_chunk,
33 const std::vector<size_t>& timing_tags,
34 ExecutionPermission permission)
35{
36 typedef AngleSetStatus Status;
37
38 if (executed_)
39 {
40 if (!async_comm_.DoneSending()) async_comm_.ClearDownstreamBuffers();
41 return AngleSetStatus::FINISHED;
42 }
43
44 // Check upstream data available
45 Status status = async_comm_.ReceiveUpstreamPsi(static_cast<int>(this->GetID()));
46
47 // Also check boundaries
48 for (auto& [bid, bndry] : ref_boundaries_)
49 if (not bndry->CheckAnglesReadyStatus(angles_, ref_group_subset_))
50 {
51 status = Status::RECEIVING;
52 break;
53 }
54
55 if (status == Status::RECEIVING) return status;
56 else if (status == Status::READY_TO_EXECUTE and permission == ExecutionPermission::EXECUTE)
57 {
58 async_comm_.InitializeLocalAndDownstreamBuffers();
59
60 log.LogEvent(timing_tags[0], Logger::EventType::EVENT_BEGIN);
61 sweep_chunk.Sweep(*this); // Execute chunk
62 log.LogEvent(timing_tags[0], Logger::EventType::EVENT_END);
63
64 // Send outgoing psi and clear local and receive buffers
65 async_comm_.SendDownstreamPsi(static_cast<int>(this->GetID()));
66 async_comm_.ClearLocalAndReceiveBuffers();
67
68 // Update boundary readiness
69 for (auto& [bid, bndry] : ref_boundaries_)
70 bndry->UpdateAnglesReadyStatus(angles_, ref_group_subset_);
71
72 executed_ = true;
73 return AngleSetStatus::FINISHED;
74 }
75 else
76 return AngleSetStatus::READY_TO_EXECUTE;
77}
