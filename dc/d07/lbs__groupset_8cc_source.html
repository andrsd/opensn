127lbs::LBSGroupset::LBSGroupset(const InputParameters& params,
128 const int id,
129 const LBSSolver& lbs_solver)
130 : Object(params), id_(id)
131{
132 const std::string fname = __FUNCTION__;
133
134 // Add groups
135 const auto groups_from_to = params.GetParamVectorValue<size_t>("groups_from_to");
136 ChiInvalidArgumentIf(groups_from_to.size() != 2,
137 "Parameter \"groups_from_to\" can only have 2 entries");
138
139 const size_t from = groups_from_to[0];
140 const size_t to = groups_from_to[1];
141 ChiInvalidArgumentIf(to < from, "\"to\" field is less than the \"from\" field.");
142
143 try
144 {
145 for (size_t g = from; g <= to; ++g)
146 {
147 groups_.push_back(lbs_solver.Groups().at(g));
148 } // for g
149 }
150 catch (const std::exception& exc)
151 {
152 throw std::invalid_argument("Attempting to group to groupset that is not"
153 "part of the solver.");
154 }
155
156 master_num_grp_subsets_ = params.GetParamValue<int>("groupset_num_subsets");
157
158 // Add quadrature
159 const size_t quad_handle = params.GetParamValue<size_t>("angular_quadrature_handle");
160 quadrature_ = GetStackItemPtr<AngularQuadrature>(angular_quadrature_stack, quad_handle, fname);
161
162 // Angle aggregation
163 const auto angle_agg_typestr = params.GetParamValue<std::string>("angle_aggregation_type");
164 if (angle_agg_typestr == "polar") angleagg_method_ = AngleAggregationType::POLAR;
165 else if (angle_agg_typestr == "single")
166 angleagg_method_ = AngleAggregationType::SINGLE;
167 else if (angle_agg_typestr == "azimuthal")
168 angleagg_method_ = AngleAggregationType::AZIMUTHAL;
169
170 master_num_ang_subsets_ = params.GetParamValue<int>("angle_aggregation_num_subsets");
171
172 // Inner solver
173 const auto inner_linear_method = params.GetParamValue<std::string>("inner_linear_method");
174 if (inner_linear_method == "richardson") iterative_method_ = IterativeMethod::KRYLOV_RICHARDSON;
175 else if (inner_linear_method == "gmres")
176 iterative_method_ = IterativeMethod::KRYLOV_GMRES;
177 else if (inner_linear_method == "bicgstab")
178 iterative_method_ = IterativeMethod::KRYLOV_BICGSTAB;
179
180 allow_cycles_ = params.GetParamValue<bool>("allow_cycles");
181 residual_tolerance_ = params.GetParamValue<double>("l_abs_tol");
182 max_iterations_ = params.GetParamValue<int>("l_max_its");
183
184 // Misc.
185 log_sweep_events_ = params.GetParamValue<bool>("log_sweep_events");
186
187 // DSA
188 apply_wgdsa_ = params.GetParamValue<bool>("apply_wgdsa");
189 apply_tgdsa_ = params.GetParamValue<bool>("apply_tgdsa");
190
191 wgdsa_tol_ = params.GetParamValue<double>("wgdsa_l_abs_tol");
192 tgdsa_tol_ = params.GetParamValue<double>("tgdsa_l_abs_tol");
193
194 wgdsa_max_iters_ = params.GetParamValue<int>("wgdsa_l_max_its");
195 tgdsa_max_iters_ = params.GetParamValue<int>("tgdsa_l_max_its");
196
197 wgdsa_verbose_ = params.GetParamValue<bool>("wgdsa_verbose");
198 tgdsa_verbose_ = params.GetParamValue<bool>("tgdsa_verbose");
199
200 wgdsa_string_ = params.GetParamValue<std::string>("wgdsa_petsc_options");
201 tgdsa_string_ = params.GetParamValue<std::string>("tgdsa_petsc_options");
202}

205lbs::LBSGroupset::BuildDiscMomOperator(unsigned int scattering_order,
206 lbs::GeometryType geometry_type)
207{
208 if (geometry_type == lbs::GeometryType::ONED_SLAB ||
209 geometry_type == lbs::GeometryType::ONED_CYLINDRICAL ||
210 geometry_type == lbs::GeometryType::ONED_SPHERICAL)
211 {
212 quadrature_->BuildDiscreteToMomentOperator(scattering_order, 1);
213 }
214 else if (geometry_type == lbs::GeometryType::TWOD_CARTESIAN ||
215 geometry_type == lbs::GeometryType::TWOD_CYLINDRICAL)
216 {
217 quadrature_->BuildDiscreteToMomentOperator(scattering_order, 2);
218 }
219 else if (geometry_type == lbs::GeometryType::THREED_CARTESIAN)
220 {
221 quadrature_->BuildDiscreteToMomentOperator(scattering_order, 3);
222 }

261void
262lbs::LBSGroupset::PrintSweepInfoFile(size_t ev_tag, const std::string& file_name)
263{
264 if (not log_sweep_events_) return;
265
266 std::ofstream ofile;
267 ofile.open(file_name, std::ofstream::out);
268
269 ofile << "Groupset Sweep information "
270 << "location " << opensn::mpi_comm.rank() << "\n";
271
272 // Print all anglesets
273 for (int q = 0; q < angle_agg_->angle_set_groups.size(); ++q)
274 {
275 ofile << "Angle-set group " << q << ":\n";
276 auto& ang_set_grp = angle_agg_->angle_set_groups[q];
277 int num_ang_sets_per_grp = (int)ang_set_grp.AngleSets().size();
278 for (int as = 0; as < num_ang_sets_per_grp; ++as)
279 {
280 auto ang_set = ang_set_grp.AngleSets()[as];
281
282 int ang_set_num = as + q * num_ang_sets_per_grp;
283
284 ofile << " Angle-set " << ang_set_num << " angles [# varphi theta]:\n";
285
286 for (auto& ang_num : ang_set->GetAngleIndices())
287 {
288 const auto& angle = quadrature_->abscissae_[ang_num];
289
290 ofile << " " << ang_num << " " << angle.phi << " " << angle.theta << "\n";
291 }
292 }
293 }
294
295 // Print event history
296 ofile << log.PrintEventHistory(ev_tag);
297
298 ofile.close();
299}
