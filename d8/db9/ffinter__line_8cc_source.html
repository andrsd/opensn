14void
15FieldFunctionInterpolationLine::Initialize()
16{
17 log.Log0Verbose1() << "Initializing line interpolator.";
18 // Check for empty FF-list
19 if (field_functions_.empty())
20 throw std::logic_error("Unassigned field function in line "
21 "field function interpolator.");
22
23 // Create points;
24 const Vector3 vif = pf_ - pi_;
25 delta_d_ = vif.Norm() / (number_of_points_ - 1);
26
27 const auto omega = vif.Normalized();
28
29 interpolation_points_.push_back(pi_);
30 for (int k = 1; k < (number_of_points_); k++)
31 interpolation_points_.push_back(pi_ + omega * delta_d_ * k);
32
33 // Loop over contexts
34 const size_t num_ff = field_functions_.size();
35 for (size_t ff = 0; ff < num_ff; ff++)
36 {
37 ff_contexts_.emplace_back();
38 auto& ff_context = ff_contexts_.back();
39
40 ff_context.ref_ff = field_functions_[ff];
41 const auto& sdm = ff_context.ref_ff->GetSpatialDiscretization();
42 const auto& grid = sdm.Grid();
43
44 ff_context.interpolation_points_ass_cell.assign(number_of_points_, 0);
45 ff_context.interpolation_points_has_ass_cell.assign(number_of_points_, false);
46
47 // Find a home for each point
48 for (const auto& cell : grid.local_cells)
49 {
50 for (int p = 0; p < number_of_points_; p++)
51 {
52 const auto& point = interpolation_points_[p];
53 if (grid.CheckPointInsideCell(cell, point))
54 {
55 ff_context.interpolation_points_ass_cell[p] = cell.local_id_;
56 ff_context.interpolation_points_has_ass_cell[p] = true;
57 }
58 } // for point p
59 } // for cell
60 } // for ff
61
62 log.Log0Verbose1() << "Finished initializing interpolator.";
63}

65void
66FieldFunctionInterpolationLine::Execute()
67{
68 log.Log0Verbose1() << "Executing line interpolator.";
69 for (int ff = 0; ff < field_functions_.size(); ff++)
70 {
71 auto& ff_ctx = ff_contexts_[ff];
72 const auto& ref_ff = *ff_ctx.ref_ff;
73 const auto& sdm = ref_ff.GetSpatialDiscretization();
74 const auto& grid = sdm.Grid();
75
76 const auto& uk_man = ref_ff.GetUnknownManager();
77 const auto uid = 0;
78 const auto cid = ref_component_;
79
80 const auto field_data = ref_ff.GetGhostedFieldVector();
81
82 ff_ctx.interpolation_points_values.assign(number_of_points_, 0.0);
83 for (int p = 0; p < number_of_points_; ++p)
84 {
85 if (not ff_ctx.interpolation_points_has_ass_cell[p]) continue;
86
87 const auto cell_local_index = ff_ctx.interpolation_points_ass_cell[p];
88 const auto& cell = grid.local_cells[cell_local_index];
89 const auto& cell_mapping = sdm.GetCellMapping(cell);
90 const size_t num_nodes = cell_mapping.NumNodes();
91
92 std::vector<double> shape_function_vals(num_nodes, 0.0);
93 cell_mapping.ShapeValues(interpolation_points_[p], shape_function_vals);
94
95 double point_value = 0.0;
96 for (size_t i = 0; i < num_nodes; ++i)
97 {
98 const int64_t imap = sdm.MapDOFLocal(cell, i, uk_man, uid, cid);
99
100 point_value += shape_function_vals[i] * field_data[imap];
101 } // for node i
102 ff_ctx.interpolation_points_values[p] = point_value;
103 } // for p
104 } // for ff
105}
