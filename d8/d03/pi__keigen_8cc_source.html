88void
89XXPowerIterationKEigen::Execute()
90{
91 double F_prev = 1.0;
92 k_eff_ = 1.0;
93 double k_eff_prev = 1.0;
94 double k_eff_change = 1.0;
95
96 // Start power iterations
97 int nit = 0;
98 bool converged = false;
99 while (nit < max_iters_)
100 {
101 // Set the fission source
102 SetLBSFissionSource(phi_old_local_, false);
103 Scale(q_moments_local_, 1.0 / k_eff_);
104
105 // This solves the inners for transport
106 primary_ags_solver_->Setup();
107 primary_ags_solver_->Solve();
108
109 // Recompute k-eigenvalue
110 double F_new = lbs_solver_.ComputeFissionProduction(phi_new_local_);
111 k_eff_ = F_new / F_prev * k_eff_;
112 double reactivity = (k_eff_ - 1.0) / k_eff_;
113
114 // Check convergence, bookkeeping
115 k_eff_change = fabs(k_eff_ - k_eff_prev) / k_eff_;
116 k_eff_prev = k_eff_;
117 F_prev = F_new;
118 nit += 1;
119
120 if (k_eff_change < std::max(k_tolerance_, 1.0e-12)) converged = true;
121
122 // Print iteration summary
123 if (lbs_solver_.Options().verbose_outer_iterations)
124 {
125 std::stringstream k_iter_info;
126 k_iter_info << program_timer.GetTimeString() << " "
127 << " Iteration " << std::setw(5) << nit << " k_eff " << std::setw(11)
128 << std::setprecision(7) << k_eff_ << " k_eff change " << std::setw(12)
129 << k_eff_change << " reactivity " << std::setw(10) << reactivity * 1e5;
130 if (converged) k_iter_info << " CONVERGED\n";
131
132 log.Log() << k_iter_info.str();
133 }
134
135 if (converged) break;
136 } // for k iterations
137
138 // Print summary
139 log.Log() << "\n";
140 log.Log() << " Final k-eigenvalue : " << std::setprecision(7) << k_eff_;
141 log.Log() << " Final change : " << std::setprecision(6) << k_eff_change
142 << " (num_TrOps:" << front_wgs_context_->counter_applications_of_inv_op_ << ")"
143 << "\n";
144 log.Log() << "\n";
145
146 if (lbs_solver_.Options().use_precursors)
147 {
148 lbs_solver_.ComputePrecursors();
149 Scale(lbs_solver_.PrecursorsNewLocal(), 1.0 / k_eff_);
150 }
151
152 lbs_solver_.UpdateFieldFunctions();
153
154 log.Log() << "LinearBoltzmann::KEigenvalueSolver execution completed\n\n";
155}
