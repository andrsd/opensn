116{

127void
128SweepChunkPWLRZ::KernelFEMRZVolumetricGradientTerm()
129{
130 const auto& G = *G_;
131 const auto& Maux = *Maux_;
132
133 for (int i = 0; i < cell_num_nodes_; ++i)
134 for (int j = 0; j < cell_num_nodes_; ++j)
135 {
136 Amat_[i][j] = omega_.Dot(G[i][j]) + fac_streaming_operator_ * Maux[i][j];
137 const auto jr = grid_fe_view_.MapDOFLocal(*cell_, j, unknown_manager_, polar_level_, gs_gi_);
138 for (int gsg = 0; gsg < gs_ss_size_; ++gsg)
139 b_[gsg][i] += fac_streaming_operator_ * Maux[i][j] * psi_sweep_[jr + gsg];
140 }

143void
144SweepChunkPWLRZ::KernelFEMRZUpwindSurfaceIntegrals()
145{
146 const size_t f = sweep_dependency_interface_.current_face_idx_;
147 if (sweep_dependency_interface_.on_boundary_)
148 {
149 const auto& face_normal = cell_->faces_[f].normal_;
150 //------------------------------------------------------------------
151 // determine whether incoming direction is incident on the point
152 // of symmetry or on the axis of symmetry
153 // N.B.: a face is considered to be on the point/axis of symmetry
154 // if all are true:
155 // 1. the face normal is antiparallel to $\vec{e}_{d}$
156 // 2. all vertices of the face exhibit $v_{d} = 0$
157 // with $d = 2$ for 1D geometries and $d = 0$ for 2D geometries.
158 // Thanks to the verifications performed during initialisation,
159 // at this point it is necessary to confirm only the orientation.
160 const bool incident_on_symmetric_boundary =
161 (face_normal.Dot(normal_vector_boundary_) < -0.999999);
162 if (incident_on_symmetric_boundary) return;
163 }
164
165 const auto& M_surf_f = (*M_surf_)[f];
166 const double mu = face_mu_values_[f];
167 const size_t num_face_nodes = cell_mapping_->NumFaceNodes(f);
168 for (int fi = 0; fi < num_face_nodes; ++fi)
169 {
170 const int i = cell_mapping_->MapFaceNode(f, fi);
171 for (int fj = 0; fj < num_face_nodes; ++fj)
172 {
173 const int j = cell_mapping_->MapFaceNode(f, fj);
174
175 const double* psi = sweep_dependency_interface_.GetUpwindPsi(fj);
176
177 const double mu_Nij = -mu * M_surf_f[i][j];
178 Amat_[i][j] += mu_Nij;
179 for (int gsg = 0; gsg < gs_ss_size_; ++gsg)
180 b_[gsg][i] += psi[gsg] * mu_Nij;
181 } // for face node j
182 } // for face node i
183}
