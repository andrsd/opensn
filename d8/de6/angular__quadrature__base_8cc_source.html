14void
15AngularQuadrature::OptimizeForPolarSymmetry(const double normalization)
16{
17 std::vector<QuadraturePointPhiTheta> new_abscissae;
18 std::vector<double> new_weights;
19 std::vector<Vector3> new_omegas;
20
21 const size_t num_dirs = omegas_.size();
22 double weight_sum = 0.0;
23 for (size_t d = 0; d < num_dirs; ++d)
24 if (omegas_[d].z > 0.0)
25 {
26 new_abscissae.emplace_back(abscissae_[d]);
27 new_weights.emplace_back(weights_[d]);
28 new_omegas.emplace_back(omegas_[d]);
29 weight_sum += weights_[d];
30 }
31
32 if (normalization > 0.0)
33 for (double& w : new_weights)
34 w *= normalization / weight_sum;
35
36 abscissae_ = std::move(new_abscissae);
37 weights_ = std::move(new_weights);
38 omegas_ = std::move(new_omegas);
39}

42AngularQuadrature::MakeHarmonicIndices(unsigned int scattering_order, int dimension)
43{
44 m_to_ell_em_map_.clear();
45
46 if (dimension == 1)
47 for (int ell = 0; ell <= scattering_order; ell++)
48 m_to_ell_em_map_.emplace_back(ell, 0);
49 else if (dimension == 2)
50 for (int ell = 0; ell <= scattering_order; ell++)
51 for (int m = -ell; m <= ell; m += 2)
52 m_to_ell_em_map_.emplace_back(ell, m);
53 else if (dimension == 3)
54 for (int ell = 0; ell <= scattering_order; ell++)
55 for (int m = -ell; m <= ell; m++)
56 m_to_ell_em_map_.emplace_back(ell, m);

105AngularQuadrature::BuildMomentToDiscreteOperator(unsigned int scattering_order, int dimension)
106{
107 if (m2d_op_built_) return;
108
109 m2d_op_.clear();
110 MakeHarmonicIndices(scattering_order, dimension);
111
112 const size_t num_angles = abscissae_.size();
113 const size_t num_moms = m_to_ell_em_map_.size();
114
115 const auto normalization = std::accumulate(weights_.begin(), weights_.end(), 0.0);
116
117 for (const auto& ell_em : m_to_ell_em_map_)
118 {
119 std::vector<double> cur_mom;
120 cur_mom.reserve(num_angles);
121
122 for (int n = 0; n < num_angles; n++)
123 {
124 const auto& cur_angle = abscissae_[n];
125 double value = ((2.0 * ell_em.ell + 1.0) / normalization) *
126 Ylm(ell_em.ell, ell_em.m, cur_angle.phi, cur_angle.theta);
127 cur_mom.push_back(value);
128 }
129
130 m2d_op_.push_back(cur_mom);
131 } // for m
132 m2d_op_built_ = true;
133
134 // Verbose printout
135 std::stringstream outs;
136
137 outs << "\nQuadrature m2d operator:\n";

185AngularQuadratureCustom::AngularQuadratureCustom(std::vector<double>& azimuthal,
186 std::vector<double>& polar,
187 std::vector<double>& in_weights,
188 bool verbose)
189{
190 size_t Na = azimuthal.size();
191 size_t Np = polar.size();
192 size_t Nw = in_weights.size();
193
194 if ((Na - Np != 0) or (Na - Nw != 0))
195 {
196 log.LogAllError() << "AngularQuadrature::InitializeWithCustom: supplied"
197 " vectors need to be of equal length.";
198 Exit(EXIT_FAILURE);
199 }
200
201 // Create angle pairs
202 std::stringstream ostr;
203 double weight_sum = 0.0;
204
205 for (unsigned int i = 0; i < Na; i++)
206 {
207 const auto abscissa = QuadraturePointPhiTheta(azimuthal[i], polar[i]);
208
209 abscissae_.push_back(abscissa);
210
211 const double weight = in_weights[i];
212 weights_.push_back(weight);
213 weight_sum += weight;
214
215 if (verbose)
216 {
217 char buf[200];
218 snprintf(buf,
219 200,
220 "Varphi=%.2f Theta=%.2f Weight=%.3e\n",
221 abscissa.phi * 180.0 / M_PI,
222 abscissa.theta * 180.0 / M_PI,
223 weight);
224 ostr << buf;
225 }
226 }
227
228 // Create omega list
229 for (const auto& qpoint : abscissae_)
230 {
231 Vector3 new_omega;
232 new_omega.x = sin(qpoint.theta) * cos(qpoint.phi);
233 new_omega.y = sin(qpoint.theta) * sin(qpoint.phi);
234 new_omega.z = cos(qpoint.theta);
235
236 omegas_.push_back(new_omega);
237 }
238
239 if (verbose)
240 {
241 log.Log() << ostr.str() << "\n"
242 << "Weight sum=" << weight_sum;
243 }
244}
