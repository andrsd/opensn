27/**A ParameterBlock is a conceptually simple data structure that supports
28 * a hierarchy of primitive parameters. There really are just 4 member variables
29 * on a ParameterBlock object, they are 1) the type (as an enum), 2) the
30 * name of the block, 3) a pointer to a value (which can only be a primitive
31 * type), and 4) a vector of child parameters.
32 *
33 * If a ParameterBlock has a primitive type, i.e., BOOLEAN, FLOAT, STRING, or
34 * INTEGER, then the value_ptr will contain a pointer to the value of a
35 * primitive type. Otherwise, for types ARRAY and BLOCK, the ParameterBlock
36 * will not have a value_ptr and instead the vector member will contain
37 * sub-parameters.*/

225 /**Converts the parameters of an array-type parameter block to a vector of
226 * primitive types and returns it.*/
227 template <typename T>
228 std::vector<T> GetVectorValue() const
229 {
230 if (Type() != ParameterBlockType::ARRAY)
231 throw std::logic_error(error_origin_scope_ + std::string(__PRETTY_FUNCTION__) +
232 ": Invalid type requested for parameter of type " +
233 ParameterBlockTypeName(Type()));
234
235 std::vector<T> vec;
236 if (parameters_.empty()) return vec;
237
238 // Check the first sub-param is of the right type
239 const auto& front_param = parameters_.front();
240
241 // Check that all other parameters are of the required type
242 for (const auto& param : parameters_)
243 if (param.Type() != front_param.Type())
244 throw std::logic_error(error_origin_scope_ + " " + std::string(__PRETTY_FUNCTION__) +
245 ": Parameter \"" + name_ +
246 "\", cannot construct vector from block because "
247 "the sub_parameters do not all have the correct type. param->" +
248 ParameterBlockTypeName(param.Type()) + " vs param0->" +
249 ParameterBlockTypeName(front_param.Type()));
250
251 const size_t num_params = parameters_.size();
252 for (size_t k = 0; k < num_params; ++k)
253 {
254 const auto& param = GetParam(k);
255 vec.push_back(param.GetValue<T>());
256 }
257
258 return vec;
259 }
