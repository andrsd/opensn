45LagrangeBaseMapping::MatDbl
46LagrangeTriangleMapping::RefJacobian(const Vec3& qpoint) const
47{
48 MatDbl J(3, VecDbl(3, 0.0));
49 for (size_t i = 0; i < num_nodes_; ++i)
50 {
51 const Vec3 grad_shape_i = RefGradShape(i, qpoint);
52 const auto& node_i = node_locations_[i];
53 const double x_i = node_i.x;
54 const double y_i = node_i.y;
55
56 // Loops unrolled for performance
57 J[0][0] += grad_shape_i.x * x_i;
58 J[0][1] += grad_shape_i.y * x_i;
59 J[1][0] += grad_shape_i.x * y_i;
60 J[1][1] += grad_shape_i.y * y_i;
61 }
62 J[2][2] = 1.0;
63
64 return J;
65}

67std::pair<double, LagrangeBaseMapping::Vec3>
68LagrangeTriangleMapping::RefFaceJacobianDeterminantAndNormal(size_t face_index,
69 const Vec3& qpoint_face) const
70{
71 // x = sum_i N_i x_i
72 const auto& x0 = node_locations_[face_node_mappings_[face_index][0]];
73 const auto& x1 = node_locations_[face_node_mappings_[face_index][1]];
74 Vec3 dx_dxbar;
75 for (size_t i = 0; i < 2; ++i)
76 {
77 if (i == 0) dx_dxbar += -0.5 * x0;
78 if (i == 1) dx_dxbar += 0.5 * x1;
79 }
80
81 const auto cross = dx_dxbar.Cross(Vec3(0.0, 0.0, 1.0));
82
83 return {dx_dxbar.Norm(), cross.Normalized()};
