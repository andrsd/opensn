46 const auto& matid_to_src_map = lbs_solver_.GetMatID2IsoSrcMap();
47
48 const size_t num_moments = lbs_solver_.NumMoments();
49 const auto& ext_src_moments_local = lbs_solver_.ExtSrcMomentsLocal();

53 // Loop over local cells
54 const auto& grid = lbs_solver_.Grid();

61 // Obtain xs
62 const auto& xs = transport_view.XS();
63
64 std::shared_ptr<IsotropicMultiGrpSource> P0_src = nullptr;

66 P0_src = matid_to_src_map.at(cell.material_id_);
67
68 const auto& S = xs.TransferMatrices();

73 // Loop over nodes

77 // Loop over moments

79 {

81
82 size_t uk_map = transport_view.MapDOF(i, m, 0); // unknown map
83
84 const double* phi = &phi_local[uk_map];
85
86 // Declare moment src
87 if (P0_src and ell == 0) fixed_src_moments_ = P0_src->source_value_g_.data();
88 else
89 fixed_src_moments_ = default_zero_src_.data();
90
91 if (lbs_solver_.Options().use_src_moments)
92 fixed_src_moments_ = &ext_src_moments_local[uk_map];
93
94 // Loop over groupset groups
95 for (size_t g = gs_i_; g <= gs_f_; ++g)
96 {
97 g_ = g;
98
99 double rhs = 0.0;
100
101 // Apply fixed sources
102 if (apply_fixed_src_) rhs += this->AddSourceMoments();
103
104 // Apply scattering sources
105 if (ell < S.size())
106 {
107 const auto& S_ell = S[ell];
108 // Add Across GroupSet Scattering (AGS)
109 if (apply_ags_scatter_src_)
110 for (const auto& [_, gp, sigma_sm] : S_ell.Row(g))
111 if (gp < gs_i_ or gp > gs_f_) rhs += sigma_sm * phi[gp];
112
113 // Add Within GroupSet Scattering (WGS)
114 if (apply_wgs_scatter_src_)
115 for (const auto& [_, gp, sigma_sm] : S_ell.Row(g))
116 if (gp >= gs_i_ and gp <= gs_f_)
117 {
118 if (suppress_wg_scatter_src_ and g_ == gp) continue;
119 rhs += sigma_sm * phi[gp];
120 }
121 }
122
123 // Apply fission sources
124 const bool fission_avail = ell == 0 and xs.IsFissionable();
125
126 if (fission_avail)
127 {
128 const auto& F_g = F[g];
129 if (apply_ags_fission_src_)
130 for (size_t gp = first_grp_; gp <= last_grp_; ++gp)
131 if (gp < gs_i_ or gp > gs_f_) rhs += F_g[gp] * phi[gp];
132
133 if (apply_wgs_fission_src_)
134 for (size_t gp = gs_i_; gp <= gs_f_; ++gp)
135 rhs += F_g[gp] * phi[gp];
136
137 if (lbs_solver_.Options().use_precursors)
138 rhs += this->AddDelayedFission(precursors, nu_delayed_sigma_f, &phi_local[uk_map]);
139 }
140
141 // Add to destination vector
142 destination_q[uk_map + g] += rhs;
143
144 } // for g
145 } // for m
146 } // for dof i
147 } // for cell
148
149 AddAdditionalSources(groupset, destination_q, phi_local, source_flags);
150
151 log.LogEvent(source_event_tag, Logger::EventType::EVENT_END);

160double
161SourceFunction::AddDelayedFission(const PrecursorList& precursors,
162 const std::vector<double>& nu_delayed_sigma_f,
163 const double* phi) const
164{
165 double value = 0.0;
166 if (apply_ags_fission_src_)
167 for (size_t gp = first_grp_; gp <= last_grp_; ++gp)
168 if (gp < gs_i_ or gp > gs_f_)
169 for (const auto& precursor : precursors)
170 value += precursor.emission_spectrum[g_] * precursor.fractional_yield *
171 nu_delayed_sigma_f[gp] * phi[gp];
172
173 if (apply_wgs_fission_src_)
174 for (size_t gp = gs_i_; gp <= gs_f_; ++gp)
175 for (const auto& precursor : precursors)
176 value += precursor.emission_spectrum[g_] * precursor.fractional_yield *
177 nu_delayed_sigma_f[gp] * phi[gp];
178
179 return value;
180}

183SourceFunction::AddPointSources(LBSGroupset& groupset,
184 std::vector<double>& destination_q,
185 const std::vector<double>&,

187{
188 const bool apply_fixed_src = (source_flags & APPLY_FIXED_SOURCES);

190 const auto& cell_transport_views = lbs_solver_.GetCellTransportViews();
191
192 const auto gs_i = static_cast<size_t>(groupset.groups_.front().id_);

194
195 // Apply point sources
196 if (not lbs_solver_.Options().use_src_moments and apply_fixed_src)
197 for (const auto& point_source : lbs_solver_.PointSources())
198 {
199 const auto& info_list = point_source.ContainingCellsInfo();
200 for (const auto& info : info_list)
201 {
202 auto& transport_view = cell_transport_views[info.cell_local_id];
203
204 const auto& strength = point_source.Strength();
205 const auto& node_weights = info.node_weights;
206 const double vol_w = info.volume_weight;
207
208 const int num_nodes = transport_view.NumNodes();
209 for (int i = 0; i < num_nodes; ++i)
210 {
211 const size_t uk_map = transport_view.MapDOF(i, 0, 0);
212 for (size_t g = gs_i; g <= gs_f; ++g)
213 destination_q[uk_map + g] += strength[g] * node_weights[i] * vol_w;
214 } // for node i
215 } // for cell
216 } // for point source
217}
