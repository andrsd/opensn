58double
59PieceWiseLinearSlabMapping::ShapeValue(const int i, const Vector3& xyz) const
60{
61 const auto& p0 = ref_grid_.vertices[v0i_];
62 const auto& p1 = ref_grid_.vertices[v1i_];
63 Vector3 xyz_ref = xyz - p0;
64
65 Vector3 v01 = p1 - p0;
66
67 double xi = v01.Dot(xyz_ref) / v01.Norm() / h_;
68
69 if ((xi >= -1.0e-6) and (xi <= 1.0 + 1.0e-6))
70 {
71 if (i == 0) return 1.0 - xi;
72 else
73 return xi;
74 } // if in cell
75
76 return 0.0;

121VolumetricQuadraturePointData
122PieceWiseLinearSlabMapping::MakeVolumetricQuadraturePointData() const
123{
124 // Determine number of internal qpoints
125 size_t ttl_num_vol_qpoints = volume_quadrature_.qpoints_.size();
126
127 // Declare necessary vars
128 std::vector<unsigned int> V_quadrature_point_indices;

132 VecDbl V_JxW;
133 size_t V_num_nodes;
134
135 // Init volumetric quadrature
136 V_quadrature_point_indices.reserve(ttl_num_vol_qpoints);
137 for (unsigned int qp = 0; qp < ttl_num_vol_qpoints; ++qp)
138 V_quadrature_point_indices.push_back(qp);
139
140 V_shape_value.reserve(num_nodes_);
141 V_shape_grad.reserve(num_nodes_);
142 for (size_t i = 0; i < num_nodes_; i++)
143 {
144 VecDbl node_shape_value;
145 VecVec3 node_shape_grad;
146
147 node_shape_value.reserve(ttl_num_vol_qpoints);
148 node_shape_grad.reserve(ttl_num_vol_qpoints);
149
150 for (const auto& qpoint : volume_quadrature_.qpoints_)
151 {
152 node_shape_value.push_back(SlabShape(i, qpoint));
153 node_shape_grad.emplace_back(0.0, 0.0, SlabGradShape(i));
154 } // for qp
155
156 V_shape_value.push_back(node_shape_value);
157 V_shape_grad.push_back(node_shape_grad);
158 } // for i
159
160 V_JxW.reserve(ttl_num_vol_qpoints);
161 V_qpoints_xyz.reserve(ttl_num_vol_qpoints);
162 const double J = h_;
163 for (size_t qp = 0; qp < ttl_num_vol_qpoints; ++qp)
164 {
165 const double w = volume_quadrature_.weights_[qp];
166 V_JxW.push_back(J * w);
167
168 const double qp_xyz_tilde = volume_quadrature_.qpoints_[qp][0];
169 V_qpoints_xyz.push_back(v0_ + J * Vector3(0.0, 0.0, qp_xyz_tilde));
170 } // for qp
171
172 V_num_nodes = num_nodes_;
173
174 return VolumetricQuadraturePointData(V_quadrature_point_indices,
175 V_qpoints_xyz,
176 V_shape_value,
177 V_shape_grad,
178 V_JxW,
179 face_node_mappings_,
180 V_num_nodes);
