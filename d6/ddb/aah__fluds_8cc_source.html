47AAH_FLUDS::NLOutgoingPsi(int outb_face_counter, int face_dof, int n)
48{
49 if (outb_face_counter > common_data_.nonlocal_outb_face_deplocI_slot.size())
50 {
51 log.LogAllError() << "Invalid number of outb_face_counter " << outb_face_counter
52 << " max allowed " << common_data_.nonlocal_outb_face_deplocI_slot.size();
53 Exit(EXIT_FAILURE);
54 }
55
56 int depLocI = common_data_.nonlocal_outb_face_deplocI_slot[outb_face_counter].first;
57 int slot = common_data_.nonlocal_outb_face_deplocI_slot[outb_face_counter].second;
58 int nonlocal_psi_Gn_blockstride = common_data_.deplocI_face_dof_count[depLocI];
59
60 int index =
61 nonlocal_psi_Gn_blockstride * num_groups_ * n + slot * num_groups_ + face_dof * num_groups_;
62
63 if ((index < 0) || (index > deplocI_outgoing_psi_[depLocI].size()))
64 {
65 log.LogAllError() << "Invalid index " << index << " encountered in non-local outgoing Psi"
66 << " max allowed " << deplocI_outgoing_psi_[depLocI].size();
67 Exit(EXIT_FAILURE);
68 }
69
70 return &deplocI_outgoing_psi_[depLocI][index];
