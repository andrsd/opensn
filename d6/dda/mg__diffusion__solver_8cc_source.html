193void
194Solver::Initialize_Materials(std::set<int>& material_ids)
195{
196 log.Log0Verbose1() << "Initializing Materials";
197
198 std::stringstream materials_list;
199
200 //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% Process materials found
201 const size_t num_physics_mats = material_stack.size();
202 bool first_material_read = true;
203
204 for (const int& mat_id : material_ids)
205 {
206 auto current_material = GetStackItemPtr(material_stack, mat_id, __FUNCTION__);
207 materials_list << "Material id " << mat_id;
208
209 // Check valid ids
210 if (mat_id < 0)
211 {
212 throw std::logic_error("MG-diff-InitMaterials: Cells encountered with no assigned material.");
213 }
214 if (static_cast<size_t>(mat_id) >= num_physics_mats)
215 {
216 throw std::logic_error("MG-diff-InitMaterials: Cells encountered with "
217 "material id that matches no material in physics material library.");
218 }
219
220 // Extract properties
221 using MatProperty = PropertyType;
222 bool found_transport_xs = false;
223 for (const auto& property : current_material->properties_)
224 {
225 if (property->Type() == MatProperty::TRANSPORT_XSECTIONS)
226 {
227 auto transp_xs = std::static_pointer_cast<MultiGroupXS>(property);
228 matid_to_xs_map[mat_id] = transp_xs;
229 found_transport_xs = true;
230 if (first_material_read) num_groups_ = transp_xs->NumGroups();
231
232 } // transport xs
233 if (property->Type() == MatProperty::ISOTROPIC_MG_SOURCE)
234 {
235 auto mg_source = std::static_pointer_cast<IsotropicMultiGrpSource>(property);
236
237 if (mg_source->source_value_g_.size() < num_groups_)
238 {
239 log.LogAllWarning() << "MG-Diff-InitMaterials: Isotropic Multigroup source specified "
240 "in "
241 << "material \"" << current_material->name_ << "\" has fewer "
242 << "energy groups than called for in the simulation. "
243 << "Source will be ignored.";
244 }
245 else { matid_to_src_map[mat_id] = mg_source; }
246 } // P0 source
247 } // for property
248
249 // Check valid property
250 if (!found_transport_xs)
251 {
252 log.LogAllError() << "MG-Diff-InitMaterials: Found no transport cross-section property "
253 "for "
254 << "material \"" << current_material->name_ << "\".";
255 Exit(EXIT_FAILURE);
256 }
257 // Check number of groups legal
258 if (matid_to_xs_map[mat_id]->NumGroups() != num_groups_)
259 {
260 log.LogAllError() << "MG-Diff-InitMaterials: Found material \"" << current_material->name_
261 << "\" has " << matid_to_xs_map[mat_id]->NumGroups() << " groups and "
262 << "the simulation has " << num_groups_ << " groups. The material "
263 << "must have the same number of groups.";
264 Exit(EXIT_FAILURE);
265 }
266
267 // Check number of moments
268 if (matid_to_xs_map[mat_id]->ScatteringOrder() > 1)
269 {
270 log.Log0Warning() << "MG-Diff-InitMaterials: Found material \"" << current_material->name_
271 << "\" has a scattering order of "
272 << matid_to_xs_map[mat_id]->ScatteringOrder() << " and"
273 << " the simulation has a scattering order of One (MG-Diff)"
274 << " The higher moments will therefore not be used.";
275 }
276
277 materials_list << " number of moments " << matid_to_xs_map[mat_id]->ScatteringOrder() + 1
278 << "\n";
279
280 first_material_read = false;
281 } // for material id
282
283 log.Log() << "Materials Initialized:\n" << materials_list.str() << "\n";
284
285 opensn::mpi_comm.barrier();
286
287 //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% Compute last fast group
288 // initialize last fast group
289 log.Log() << "Computing last fast group.";
290 unsigned int lfg = num_groups_;
291
292 if (num_groups_ > 1)
293 {
294 // loop over all materials
295 for (const auto& mat_id_xs : matid_to_xs_map)
296 {
297 // get the P0 transfer matrix
298 const auto& S = mat_id_xs.second->TransferMatrix(0);
299 // loop over all row of the transfer matrix
300 const int G = static_cast<int>(num_groups_);
301 for (int g = G - 1; g >= 0; --g)
302 {
303 for (const auto& [row_g, gp, sigma_sm] : S.Row(g))
304 {
305 if ((std::fabs(sigma_sm) > 1e-10) && (gp > row_g))
306 lfg = std::min(lfg, static_cast<unsigned int>(row_g));
307 }
308 }
309 } // loop on materials
310 } // if num_groups>1
311
312 last_fast_group_ = lfg;
313
314 //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% Compute two-grid params
315 do_two_grid_ = basic_options_("do_two_grid").BoolValue();
316 if ((lfg == num_groups_) && do_two_grid_)
317 {
318 log.Log0Error() << "Two-grid is not possible with no upscattering.";
319 do_two_grid_ = false;
320 Exit(EXIT_FAILURE);
321 }
322 if (do_two_grid_)
323 {
324 log.Log() << "Compute_TwoGrid_Params";
325 Compute_TwoGrid_Params();
326 }

329void
330Solver::Compute_TwoGrid_Params()
331{
332 // loop over all materials
333 for (const auto& mat_id_xs : matid_to_xs_map)
334 {
335
336 // get the P0 transfer matrix and total XS
337 const auto& isotropic_transfer_matrix = mat_id_xs.second->TransferMatrix(0);
338 const auto& sigma_t = mat_id_xs.second->SigmaTotal();
339 const auto& diffusion_coeff = mat_id_xs.second->DiffusionCoefficient();
340
341 // put P0 transfer matrix in nicer form
342 MatDbl S(num_groups_, VecDbl(num_groups_, 0.0));
343 for (unsigned int g = 0; g < num_groups_; ++g)
344 for (const auto& [row_g, gprime, sigma] : isotropic_transfer_matrix.Row(g))
345 S[g][gprime] = sigma;
346
347 // (L+D) e_new = -U e_old
348 // original matrix = diag(total) - scattering
349 // so L+D = diag(removal) - tril(scattering)
350 // and U = -triu(scattering)
351 MatDbl A(num_groups_, VecDbl(num_groups_, 0.0));
352 MatDbl B(num_groups_, VecDbl(num_groups_, 0.0));
353 for (unsigned int g = 0; g < num_groups_; ++g)
354 {
355 A[g][g] = sigma_t[g] - S[g][g];
356 for (unsigned int gp = 0; gp < g; ++gp)
357 A[g][gp] = -S[g][gp];
358 for (unsigned int gp = g + 1; gp < num_groups_; ++gp)
359 B[g][gp] = S[g][gp];
360 }
361 MatDbl Ainv = Inverse(A);
362 // finally, obtain the iteration matrix
363 MatDbl C_ = MatMul(Ainv, B);
364 // Perform power iteration
365 VecDbl E(num_groups_, 1.0);
366 double rho = PowerIteration(C_, E, 10000, 1.0e-12);
367
368 // Compute two-grid diffusion quantities
369 // normalize spectrum
370 std::vector<double> spectrum(num_groups_, 1.0);
371 double sum = 0.0;
372 for (unsigned int g = 0; g < num_groups_; ++g)
373 sum += std::fabs(E[g]);
374 for (unsigned int g = 0; g < num_groups_; ++g)
375 spectrum[g] = std::fabs(E[g]) / sum;
376 // D ave and Sigma_a ave
377 double collapsed_D = 0.0;
378 double collapsed_sig_a = 0.0;
379 for (unsigned int g = last_fast_group_; g < num_groups_; ++g)
380 {
381 collapsed_D += diffusion_coeff[g] * spectrum[g];
382 collapsed_sig_a += sigma_t[g] * spectrum[g];
383 for (unsigned int gp = last_fast_group_; gp < num_groups_; ++gp)
384 collapsed_sig_a -= S[g][gp] * spectrum[gp];
385 }
386 // Verbose output the spectrum
387 log.Log0Verbose1() << "Fundamental eigen-value: " << rho;
388 std::stringstream outstr;
389 for (auto& xi : spectrum)
390 outstr << xi << '\n';
391 log.Log0Verbose1() << outstr.str(); // jcr verbose1
392
393 // std::stringstream outstr2;
394 // for (auto &xi: diffusion_coeff)
395 // outstr2 << xi << '\n';
396 // chi::log.Log0Verbose0() << outstr2.str(); // jcr verbose1
397 //
398 // std::cout << "collapsed = " << collapsed_sig_a
399 // <<", "<< collapsed_D << std::endl;
400 // chi::Exit(12345);
401
402 const auto mat_id = mat_id_xs.first;
403 map_mat_id_2_tginfo.insert(
404 std::make_pair(mat_id, TwoGridCollapsedInfo{collapsed_D, collapsed_sig_a, spectrum}));
405
406 } // end loop over materials
407}

409void
410Solver::Compute_TwoGrid_VolumeFractions()
411{
412 const auto& grid = *grid_ptr_;
413 const auto& sdm = *sdm_ptr_;
414
415 const size_t ncells = grid.local_cells.size();
416 VF_.resize(ncells);
417
418 int counter = 0;
419 for (const auto& cell : grid.local_cells)
420 {
421 const auto& cell_mapping = sdm.GetCellMapping(cell);
422 const auto qp_data = cell_mapping.MakeVolumetricQuadraturePointData();
423 const size_t num_nodes = cell_mapping.NumNodes();
424
425 VF_[counter].resize(num_nodes, 0.0);
426
427 for (size_t i = 0; i < num_nodes; ++i)
428 {
429 double vol_frac_shape_i = 0.0;
430 for (size_t qp : qp_data.QuadraturePointIndices())
431 vol_frac_shape_i += qp_data.ShapeValue(i, qp) * qp_data.JxW(qp);
432 vol_frac_shape_i /= cell_mapping.CellVolume();
433 VF_[counter][i] = vol_frac_shape_i;
434 } // for i
435
436 counter++;
437 } // for cell
438}

440void
441Solver::Set_BCs(const std::vector<uint64_t>& globl_unique_bndry_ids)
442{
443 log.Log0Verbose1() << "Setting Boundary Conditions";
444
445 uint64_t max_boundary_id = 0;
446 for (const auto& id : globl_unique_bndry_ids)
447 max_boundary_id = std::max(id, max_boundary_id);
448
449 log.Log() << "Max boundary id identified: " << max_boundary_id;
450
451 for (int bndry = 0; bndry < (max_boundary_id + 1); ++bndry)
452 {
453 if (boundary_preferences_.find(bndry) != boundary_preferences_.end())
454 {
455 BoundaryInfo bndry_info = boundary_preferences_.at(bndry);
456 auto& bndry_vals = bndry_info.second;
457
458 switch (bndry_info.first)
459 {
460 case BoundaryType::Reflecting: // ------------- REFLECTING
461 {
462 boundaries_.push_back({BoundaryType::Reflecting});
463 log.Log() << "Boundary " << bndry << " set to reflecting.";
464 break;
465 }
466 case BoundaryType::Robin: // ------------- ROBIN
467 {
468 if (bndry_vals.size() != 3)
469 throw std::logic_error(std::string(__PRETTY_FUNCTION__) +
470 " Robin needs 3 values in bndry vals.");
471 boundaries_.push_back(Boundary{BoundaryType::Robin, bndry_vals});
472 log.Log() << "Boundary " << bndry << " set to robin.";
473 break;
474 }
475 case BoundaryType::Vacuum: // ------------- VACUUM
476 {
477 auto ng = mg_diffusion::Solver::num_groups_;
478 std::vector<double> a_values(ng, 0.25);
479 std::vector<double> b_values(ng, 0.5);
480 std::vector<double> f_values(ng, 0.0);
481 boundaries_.push_back(Boundary{BoundaryType::Robin, {a_values, b_values, f_values}});
482 log.Log() << "Boundary " << bndry << " set to vacuum.";
483 break;
484 }
485 case BoundaryType::Neumann: // ------------- NEUMANN
486 {
487 if (bndry_vals.size() != 3)
488 throw std::logic_error(std::string(__PRETTY_FUNCTION__) +
489 " Neumann needs 3 values in bndry vals.");
490 boundaries_.push_back(Boundary{BoundaryType::Robin, bndry_vals});
491 log.Log() << "Boundary " << bndry << " set to neumann.";
492 break;
493 }
494 } // switch boundary type
495 }
496 else
497 {
498 auto ng = mg_diffusion::Solver::num_groups_;
499 std::vector<double> a_values(ng, 0.25);
500 std::vector<double> b_values(ng, 0.5);
501 std::vector<double> f_values(ng, 0.0);
502 boundaries_.push_back({BoundaryType::Robin, {a_values, b_values, f_values}});
503 log.Log0Verbose1() << "No boundary preference found for boundary index " << bndry
504 << "Vacuum boundary added as default.";
505 }
506 } // for bndry

509void
510Solver::Assemble_A_bext()
511{
512 const auto& grid = *grid_ptr_;
513 const auto& sdm = *sdm_ptr_;
514
515 // Assemble the system
516 unsigned int i_two_grid = do_two_grid_ ? 1 : 0;
517
518 for (const auto& cell : grid.local_cells)
519 {
520 const auto& cell_mapping = sdm.GetCellMapping(cell);
521 const auto qp_data = cell_mapping.MakeVolumetricQuadraturePointData();
522 const size_t num_nodes = cell_mapping.NumNodes();
523
524 const auto& xs = matid_to_xs_map.at(cell.material_id_);
525 const auto& D = xs->DiffusionCoefficient();
526 const auto& sigma_r = xs->SigmaRemoval();
527
528 const auto& qext = matid_to_src_map.at(cell.material_id_);
529 double collapsed_D = 0., collapsed_sig_a = 0.;
530 if (do_two_grid_)
531 {
532 const auto& xstg = map_mat_id_2_tginfo.at(cell.material_id_);
533 collapsed_D = xstg.collapsed_D;
534 collapsed_sig_a = xstg.collapsed_sig_a;
535 }
536
537 std::vector<VecDbl> rhs_cell;
538 rhs_cell.resize(num_groups_);
539 for (uint g = 0; g < num_groups_; ++g)
540 rhs_cell[g].resize(num_nodes, 0.0);
541
542 std::vector<MatDbl> Acell;
543 Acell.resize(num_groups_ + i_two_grid);
544 for (uint g = 0; g < num_groups_ + i_two_grid; ++g)
545 Acell[g].resize(num_nodes, VecDbl(num_nodes, 0.0));
546
547 for (size_t i = 0; i < num_nodes; ++i)
548 {
549 for (size_t j = 0; j < num_nodes; ++j)
550 {
551 double entry_mij = 0.0;
552 double entry_kij = 0.0;
553 for (size_t qp : qp_data.QuadraturePointIndices())
554 {
555 entry_mij += qp_data.ShapeValue(i, qp) * qp_data.ShapeValue(j, qp) * qp_data.JxW(qp);
556
557 entry_kij += qp_data.ShapeGrad(i, qp).Dot(qp_data.ShapeGrad(j, qp)) * qp_data.JxW(qp);
558 } // for qp
559 for (uint g = 0; g < num_groups_; ++g)
560 Acell[g][i][j] = entry_mij * sigma_r[g] + entry_kij * D[g];
561
562 if (do_two_grid_)
563 Acell[num_groups_][i][j] = entry_mij * collapsed_sig_a + entry_kij * collapsed_D;
564 } // for j
565 double entry_rhsi = 0.0;
566 for (size_t qp : qp_data.QuadraturePointIndices())
567 entry_rhsi += qp_data.ShapeValue(i, qp) * qp_data.JxW(qp);
568 for (uint g = 0; g < num_groups_; ++g)
569 rhs_cell[g][i] = entry_rhsi * (qext->source_value_g_[g]);
570 } // for i
571
572 // Deal with BC (all based on variations of Robin)
573 const size_t num_faces = cell.faces_.size();
574 for (size_t f = 0; f < num_faces; ++f)
575 {
576 const auto& face = cell.faces_[f];
577 // not a boundary face
578 if (face.has_neighbor_) continue;
579
580 auto& bndry = boundaries_[face.neighbor_id_];
581
582 // Robin boundary
583 // for two-grid, it is homogenous Robin
584 if (bndry.type_ == BoundaryType::Robin)
585 {
586 const auto qp_face_data = cell_mapping.MakeSurfaceQuadraturePointData(f);
587 const size_t num_face_nodes = face.vertex_ids_.size();
588
589 auto& aval = bndry.mg_values_[0];
590 auto& bval = bndry.mg_values_[1];
591 auto& fval = bndry.mg_values_[2];
592 if (do_two_grid_)
593 {
594 aval.push_back(0.25);
595 bval.push_back(0.5);
596 fval.push_back(0.0);
597 }
598
599 // sanity check, Assert if b=0
600 for (uint g = 0; g < num_groups_ + i_two_grid; ++g)
601 {
602 if (std::fabs(bval[g]) < 1e-8)
603 throw std::logic_error("if b=0, this is a Dirichlet BC, not a Robin BC");
604 }
605
606 // true Robin when a!=0, otherwise, it is a Neumann:
607 // only do this part if true Robin (i.e., a!=0)
608 for (uint g = 0; g < num_groups_ + i_two_grid; ++g)
609 {
610 if (std::fabs(aval[g]) > 1.0e-8)
611 {
612 // loop over nodes of that face
613 for (size_t fi = 0; fi < num_face_nodes; ++fi)
614 {
615 const uint i = cell_mapping.MapFaceNode(f, fi);
616
617 double entry_rhsi = 0.0;
618 for (size_t qp : qp_face_data.QuadraturePointIndices())
619 entry_rhsi += qp_face_data.ShapeValue(i, qp) * qp_face_data.JxW(qp);
620 if (g < num_groups_) // check due to two-grid
621 rhs_cell[g][i] += fval[g] / bval[g] * entry_rhsi;
622
623 for (size_t fj = 0; fj < num_face_nodes; ++fj)
624 {
625 const uint j = cell_mapping.MapFaceNode(f, fj);
626 double entry_aij = 0.0;
627 for (size_t qp : qp_face_data.QuadraturePointIndices())
628 entry_aij += qp_face_data.ShapeValue(i, qp) * qp_face_data.ShapeValue(j, qp) *
629 qp_face_data.JxW(qp);
630 Acell[g][i][j] += aval[g] / bval[g] * entry_aij;
631 } // for fj
632 } // for fi
633 } // end true Robin
634 } // for g
635 } // if Robin
636 } // for face f
637
638 // Develop node mapping
639 std::vector<int64_t> imap(num_nodes, 0); // node-mapping
640 for (size_t i = 0; i < num_nodes; ++i)
641 imap[i] = sdm.MapDOF(cell, i);
642
643 // Assembly into system
644 for (uint g = 0; g < num_groups_; ++g)
645 for (size_t i = 0; i < num_nodes; ++i)

647
648 for (uint g = 0; g < num_groups_ + i_two_grid; ++g)
649 for (size_t i = 0; i < num_nodes; ++i)
650 for (size_t j = 0; j < num_nodes; ++j)
651 MatSetValue(A_[g], imap[i], imap[j], Acell[g][i][j], ADD_VALUES);
652
653 } // for cell
654
655 log.Log() << "Global assembly";
656
657 for (uint g = 0; g < num_groups_; ++g)
658 {
659 VecAssemblyBegin(bext_[g]);
660 VecAssemblyEnd(bext_[g]);
661 }
662 for (uint g = 0; g < num_groups_ + i_two_grid; ++g)
663 {
664 MatAssemblyBegin(A_[g], MAT_FINAL_ASSEMBLY);
665 MatAssemblyEnd(A_[g], MAT_FINAL_ASSEMBLY);

683void
684Solver::Execute()
685{
686 log.Log() << "\nExecuting CFEM Multigroup Diffusion solver";
687
688 // Create Krylov Solver
689 // setup KSP once for all
690 petsc_solver_ = CreateCommonKrylovSolverSetup(A_.front(),
691 TextName(),
692 KSPCG,
693 PCGAMG,
694 basic_options_("residual_tolerance").FloatValue(),
695 basic_options_("max_inner_iters").IntegerValue());
696
697 KSPSetApplicationContext(petsc_solver_.ksp, (void*)&my_app_context_);
698 KSPMonitorCancel(petsc_solver_.ksp);
699 KSPMonitorSet(petsc_solver_.ksp, &mg_diffusion::MGKSPMonitor, nullptr, nullptr);
700
701 int64_t iverbose = basic_options_("verbose_level").IntegerValue();
702 my_app_context_.verbose = iverbose > 1 ? PETSC_TRUE : PETSC_FALSE;
703 // if (my_app_context.verbose == PETSC_TRUE)
704 // cout << "--context TRUE" << endl;
705 // if (my_app_context.verbose == PETSC_FALSE)
706 // cout << "--context FALSE" << endl;
707 // std::cout << "STOP" << std::endl; std::cin.get();
708
709 // shortcuts
710 // unsigned int lfg = mg_diffusion::Solver::last_fast_group;
711 // unsigned int ng = mg_diffusion::Solver::num_groups;
712
713 // Solve fast groups:
714 for (unsigned int g = 0; g < last_fast_group_; ++g)
715 {
716 mg_diffusion::Solver::Assemble_RHS(g, iverbose);
717 mg_diffusion::Solver::SolveOneGroupProblem(g, iverbose);
718 }
719
720 // Solve thermal groups:
721 unsigned int thermal_iteration = 0;
722 // max # of thermal iterations
723 int64_t max_thermal_iters = basic_options_("max_thermal_iters").IntegerValue();
724 // max thermal error between two successive iterates
725 double thermal_tol = basic_options_("thermal_flux_tolerance").FloatValue();
726 // computed error
727 double thermal_error_all;
728 double thermal_error_g;
729
730 do
731 {
732 thermal_error_all = 0.0;
733 for (unsigned int g = last_fast_group_; g < num_groups_; ++g)
734 {
735 // conpute rhs src
736 mg_diffusion::Solver::Assemble_RHS(g, iverbose);
737 // copy solution
738 VecCopy(x_[g], x_old_[g]);
739 // solve group g for new solution
740 mg_diffusion::Solver::SolveOneGroupProblem(g, iverbose);
741 // compute L2 norm of thermal error for current g (requires one more copy)
742 VecCopy(x_[g], thermal_dphi_);
743 VecAXPY(thermal_dphi_, -1.0, x_old_[g]);
744 VecNorm(thermal_dphi_, NORM_2, &thermal_error_g);
745 thermal_error_all = std::max(thermal_error_all, thermal_error_g);
746 }
747 // perform two-grid
748 if (do_two_grid_)
749 {
750 mg_diffusion::Solver::Assemble_RHS_TwoGrid(iverbose);
751 mg_diffusion::Solver::SolveOneGroupProblem(num_groups_, iverbose);
752 mg_diffusion::Solver::Update_Flux_With_TwoGrid(iverbose);
753 }
754
755 if (iverbose > 0)
756 log.Log() << " --thermal iteration = " << std::setw(5) << std::right << thermal_iteration
757 << ", Error=" << std::setw(11) << std::right << std::scientific
758 << std::setprecision(7) << thermal_error_all << std::endl;
759
760 ++thermal_iteration;
761 } while ((thermal_error_all > thermal_tol) && (thermal_iteration < max_thermal_iters));
762
763 if (iverbose > 0)
764 {
765 if (thermal_error_all < thermal_tol)
766 std::cout << "\nThermal iterations converged for fixed-source problem" << std::endl;
767 else
768 std::cout << "\nThermal iterations NOT converged for fixed-source problem" << std::endl;
769 }
770
771 UpdateFieldFunctions();
772 log.Log() << "Done solving multi-group diffusion";
773}

775void
776Solver::Assemble_RHS(const unsigned int g, const int64_t verbose)
777{
778 if (verbose > 2) log.Log() << "\nAssemblying RHS for group " + std::to_string(g);
779
780 // copy the external source vector for group g into b
781 VecSet(b_, 0.0);
782 VecCopy(bext_[g], b_);
783
784 const auto& sdm = *mg_diffusion::Solver::sdm_ptr_;
785 // compute inscattering term
786 for (const auto& cell : mg_diffusion::Solver::grid_ptr_->local_cells)
787 {
788 const auto& cell_mapping = sdm.GetCellMapping(cell);
789 const auto qp_data = cell_mapping.MakeVolumetricQuadraturePointData();
790 const size_t num_nodes = cell_mapping.NumNodes();
791
792 const auto& xs = matid_to_xs_map.at(cell.material_id_);
793 const auto& S = xs->TransferMatrix(0);
794
795 for (const auto& [row_g, gprime, sigma_sm] : S.Row(g))
796 {
797 if (gprime != g) // g and row_g are the same, maybe different int types
798 {
799 const double* xlocal;
800 VecGetArrayRead(x_[gprime], &xlocal);
801
802 for (size_t i = 0; i < num_nodes; ++i)
803 {
804 const int64_t imap = sdm.MapDOF(cell, i);
805 double inscatter_g = 0.0;
806
807 for (size_t j = 0; j < num_nodes; ++j)
808 {
809 const int64_t jmap = sdm.MapDOFLocal(cell, j);
810
811 // get flux at node j
812 const double flxj_gp = xlocal[jmap];
813 for (size_t qp : qp_data.QuadraturePointIndices())
814 inscatter_g += sigma_sm * flxj_gp * qp_data.ShapeValue(i, qp) *
815 qp_data.ShapeValue(j, qp) * qp_data.JxW(qp);
816 } // for j
817 // add inscattering value to vector
818 VecSetValue(b_, imap, inscatter_g, ADD_VALUES);
819 } // for i
820 VecRestoreArrayRead(x_[gprime], &xlocal);
821 } // if gp!=g
822 } // for gprime
823 } // for cell
824
825 VecAssemblyBegin(b_);
826 VecAssemblyEnd(b_);
827}

843void
844Solver::Assemble_RHS_TwoGrid(const int64_t verbose)
845{
846 if (verbose > 2) log.Log() << "\nAssemblying RHS for two-grid ";
847
848 VecSet(b_, 0.0);
849
850 const auto& sdm = *sdm_ptr_;
851 // compute inscattering term
852 for (const auto& cell : grid_ptr_->local_cells)
853 {
854 const auto& cell_mapping = sdm.GetCellMapping(cell);

856 const size_t num_nodes = cell_mapping.NumNodes();

858 const auto& S = matid_to_xs_map.at(cell.material_id_)->TransferMatrix(0);
859
860 for (unsigned g = last_fast_group_; g < num_groups_; ++g)
861 {
862 for (const auto& [row_g, gprime, sigma_sm] : S.Row(g))

866 const double* xlocal;

877 {
878 const int64_t jmap = sdm.MapDOFLocal(cell, j);
879
880 // get flux at node j
881 const double delta_flxj_gp = xlocal[jmap] - xlocal_old[jmap];
882 for (size_t qp : qp_data.QuadraturePointIndices())

892 } // for gprime
893 } // for g
894 } // for cell
895
896 VecAssemblyBegin(b_);

902void
903Solver::Update_Flux_With_TwoGrid(const int64_t verbose)
904{
905 if (verbose > 2) log.Log() << "\nUpdating Thermal fluxes from two-grid";
906
907 const auto& grid = *grid_ptr_;
908 const auto& sdm = *sdm_ptr_;
909
910 // contains two_grid flux, stored in last num_groups entry
911 const double* xlocal_tg;
912 VecGetArrayRead(x_[num_groups_], &xlocal_tg);
913
914 int counter = 0;
915 for (const auto& cell : grid.local_cells)
916 {
917 const auto& cell_mapping = sdm.GetCellMapping(cell);
918 const size_t num_nodes = cell_mapping.NumNodes();
919 const auto& xstg = map_mat_id_2_tginfo.at(cell.material_id_);
920
921 for (unsigned int g = last_fast_group_; g < num_groups_; ++g)
922 {
923 for (size_t i = 0; i < num_nodes; ++i)
924 {
925 const int64_t imap = sdm.MapDOFLocal(cell, i);
926 const int64_t imap_globl = sdm.MapDOF(cell, i);
927 const double aux = xlocal_tg[imap] * VF_[counter][i] * xstg.spectrum[g];
928 VecSetValue(x_[g], imap_globl, aux, ADD_VALUES);
929 } // i
930 } // g
931 counter++;
932 } // for cell
933
934 // finalize
935 for (unsigned int g = last_fast_group_; g < num_groups_; ++g)
936 {
937 VecAssemblyBegin(x_[g]);
938 VecAssemblyEnd(x_[g]);
939 }
940 // release two-grid flux
941 VecRestoreArrayRead(x_[num_groups_], &xlocal_tg);
942}
