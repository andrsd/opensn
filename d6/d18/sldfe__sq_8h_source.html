185/**This is a utility function that encapsulates
186 * all the necessary functionality to determine shape
187 * function coefficients and integrate accross a
188 * spherical quadrilateral.*/
189struct SimplifiedLDFESQ::FUNCTION_WEIGHT_FROM_RHO
190{
191 Quadrature& sldfesq;
192 Vertex& centroid_xy_tilde;
193 std::array<Vector3, 4>& radii_vectors_xy_tilde;
194 SphericalQuadrilateral& sq;
195
196 std::array<DynamicVector<double>, 4> rhs;
197 DynamicMatrix<double> A;
198 DynamicMatrix<double> A_inv;
199 std::array<DynamicVector<double>, 4> c_coeffs;
200 /// Legendre quadrature points
201 std::vector<QuadraturePointXYZ>& lqp;
202 /// Legendre quadrature weights
203 std::vector<double>& lqw;
204
205 FUNCTION_WEIGHT_FROM_RHO(SimplifiedLDFESQ::Quadrature& in_sldfesq,
206 Vertex& in_centroid_xy_tilde,
207 std::array<Vector3, 4>& in_radii_vectors_xy_tilde,
208 SphericalQuadrilateral& in_sq,
209 QuadratureGaussLegendre& in_legendre_quadrature)
210 : sldfesq(in_sldfesq),
211 centroid_xy_tilde(in_centroid_xy_tilde),
212 radii_vectors_xy_tilde(in_radii_vectors_xy_tilde),
213 sq(in_sq),
214 A(4, 4),
215 A_inv(4, 4),
216 lqp(in_legendre_quadrature.qpoints_),
217 lqw(in_legendre_quadrature.weights_)
218 {
219 // Init RHS
220 for (int i = 0; i < 4; ++i)
221 {
222 rhs[i] = std::vector<double>(4);
223 c_coeffs[i] = std::vector<double>(4);
224 for (int j = 0; j < 4; ++j)
225 rhs[i][i] = 1.0;
226 }
227 }
228
229 /**Computes the quadrature point locations
230 * from rho, followed by the shape-function coefficients and
231 * then the integral of the shape function to get the weights.*/
232 std::array<double, 4> operator()(const DynamicVector<double>& rho)
233 {
234 // Determine qpoints from rho
235 std::array<Vector3, 4> qpoints;
236 for (int i = 0; i < 4; ++i)
237 {
238 auto xy_tilde = centroid_xy_tilde + rho[i] * radii_vectors_xy_tilde[i];
239 auto xyz_prime = sq.rotation_matrix * xy_tilde + sq.translation_vector;
240 qpoints[i] = xyz_prime.Normalized();
241 }
242
243 // Assemble A
244 for (int i = 0; i < 4; ++i)
245 A[i] = {1.0, qpoints[i][0], qpoints[i][1], qpoints[i][2]};
246
247 // Compute A-inverse
248 A_inv = Inverse(A.elements_);
249
250 // Compute coefficients
251 for (int i = 0; i < 4; ++i)
252 c_coeffs[i] = A_inv * rhs[i];
253
254 return sldfesq.IntegrateLDFEShapeFunctions(sq, c_coeffs, lqp, lqw);
255 }
256};
