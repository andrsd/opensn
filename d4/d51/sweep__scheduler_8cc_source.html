55SweepScheduler::InitializeAlgoDOG()
56{
57 // Load all anglesets in preperation for sorting
58 // Loop over angleset groups
59 for (size_t q = 0; q < angle_agg_.angle_set_groups.size(); q++)
60 {
61 TAngleSetGroup& angleset_group = angle_agg_.angle_set_groups[q];
62
63 // Loop over anglesets in group
64 size_t num_anglesets = angleset_group.AngleSets().size();
65 for (size_t as = 0; as < num_anglesets; as++)
66 {
67 auto angleset = angleset_group.AngleSets()[as];
68 const auto& spds = dynamic_cast<const SPDS_AdamsAdamsHawkins&>(angleset->GetSPDS());
69
70 const TLEVELED_GRAPH& leveled_graph = spds.GetGlobalSweepPlanes();
71
72 // Find location depth
73 int loc_depth = -1;
74 for (size_t level = 0; level < leveled_graph.size(); level++)
75 {
76 for (size_t index = 0; index < leveled_graph[level].item_id.size(); index++)
77 {
78 if (leveled_graph[level].item_id[index] == opensn::mpi_comm.rank())
79 {
80 loc_depth = static_cast<int>(leveled_graph.size() - level);
81 break;
82 }
83 } // for locations in plane
84 } // for sweep planes
85
86 // Set up rule values
87 if (loc_depth >= 0)
88 {
89 RULE_VALUES new_rule_vals(angleset);
90 new_rule_vals.depth_of_graph = loc_depth;
91 new_rule_vals.set_index = as + q * num_anglesets;
92
93 const auto& omega = spds.Omega();
94 new_rule_vals.sign_of_omegax = (omega.x >= 0) ? 2 : 1;
95 new_rule_vals.sign_of_omegay = (omega.y >= 0) ? 2 : 1;
96 new_rule_vals.sign_of_omegaz = (omega.z >= 0) ? 2 : 1;
97
98 rule_values_.push_back(new_rule_vals);
99 }
100 else
101 {
102 log.LogAllError() << "Location depth not found in Depth-Of-Graph algorithm.";
103 Exit(EXIT_FAILURE);
104 }
105
106 } // for anglesets
107 } // for quadrants/anglesetgroups
108
109 // Init sort functions

112 bool operator()(const RULE_VALUES& a, const RULE_VALUES& b)
113 {
114 return a.depth_of_graph > b.depth_of_graph;

144 // Sort
145 std::stable_sort(rule_values_.begin(), rule_values_.end(), compare_D);
146 std::stable_sort(rule_values_.begin(), rule_values_.end(), compare_omega_x);
147 std::stable_sort(rule_values_.begin(), rule_values_.end(), compare_omega_y);
148 std::stable_sort(rule_values_.begin(), rule_values_.end(), compare_omega_z);

151void
152SweepScheduler::ScheduleAlgoDOG(SweepChunk& sweep_chunk)
153{
154 typedef ExecutionPermission ExePerm;
155 typedef AngleSetStatus Status;
156
157 log.LogEvent(sweep_event_tag_, Logger::EventType::EVENT_BEGIN);
158
159 auto ev_info = std::make_shared<Logger::EventInfo>(std::string("Sweep initiated"));
160
161 log.LogEvent(sweep_event_tag_, Logger::EventType::SINGLE_OCCURRENCE, ev_info);
162
163 // Loop till done
164 bool finished = false;
165 size_t scheduled_angleset = 0;
166 while (!finished)
167 {
168 finished = true;
169 for (auto& rule_value : rule_values_)
170 {
171 auto angleset = rule_value.angle_set;
172
173 // Query angleset status
174 // Status will here be one of the following:
175 // - RECEIVING.
176 // Meaning it is either waiting for messages or actively receiving it
177 // - READY_TO_EXECUTE.
178 // Meaning it has received all upstream data and can be executed
179 // - FINISHED.
180 // Meaning the angleset has executed its sweep chunk
181 Status status =
182 angleset->AngleSetAdvance(sweep_chunk, sweep_timing_events_tag_, ExePerm::NO_EXEC_IF_READY);
183
184 // Execute if ready and allowed
185 // If this angleset is the one scheduled to run
186 // and it is ready then it will be given permission
187 if (status == Status::READY_TO_EXECUTE)
188 {
189 std::stringstream message_i;
190 message_i << "Angleset " << angleset->GetID() << " executed on location "
191 << opensn::mpi_comm.rank();
192
193 auto ev_info_i = std::make_shared<Logger::EventInfo>(message_i.str());
194
195 log.LogEvent(sweep_event_tag_, Logger::EventType::SINGLE_OCCURRENCE, ev_info_i);
196
197 status = angleset->AngleSetAdvance(sweep_chunk, sweep_timing_events_tag_, ExePerm::EXECUTE);
198
199 std::stringstream message_f;
200 message_f << "Angleset " << angleset->GetID() << " finished on location "
201 << opensn::mpi_comm.rank();
202
203 auto ev_info_f = std::make_shared<Logger::EventInfo>(message_f.str());
204
205 log.LogEvent(sweep_event_tag_, Logger::EventType::SINGLE_OCCURRENCE, ev_info_f);
206
207 scheduled_angleset++; // Schedule the next angleset
208 }
209
210 if (status != Status::FINISHED) finished = false;
211 } // for each angleset rule
212 } // while not finished
213
214 // Receive delayed data
215 opensn::mpi_comm.barrier();
216 bool received_delayed_data = false;
217 while (not received_delayed_data)
218 {
219 received_delayed_data = true;
220
221 for (auto& angle_set_group : angle_agg_.angle_set_groups)
222 for (auto& angle_set : angle_set_group.AngleSets())
223 {
224 if (angle_set->FlushSendBuffers() == Status::MESSAGES_PENDING)
225 received_delayed_data = false;
226
227 if (not angle_set->ReceiveDelayedData()) received_delayed_data = false;
228 }
229 }
230
231 // Reset all
232 for (auto& angle_set_group : angle_agg_.angle_set_groups)
233 for (auto& angle_set : angle_set_group.AngleSets())
234 angle_set->ResetSweepBuffers();
235
236 for (auto& [bid, bndry] : angle_agg_.sim_boundaries)
237 {
238 if (bndry->Type() == BoundaryType::REFLECTING)
239 {
240 auto rbndry = std::static_pointer_cast<BoundaryReflecting>(bndry);
241 rbndry->ResetAnglesReadyStatus();
242 }
243 }
244
245 log.LogEvent(sweep_event_tag_, Logger::EventType::EVENT_END);
246}
