66ExtruderMeshGenerator::ExtruderMeshGenerator(const InputParameters& params)
67 : MeshGenerator(params),
68 top_boundary_name_(params.GetParamValue<std::string>("top_boundary_name")),
69 bottom_boundary_name_(params.GetParamValue<std::string>("bottom_boundary_name"))
70{
71 const auto& layers_param = params.GetParam("layers");
72
73 double current_z_level = 0.0;
74 for (const auto& layer_block : layers_param)
75 {
76 auto valid_params = ExtrusionLayer::GetInputParameters();
77 valid_params.SetErrorOriginScope("ExtruderMeshGenerator:\"layers\"");
78 valid_params.AssignParameters(layer_block);
79
80 const int h_and_z_config = int(layer_block.Has("h")) + int(layer_block.Has("z"));
81
82 if (h_and_z_config != 1)
83 ChiInvalidArgument("For an ExtrusionLayer either \"h\" or \"z\" must"
84 "be specified and also not both.");
85
86 auto n = valid_params.GetParamValue<uint32_t>("n");
87 double h;
88 if (layer_block.Has("h")) h = valid_params.GetParamValue<double>("h");
89 else
90 {
91 double z = valid_params.GetParamValue<double>("z");
92 ChiInvalidArgumentIf(z <= current_z_level,
93 "For extrusion layers, the \"z\" coordinates must "
94 "be monotonically increasing.");
95 h = z - current_z_level;
96 }
97
98 current_z_level += h;
99
100 layers_.push_back(ExtrusionLayer{h, n});
101
102 log.Log0Verbose1() << "Layer " << layer_block.Name() << " height=" << h
103 << " num_sub_layers=" << n << " top-z=" << current_z_level;
104 } // layer_block in layers_param
105}

108ExtruderMeshGenerator::GenerateUnpartitionedMesh(std::unique_ptr<UnpartitionedMesh> input_umesh)
109{
110 log.Log0Verbose1() << "ExtruderMeshGenerator::GenerateUnpartitionedMesh";
111 const Vector3 khat(0.0, 0.0, 1.0);
112
113 ChiInvalidArgumentIf(not(input_umesh->GetMeshAttributes() & DIMENSION_2),
114 "Input mesh is not 2D. A 2D mesh is required for extrusion");
115
116 const auto& template_vertices = input_umesh->GetVertices();
117 const auto& template_cells = input_umesh->GetRawCells();
118
119 const size_t num_template_vertices = template_vertices.size();
120 const size_t num_template_cells = template_cells.size();
121
122 ChiLogicalErrorIf(template_vertices.empty(), "Input mesh has no vertices.");
123 ChiLogicalErrorIf(template_cells.empty(), "Input mesh has no cells.");
124
125 // Check cells
126 for (const auto& template_cell_ptr : template_cells)
127 {
128 const auto& template_cell = *template_cell_ptr;
129 if (template_cell.type != CellType::POLYGON)
130 throw std::logic_error("ExtruderMeshGenerator: "
131 "Template cell error. Not of base type POLYGON");
132
133 // Check cell not inverted
134 const auto& v0 = template_cell.centroid;
135 const auto& v1 = template_vertices[template_cell.vertex_ids[0]];
136 const auto& v2 = template_vertices[template_cell.vertex_ids[1]];
137
138 auto v01 = v1 - v0;
139 auto v02 = v2 - v0;
140
141 if (v01.Cross(v02).Dot(khat) < 0.0)
142 throw std::logic_error("Extruder attempting to extrude a template"
143 " cell with a normal pointing downward. This"
144 " causes erratic behavior and needs to be"
145 " corrected.");
146 }
147
148 auto umesh = std::make_unique<UnpartitionedMesh>();
149
150 // Update boundary maps
151 auto& umesh_bndry_map = umesh->GetMeshOptions().boundary_id_map;
152 umesh_bndry_map = input_umesh->GetMeshOptions().boundary_id_map;
153
154 const uint64_t zmax_bndry_id = umesh->MakeBoundaryID(top_boundary_name_);
155 umesh_bndry_map[zmax_bndry_id] = top_boundary_name_;
156 const uint64_t zmin_bndry_id = umesh->MakeBoundaryID(bottom_boundary_name_);
157 umesh_bndry_map[zmin_bndry_id] = bottom_boundary_name_;
158
159 // Setup z-levels
160 double current_z = 0.0;
161 std::vector<double> z_levels = {current_z};
162 for (const auto& layer : layers_)
163 {
164 const double dz = layer.height_ / layer.num_sub_layers_;
165 for (uint32_t i = 0; i < layer.num_sub_layers_; ++i)
166 z_levels.push_back(current_z += dz);
167 }
168
169 // Build vertices
170 typedef Vector3 Vec3;
171 auto& extruded_vertices = umesh->GetVertices();
172 for (const double z_level : z_levels)
173 for (const auto& template_vertex : template_vertices)
174 extruded_vertices.push_back(Vec3(template_vertex.x, template_vertex.y, z_level));
175
176 // Build cells
177 size_t k = 0;
178 for (const auto& layer : layers_)
179 {
180 for (uint32_t n = 0; n < layer.num_sub_layers_; ++n)
181 {
182 size_t tc_counter = 0;
183 for (const auto& template_cell : template_cells)
184 {
185 // Determine cell sub-type
186 CellType extruded_subtype;
187 // clang-format off
188 switch (template_cell->sub_type)
189 {
190 case CellType::TRIANGLE: extruded_subtype = CellType::WEDGE; break;
191 case CellType::QUADRILATERAL: extruded_subtype = CellType::HEXAHEDRON; break;
192 default: extruded_subtype = CellType::POLYHEDRON;
193 }
194 // clang-format on
195
196 // Create new cell
197 auto new_cell_ptr =
198 new UnpartitionedMesh::LightWeightCell(CellType::POLYHEDRON, extruded_subtype);
199 auto& new_cell = *new_cell_ptr;
200
201 new_cell.material_id = template_cell->material_id;
202
203 // Build vertices
204 const size_t tc_num_verts = template_cell->vertex_ids.size();
205 new_cell.vertex_ids.reserve(2 * tc_num_verts);
206 for (const auto tc_vid : template_cell->vertex_ids)
207 new_cell.vertex_ids.push_back(tc_vid + k * num_template_vertices);
208 for (const auto tc_vid : template_cell->vertex_ids)
209 new_cell.vertex_ids.push_back(tc_vid + (k + 1) * num_template_vertices);
210
211 // Create side faces
212 for (const auto& tc_face : template_cell->faces)
213 {
214 UnpartitionedMesh::LightWeightFace new_face;
215
216 // clang-format off
217 new_face.vertex_ids.resize(4, -1);
218 new_face.vertex_ids[0] = tc_face.vertex_ids[0] + k * num_template_vertices;
219 new_face.vertex_ids[1] = tc_face.vertex_ids[1] + k * num_template_vertices;
220 new_face.vertex_ids[2] = tc_face.vertex_ids[1] + (k + 1) * num_template_vertices;
221 new_face.vertex_ids[3] = tc_face.vertex_ids[0] + (k + 1) * num_template_vertices;
222 // clang-format on
223
224 if (tc_face.has_neighbor)
225 {
226 new_face.neighbor = num_template_cells * k + tc_face.neighbor;
227 new_face.has_neighbor = true;
228 }
229 else
230 {
231 new_face.neighbor = tc_face.neighbor;
232 new_face.has_neighbor = false;
233 }
234
235 new_cell.faces.push_back(std::move(new_face));
236 } // for tc face
237
238 // Create top and bottom faces
239 // Top face
240 {
241 UnpartitionedMesh::LightWeightFace new_face;
242
243 new_face.vertex_ids.reserve(template_cell->vertex_ids.size());
244 for (auto vid : template_cell->vertex_ids)
245 new_face.vertex_ids.push_back(vid + (k + 1) * num_template_vertices);
246
247 if (k == (z_levels.size() - 2))
248 {
249 new_face.neighbor = zmax_bndry_id;
250 new_face.has_neighbor = false;
251 }
252 else
253 {
254 new_face.neighbor = num_template_cells * (k + 1) + tc_counter;
255 new_face.has_neighbor = true;
256 }
257
258 new_cell.faces.push_back(std::move(new_face));
259 }
260
261 // Bottom face
262 {
263 UnpartitionedMesh::LightWeightFace new_face;
264
265 new_face.vertex_ids.reserve(template_cell->vertex_ids.size());
266 auto& vs = template_cell->vertex_ids;
267 for (auto vid = vs.rbegin(); vid != vs.rend(); ++vid)
268 new_face.vertex_ids.push_back((*vid) + k * num_template_vertices);
269
270 if (k == 0)
271 {
272 new_face.neighbor = zmin_bndry_id;
273 new_face.has_neighbor = false;
274 }
275 else
276 {
277 new_face.neighbor = num_template_cells * (k - 1) + tc_counter;
278 new_face.has_neighbor = true;
279 }
280
281 new_cell.faces.push_back(std::move(new_face));
282 }
283 umesh->GetRawCells().push_back(new_cell_ptr);
284
285 ++tc_counter;
286 } // for template cell
287 ++k;
288 } // for sub-layer n
289 } // for layer
290
291 umesh->SetAttributes(DIMENSION_3 | EXTRUDED);
292
293 umesh->ComputeCentroidsAndCheckQuality();
294 umesh->BuildMeshConnectivity();
295
296 log.Log0Verbose1() << "ExtruderMeshGenerator::GenerateUnpartitionedMesh Done";
297 return umesh;
298}
