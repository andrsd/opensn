135
136 for (size_t i = 0; i < num_nodes; ++i)
137 {
138 std::vector<double> input_params;
139 const auto& node_i_xyz = nodes_xyz[i];
140 for (size_t d = 0; d < 3; ++d)
141 input_params.push_back(node_i_xyz[d]);
142 input_params.push_back(static_cast<double>(cell.material_id_));
143
144 for (size_t k = 0; k < num_deps; ++k)
145 {
146 const auto& dep_ff = dependent_ffs_[k];
147 const double value = dep_ff->Evaluate(cell, node_i_xyz, dependent_field_ref_component_[k]);
148 input_params.push_back(value);
149 }
150
151 std::vector<double> output_params = function_ptr_->Evaluate(input_params);
152
153 ChiLogicalErrorIf(output_params.size() != result_component_references_.size(),
154 "Calling function number of output values not matching");
155
156 size_t k = 0;
157 for (uint c : result_component_references_)
158 {
159 cint64_t dof_map = sdm.MapDOFLocal(cell, i, uk_man, 0, c);
160
161 primary_ff_->FieldVector()[dof_map] = output_params[k++];
162 }
163
164 } // for node i
165 } // for cell
166}
