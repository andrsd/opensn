97CreateCommonKrylovSolverSetup(Mat ref_matrix,
98 const std::string& in_solver_name,
99 const std::string& in_solver_type,
100 const std::string& in_preconditioner_type,
101 double in_relative_residual_tolerance,
102 int64_t in_maximum_iterations)
103{
104 PETScSolverSetup setup;
105
106 KSPCreate(opensn::mpi_comm, &setup.ksp);
107 KSPSetOperators(setup.ksp, ref_matrix, ref_matrix);
108 KSPSetType(setup.ksp, in_solver_type.c_str());
109
110 KSPSetOptionsPrefix(setup.ksp, in_solver_name.c_str());
111
112 KSPGetPC(setup.ksp, &setup.pc);
113 PCSetType(setup.pc, in_preconditioner_type.c_str());
114
115 KSPSetTolerances(
116 setup.ksp, 1.e-50, in_relative_residual_tolerance, 1.0e50, in_maximum_iterations);
117 KSPSetInitialGuessNonzero(setup.ksp, PETSC_TRUE);
118
119 KSPSetConvergenceTest(setup.ksp, &RelativeResidualConvergenceTest, nullptr, nullptr);
120 KSPSetFromOptions(setup.ksp);
121
122 KSPMonitorSet(setup.ksp, &KSPMonitorRelativeToRHS, nullptr, nullptr);
123
124 return setup;

127PetscErrorCode
128RelativeResidualConvergenceTest(
129 KSP ksp, PetscInt, PetscReal rnorm, KSPConvergedReason* convergedReason, void*)
130{
131 // Compute rhs norm
132 Vec Rhs;
133 KSPGetRhs(ksp, &Rhs);
134 double rhs_norm;
135 VecNorm(Rhs, NORM_2, &rhs_norm);
136 if (rhs_norm < 1.0e-12) rhs_norm = 1.0;
137
138 // Compute test criterion
139 double tol;
140 int64_t maxIts;
141 KSPGetTolerances(ksp, nullptr, &tol, nullptr, &maxIts);
142
143 double relative_residual = rnorm / rhs_norm;
144
145 if (relative_residual < tol) *convergedReason = KSP_CONVERGED_RTOL;
146
147 return KSP_CONVERGED_ITERATING;
148}

263void
264CopyGlobalVecToSTLvector(Vec x,
265 const std::vector<int64_t>& global_indices,
266 std::vector<double>& data)
267{
268 // Populating local indices
269 size_t N = global_indices.size();
270 std::vector<int64_t> local_indices(N, 0);
271 size_t counter = 0;
272 for (auto val : global_indices)
273 {
274 local_indices[counter] = counter;
275 ++counter;
276 }
277
278 // Creating PETSc vector
279 Vec local_vec;
280 VecCreateSeq(PETSC_COMM_SELF, global_indices.size() + 1, &local_vec);
281 VecSet(local_vec, 0.0);
282
283 // Create and transfer index sets
284 IS global_set;
285 IS local_set;
286 ISCreateGeneral(PETSC_COMM_SELF, N, global_indices.data(), PETSC_COPY_VALUES, &global_set);
287 ISCreateGeneral(PETSC_COMM_SELF, N, local_indices.data(), PETSC_COPY_VALUES, &local_set);
288 VecScatter scat;
289 VecScatterCreate(x, global_set, local_vec, local_set, &scat);
290 VecScatterBegin(scat, x, local_vec, INSERT_VALUES, SCATTER_FORWARD);
291 VecScatterEnd(scat, x, local_vec, INSERT_VALUES, SCATTER_FORWARD);
292
293 // Copy to STL
294 data.clear();
295 data.resize(N, 0.0);
296 const double* x_ref;
297 VecGetArrayRead(local_vec, &x_ref);
298
299 std::copy(x_ref, x_ref + N, data.begin());
300
301 VecRestoreArrayRead(x, &x_ref);
302
303 // Cleanup
304 ISDestroy(&global_set);
305 ISDestroy(&local_set);
306
307 VecDestroy(&local_vec);
308}
