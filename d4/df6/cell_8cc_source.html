124int
125CellFace::GetNeighborAssociatedFace(const MeshContinuum& grid) const
126{
127 const auto& cur_face = *this; // just for readability
128 // Check index validity
129 if (not cur_face.has_neighbor_)
130 {
131 std::stringstream outstr;
132 outstr << "Invalid cell index encountered in call to "
133 << "CellFace::GetNeighborAssociatedFace. Index points "
134 << "to a boundary";
135 throw std::logic_error(outstr.str());
136 }
137
138 const auto& adj_cell = grid.cells[cur_face.neighbor_id_];
139
140 int associated_face = -1;
141 std::set<uint64_t> cfvids(cur_face.vertex_ids_.begin(),
142 cur_face.vertex_ids_.end()); // cur_face vertex ids
143
144 // Loop over adj cell faces
145 int af = -1;
146 for (const auto& adj_face : adj_cell.faces_)
147 {
148 ++af;
149 std::set<uint64_t> afvids(adj_face.vertex_ids_.begin(),
150 adj_face.vertex_ids_.end()); // adj_face vertex ids
151
152 if (afvids == cfvids)
153 {
154 associated_face = af;
155 break;
156 }
157 }
158
159 // Check associated face validity
160 if (associated_face < 0)
161 {
162 std::stringstream outstr;
163 outstr << "Could not find associated face in call to "
164 << "CellFace::GetNeighborAssociatedFace.\n"
165 << "Reference face with centroid at: " << cur_face.centroid_.PrintS() << "\n"
166 << "Adjacent cell: " << adj_cell.global_id_ << "\n";
167 for (size_t afi = 0; afi < adj_cell.faces_.size(); afi++)
168 {
169 outstr << "Adjacent cell face " << afi << " centroid "
170 << adj_cell.faces_[afi].centroid_.PrintS();
171 }
172 throw std::runtime_error(outstr.str());
173 }
174
175 return associated_face;
176}

178double
179CellFace::ComputeFaceArea(const MeshContinuum& grid) const
180{
181 if (vertex_ids_.size() <= 1) return 1.0;
182 else if (vertex_ids_.size() == 2)
183 {
184 const auto& v0 = grid.vertices[vertex_ids_[0]];
185 const auto& v1 = grid.vertices[vertex_ids_[1]];
186
187 return (v1 - v0).Norm();
188 }
189 else
190 {
191 double area = 0.0;
192 auto& v2 = centroid_;
193 const auto num_verts = vertex_ids_.size();
194 for (uint64_t v = 0; v < num_verts; ++v)
195 {
196 uint64_t vid0 = vertex_ids_[v];
197 uint64_t vid1 = (v < (num_verts - 1)) ? vertex_ids_[v + 1] : vertex_ids_[0];
198
199 const auto& v0 = grid.vertices[vid0];
200 const auto& v1 = grid.vertices[vid1];
201
202 auto v01 = v1 - v0;
203 auto v02 = v2 - v0;
204
205 Matrix3x3 J;
206 J.SetColJVec(0, v01);
207 J.SetColJVec(1, v02);
208 J.SetColJVec(2, Vector3(1.0, 1.0, 1.0));
209
210 area += 0.5 * std::fabs(J.Det());
211 }
212
213 return area;
214 }
215}

370void
371Cell::RecomputeCentroidsAndNormals(const MeshContinuum& grid)
372{
373 const auto k_hat = Vector3(0, 0, 1);
374
375 centroid_ = Vector3(0, 0, 0);
376 for (uint64_t vid : vertex_ids_)
377 centroid_ += grid.vertices[vid];
378 centroid_ /= static_cast<double>(vertex_ids_.size());
379
380 for (auto& face : faces_)
381 {
382 face.RecomputeCentroid(grid);
383
384 if (cell_type_ == CellType::POLYGON)
385 {
386 const auto v0 = grid.vertices[face.vertex_ids_[0]];
387 const auto v1 = grid.vertices[face.vertex_ids_[1]];
388
389 const auto v01 = v1 - v0;
390
391 face.normal_ = v01.Cross(k_hat).Normalized();
392 }
393 else if (cell_type_ == CellType::POLYHEDRON)
394 {
395 // A face of a polyhedron can itself be a polygon
396 // which can be multifaceted. Here we need the
397 // average normal over all the facets computed
398 // using an area-weighted average.
399 const size_t num_face_verts = face.vertex_ids_.size();
400 double total_area = 0.0;
401 auto weighted_normal = Vector3(0, 0, 0);
402 for (size_t fv = 0; fv < num_face_verts; ++fv)
403 {
404 size_t fvp1 = (fv < (num_face_verts - 1)) ? fv + 1 : 0;
405
406 uint64_t fvid_m = face.vertex_ids_[fv];
407 uint64_t fvid_p = face.vertex_ids_[fvp1];
408
409 auto leg_m = grid.vertices[fvid_m] - face.centroid_;
410 auto leg_p = grid.vertices[fvid_p] - face.centroid_;
411
412 auto vn = leg_m.Cross(leg_p);
413
414 double area = 0.5 * vn.Norm();
415 total_area += area;
416
417 weighted_normal = weighted_normal + area * vn.Normalized();
418 }
419 weighted_normal = weighted_normal / total_area;
420
421 face.normal_ = weighted_normal.Normalized();
422 }
423 }
