254double
255Logger::ProcessEvent(size_t ev_tag, Logger::EventOperation ev_operation)
256{
257 if (ev_tag >= repeating_events.size()) return 0.0;
258
259 RepeatingEvent& ref_rep_event = repeating_events[ev_tag];
260
261 double ret_val = 0.0;
262 switch (ev_operation)
263 {
264 case EventOperation::NUMBER_OF_OCCURRENCES:
265 {
266 for (auto& event : ref_rep_event.Events())
267 {
268 if ((event.ev_type == EventType::EVENT_CREATED) or
269 (event.ev_type == EventType::SINGLE_OCCURRENCE) or
270 (event.ev_type == EventType::EVENT_BEGIN))
271 ret_val += 1.0;
272 } // for events
273 break;
274 }
275 case EventOperation::TOTAL_DURATION:
276 {
277 double start_time = 0.0;
278 for (auto& event : ref_rep_event.Events())
279 {
280 if (event.ev_type == EventType::EVENT_BEGIN) start_time = event.ev_time;
281 if (event.ev_type == EventType::EVENT_END) ret_val += event.ev_time - start_time;
282 } // for events
283 ret_val *= 1000.0;
284 break;
285 }
286 case EventOperation::AVERAGE_DURATION:
287 {
288 double start_time = 0.0;
289 int counter = 0;
290 for (auto& event : ref_rep_event.Events())
291 {
292 if (event.ev_type == EventType::EVENT_BEGIN) start_time = event.ev_time;
293
294 if (event.ev_type == EventType::EVENT_END)
295 {
296 ret_val += event.ev_time - start_time;
297 counter++;
298 }
299 } // for events
300 ret_val /= (1000.0 * counter);
301 break;
302 }
303 case EventOperation::MAX_VALUE:
304 {
305 ret_val = 0.0;
306 for (auto& event : ref_rep_event.Events())
307 {
308 if ((event.ev_type == EventType::SINGLE_OCCURRENCE) or
309 (event.ev_type == EventType::EVENT_BEGIN) or (event.ev_type == EventType::EVENT_END))
310 {
311 if (event.ev_info != nullptr) ret_val = std::max(event.ev_info->arb_value, ret_val);
312 }
313 } // for events
314 break;
315 }
316 case EventOperation::AVERAGE_VALUE:
317 {
318 ret_val = 0.0;
319 int count = 0;
320 for (auto& event : ref_rep_event.Events())
321 {
322 if ((event.ev_type == EventType::SINGLE_OCCURRENCE) or
323 (event.ev_type == EventType::EVENT_BEGIN) or (event.ev_type == EventType::EVENT_END))
324 {
325 if (event.ev_info != nullptr)
326 {
327 ret_val += event.ev_info->arb_value;
328 ++count;
329 }
330 }
331 } // for events
332 if (count == 0) count = 1;
333 ret_val /= count;
334 break;
335 }
336 } // switch
337
338 return ret_val;
339}
