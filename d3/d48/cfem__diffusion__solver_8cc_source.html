47void
48Solver::Initialize()
49{
50 const std::string fname = "Solver::Initialize";
51 log.Log() << "\n"
52 << program_timer.GetTimeString() << " " << TextName()
53 << ": Initializing CFEM Diffusion solver ";
54
55 // Get grid
56 grid_ptr_ = GetCurrentHandler().GetGrid();
57 const auto& grid = *grid_ptr_;
58 if (grid_ptr_ == nullptr)
59 throw std::logic_error(std::string(__PRETTY_FUNCTION__) + " No grid defined.");
60
61 log.Log() << "Global num cells: " << grid.GetGlobalNumberOfCells();
62
63 // BIDs
64 auto globl_unique_bndry_ids = grid.GetDomainUniqueBoundaryIDs();
65
66 const auto& grid_boundary_id_map = grid_ptr_->GetBoundaryIDMap();
67 for (uint64_t bndry_id : globl_unique_bndry_ids)
68 {
69 if (grid_boundary_id_map.count(bndry_id) == 0)
70 throw std::logic_error(fname + ": Boundary id " + std::to_string(bndry_id) +
71 " does not have a name-assignment.");
72
73 const auto& bndry_name = grid_boundary_id_map.at(bndry_id);
74 if (boundary_preferences_.find(bndry_name) != boundary_preferences_.end())
75 {
76 BoundaryInfo bndry_info = boundary_preferences_.at(bndry_name);
77 auto& bndry_vals = bndry_info.second;
78 switch (bndry_info.first)
79 {
80 case BoundaryType::Reflecting:
81 {
82 boundaries_.insert(
83 std::make_pair(bndry_id, Boundary{BoundaryType::Reflecting, {0., 0., 0.}}));
84 log.Log() << "Boundary " << bndry_name << " set to reflecting.";
85 break;
86 }
87 case BoundaryType::Dirichlet:
88 {
89 if (bndry_vals.empty()) bndry_vals.resize(1, 0.0);
90 boundaries_.insert(
91 std::make_pair(bndry_id, Boundary{BoundaryType::Dirichlet, {bndry_vals[0], 0., 0.}}));
92 log.Log() << "Boundary " << bndry_name << " set to dirichlet.";
93 break;
94 }
95 case BoundaryType::Robin:
96 {
97 if (bndry_vals.size() != 3)
98 throw std::logic_error(std::string(__PRETTY_FUNCTION__) +
99 " Robin needs 3 values in bndry vals.");
100 boundaries_.insert(std::make_pair(
101 bndry_id,
102 Boundary{BoundaryType::Robin, {bndry_vals[0], bndry_vals[1], bndry_vals[2]}}));
103 log.Log() << "Boundary " << bndry_name << " set to robin." << bndry_vals[0] << ","
104 << bndry_vals[1] << "," << bndry_vals[2];
105 break;
106 }
107 case BoundaryType::Vacuum:
108 {
109 boundaries_.insert(
110 std::make_pair(bndry_id, Boundary{BoundaryType::Robin, {0.25, 0.5, 0.}}));
111 log.Log() << "Boundary " << bndry_name << " set to vacuum.";
112 break;
113 }
114 case BoundaryType::Neumann:
115 {
116 if (bndry_vals.size() != 3)
117 throw std::logic_error(std::string(__PRETTY_FUNCTION__) +
118 " Neumann needs 3 values in bndry vals.");
119 boundaries_.insert(std::make_pair(
120 bndry_id, Boundary{BoundaryType::Robin, {0., bndry_vals[0], bndry_vals[1]}}));
121 log.Log() << "Boundary " << bndry_name << " set to neumann." << bndry_vals[0];
122 break;
123 }
124 } // switch boundary type
125 }
126 else
127 {
128 boundaries_.insert(std::make_pair(bndry_id, Boundary{BoundaryType::Dirichlet, {0., 0., 0.}}));
129 log.Log0Verbose1() << "No boundary preference found for boundary index " << bndry_name
130 << "Dirichlet boundary added with zero boundary value.";
131 }
132 } // for bndry
133
134 // Make SDM
135 sdm_ptr_ = PieceWiseLinearContinuous::New(*grid_ptr_);
136 const auto& sdm = *sdm_ptr_;
137
138 const auto& OneDofPerNode = sdm.UNITARY_UNKNOWN_MANAGER;
139 num_local_dofs_ = sdm.GetNumLocalDOFs(OneDofPerNode);
140 num_globl_dofs_ = sdm.GetNumGlobalDOFs(OneDofPerNode);
141
142 log.Log() << "Num local DOFs: " << num_local_dofs_;
143 log.Log() << "Num globl DOFs: " << num_globl_dofs_;
144
145 // Initializes Mats and Vecs
146 const auto n = static_cast<int64_t>(num_local_dofs_);
147 const auto N = static_cast<int64_t>(num_globl_dofs_);
148
149 A_ = CreateSquareMatrix(n, N);
150 x_ = CreateVector(n, N);
151 b_ = CreateVector(n, N);
152
153 std::vector<int64_t> nodal_nnz_in_diag;
154 std::vector<int64_t> nodal_nnz_off_diag;
155 sdm.BuildSparsityPattern(nodal_nnz_in_diag, nodal_nnz_off_diag, OneDofPerNode);
156
157 InitMatrixSparsity(A_, nodal_nnz_in_diag, nodal_nnz_off_diag);
158
159 if (field_functions_.empty())
160 {
161 std::string solver_name;
162 if (not TextName().empty()) solver_name = TextName() + "-";
163
164 std::string text_name = solver_name + "phi";
165
166 auto initial_field_function =
167 std::make_shared<FieldFunctionGridBased>(text_name, sdm_ptr_, Unknown(UnknownType::SCALAR));
168
169 field_functions_.push_back(initial_field_function);
170 field_function_stack.push_back(initial_field_function);
171 } // if not ff set
172}

174void
175Solver::Execute()
176{
177 log.Log() << "\nExecuting CFEM Diffusion solver";
178
179 const auto& grid = *grid_ptr_;
180 const auto& sdm = *sdm_ptr_;
181
182 // Assemble the system
183 log.Log() << "Assembling system: ";
184 for (const auto& cell : grid.local_cells)
185 {
186 const auto& cell_mapping = sdm.GetCellMapping(cell);
187 const auto qp_data = cell_mapping.MakeVolumetricQuadraturePointData();
188
189 const auto imat = cell.material_id_;
190 const size_t num_nodes = cell_mapping.NumNodes();
191 MatDbl Acell(num_nodes, VecDbl(num_nodes, 0.0));
192 VecDbl cell_rhs(num_nodes, 0.0);
193
194 for (size_t i = 0; i < num_nodes; ++i)
195 {
196 for (size_t j = 0; j < num_nodes; ++j)
197 {
198 double entry_aij = 0.0;
199 for (size_t qp : qp_data.QuadraturePointIndices())
200 {
201 entry_aij += (d_coef_function_->Evaluate(imat, qp_data.QPointXYZ(qp)) *
202 qp_data.ShapeGrad(i, qp).Dot(qp_data.ShapeGrad(j, qp)) +
203 sigma_a_function_->Evaluate(imat, qp_data.QPointXYZ(qp)) *
204 qp_data.ShapeValue(i, qp) * qp_data.ShapeValue(j, qp)) *
205 qp_data.JxW(qp);
206 } // for qp
207 Acell[i][j] = entry_aij;
208 } // for j
209 for (size_t qp : qp_data.QuadraturePointIndices())
210 cell_rhs[i] += q_ext_function_->Evaluate(imat, qp_data.QPointXYZ(qp)) *
211 qp_data.ShapeValue(i, qp) * qp_data.JxW(qp);
212 } // for i
213
214 // Flag nodes for being on a boundary
215 std::vector<int> dirichlet_count(num_nodes, 0);
216 std::vector<double> dirichlet_value(num_nodes, 0.0);
217
218 const size_t num_faces = cell.faces_.size();
219 for (size_t f = 0; f < num_faces; ++f)
220 {
221 const auto& face = cell.faces_[f];
222 // not a boundary face
223 if (face.has_neighbor_) continue;
224
225 const auto& bndry = boundaries_[face.neighbor_id_];
226
227 // Robin boundary
228 if (bndry.type_ == BoundaryType::Robin)
229 {
230 const auto qp_face_data = cell_mapping.MakeSurfaceQuadraturePointData(f);
231 const size_t num_face_nodes = face.vertex_ids_.size();
232
233 const auto& aval = bndry.values_[0];
234 const auto& bval = bndry.values_[1];
235 const auto& fval = bndry.values_[2];
236
237 log.Log0Verbose1() << "Boundary set as Robin with a,b,f = (" << aval << "," << bval << ","
238 << fval << ") ";
239 // true Robin when a!=0, otherwise, it is a Neumann:
240 // Assert if b=0
241 if (std::fabs(bval) < 1e-8)
242 throw std::logic_error("if b=0, this is a Dirichlet BC, not a Robin BC");
243
244 // loop over nodes of that face
245 for (size_t fi = 0; fi < num_face_nodes; ++fi)
246 {
247 const uint i = cell_mapping.MapFaceNode(f, fi);
248
249 double entry_rhsi = 0.0;
250 for (size_t qp : qp_face_data.QuadraturePointIndices())
251 entry_rhsi += qp_face_data.ShapeValue(i, qp) * qp_face_data.JxW(qp);
252 cell_rhs[i] += fval / bval * entry_rhsi;
253
254 // only do this part if true Robin (i.e., a!=0)
255 if (std::fabs(aval) > 1.0e-8)
256 {
257 for (size_t fj = 0; fj < num_face_nodes; ++fj)
258 {
259 const uint j = cell_mapping.MapFaceNode(f, fj);
260
261 double entry_aij = 0.0;
262 for (size_t qp : qp_face_data.QuadraturePointIndices())
263 entry_aij += qp_face_data.ShapeValue(i, qp) * qp_face_data.ShapeValue(j, qp) *
264 qp_face_data.JxW(qp);
265 Acell[i][j] += aval / bval * entry_aij;
266 } // for fj
267 } // end true Robin
268 } // for fi
269 } // if Robin
270
271 // Dirichlet boundary
272 if (bndry.type_ == BoundaryType::Dirichlet)
273 {
274 const size_t num_face_nodes = face.vertex_ids_.size();
275
276 const auto& boundary_value = bndry.values_[0];
277
278 // loop over nodes of that face
279 for (size_t fi = 0; fi < num_face_nodes; ++fi)
280 {
281 const uint i = cell_mapping.MapFaceNode(f, fi);
282 dirichlet_count[i] += 1;
283 dirichlet_value[i] += boundary_value;
284 } // for fi
285 } // if Dirichlet
286
287 } // for face f
288
289 // Develop node mapping
290 std::vector<int64_t> imap(num_nodes, 0); // node-mapping
291 for (size_t i = 0; i < num_nodes; ++i)
292 imap[i] = sdm.MapDOF(cell, i);
293
294 // Assembly into system
295 for (size_t i = 0; i < num_nodes; ++i)
296 {
297 if (dirichlet_count[i] > 0) // if Dirichlet boundary node
298 {
299 MatSetValue(A_, imap[i], imap[i], 1.0, ADD_VALUES);
300 // because we use CFEM, a given node is common to several faces
301 const double aux = dirichlet_value[i] / dirichlet_count[i];
302 VecSetValue(b_, imap[i], aux, ADD_VALUES);
303 }
304 else
305 {
306 for (size_t j = 0; j < num_nodes; ++j)
307 {
308 if (dirichlet_count[j] == 0) // not related to a dirichlet node
309 MatSetValue(A_, imap[i], imap[j], Acell[i][j], ADD_VALUES);
310 else
311 {
312 const double aux = dirichlet_value[j] / dirichlet_count[j];
313 cell_rhs[i] -= Acell[i][j] * aux;
314 }
315 } // for j
316 VecSetValue(b_, imap[i], cell_rhs[i], ADD_VALUES);
317 }
318 } // for i
319 } // for cell
320
321 log.Log() << "Global assembly";
322
323 MatAssemblyBegin(A_, MAT_FINAL_ASSEMBLY);
324 MatAssemblyEnd(A_, MAT_FINAL_ASSEMBLY);
325 VecAssemblyBegin(b_);
326 VecAssemblyEnd(b_);
327
328 log.Log() << "Done global assembly";
329
330 // Create Krylov Solver
331 log.Log() << "Solving: ";
332 auto petsc_solver =
333 CreateCommonKrylovSolverSetup(A_,
334 TextName(),
335 KSPCG,
336 PCGAMG,
337 basic_options_("residual_tolerance").FloatValue(),
338 basic_options_("max_iters").IntegerValue());
339
340 // Solve
341 KSPSolve(petsc_solver.ksp, b_, x_);
342
343 UpdateFieldFunctions();
344
345 log.Log() << "Done solving";
346}
