56void
57LagrangeDiscontinuous::OrderNodes()
58{
59 const std::string fname = __FUNCTION__;
60 Timer t_stage[6];
61
62 t_stage[0].Reset();
63 // Check cell views avail
64 size_t num_loc_cells = ref_grid_.local_cells.size();
65
66 // Get local DOF count and set cell_local_block_address
67 cell_local_block_address_.resize(num_loc_cells, 0);
68
69 uint64_t local_node_count = 0;
70 for (const auto& cell : ref_grid_.local_cells)
71 {
72 const auto& cell_mapping = GetCellMapping(cell);
73 cell_local_block_address_[cell.local_id_] = static_cast<int64_t>(local_node_count);
74 local_node_count += cell_mapping.NumNodes();
75 }
76
77 // Allgather node_counts
78 mpi_comm.all_gather(local_node_count, locJ_block_size_);
79
80 // Assign
81 // local_block_address
82 uint64_t running_block_address = 0;
83 for (int locI = 0; locI < opensn::mpi_comm.size(); ++locI)
84 {
85 if (locI == opensn::mpi_comm.rank())
86 local_block_address_ = static_cast<int64_t>(running_block_address);
87
88 running_block_address += locJ_block_size_[locI];
89 }
90 const uint64_t global_node_count = running_block_address;
91
92 local_base_block_size_ = local_node_count;
93 globl_base_block_size_ = global_node_count;
94
95 // Collect ghost cell ids needing block addresses
96 std::map<int, std::vector<uint64_t>> ghost_cell_ids_consolidated;
97
98 for (uint64_t global_id : ref_grid_.cells.GetGhostGlobalIDs())
99 {
100 const auto& cell = ref_grid_.cells[global_id];
101 const int locI = static_cast<int>(cell.partition_id_);
102
103 std::vector<uint64_t>& locI_cell_id_list = ghost_cell_ids_consolidated[locI];
104
105 locI_cell_id_list.push_back(cell.global_id_);
106 }
107
108 // AllToAll to get query cell-ids
109 const std::map<int, std::vector<uint64_t>> query_ghost_cell_ids_consolidated =
110 MapAllToAll(ghost_cell_ids_consolidated);
111
112 // Map all query cell-ids
113 std::map<int, std::vector<uint64_t>> mapped_ghost_cell_ids_consolidated;
114 for (const auto& [pid, cell_id_list] : query_ghost_cell_ids_consolidated)
115 {
116 std::vector<uint64_t>& map_list = mapped_ghost_cell_ids_consolidated[pid];
117
118 for (uint64_t cell_global_id : cell_id_list)
119 {
120 const auto& cell = ref_grid_.cells[cell_global_id];
121
122 const uint64_t cell_block_address =
123 local_block_address_ + cell_local_block_address_[cell.local_id_];
124 map_list.push_back(cell_block_address);
125 }
126 }
127
128 // Communicate back the mapping
129 const std::map<int, std::vector<uint64_t>> global_id_mapping =
130 MapAllToAll(mapped_ghost_cell_ids_consolidated);
131
132 // Process global id mapping
133 for (const auto& [pid, mapping_list] : global_id_mapping)
134 {
135 const auto& global_id_list = ghost_cell_ids_consolidated.at(pid);
136
137 if (mapping_list.size() != global_id_list.size())
138 throw std::logic_error(fname + ": Ghost cell mapping error.");
139
140 const size_t list_size = mapping_list.size();
141 for (size_t k = 0; k < list_size; ++k)
142 neighbor_cell_block_address_.emplace_back(global_id_list[k],
143 static_cast<int64_t>(mapping_list[k]));
144 }
145
146 // Print info
147 log.LogAllVerbose2() << "Local dof count, start, total " << local_node_count << " "
148 << local_block_address_ << " " << global_node_count;
149}

151void
152LagrangeDiscontinuous::BuildSparsityPattern(std::vector<int64_t>& nodal_nnz_in_diag,
153 std::vector<int64_t>& nodal_nnz_off_diag,
154 const UnknownManager& unknown_manager) const
155{
156 //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% LOCAL CONNECTIVITY
157 size_t local_dof_count = local_base_block_size_;
158
159 nodal_nnz_in_diag.clear();
160 nodal_nnz_off_diag.clear();
161
162 nodal_nnz_in_diag.resize(local_dof_count, 0);
163 nodal_nnz_off_diag.resize(local_dof_count, 0);
164
165 int lc = 0;
166 for (const auto& cell : ref_grid_.local_cells)
167 {
168 const auto& cell_mapping = GetCellMapping(cell);
169 size_t num_nodes = cell_mapping.NumNodes();
170
171 // Self connection
172 for (int i = 0; i < num_nodes; ++i)
173 {
174 int64_t ir = cell_local_block_address_[lc] + i;
175 nodal_nnz_in_diag[ir] += static_cast<int64_t>(num_nodes);
176 }
177
178 // Local adjacent cell connections
179 for (auto& face : cell.faces_)
180 {
181 if (face.has_neighbor_ and face.IsNeighborLocal(ref_grid_))
182 {
183 const auto& adj_cell = ref_grid_.cells[face.neighbor_id_];
184 const auto& adj_cell_mapping = GetCellMapping(adj_cell);
185
186 for (int i = 0; i < num_nodes; ++i)
187 {
188 int64_t ir = cell_local_block_address_[lc] + i;
189 nodal_nnz_in_diag[ir] += static_cast<int64_t>(adj_cell_mapping.NumNodes());
190 }
191 }
192 } // for face
193 ++lc;
194 } // for local cell
195
196 //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% NEIGHBORING CONNECTIVITY
197 lc = 0;
198 for (auto& cell : ref_grid_.local_cells)
199 {
200 const auto& cell_mapping = GetCellMapping(cell);
201
202 // Local adjacent cell connections
203 for (auto& face : cell.faces_)
204 {
205 if (face.has_neighbor_ and (not face.IsNeighborLocal(ref_grid_)))
206 {
207 const auto& adj_cell = ref_grid_.cells[face.neighbor_id_];
208 const auto& adj_cell_mapping = GetCellMapping(adj_cell);
209
210 for (int i = 0; i < cell_mapping.NumNodes(); ++i)
211 {
212 int64_t ir = cell_local_block_address_[lc] + i;
213 nodal_nnz_off_diag[ir] += static_cast<int64_t>(adj_cell_mapping.NumNodes());
214 }
215 }
216 }
217 ++lc;
218 } // for local cell
219
220 // Spacing according to unknown manager
221 auto backup_nnz_in_diag = nodal_nnz_in_diag;
222 auto backup_nnz_off_diag = nodal_nnz_off_diag;
223
224 unsigned int N = unknown_manager.GetTotalUnknownStructureSize();
225
226 nodal_nnz_in_diag.clear();
227 nodal_nnz_off_diag.clear();
228
229 nodal_nnz_in_diag.resize(local_base_block_size_ * N, 0);
230 nodal_nnz_off_diag.resize(local_base_block_size_ * N, 0);
231
232 if (unknown_manager.dof_storage_type_ == UnknownStorageType::NODAL)
233 {
234 int ir = -1;
235 for (int i = 0; i < local_base_block_size_; ++i)
236 {
237 for (int j = 0; j < N; ++j)
238 {
239 ++ir;
240 nodal_nnz_in_diag[ir] = backup_nnz_in_diag[i];
241 nodal_nnz_off_diag[ir] = backup_nnz_off_diag[i];
242 } // for j
243 } // for i
244 }
245 else if (unknown_manager.dof_storage_type_ == UnknownStorageType::BLOCK)
246 {
247 int ir = -1;
248 for (int j = 0; j < N; ++j)
249 {
250 for (int i = 0; i < local_base_block_size_; ++i)
251 {
252 ++ir;
253 nodal_nnz_in_diag[ir] = backup_nnz_in_diag[i];
254 nodal_nnz_off_diag[ir] = backup_nnz_off_diag[i];
255 } // for i
256 } // for j
257 }
258
259 opensn::mpi_comm.barrier();
260}

262int64_t
263LagrangeDiscontinuous::MapDOF(const Cell& cell,
264 const unsigned int node,
265 const UnknownManager& unknown_manager,
266 const unsigned int unknown_id,
267 const unsigned int component) const
268{
269 auto storage = unknown_manager.dof_storage_type_;
270
271 size_t num_unknowns = unknown_manager.GetTotalUnknownStructureSize();
272 size_t block_id = unknown_manager.MapUnknown(unknown_id, component);
273
274 if (cell.partition_id_ == opensn::mpi_comm.rank())
275 {
276 if (storage == UnknownStorageType::BLOCK)
277 {
278 int64_t address = static_cast<int64_t>(local_block_address_ * num_unknowns) +
279 cell_local_block_address_[cell.local_id_] +
280 local_base_block_size_ * block_id + node;
281 return address;
282 }
283 else if (storage == UnknownStorageType::NODAL)
284 {
285 int64_t address = static_cast<int64_t>(local_block_address_ * num_unknowns) +
286 cell_local_block_address_[cell.local_id_] * num_unknowns +
287 node * num_unknowns + block_id;
288 return address;
289 }
290 }
291 else
292 {
293 int index = 0;
294 bool found = false;
295 for (auto neighbor_info : neighbor_cell_block_address_)
296 {
297 if (neighbor_info.first == cell.global_id_)
298 {
299 found = true;
300 break;
301 }
302 ++index;
303 }
304
305 if (!found)
306 {
307 log.LogAllError() << "SpatialDiscretization_PWL::MapDFEMDOF. Mapping failed for cell "
308 << "with global index " << cell.global_id_ << " and partition-ID "
309 << cell.partition_id_;
310 Exit(EXIT_FAILURE);
311 }
312
313 if (storage == UnknownStorageType::BLOCK)
314 {
315 int64_t address = static_cast<int64_t>(neighbor_cell_block_address_[index].second) +
316 locJ_block_size_[cell.partition_id_] * block_id + node;
317 return address;
318 }
319 else if (storage == UnknownStorageType::NODAL)
320 {
321 int64_t address =
322 static_cast<int64_t>(neighbor_cell_block_address_[index].second * num_unknowns) +
323 node * num_unknowns + block_id;
324 return address;
325 }
326 }
327
328 return -1;
329}
