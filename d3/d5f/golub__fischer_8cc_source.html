63void
64GolubFischer::MCA(Tvecdbl& in_mell, Tvecdbl& a, Tvecdbl& b, Tvecdbl& c)
65{
66 log.Log(Logger::LOG_LVL::LOG_0VERBOSE_2) << "MCA Start" << '\n';
67
68 int N = in_mell.size() - 1;
69 int n = (N + 1) / 2;
70
71 log.Log(Logger::LOG_LVL::LOG_0VERBOSE_2) << "N " << N << " n " << n << '\n';
72 log.Log(Logger::LOG_LVL::LOG_0VERBOSE_2) << "alpha, beta" << '\n';
73
74 alpha_.resize(n + 1, 0.0);
75 beta_.resize(n + 1, 0.0);
76
77 std::vector<std::vector<double>> sigma(n + 1, std::vector<double>(2 * n + 1, 0.0));
78
79 for (int ell = 0; ell < 2 * n; ell++)
80 {
81 sigma[0][ell] = in_mell[ell];
82 }
83
84 alpha_[0] = a[0] + c[0] * sigma[0][1] / sigma[0][0];
85 beta_[0] = in_mell[0];
86
87 log.Log(Logger::LOG_LVL::LOG_0VERBOSE_2) << 0 << " " << alpha_[0] << " " << beta_[0] << "\n";
88
89 for (int k = 1; k < n + 1; k++)
90 {
91 for (int ell = k; ell < (2 * n - k + 1); ell++)
92 {
93 double sigmakm2ell = 0.0;
94
95 if (k == 1) { sigmakm2ell = 0.0; }
96 else { sigmakm2ell = sigma[k - 2][ell]; }
97 sigma[k][ell] = c[ell] * sigma[k - 1][ell + 1] -
98 (alpha_[k - 1] - a[ell]) * sigma[k - 1][ell] - beta_[k - 1] * sigmakm2ell +
99 b[ell] * sigma[k - 1][ell - 1];
100 }
101 alpha_[k] = a[k] - c[k - 1] * (sigma[k - 1][k] / sigma[k - 1][k - 1]) +
102 c[k] * (sigma[k][k + 1] / sigma[k][k]);
103 beta_[k] = c[k - 1] * sigma[k][k] / sigma[k - 1][k - 1];
104
105 log.Log(Logger::LOG_LVL::LOG_0VERBOSE_2) << k << " " << alpha_[k] << " " << beta_[k] << "\n";
106 }
107
108 log.Log(Logger::LOG_LVL::LOG_0VERBOSE_2) << "Done" << '\n';
109}

112GolubFischer::RootsOrtho(int& N, Tvecdbl& in_alpha, Tvecdbl& in_beta)
113{

115
116 int maxiters = 1000;
117 double tol = 1.0e-6;
118 double adder = 0.999 * 2 / std::max(N - 1, 1);

120 log.Log(Logger::LOG_LVL::LOG_0VERBOSE_2) << "Check 1: Init guess" << '\n';

126

128 {

130 log.Log(Logger::LOG_LVL::LOG_0VERBOSE_2) << "x[" << i << "]=" << xn[i] << "\n";
131 }
132
133 Tvecdbl norm;
134 norm.resize(N + 1, 0.0);
135 log.Log(Logger::LOG_LVL::LOG_0VERBOSE_2) << "Check 2 " << in_beta[0] << '\n';
136 norm[0] = in_beta[0];
137 log.Log(Logger::LOG_LVL::LOG_0VERBOSE_2) << "Check 3a norms" << '\n';
138 log.Log(Logger::LOG_LVL::LOG_0VERBOSE_2) << norm[0] << '\n';
139 for (int i = 1; i < (N + 1); i++)
140 {
141 norm[i] = in_beta[i] * norm[i - 1];
142 log.Log(Logger::LOG_LVL::LOG_0VERBOSE_2) << norm[i] << '\n';
143 }
144
145 log.Log(Logger::LOG_LVL::LOG_0VERBOSE_2) << "Check 3" << '\n';
146
147 for (int k = 0; k < N; k++)
148 {
149 int i = 0;
150
151 while (i < maxiters)
152 {
153 double xold = xn[k];
154 double a = Ortho(N, xold, in_alpha, in_beta);
155 double b = dOrtho(N, xold, in_alpha, in_beta);
156 double c = 0;
157
158 for (int j = 0; j < k; j++)
159 {
160 c = c + (1.0 / (xold - xn[j]));
161 }
162
163 double xnew = xold - (a / (b - a * c));
164 if (std::isnan(xnew))
165 {
166 log.Log(Logger::LOG_LVL::LOG_0VERBOSE_2)
167 << "xnew " << i << " " << xnew << " y=" << a << std::endl;
168 Exit(EXIT_FAILURE);
169 }
170
171 double res = std::fabs(xnew - xold);
172 xn[k] = xnew;
173
174 log.Log(Logger::LOG_LVL::LOG_0VERBOSE_2)
175 << "xnew " << i << " " << xnew << " y=" << a << std::endl;
176
177 if (res < tol) { break; }
178
179 i = i + 1;
180 } // while
181
182 } // for k
183
184 for (int i = 0; i < N - 1; i++)
185 {
186 for (int j = 0; j < N - i - 1; j++)
187 {
188 if (xn[j] > xn[j + 1])
189 {
190 double tempx = xn[j + 1];
191 double tempw = wn[j + 1];
192 xn[j + 1] = xn[j];
193 wn[j + 1] = wn[j];
194 xn[j] = tempx;
195 wn[j] = tempw;
196 }
197 }
198 }
199
200 for (int i = 0; i < N; i++)
201 {
202 wn[i] = 0.0;
203
204 for (int k = 0; k < N; k++)
205 {
206 wn[i] += Ortho(k, xn[i], in_alpha, in_beta) * Ortho(k, xn[i], in_alpha, in_beta) / norm[k];
207 }
208
209 wn[i] = 1.0 / wn[i];
210 }
211 for (int i = 0; i < N; i++)
212 {
213 xn_wn_[i].first = xn[i];
214 xn_wn_[i].second = wn[i];
215 }

217 log.Log(Logger::LOG_LVL::LOG_0VERBOSE_2) << "Done" << '\n';
218}

220double
221GolubFischer::Ortho(int ell, double x, Tvecdbl& in_alpha, Tvecdbl& in_beta)
222{
223 if (ell == 0) { return 1; }
224
225 if (ell == 1) { return (x - in_alpha[0]); }
226
227 double Pnm1 = 1.0;
228 double Pn = (x - in_alpha[0]);
229 double Pnp1 = 0.0;
230
231 for (int n = 2; n < ell + 1; n++)
232 {
233 int ns = n - 1;
234 Pnp1 = (x - in_alpha[ns]) * Pn - in_beta[ns] * Pnm1;
235 Pnm1 = Pn;
236 Pn = Pnp1;
237 }
238
239 return Pnp1;
240}
