154 // Copy row indexes and values into vector of pairs

156 target.reserve(indices.size());
157
158 auto index = indices.begin();
159 auto value = values.begin();
160 for (; index != indices.end(); ++index, ++value)
161 {
162 target.emplace_back(*index, *value);

169 {
170 return a.first < b.first;
171 }
172 } compare_index;
173
174 // Sort
175 std::stable_sort(target.begin(), target.end(), compare_index);
176
177 // Copy back
178 indices.clear();
179 values.clear();
