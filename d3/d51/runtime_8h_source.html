81/**Attempts to retrieve an object of base-type `shared_ptr<T>` at the given
82 * handle. It then attempts to cast it to type `shared_ptr<R>` and, if
83 * successful, will return a reference of type R&.
84 * \n
85 * \n
86 * Example usage:
87 *
88 * \code
89 * const auto& surf_mesh = GetStackItem<SurfaceMesh>(object_stack, surface_hndl);
90 // Returns SurfaceMesh&
91 * \endcode
92 * */
93template <class R, class T>
94static R&
95GetStackItem(std::vector<std::shared_ptr<T>>& stack,
96 const size_t handle,
97 const std::string& calling_function_name = "Unknown")
98{
99 try
100 {
101 std::shared_ptr<T>& item = stack.at(handle);
102 std::shared_ptr<R> ret_item = std::dynamic_pointer_cast<R>(item);
103 if (not ret_item)
104 throw std::logic_error("GetStackItem: Invalid return type used. Calling function: " +
105 calling_function_name);
106 return *ret_item;
107 }
108 catch (const std::out_of_range& oor)
109 {
110 throw std::out_of_range("GetStackItem: Invalid handle used. Calling function: " +
111 calling_function_name);
112 }
113}

115/**Attempts to obtain object of type `shared_ptr<T>` at the given
116 * handle of a stack with parent type P.
117 * \n
118 * \n
119 * Example usage:
120 *
121 * \code
122 * auto surf_mesh_ptr =
123 * GetStackItemPtrAsType<SurfaceMesh>(object_stack, surf_mesh_hndle, fname);
124 // Returns std::shared_ptr<SurfaceMesh>
125 * \endcode
126 * */
127template <class T, class P>
128static std::shared_ptr<T>
129GetStackItemPtrAsType(std::vector<std::shared_ptr<P>>& stack,
130 const size_t handle,
131 const std::string& calling_function_name = "Unknown")
132{
133 std::shared_ptr<P> item_type_P;
134 try
135 {
136 item_type_P = stack.at(handle);
137 }
138 catch (const std::out_of_range& oor)
139 {
140 throw std::out_of_range("GetStackItem: Invalid handle used. Calling function: " +
141 calling_function_name);
142 }
143
144 auto item_type_T = std::dynamic_pointer_cast<T>(item_type_P);
145 if (not item_type_T)
146 throw std::logic_error(calling_function_name + "Failed to cast to requested type");
147
148 return item_type_T;
149}
