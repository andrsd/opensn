64WGSLinearSolver::SetSystem()
65{
66 if (IsSystemSet()) return;
67
68 x_ = CreateVector(static_cast<int64_t>(num_local_dofs_), static_cast<int64_t>(num_global_dofs_));
69
70 VecSet(x_, 0.0);

75 static_cast<int64_t>(num_local_dofs_),

77 static_cast<int64_t>(num_global_dofs_),

80 &A_);

84

115void
116WGSLinearSolver::SetInitialGuess()
117{
118 // If the norm of the initial guess is large enough, the initial guess will be used, otherwise it
119 // is assumed to be zero.
120
121 auto gs_context_ptr = std::dynamic_pointer_cast<WGSContext>(context_ptr_);
122
123 auto& groupset = gs_context_ptr->groupset_;
124 auto& lbs_solver = gs_context_ptr->lbs_solver_;
125
126 lbs_solver.SetGSPETScVecFromPrimarySTLvector(groupset, x_, PhiSTLOption::PHI_OLD);
127
128 double init_guess_norm = 0.0;
129 VecNorm(x_, NORM_2, &init_guess_norm);
130
131 if (init_guess_norm > 1.0e-10)
132 {
133 KSPSetInitialGuessNonzero(ksp_, PETSC_TRUE);
134 if (gs_context_ptr->log_info_) log.Log() << "Using phi_old as initial guess.";
135 }

138void
139WGSLinearSolver::SetRHS()
140{
141 auto gs_context_ptr = std::dynamic_pointer_cast<WGSContext>(context_ptr_);
142
143 auto& groupset = gs_context_ptr->groupset_;
144 auto& lbs_solver = gs_context_ptr->lbs_solver_;
145
146 if (gs_context_ptr->log_info_) log.Log() << program_timer.GetTimeString() << " Computing b";
147
148 // SetSource for RHS
149 saved_q_moments_local_ = lbs_solver.QMomentsLocal();
150
151 const bool single_richardson =
152 iterative_method_ == "richardson" and tolerance_options_.maximum_iterations == 1;
153
154 if (not single_richardson)
155 {
156 const auto scope = gs_context_ptr->rhs_src_scope_ | ZERO_INCOMING_DELAYED_PSI;
157 gs_context_ptr->set_source_function_(
158 groupset, lbs_solver.QMomentsLocal(), lbs_solver.PhiOldLocal(), scope);
159
160 // Apply transport operator
161 gs_context_ptr->ApplyInverseTransportOperator(scope);
162
163 // Assemble PETSc vector
164 lbs_solver.SetGSPETScVecFromPrimarySTLvector(groupset, b_, PhiSTLOption::PHI_NEW);
165
166 // Compute RHS norm
167 VecNorm(b_, NORM_2, &context_ptr_->rhs_norm);
168
169 // Compute precondition RHS norm
170 PC pc;
171 KSPGetPC(ksp_, &pc);
172 Vec temp_vec;
173 VecDuplicate(b_, &temp_vec);
174 PCApply(pc, b_, temp_vec);
175 VecNorm(temp_vec, NORM_2, &context_ptr_->rhs_preconditioned_norm);
176 VecDestroy(&temp_vec);
177 }
178 // If we have a single richardson iteration then the user probably wants
179 // only a single sweep. Therefore, we are going to combine the scattering
180 // source (normally included in the lhs_src_scope) into the sweep for the
181 // RHS, and just suppress the kspsolve part.
182 else
183 {
184 const auto scope = gs_context_ptr->rhs_src_scope_ | gs_context_ptr->lhs_src_scope_;
185 gs_context_ptr->set_source_function_(
186 groupset, lbs_solver.QMomentsLocal(), lbs_solver.PhiOldLocal(), scope);
187
188 // Apply transport operator
189 gs_context_ptr->ApplyInverseTransportOperator(scope);
190
191 // Assemble PETSc vector
192 lbs_solver.SetGSPETScVecFromPrimarySTLvector(groupset, x_, PhiSTLOption::PHI_NEW);
193
194 // Compute RHS norm
195 VecNorm(x_, NORM_2, &context_ptr_->rhs_norm);
196
197 // Compute precondition RHS norm
198 PC pc;
199 KSPGetPC(ksp_, &pc);
200 Vec temp_vec;
201 VecDuplicate(x_, &temp_vec);
202 PCApply(pc, x_, temp_vec);
203 VecNorm(temp_vec, NORM_2, &context_ptr_->rhs_preconditioned_norm);
204 VecDestroy(&temp_vec);
205
206 SetKSPSolveSuppressionFlag(true);
207 }
208}

210void
211WGSLinearSolver::PostSolveCallback()
212{
213 // We simply restore the q_moments_local vector.
214
215 // Get convergence reason
216 if (not GetKSPSolveSuppressionFlag())
217 {
218 KSPConvergedReason reason;
219 KSPGetConvergedReason(ksp_, &reason);
220 if (reason != KSP_CONVERGED_RTOL and reason != KSP_DIVERGED_ITS)
221 log.Log0Warning() << "Krylov solver failed. "
222 << "Reason: " << GetPETScConvergedReasonstring(reason);
223 }
224
225 // Copy x to local solution
226 auto gs_context_ptr = std::dynamic_pointer_cast<WGSContext>(context_ptr_);
227
228 auto& groupset = gs_context_ptr->groupset_;
229 auto& lbs_solver = gs_context_ptr->lbs_solver_;
230
231 lbs_solver.SetPrimarySTLvectorFromGSPETScVec(groupset, x_, PhiSTLOption::PHI_NEW);
232 lbs_solver.SetPrimarySTLvectorFromGSPETScVec(groupset, x_, PhiSTLOption::PHI_OLD);
233
234 // Restore saved q_moms
235 lbs_solver.QMomentsLocal() = saved_q_moments_local_;
236
237 // Context specific callback
238 gs_context_ptr->PostSolveCallback();
239}
