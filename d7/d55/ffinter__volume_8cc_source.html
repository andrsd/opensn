69 node_dof_values[i] = field_data[imap];
70 } // for i
71
72 if (cell_local_id == cell_local_ids_inside_logvol_.front())
73 {
74 local_max = node_dof_values.front();

76 }
77
78 for (size_t i = 0; i < num_nodes; ++i)
79 {
80 local_max = std::fmax(node_dof_values[i], local_max);
81 local_min = std::fmin(node_dof_values[i], local_min);
82 }

92 function_value = oper_function_->Evaluate(ff_value, cell.material_id_);
93
94 local_volume += qp_data.JxW(qp);
95 local_sum += function_value * qp_data.JxW(qp);
96 local_max = std::fmax(ff_value, local_max);
97 local_min = std::fmin(ff_value, local_min);
98 } // for qp
99 } // for cell-id
100

102 {
103 mpi_comm.all_reduce(local_sum, op_value_, mpi::op::sum<double>());
104 }
105 if (op_type_ == Operation::OP_AVG or op_type_ == Operation::OP_AVG_FUNC)

111 double global_volume = global_data[0];

114 }
