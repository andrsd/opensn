90std::pair<std::set<uint32_t>, std::set<uint32_t>>
91SpatialDiscretization::MakeCellInternalAndBndryNodeIDs(const Cell& cell) const
92{
93 const auto& cell_mapping = GetCellMapping(cell);
94 const size_t num_faces = cell.faces_.size();
95 const size_t num_nodes = cell_mapping.NumNodes();
96
97 // Determine which nodes are on the boundary
98 std::set<uint32_t> boundary_nodes;
99 for (size_t f = 0; f < num_faces; ++f)
100 {
101 if (not cell.faces_[f].has_neighbor_)
102 {
103 const size_t num_face_nodes = cell_mapping.NumFaceNodes(f);
104 for (size_t fi = 0; fi < num_face_nodes; ++fi)
105 boundary_nodes.insert(cell_mapping.MapFaceNode(f, fi));
106 }
107 } // for f
108
109 // Determine non-boundary nodes
110 std::set<uint32_t> internal_nodes;
111 for (size_t i = 0; i < num_nodes; ++i)
112 if (boundary_nodes.find(i) == boundary_nodes.end()) internal_nodes.insert(i);
113
114 return {internal_nodes, boundary_nodes};
115}

117std::vector<std::vector<std::vector<int>>>
118SpatialDiscretization::MakeInternalFaceNodeMappings(const double tolerance) const
119{
120 typedef std::vector<int> FaceAdjMapping;
121 typedef std::vector<FaceAdjMapping> PerFaceAdjMapping;
122 typedef std::vector<PerFaceAdjMapping> CellAdjMapping;
123
124 const auto& grid = this->ref_grid_;
125
126 CellAdjMapping cell_adj_mapping;
127 for (const auto& cell : grid.local_cells)
128 {
129 const auto& cell_mapping = this->GetCellMapping(cell);
130 const auto& node_locations = cell_mapping.GetNodeLocations();
131 const size_t num_faces = cell.faces_.size();
132
133 PerFaceAdjMapping per_face_adj_mapping;
134
135 for (size_t f = 0; f < num_faces; ++f)
136 {
137 const auto& face = cell.faces_[f];
138 const auto num_face_nodes = cell_mapping.NumFaceNodes(f);
139 FaceAdjMapping face_adj_mapping(num_face_nodes, -1);
140 if (face.has_neighbor_)
141 {
142 const auto& adj_cell = grid.cells[face.neighbor_id_];
143 const auto& adj_cell_mapping = this->GetCellMapping(adj_cell);
144 const auto& adj_node_locations = adj_cell_mapping.GetNodeLocations();
145 const size_t adj_num_nodes = adj_cell_mapping.NumNodes();
146
147 for (size_t fi = 0; fi < num_face_nodes; ++fi)
148 {
149 const int i = cell_mapping.MapFaceNode(f, fi);
150 const auto& ivec3 = node_locations[i];
151
152 for (size_t ai = 0; ai < adj_num_nodes; ++ai)
153 {
154 const auto& aivec3 = adj_node_locations[ai];
155 if ((ivec3 - aivec3).NormSquare() < tolerance)
156 {
157 face_adj_mapping[fi] = static_cast<int>(ai);
158 break;
159 }
160 } // for ai
161 if (face_adj_mapping[fi] < 0) throw std::logic_error("Face node mapping failed");
162 } // for fi
163 } // if internal face
164
165 per_face_adj_mapping.push_back(std::move(face_adj_mapping));
166 } // for face
167
168 cell_adj_mapping.push_back(std::move(per_face_adj_mapping));
169 } // for cell
170
171 return cell_adj_mapping;
172}

176 std::vector<double>& to_vector,
177 const UnknownManager& from_vec_uk_structure,
178 const unsigned int from_vec_uk_id,
179 const UnknownManager& to_vec_uk_structure,
180 const unsigned int to_vec_uk_id) const
181{
182 const std::string fname = "spatial_discretization::"
183 "CopyVectorWithUnknownScope";
184 const auto& ukmanF = from_vec_uk_structure;
185 const auto& ukmanT = to_vec_uk_structure;
186 const auto& ukidF = from_vec_uk_id;
187 const auto& ukidT = to_vec_uk_id;
188 try
189 {
190 const auto& ukA = from_vec_uk_structure.unknowns_.at(from_vec_uk_id);
191 const auto& ukB = to_vec_uk_structure.unknowns_.at(to_vec_uk_id);
192
193 if (ukA.num_components_ != ukB.num_components_)
194 throw std::logic_error(fname + " Unknowns do not have the "
195 "same number of components");
196
197 const size_t num_comps = ukA.num_components_;
198
199 for (const auto& cell : ref_grid_.local_cells)
200 {
201 const auto& cell_mapping = this->GetCellMapping(cell);
202 const size_t num_nodes = cell_mapping.NumNodes();
203
204 for (size_t i = 0; i < num_nodes; ++i)
205 {
206 for (size_t c = 0; c < num_comps; ++c)
207 {
208 const int64_t fmap = MapDOFLocal(cell, i, ukmanF, ukidF, c);
209 const int64_t imap = MapDOFLocal(cell, i, ukmanT, ukidT, c);
210
211 to_vector[imap] = from_vector[fmap];
212 } // for component c
213 } // for node i
214 } // for cell
215 }
216 catch (const std::out_of_range& oor)
217 {
218 throw std::out_of_range(fname + ": either from_vec_uk_id or to_vec_uk_id is "
