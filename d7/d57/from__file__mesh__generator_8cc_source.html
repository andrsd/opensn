49FromFileMeshGenerator::GenerateUnpartitionedMesh(std::unique_ptr<UnpartitionedMesh> input_umesh)
50{
51 ChiInvalidArgumentIf(input_umesh != nullptr,
52 "FromFileMeshGenerator can not be preceded by another"
53 " mesh generator because it cannot process an input mesh");
54
55 UnpartitionedMesh::Options options;
56 options.file_name = filename_;
57 options.scale = scale_;
58 options.material_id_fieldname = material_id_fieldname_;
59 options.boundary_id_fieldname = boundary_id_fieldname_;
60
61 const std::filesystem::path filepath(filename_);
62 const std::string extension = filepath.extension();
63
64 auto umesh = std::make_unique<UnpartitionedMesh>();
65
66 log.Log() << "FromFileMeshGenerator: Generating UnpartitionedMesh";
67
68 if (extension == ".obj") umesh->ReadFromWavefrontOBJ(options);
69 else if (extension == ".msh")
70 umesh->ReadFromMsh(options);
71 else if (extension == ".e")
72 umesh->ReadFromExodus(options);
73 else if (extension == ".vtu")
74 umesh->ReadFromVTU(options);
75 else if (extension == ".pvtu")
76 umesh->ReadFromPVTU(options);
77 else if (extension == ".case")
78 umesh->ReadFromEnsightGold(options);
79 else
80 ChiInvalidArgument("Unsupported file type \"" + extension +
81 "\". Supported types limited to"
82 ".obj, .msh, .e, .vtu, .pvtu, .case.");
83
84 log.Log() << "FromFileMeshGenerator: Done generating UnpartitionedMesh";
85 return umesh;
86}
