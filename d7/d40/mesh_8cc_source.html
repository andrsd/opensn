31size_t
32CreateUnpartitioned1DOrthoMesh(std::vector<double>& vertices)
33{
34 ChiLogicalErrorIf(vertices.empty(), "Empty vertex list.");
35
36 // Get current mesh handler
37 auto& handler = GetCurrentHandler();
38
39 // Reorient 1D verts along z
40 std::vector<Vertex> zverts;
41 zverts.reserve(vertices.size());
42 for (double z_coord : vertices)
43 zverts.emplace_back(0.0, 0.0, z_coord);
44
45 // Create unpartitioned mesh
46 auto umesh = std::make_shared<UnpartitionedMesh>();
47
48 umesh->GetMeshAttributes() = DIMENSION_1 | ORTHOGONAL;
49
50 // Create vertices
51 size_t Nz = vertices.size();
52
53 umesh->GetMeshOptions().ortho_Nx = 1;
54 umesh->GetMeshOptions().ortho_Ny = 1;
55 umesh->GetMeshOptions().ortho_Nz = Nz - 1;
56 umesh->GetMeshOptions().boundary_id_map[4] = "ZMAX";
57 umesh->GetMeshOptions().boundary_id_map[5] = "ZMIN";
58
59 umesh->GetVertices().reserve(Nz);
60 for (auto& vertex : zverts)
61 umesh->GetVertices().push_back(vertex);
62
63 // Create cells
64 for (size_t c = 0; c < (zverts.size() - 1); ++c)
65 {
66 auto cell = new UnpartitionedMesh::LightWeightCell(CellType::SLAB, CellType::SLAB);
67
68 cell->vertex_ids = {c, c + 1};
69
70 UnpartitionedMesh::LightWeightFace left_face;
71 UnpartitionedMesh::LightWeightFace rite_face;
72
73 left_face.vertex_ids = {c};
74 rite_face.vertex_ids = {c + 1};
75
76 if (c == 0) left_face.neighbor = 5 /*ZMIN*/;
77 if (c == (zverts.size() - 2)) rite_face.neighbor = 4 /*ZMAX*/;
78
79 cell->faces.push_back(left_face);
80 cell->faces.push_back(rite_face);
81
82 umesh->AddCell(cell);
83 }
84
85 umesh->ComputeCentroidsAndCheckQuality();
86 umesh->BuildMeshConnectivity();
87
88 unpartitionedmesh_stack.push_back(umesh);
89
90 // Create meshers

100CreateUnpartitioned2DOrthoMesh(std::vector<double>& vertices_1d_x,

107

110
111 umesh->GetMeshAttributes() = DIMENSION_2 | ORTHOGONAL;
112
113 // Create vertices
114 size_t Nx = vertices_1d_x.size();
115 size_t Ny = vertices_1d_y.size();
116
117 umesh->GetMeshOptions().ortho_Nx = Nx - 1;
118 umesh->GetMeshOptions().ortho_Ny = Ny - 1;
119 umesh->GetMeshOptions().ortho_Nz = 1;
120 umesh->GetMeshOptions().boundary_id_map[0] = "XMAX";
121 umesh->GetMeshOptions().boundary_id_map[1] = "XMIN";
122 umesh->GetMeshOptions().boundary_id_map[2] = "YMAX";
123 umesh->GetMeshOptions().boundary_id_map[3] = "YMIN";
124
125 typedef std::vector<uint64_t> VecIDs;
126 std::vector<VecIDs> vertex_ij_to_i_map(Ny, VecIDs(Nx));
127 umesh->GetVertices().reserve(Nx * Ny);
128 uint64_t k = 0;
129 for (size_t i = 0; i < Ny; ++i)
130 {
131 for (size_t j = 0; j < Nx; ++j)
132 {
133 vertex_ij_to_i_map[i][j] = k++;
134 umesh->GetVertices().emplace_back(vertices_1d_x[j], vertices_1d_y[i], 0.0);
135 } // for j
136 } // for i
137
138 // Create cells
139 auto& vmap = vertex_ij_to_i_map;
140 for (size_t i = 0; i < (Ny - 1); ++i)
141 {
142 for (size_t j = 0; j < (Nx - 1); ++j)
143 {
144 auto cell =
145 new UnpartitionedMesh::LightWeightCell(CellType::POLYGON, CellType::QUADRILATERAL);
146
147 // vertex ids: face ids:
148 // 2
149 // 3---2 x---x
150 // | | 3| |1
151 // 0---1 x---x
152 // 0
153
154 cell->vertex_ids = {vmap[i][j], vmap[i][j + 1], vmap[i + 1][j + 1], vmap[i + 1][j]};
155
156 for (int v = 0; v < 4; ++v)
157 {
158 UnpartitionedMesh::LightWeightFace face;
159
160 if (v < 3)
161 face.vertex_ids = std::vector<uint64_t>{cell->vertex_ids[v], cell->vertex_ids[v + 1]};
162 else
163 face.vertex_ids = std::vector<uint64_t>{cell->vertex_ids[v], cell->vertex_ids[0]};
164
165 // boundary logic
166 if (j == (Nx - 2) and v == 1) face.neighbor = 0 /*XMAX*/;
167 if (j == 0 and v == 3) face.neighbor = 1 /*XMIN*/;
168 if (i == (Ny - 2) and v == 2) face.neighbor = 2 /*YMAX*/;
169 if (i == 0 and v == 0) face.neighbor = 3 /*YMIN*/;
170
171 cell->faces.push_back(face);
172 }
173
174 umesh->AddCell(cell);
175 } // for j
176 } // for i
177
178 umesh->ComputeCentroidsAndCheckQuality();
179 umesh->BuildMeshConnectivity();
180
181 unpartitionedmesh_stack.push_back(umesh);
182
183 // Create meshers
184 handler.SetSurfaceMesher(std::make_shared<SurfaceMesherPredefined>());
185 handler.SetVolumeMesher(std::make_shared<VolumeMesherPredefinedUnpartitioned>(umesh));

189 return unpartitionedmesh_stack.size() - 1;
