69void
70SplitFileMeshGenerator::Execute()
71{
72 const int num_mpi = opensn::mpi_comm.size();
73 const int num_parts = num_mpi == 1 ? num_parts_ : num_mpi;
74
75 if (opensn::mpi_comm.rank() == 0 and (not read_only_))
76 {
77 // Execute all input generators
78 // Note these could be empty
79 std::unique_ptr<UnpartitionedMesh> current_umesh = nullptr;
80 for (auto mesh_generator_ptr : inputs_)
81 {
82 auto new_umesh = mesh_generator_ptr->GenerateUnpartitionedMesh(std::move(current_umesh));
83 current_umesh = std::move(new_umesh);
84 }
85
86 // Generate final umesh
87 current_umesh = GenerateUnpartitionedMesh(std::move(current_umesh));
88
89 log.Log() << "Writing split-mesh with " << num_parts << " parts";
90 const auto cell_pids = PartitionMesh(*current_umesh, num_parts);
91 WriteSplitMesh(cell_pids, *current_umesh, num_parts);
92 log.Log() << "Split-mesh with " << num_parts << " parts successfully created";
93 } // if home location
94
95 // Other locations wait here for files to be written
96 opensn::mpi_comm.barrier();
97
98 if (opensn::mpi_comm.size() == num_parts)
99 {
100 log.Log() << "Reading split-mesh";
101 auto mesh_info = ReadSplitMesh();
102
103 auto grid_ptr = SetupLocalMesh(mesh_info);
104
105 auto new_mesher = std::make_shared<VolumeMesher>(VolumeMesherType::UNPARTITIONED);
106 new_mesher->SetContinuum(grid_ptr);
107
108 if (current_mesh_handler < 0) PushNewHandlerAndGetIndex();
109
110 auto& cur_hndlr = GetCurrentHandler();
111 cur_hndlr.SetVolumeMesher(new_mesher);
112 log.Log() << "Done reading split-mesh files";
113 }
114 else
115 {
116 log.Log0Warning() << "After creating a split-mesh with mpi-processes < "
117 "num_parts the program will now auto terminate. This is not an error "
118 "and is the default behavior for the SplitFileMeshGenerator.\n"
119 << log.GetTimingBlock(opensn::name).MakeGraphString();
120 Exit(EXIT_SUCCESS);
121 }
122
123 opensn::mpi_comm.barrier();
124}

126void
127SplitFileMeshGenerator::WriteSplitMesh(const std::vector<int64_t>& cell_pids,
128 const UnpartitionedMesh& umesh,
129 int num_parts)
130{
131 const std::filesystem::path dir_path = std::filesystem::absolute(split_mesh_dir_path_);
132
133 const auto parent_path = dir_path.parent_path();
134 ChiInvalidArgumentIf(not std::filesystem::exists(parent_path),
135 "Parent path " + parent_path.string() + " does not exist");
136
137 bool root_dir_created = true;
138 if (not std::filesystem::exists(dir_path))
139 root_dir_created = std::filesystem::create_directories(dir_path);
140
141 ChiLogicalErrorIf(not root_dir_created, "Failed to create directory " + dir_path.string());
142
143 const auto& vertex_subs = umesh.GetVertextCellSubscriptions();
144 const auto& raw_cells = umesh.GetRawCells();
145 const auto& raw_vertices = umesh.GetVertices();
146
147 auto& t_write = log.CreateOrGetTimingBlock("FileMeshGenerator::WriteSplitMesh");
148 auto& t_sorting = log.CreateOrGetTimingBlock("Sorting data", "FileMeshGenerator::WriteSplitMesh");
149 auto& t_cells = log.CreateOrGetTimingBlock("WriteCells", "FileMeshGenerator::WriteSplitMesh");
150 auto& t_verts = log.CreateOrGetTimingBlock("WriteVerts", "FileMeshGenerator::WriteSplitMesh");
151 auto& t_serialize = log.CreateOrGetTimingBlock("Serialize");
152
153 uint64_t aux_counter = 0;
154 for (int pid = 0; pid < num_parts; ++pid)
155 {
156 t_write.TimeSectionBegin();
157 const std::filesystem::path file_path =
158 dir_path.string() + "/" + split_file_prefix_ + "_" + std::to_string(pid) + ".cmesh";
159
160 std::ofstream ofile(file_path.string(), std::ios_base::binary | std::ios_base::out);
161
162 ChiLogicalErrorIf(not ofile.is_open(), "Failed to open " + file_path.string());
163
164 // Appropriate cells and vertices to the current part being writting
165 t_sorting.TimeSectionBegin();
166
167 std::vector<uint64_t> local_cells_needed;
168 std::set<uint64_t> cells_needed;
169 std::set<uint64_t> vertices_needed;
170 {
171 local_cells_needed.reserve(raw_cells.size() / num_parts);
172 {
173 uint64_t cell_global_id = 0;
174 for (auto cell_pid : cell_pids)
175 {
176 if (cell_pid == pid) local_cells_needed.push_back(cell_global_id);
177 ++cell_global_id;
178 }
179 }
180
181 for (uint64_t cell_global_id : local_cells_needed)
182 {
183 cells_needed.insert(cell_global_id);
184
185 const auto& raw_cell = *raw_cells[cell_global_id];
186
187 for (uint64_t vid : raw_cell.vertex_ids)
188 {
189 vertices_needed.insert(vid);
190 for (uint64_t ghost_gid : vertex_subs[vid])
191 {
192 if (ghost_gid == cell_global_id) continue;
193 cells_needed.insert(ghost_gid);
194
195 const auto& ghost_raw_cell = *raw_cells[ghost_gid];
196 for (uint64_t gvid : ghost_raw_cell.vertex_ids)
197 vertices_needed.insert(gvid);
198 }
199 }
200 }
201 }
202 t_sorting.TimeSectionEnd();
203
204 if (verbosity_level_ >= 2)
205 log.Log() << "Writing part " << pid << " num_local_cells=" << local_cells_needed.size();
206
207 // Write mesh attributes and general info
208 const auto& mesh_options = umesh.GetMeshOptions();
209
210 WriteBinaryValue(ofile, num_parts); // int
211
212 WriteBinaryValue(ofile, static_cast<int>(umesh.GetMeshAttributes())); // int
213 WriteBinaryValue(ofile, mesh_options.ortho_Nx); // size_t
214 WriteBinaryValue(ofile, mesh_options.ortho_Ny); // size_t
215 WriteBinaryValue(ofile, mesh_options.ortho_Nz); // size_t
216
217 WriteBinaryValue(ofile, raw_vertices.size()); // size_t
218
219 // Write the boundary map
220 const auto& bndry_map = mesh_options.boundary_id_map;
221 WriteBinaryValue(ofile, bndry_map.size()); // size_t
222 for (const auto& [bid, bname] : bndry_map)
223 {
224 WriteBinaryValue(ofile, bid); // uint64_t
225 const size_t num_chars = bname.size();
226 WriteBinaryValue(ofile, num_chars); // size_t
227 ofile.write(bname.data(), static_cast<int>(num_chars)); // characters
228 }
229
230 // Write how many cells and vertices in file
231 WriteBinaryValue(ofile, cells_needed.size()); // size_t
232 WriteBinaryValue(ofile, vertices_needed.size()); // size_t
233
234 // Write cells
235 const size_t BUFFER_SIZE = 4096 * 2;
236 ByteArray serial_data;
237 serial_data.Data().reserve(BUFFER_SIZE * 2);
238 for (const auto& cell_global_id : cells_needed)
239 {
240 t_cells.TimeSectionBegin();
241 t_serialize.TimeSectionBegin();
242 const auto& cell = *raw_cells[cell_global_id];
243 serial_data.Write(static_cast<int>(cell_pids[cell_global_id])); // int
244 serial_data.Write(cell_global_id); // uint64_t
245 SerializeCell(cell, serial_data);
246 t_serialize.TimeSectionEnd();
247 if (serial_data.Size() > BUFFER_SIZE)
248 {
249 ofile.write((char*)serial_data.Data().data(), static_cast<int>(serial_data.Size()));
250 const size_t cap = serial_data.Data().capacity();
251 serial_data.Clear();
252 serial_data.Data().reserve(cap);
253 }
254 t_cells.TimeSectionEnd();
255 }
256 if (serial_data.Size() > 0)
257 {
258 ofile.write((char*)serial_data.Data().data(), static_cast<int>(serial_data.Size()));
259 serial_data.Clear();
260 }
261
262 // Write vertices
263 t_verts.TimeSectionBegin();
264 for (const uint64_t vid : vertices_needed)
265 {
266 serial_data.Write(vid); // uint64_t
267 serial_data.Write(raw_vertices[vid]);
268 if (serial_data.Size() > BUFFER_SIZE)
269 {
270 ofile.write((char*)serial_data.Data().data(), static_cast<int>(serial_data.Size()));
271 serial_data.Clear();
272 }
273 }
274 if (serial_data.Size() > 0)
275 {
276 ofile.write((char*)serial_data.Data().data(), static_cast<int>(serial_data.Size()));
277 serial_data.Clear();
278 }
279 t_verts.TimeSectionEnd();
280
281 ofile.close();
282 t_write.TimeSectionEnd();
283
284 const double fraction_complete = static_cast<double>(pid) / static_cast<double>(num_parts);
285 if (fraction_complete >= static_cast<double>(aux_counter + 1) * 0.1)
286 {
287 if (verbosity_level_ >= 1)
288 log.Log() << program_timer.GetTimeString() << " Surpassing part " << pid << " of "
289 << num_parts << " (" << (aux_counter + 1) * 10 << "%)";
290 ++aux_counter;
291 }
292 } // for p
293}

317SplitFileMeshGenerator::SplitMeshInfo
318SplitFileMeshGenerator::ReadSplitMesh()
319{
320 const int pid = opensn::mpi_comm.rank();
321 const std::filesystem::path dir_path = std::filesystem::absolute(split_mesh_dir_path_);
322 const std::filesystem::path file_path =
323 dir_path.string() + "/" + split_file_prefix_ + "_" + std::to_string(pid) + ".cmesh";
324
325 SplitMeshInfo info_block;
326 auto& cells = info_block.cells_;
327 auto& vertices = info_block.vertices_;
328 std::ifstream ifile(file_path, std::ios_base::binary | std::ios_base::in);
329
330 ChiLogicalErrorIf(not ifile.is_open(), "Failed to open " + file_path.string());
331
332 // Read mesh attributes and general info
333 const size_t file_num_parts = ReadBinaryValue<int>(ifile);
334
335 ChiLogicalErrorIf(opensn::mpi_comm.size() != file_num_parts,
336 "Split mesh files with prefix \"" + split_file_prefix_ +
337 "\" has been created with " + std::to_string(file_num_parts) +
338 " parts but is now being read with " +
339 std::to_string(opensn::mpi_comm.size()) + " processes.");
340
341 info_block.mesh_attributes_ = ReadBinaryValue<int>(ifile);
342 info_block.ortho_Nx_ = ReadBinaryValue<size_t>(ifile);
343 info_block.ortho_Ny_ = ReadBinaryValue<size_t>(ifile);
344 info_block.ortho_Nz_ = ReadBinaryValue<size_t>(ifile);
345
346 info_block.num_global_vertices_ = ReadBinaryValue<size_t>(ifile);
347
348 // Read boundary map
349 const size_t num_bndries = ReadBinaryValue<size_t>(ifile);
350 for (size_t b = 0; b < num_bndries; ++b)
351 {
352 const uint64_t bid = ReadBinaryValue<uint64_t>(ifile);
353 const size_t num_chars = ReadBinaryValue<size_t>(ifile);
354 std::string bname(num_chars, ' ');
355 ifile.read(bname.data(), static_cast<int>(num_chars));
356
357 info_block.boundary_id_map_.insert(std::make_pair(bid, bname));
358 }
359
360 // Write how many cells and vertices in file
361 const size_t num_cells = ReadBinaryValue<size_t>(ifile);
362 const size_t num_vertices = ReadBinaryValue<size_t>(ifile);
363
364 // Read the cells
365 for (size_t c = 0; c < num_cells; ++c)
366 {
367 const int cell_pid = ReadBinaryValue<int>(ifile);
368 const uint64_t cell_gid = ReadBinaryValue<uint64_t>(ifile);
369 const CellType cell_type = ReadBinaryValue<CellType>(ifile);
370 const CellType cell_sub_type = ReadBinaryValue<CellType>(ifile);
371
372 UnpartitionedMesh::LightWeightCell new_cell(cell_type, cell_sub_type);
373
374 new_cell.centroid = ReadBinaryValue<Vector3>(ifile);
375 new_cell.material_id = ReadBinaryValue<int>(ifile);
376
377 const size_t num_vids = ReadBinaryValue<size_t>(ifile);
378 for (size_t v = 0; v < num_vids; ++v)
379 new_cell.vertex_ids.push_back(ReadBinaryValue<uint64_t>(ifile));
380
381 const size_t num_faces = ReadBinaryValue<size_t>(ifile);
382 for (size_t f = 0; f < num_faces; ++f)
383 {
384 UnpartitionedMesh::LightWeightFace new_face;
385 const size_t num_face_vids = ReadBinaryValue<size_t>(ifile);
386 for (size_t v = 0; v < num_face_vids; ++v)
387 new_face.vertex_ids.push_back(ReadBinaryValue<uint64_t>(ifile));
388
389 new_face.has_neighbor = ReadBinaryValue<bool>(ifile);
390 new_face.neighbor = ReadBinaryValue<uint64_t>(ifile);
391
392 new_cell.faces.push_back(std::move(new_face));
393 } // for f
394
395 cells.insert(std::make_pair(CellPIDGID(cell_pid, cell_gid), std::move(new_cell)));
396 } // for cell c
397
398 // Read the vertices
399 for (size_t v = 0; v < num_vertices; ++v)
400 {
401 const uint64_t vid = ReadBinaryValue<uint64_t>(ifile);
402 const Vector3 vertex = ReadBinaryValue<Vector3>(ifile);
403 vertices.insert(std::make_pair(vid, vertex));
404 } // for vertex v
405
406 ifile.close();
407
408 return info_block;
409}

411std::shared_ptr<MeshContinuum>
412SplitFileMeshGenerator::SetupLocalMesh(SplitMeshInfo& mesh_info)
413{
414 auto grid_ptr = MeshContinuum::New();
415
416 grid_ptr->GetBoundaryIDMap() = mesh_info.boundary_id_map_;
417
418 auto& cells = mesh_info.cells_;
419 auto& vertices = mesh_info.vertices_;
420
421 for (const auto& [vid, vertex] : vertices)
422 grid_ptr->vertices.Insert(vid, vertex);
423
424 for (const auto& [pidgid, raw_cell] : cells)
425 {
426 const auto& [cell_pid, cell_global_id] = pidgid;
427 auto cell = SetupCell(raw_cell, cell_global_id, cell_pid, STLVertexListHelper(vertices));
428
429 grid_ptr->cells.push_back(std::move(cell));
430 }
431
432 SetGridAttributes(*grid_ptr,
433 static_cast<MeshAttributes>(mesh_info.mesh_attributes_),
434 {mesh_info.ortho_Nx_, mesh_info.ortho_Ny_, mesh_info.ortho_Nz_});
435
436 grid_ptr->SetGlobalVertexCount(mesh_info.num_global_vertices_);
437
438 ComputeAndPrintStats(*grid_ptr);
439
440 return grid_ptr;
441}
